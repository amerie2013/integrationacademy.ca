{\LARGE\bfseries\color{titleblue}Worksheet 4.2 --- Concavity \& the Second Derivative}\\[2pt]
{\small Grade 12 Calculus and Vectors (MCV4U) \textperiodcentered{} Unit 4: Curve Sketching}
\end{center}

\begin{ideabox}
$f''>0$ concave up, $f''<0$ concave down; inflection where $f''$ changes sign. 2nd-derivative test: $f''(c)>0$ min, $f''(c)<0$ max.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Concave up: $f''>0$; down: $f''<0$.
  \item Inflection: $f''$ changes sign.
  \item 2nd-derivative test classifies critical points.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{What concavity means}\textbf{Concave up} curves like a cup ($f''>0$); \textbf{concave down} curves like a cap ($f''<0$). The second derivative reads the bend.
\lh{Inflection points}Where concavity changes sign is an \textbf{inflection point} — the curve switches from cupping up to down or vice versa.
\lh{The second derivative test}At a critical number, $f''>0$ signals a local minimum and $f''<0$ a local maximum.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Always concave up}
Concavity of $f(x)=x^2$.\soln $f''=2>0$: concave up everywhere.
\end{example}

\begin{example}{ex1}{Example 2 --- Switches}
Concavity of $f(x)=x^3$.\soln $f''=6x$: down for $x<0$, up for $x>0$, inflection at $(0,0)$:\eplot{-2.5}{2.5}{-8}{8}{\addplot[exblue,very thick,domain=-2:2]{x^3};\addplot[qorange,very thick,domain=-1.3:1.3]{6*x};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(0,0)};}
\end{example}

\begin{example}{ex2}{Example 3 --- 2nd-derivative test}
Classify the critical point of $f(x)=x^2-4x$.\soln $f'=2x-4=0\Rightarrow x=2$; $f''=2>0$: local min.
\end{example}

\begin{example}{ex3}{Example 4 --- Inflection}
Inflection point of $f(x)=x^3-3x^2$.\soln $f''=6x-6=0\Rightarrow x=1$; $f(1)=-2$: $(1,-2)$.
\end{example}

\begin{example}{ex4}{Example 5 --- Classify all}
Use $f''$ on $f(x)=x^4-2x^2$.\soln $f''=12x^2-4$; $f''(0)=-4$ (max), $f''(\pm1)=8$ (min).
\end{example}

\begin{example}{ex5}{Example 6 --- Concave down}
Concavity of $f(x)=-x^2$.\soln $f''=-2<0$: concave down everywhere.
\end{example}

\begin{example}{ex6}{Example 7 --- Concave up where}
Where is $f(x)=x^3$ concave up?\soln $6x>0\Rightarrow x>0$.
\end{example}

\begin{example}{ex7}{Example 8 --- 2nd-derivative test}
Classify the critical point of $f(x)=x^2+6x$.\soln $x=-3$; $f''=2>0$: local min.
\end{example}

\begin{example}{ex8}{Example 9 --- Inflection}
Inflection point of $f(x)=x^3-6x^2$.\soln $f''=6x-12=0\Rightarrow x=2$; $f(2)=-16$: $(2,-16)$.
\end{example}
