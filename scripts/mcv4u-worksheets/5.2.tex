\begin{ideabox}
Write an equation relating the quantities, differentiate both sides with respect to $t$, then substitute the instant and solve for the unknown rate.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Identify known/unknown rates.
  \item Relate the quantities; differentiate w.r.t. $t$ (chain rule).
  \item Substitute the instant's values; solve.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Rates linked in time}In a \textbf{related rates} problem several quantities change together over time, and their rates are connected by an equation.
\lh{Differentiate implicitly}Differentiate the relating equation with respect to time $t$, so each variable's rate appears through the chain rule.
\lh{Substitute last}Plug in the known values only \emph{after} differentiating, then solve for the unknown rate.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Expanding circle}
$\dfrac{dr}{dt}=2$ cm/s. Find $\dfrac{dA}{dt}$ at $r=5$.\soln $A=\pi r^2\Rightarrow\dfrac{dA}{dt}=2\pi r\dfrac{dr}{dt}=2\pi(5)(2)=20\pi\approx62.8$ cm²/s.
\end{example}

\begin{example}{ex1}{Example 2 --- Sliding ladder}
A 13 m ladder's base slides out at 2 m/s. Find the top's rate when the base is 5 m out.\soln $x^2+y^2=169$, $y=12$; $2xx'+2yy'=0\Rightarrow y'=-\dfrac{5(2)}{12}=-\dfrac56$ m/s.\rtri{5}{12}{$y$}{$x$}{$13$}
\end{example}

\begin{example}{ex2}{Example 3 --- Inflating balloon}
$\dfrac{dV}{dt}=100$ cm³/s. Find $\dfrac{dr}{dt}$ at $r=5$.\soln $V=\tfrac43\pi r^3\Rightarrow\dfrac{dV}{dt}=4\pi r^2\dfrac{dr}{dt}$; $100=4\pi(25)\dfrac{dr}{dt}\Rightarrow\dfrac{dr}{dt}=\dfrac1\pi$ cm/s.
\end{example}

\begin{example}{ex3}{Example 4 --- Growing square}
A square's side grows at 3 cm/s. Find $\dfrac{dA}{dt}$ at $s=10$.\soln $A=s^2\Rightarrow 2s\dfrac{ds}{dt}=2(10)(3)=60$ cm²/s.
\end{example}

\begin{example}{ex4}{Example 5 --- Two cars}
Car A north at 60, car B east at 80 km/h from a point. Find the rate the distance grows after 1 h.\soln $D=100$; $2DD'=2aa'+2bb'\Rightarrow D'=\dfrac{60(60)+80(80)}{100}=100$ km/h.
\end{example}

\begin{example}{ex5}{Example 6 --- Expanding circle}
$\dfrac{dr}{dt}=3$ cm/s. Find $\dfrac{dA}{dt}$ at $r=4$.\soln $2\pi(4)(3)=24\pi$ cm²/s.
\end{example}

\begin{example}{ex6}{Example 7 --- Growing square}
Side grows at 2 cm/s. Find $\dfrac{dA}{dt}$ at $s=6$.\soln $2(6)(2)=24$ cm²/s.
\end{example}

\begin{example}{ex7}{Example 8 --- Inflating balloon}
$\dfrac{dV}{dt}=36\pi$ cm³/s. Find $\dfrac{dr}{dt}$ at $r=3$.\soln $36\pi=4\pi(9)\dfrac{dr}{dt}\Rightarrow\dfrac{dr}{dt}=1$ cm/s.
\end{example}

\begin{example}{ex8}{Example 9 --- Sliding ladder}
A 10 m ladder's base slides out at 1 m/s. Find the top's rate at base $=6$ m.\soln $y=8$; $y'=-\dfrac{6(1)}{8}=-\dfrac34$ m/s.
\end{example}
