\begin{ideabox}
Rewrite as powers; differentiate again for $f''$; use implicit differentiation when $y$ is tangled with $x$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Rewrite: $\dfrac1{x^k}=x^{-k}$, $\sqrt[n]{x^m}=x^{m/n}$.
  \item Second derivative $f''=(f')'$.
  \item Implicit: differentiate both sides, attach $\tfrac{dy}{dx}$ to every $y$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Rewrite then differentiate}Rational and radical functions are easiest as powers: write $\dfrac{1}{x}=x^{-1}$ and $\sqrt{x}=x^{1/2}$, then apply the power rule.
\lh{Combine rules}These often need the chain, product, or quotient rule together — identify the structure before differentiating.
\lh{Higher-order derivatives}Differentiating again gives $f''(x)$, the rate of change of the slope, and so on for further derivatives.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Rational via powers}
Differentiate $f(x)=\dfrac1{x^2}$.\soln $x^{-2}\to-2x^{-3}=-\dfrac2{x^3}$.
\end{example}

\begin{example}{ex1}{Example 2 --- Radical via powers}
Differentiate $f(x)=\sqrt[3]{x^2}$.\soln $x^{2/3}\to\tfrac23x^{-1/3}=\dfrac2{3\sqrt[3]{x}}$.
\end{example}

\begin{example}{ex2}{Example 3 --- Second derivative}
Find $f''(x)$ for $f(x)=x^4$.\soln $f'=4x^3,\ f''=12x^2$. For $x^3$: $f,f',f''$ are $x^3,3x^2,6x$ — each the slope of the previous:\eplot{-2.5}{2.5}{-8}{8}{\addplot[exblue,very thick,domain=-2:2]{x^3};\addplot[qorange,very thick,domain=-1.6:1.6]{3*x^2};\addplot[red,very thick,domain=-1.3:1.3]{6*x};}
\end{example}

\begin{example}{ex3}{Example 4 --- Second derivative}
Find $f''(x)$ for $f(x)=x^3-3x$ and $f''(1)$.\soln $f'=3x^2-3,\ f''=6x,\ f''(1)=6$.
\end{example}

\begin{example}{ex4}{Example 5 --- Implicit differentiation}
Find $\dfrac{dy}{dx}$ for $x^2+y^2=25$.\soln $2x+2yy'=0\Rightarrow y'=-\dfrac xy$.
\end{example}

\begin{example}{ex5}{Example 6 --- Rational via powers}
Differentiate $f(x)=\dfrac1{x^4}$.\soln $x^{-4}\to-4x^{-5}=-\dfrac4{x^5}$.
\end{example}

\begin{example}{ex6}{Example 7 --- Radical via powers}
Differentiate $f(x)=x^{3/2}$.\soln $\tfrac32x^{1/2}$.
\end{example}

\begin{example}{ex7}{Example 8 --- Second derivative}
Find $f''(x)$ for $f(x)=2x^3-x^2$.\soln $f'=6x^2-2x,\ f''=12x-2$.
\end{example}

\begin{example}{ex8}{Example 9 --- Implicit differentiation}
Find $\dfrac{dy}{dx}$ for $xy=4$.\soln Product rule: $y+xy'=0\Rightarrow y'=-\dfrac yx$.
\end{example}
