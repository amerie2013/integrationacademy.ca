\begin{ideabox}
$v=s'$, $a=v'=s''$, speed $=|v|$. At rest when $v=0$; speeding up when $v,a$ share a sign.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $v(t)=s'(t)$, $a(t)=s''(t)$.
  \item At rest: $v=0$; direction: sign of $v$.
  \item Speeding up: $v,a$ same sign; slowing: opposite.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Position, velocity, acceleration}If $s(t)$ is position, then velocity is $v(t)=s'(t)$ and acceleration is $a(t)=v'(t)=s''(t)$ â€” each a derivative of the last.
\lh{Reading motion}Velocity's sign gives direction; where $v=0$ the object is momentarily at rest and may turn around.
\lh{Speeding up or slowing}The object speeds up when velocity and acceleration share a sign, and slows when their signs differ.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- v and a}
For $s(t)=t^2-4t$, find $v$ and $a$.\soln $v=2t-4$, $a=2$. Velocity is the slope of position; at rest at the lowest point ($t=2$):\eplot{-1}{5}{-5}{6}{\addplot[exblue,very thick,domain=0:4.5]{x^2-4*x};\addplot[qorange,very thick,domain=0:4.5]{2*x-4};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(2,-4)};}
\end{example}

\begin{example}{ex1}{Example 2 --- At rest}
When is $s(t)=t^2-4t$ at rest?\soln $v=2t-4=0\Rightarrow t=2$ s.
\end{example}

\begin{example}{ex2}{Example 3 --- Multiple rest times}
When is $s(t)=t^3-6t^2+9t$ at rest?\soln $v=3t^2-12t+9=3(t-1)(t-3)=0\Rightarrow t=1,3$.
\end{example}

\begin{example}{ex3}{Example 4 --- Direction \& accel}
Describe the motion of $s(t)=t^3-6t^2+9t$ at $t=0$.\soln $v(0)=9>0$ (moving right); $a=6t-12$, $a(0)=-12<0$ (decelerating).
\end{example}

\begin{example}{ex4}{Example 5 --- Speeding up?}
Is $s(t)=t^3-6t^2+9t$ speeding up at $t=4$?\soln $v(4)=9>0$, $a(4)=12>0$; same sign â‡’ speeding up.
\end{example}

\begin{example}{ex5}{Example 6 --- Velocity}
For $s(t)=t^2-6t$, find $v(t)$.\soln $2t-6$.
\end{example}

\begin{example}{ex6}{Example 7 --- At rest}
When is $s(t)=t^2-6t$ at rest?\soln $2t-6=0\Rightarrow t=3$.
\end{example}

\begin{example}{ex7}{Example 8 --- Acceleration}
For $s(t)=t^3-3t^2$, find $a(t)$.\soln $v=3t^2-6t$, $a=6t-6$.
\end{example}

\begin{example}{ex8}{Example 9 --- Max height}
A projectile has $h(t)=-5t^2+20t$. When is it highest?\soln $v=-10t+20=0\Rightarrow t=2$ s.
\end{example}
