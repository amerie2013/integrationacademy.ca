{\LARGE\bfseries\color{titleblue}Worksheet 4.1 --- Increasing/Decreasing \& the First Derivative Test}\\[2pt]
{\small Grade 12 Calculus and Vectors (MCV4U) \textperiodcentered{} Unit 4: Curve Sketching}
\end{center}

\begin{ideabox}
$f$ increases where $f'>0$ and decreases where $f'<0$. Critical numbers ($f'=0$ or undefined) are classified by how $f'$ changes sign.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Increasing: $f'>0$; decreasing: $f'<0$.
  \item Critical numbers: $f'=0$ or undefined.
  \item 1st-derivative test: $+\to-$ max, $-\to+$ min, no change ⇒ neither.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Sign of the derivative}Where $f'(x)>0$ the function \textbf{increases}; where $f'(x)<0$ it \textbf{decreases}. The derivative's sign reads the graph's direction.
\lh{Critical numbers}Points where $f'(x)=0$ or is undefined are \textbf{critical numbers} — the only places a local max or min can occur.
\lh{The first derivative test}If $f'$ changes from $+$ to $-$ at a critical number it is a local maximum; from $-$ to $+$ it is a local minimum.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Intervals}
Where is $f(x)=x^2-4x$ increasing/decreasing?\soln $f'=2x-4$; decreasing on $(-\infty,2)$, increasing on $(2,\infty)$.
\end{example}

\begin{example}{ex1}{Example 2 --- Critical numbers}
Critical numbers of $f(x)=x^3-3x$.\soln $3x^2-3=0\Rightarrow x=\pm1$. $f$ rises where $f'>0$, falls where $f'<0$:\eplot{-3}{3}{-6}{6}{\addplot[exblue,very thick,domain=-2.2:2.2]{x^3-3*x};\addplot[qorange,very thick,domain=-1.7:1.7]{3*x^2-3};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(-1,2) (1,-2)};}
\end{example}

\begin{example}{ex2}{Example 3 --- Classify}
Classify the critical points of $f(x)=x^3-3x$.\soln $x=-1$: $+\to-$, local max $(-1,2)$; $x=1$: $-\to+$, local min $(1,-2)$.
\end{example}

\begin{example}{ex3}{Example 4 --- Neither}
Classify the critical point of $f(x)=x^3$.\soln $f'=3x^2=0$ at $0$, but no sign change ⇒ neither (inflection).
\end{example}

\begin{example}{ex4}{Example 5 --- Several}
Find and classify the critical points of $f(x)=x^4-2x^2$.\soln $4x(x-1)(x+1)=0\Rightarrow x=0,\pm1$; signs $-,+,-,+$: min at $\pm1$ ($-1$), max at $0$ ($0$).
\end{example}

\begin{example}{ex5}{Example 6 --- Intervals}
Where is $f(x)=x^2-6x$ decreasing?\soln $2x-6<0\Rightarrow x<3$.
\end{example}

\begin{example}{ex6}{Example 7 --- Critical numbers}
Critical numbers of $f(x)=x^3-12x$.\soln $3x^2-12=0\Rightarrow x=\pm2$.
\end{example}

\begin{example}{ex7}{Example 8 --- Always increasing}
Is $f(x)=x^3+x$ ever decreasing?\soln $f'=3x^2+1>0$ always — never decreasing.
\end{example}

\begin{example}{ex8}{Example 9 --- Classify}
Classify the critical point of $f(x)=x^2-4x$.\soln $x=2$: $-\to+$, local min.
\end{example}
