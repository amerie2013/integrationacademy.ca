{\LARGE\bfseries\color{titleblue}Worksheet 2.2 --- Product \& Quotient Rules}\\[2pt]
{\small Grade 12 Calculus and Vectors (MCV4U) \textperiodcentered{} Unit 2: Derivative Rules}
\end{center}

\begin{ideabox}
Product: $(fg)'=f'g+fg'$. Quotient: $\left(\dfrac fg\right)'=\dfrac{f'g-fg'}{g^2}$ — top derivative first.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $(fg)'=f'g+fg'$ (not $f'g'$).
  \item $\left(\dfrac fg\right)'=\dfrac{f'g-fg'}{g^2}$.
  \item If expanding is easy, expand and use the power rule.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Products}The derivative of a product is \emph{not} the product of derivatives: $\dfrac{d}{dx}[fg]=f'g+fg'$.
\lh{Quotients}For a quotient, $\dfrac{d}{dx}\!\left[\dfrac{f}{g}\right]=\dfrac{f'g-fg'}{g^2}$ — mind the order in the numerator.
\lh{Choosing a rule}Identify the structure first: a product calls for the product rule, a fraction for the quotient rule.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Product rule}
Differentiate $f(x)=x^2(x+1)$.\soln $2x(x+1)+x^2(1)=3x^2+2x$ (check: $x^3+x^2\to3x^2+2x$).
\end{example}

\begin{example}{ex1}{Example 2 --- Two factors}
Differentiate $f(x)=(2x+1)(x^2-3)$.\soln $2(x^2-3)+(2x+1)(2x)=6x^2+2x-6$.
\end{example}

\begin{example}{ex2}{Example 3 --- Quotient rule}
Differentiate $f(x)=\dfrac{x+1}{x-1}$.\soln $\dfrac{(x-1)-(x+1)}{(x-1)^2}=\dfrac{-2}{(x-1)^2}$.
\end{example}

\begin{example}{ex3}{Example 4 --- Polynomial over linear}
Differentiate $f(x)=\dfrac{x^2}{x+1}$.\soln $\dfrac{2x(x+1)-x^2}{(x+1)^2}=\dfrac{x^2+2x}{(x+1)^2}$.
\end{example}

\begin{example}{ex4}{Example 5 --- From a table}
Given $f(2)=3,f'(2)=1,g(2)=4,g'(2)=-2$, find $(fg)'(2)$.\soln $(1)(4)+(3)(-2)=-2$.
\end{example}

\begin{example}{ex5}{Example 6 --- Product rule}
Differentiate $f(x)=x^3(x-2)$.\soln $3x^2(x-2)+x^3=4x^3-6x^2$.
\end{example}

\begin{example}{ex6}{Example 7 --- Two factors}
Differentiate $f(x)=(x+2)(x^2+1)$.\soln $(x^2+1)+(x+2)(2x)=3x^2+4x+1$.
\end{example}

\begin{example}{ex7}{Example 8 --- Quotient rule}
Differentiate $f(x)=\dfrac{x}{x+2}$.\soln $\dfrac{(x+2)-x}{(x+2)^2}=\dfrac{2}{(x+2)^2}$.
\end{example}

\begin{example}{ex8}{Example 9 --- Quotient rule}
Differentiate $f(x)=\dfrac{x-1}{x+1}$.\soln $\dfrac{(x+1)-(x-1)}{(x+1)^2}=\dfrac{2}{(x+1)^2}$.
\end{example}
