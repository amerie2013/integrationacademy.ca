\begin{ideabox}
$\displaystyle\lim_{x\to a}f(x)=L$ is the value $f$ approaches near $a$. Substitute when continuous; for $\tfrac00$ factor or rationalize first.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Direct substitution when continuous.
  \item $\tfrac00$ indeterminate $\Rightarrow$ factor / rationalize, then substitute.
  \item One-sided limits must agree; at infinity, divide by the highest power.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{What a limit is}$\displaystyle\lim_{x\to a}f(x)=L$ means $f(x)$ gets arbitrarily close to $L$ as $x$ approaches $a$ — regardless of the value (or gap) at $a$ itself.
\lh{One-sided limits}The limit exists only when the left-hand and right-hand limits agree; a jump between them means the limit does not exist.
\lh{Indeterminate forms}A form like $\tfrac00$ signals more work — factor, rationalize, or simplify first, then substitute.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Direct substitution}
$\displaystyle\lim_{x\to3}(2x+1)$.\soln Continuous; substitute: $2(3)+1=7$.
\end{example}

\begin{example}{ex1}{Example 2 --- Factor out 0/0}
$\displaystyle\lim_{x\to2}\dfrac{x^2-4}{x-2}$.\soln $\tfrac00$; factor $\dfrac{(x-2)(x+2)}{x-2}=x+2\to4$. The graph is $y=x+2$ with a hole at $(2,4)$:\eplot{-1}{5}{-1}{7}{\addplot[exblue,very thick,domain=-1:1.9]{x+2};\addplot[exblue,very thick,domain=2.1:5]{x+2};\addplot[red,only marks,mark=o,mark size=2.4pt] coordinates {(2,4)};}
\end{example}

\begin{example}{ex2}{Example 3 --- Rationalize 0/0}
$\displaystyle\lim_{x\to0}\dfrac{\sqrt{x+9}-3}{x}$.\soln Multiply by the conjugate: $\dfrac{x}{x(\sqrt{x+9}+3)}=\dfrac1{\sqrt{x+9}+3}\to\dfrac16$.
\end{example}

\begin{example}{ex3}{Example 4 --- One-sided (DNE)}
$\displaystyle\lim_{x\to0}\dfrac{|x|}{x}$.\soln Right limit $=1$, left limit $=-1$; they differ, so the limit does not exist.
\end{example}

\begin{example}{ex4}{Example 5 --- Limit at infinity}
$\displaystyle\lim_{x\to\infty}\dfrac{3x^2-x}{x^2+5}$.\soln Divide by $x^2$: $\dfrac{3-1/x}{1+5/x^2}\to3$.
\end{example}

\begin{example}{ex5}{Example 6 --- Direct substitution}
$\displaystyle\lim_{x\to4}(x^2-1)$.\soln $16-1=15$.
\end{example}

\begin{example}{ex6}{Example 7 --- Factor}
$\displaystyle\lim_{x\to3}\dfrac{x^2-9}{x-3}$.\soln $x+3\to6$.
\end{example}

\begin{example}{ex7}{Example 8 --- At infinity}
$\displaystyle\lim_{x\to\infty}\dfrac{5x-2}{x+1}$.\soln Ratio of leading coefficients $\to5$.
\end{example}

\begin{example}{ex8}{Example 9 --- Factor}
$\displaystyle\lim_{x\to1}\dfrac{x^2+x-2}{x-1}$.\soln $\dfrac{(x+2)(x-1)}{x-1}=x+2\to3$.
\end{example}
