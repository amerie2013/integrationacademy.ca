\begin{ideabox}
$\dfrac{d}{dx}x^n=nx^{n-1}$, with the constant-multiple and sum/difference rules, differentiates any polynomial. Rewrite radicals and reciprocals as powers first.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Power: $\dfrac{d}{dx}x^n=nx^{n-1}$; constant: $\dfrac{d}{dx}c=0$.
  \item Constant multiple and sum/difference rules.
  \item Rewrite $\sqrt{x}=x^{1/2}$, $\dfrac1{x^k}=x^{-k}$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The power rule}For $f(x)=x^n$ the derivative is $f'(x)=nx^{\,n-1}$ â€” the workhorse of differentiation.
\lh{Constants and sums}A constant multiple factors out, $\dfrac{d}{dx}[cf]=cf'$, and derivatives add term by term, $\dfrac{d}{dx}[f+g]=f'+g'$.
\lh{Together}These rules differentiate any polynomial instantly, one term at a time.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Power rule}
Differentiate $f(x)=x^5$.\soln $f'(x)=5x^4$.
\end{example}

\begin{example}{ex1}{Example 2 --- Whole polynomial}
Differentiate $f(x)=3x^4-2x^2+7$.\soln $f'(x)=12x^3-4x$ (the constant vanishes). $f$ and $f'$ for $x^3-3x$ illustrate the idea â€” $f'$ is zero at the turning points:\eplot{-3}{3}{-6}{6}{\addplot[exblue,very thick,domain=-2.2:2.2]{x^3-3*x};\addplot[qorange,very thick,domain=-1.7:1.7]{3*x^2-3};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(-1,0) (1,0)};}
\end{example}

\begin{example}{ex2}{Example 3 --- Negative \& fractional}
Differentiate $f(x)=\sqrt{x}+\dfrac1{x^2}$.\soln $x^{1/2}+x^{-2}\Rightarrow \tfrac12x^{-1/2}-2x^{-3}=\dfrac1{2\sqrt{x}}-\dfrac2{x^3}$.
\end{example}

\begin{example}{ex3}{Example 4 --- Slope at a point}
Slope of the tangent to $f(x)=x^3-3x$ at $x=2$.\soln $f'(x)=3x^2-3\Rightarrow f'(2)=9$.
\end{example}

\begin{example}{ex4}{Example 5 --- Horizontal tangent}
Where does $f(x)=x^3-3x$ have a horizontal tangent?\soln $3x^2-3=0\Rightarrow x=\pm1$.
\end{example}

\begin{example}{ex5}{Example 6 --- Power rule}
Differentiate $f(x)=x^7$.\soln $7x^6$.
\end{example}

\begin{example}{ex6}{Example 7 --- Polynomial}
Differentiate $f(x)=4x^3-6x^2+x-9$.\soln $12x^2-12x+1$.
\end{example}

\begin{example}{ex7}{Example 8 --- Reciprocal power}
Differentiate $f(x)=\dfrac1{x^3}$.\soln $x^{-3}\to-3x^{-4}=-\dfrac3{x^4}$.
\end{example}

\begin{example}{ex8}{Example 9 --- Evaluate a derivative}
Find $f'(1)$ for $f(x)=x^4-2x$.\soln $f'(x)=4x^3-2\Rightarrow f'(1)=2$.
\end{example}
