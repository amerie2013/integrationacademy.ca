{\LARGE\bfseries\color{titleblue}Worksheet 7.2 --- Equations of Planes}\\[2pt]
{\small Grade 12 Calculus and Vectors (MCV4U) \textperiodcentered{} Unit 7: Lines and Planes}
\end{center}

\begin{ideabox}
A plane needs a point and a normal $\vec n=(A,B,C)$. Scalar: $Ax+By+Cz=D$ (the coefficients are the normal). Get a normal from two directions via the cross product.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Scalar: $Ax+By+Cz=D$, normal $(A,B,C)$.
  \item $A(x-x_0)+B(y-y_0)+C(z-z_0)=0$.
  \item Normal from directions: $\vec n=\vec u\times\vec v$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Point and normal}A plane is fixed by a point and a \textbf{normal vector} perpendicular to it; the normal's components give the \textbf{scalar equation} $ax+by+cz+d=0$.
\lh{Vector and parametric forms}Using two direction vectors lying in the plane, $\vec r=\vec r_0+s\,\vec u+t\,\vec v$ describes every point on it.
\lh{Reading the normal}The coefficients $(a,b,c)$ in the scalar equation are exactly the components of the normal vector.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- From point \& normal}
Plane through $(1,2,3)$ with normal $(2,1,-1)$.\soln $2(x-1)+(y-2)-(z-3)=0\Rightarrow 2x+y-z=1$.
\end{example}

\begin{example}{ex1}{Example 2 --- Read the normal}
Normal of $2x+y-z=5$.\soln $(2,1,-1)$.
\end{example}

\begin{example}{ex2}{Example 3 --- Point on the plane?}
Is $(1,1,1)$ on $2x+y-z=2$?\soln $2+1-1=2$ ✓: yes.
\end{example}

\begin{example}{ex3}{Example 4 --- Normal from directions}
Plane contains $\vec u=(1,0,0),\vec v=(0,1,0)$. Find a normal.\soln $\vec n=\vec u\times\vec v=(0,0,1)$.
\end{example}

\begin{example}{ex4}{Example 5 --- Through three points}
Plane through $A(0,0,0),B(1,0,0),C(0,1,0)$.\soln $\vec{AB}\times\vec{AC}=(0,0,1)$; equation $z=0$.
\end{example}

\begin{example}{ex5}{Example 6 --- From point \& normal}
Plane through $(0,0,0)$ with normal $(1,2,3)$.\soln $x+2y+3z=0$.
\end{example}

\begin{example}{ex6}{Example 7 --- Read the normal}
Normal of $3x-y+4z=7$.\soln $(3,-1,4)$.
\end{example}

\begin{example}{ex7}{Example 8 --- Point on the plane?}
Is $(2,0,1)$ on $x+y+z=3$?\soln $2+0+1=3$ ✓: yes.
\end{example}

\begin{example}{ex8}{Example 9 --- Vertical plane}
Plane through $(2,0,0)$ with normal $(1,0,0)$.\soln $x-2=0\Rightarrow x=2$.
\end{example}
