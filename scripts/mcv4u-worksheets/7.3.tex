{\LARGE\bfseries\color{titleblue}Worksheet 7.3 --- Intersections of Lines \& Planes}\\[2pt]
{\small Grade 12 Calculus and Vectors (MCV4U) \textperiodcentered{} Unit 7: Lines and Planes}
\end{center}

\begin{ideabox}
Line–plane: substitute the line into the plane and solve for $t$. Compare directions/normals to detect parallel, skew, or coincident.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Line–plane: substitute $x,y,z$; solve for $t$; back-substitute.
  \item Line $\parallel$ plane when $\vec d\cdot\vec n=0$.
  \item Planes $\parallel$ when normals are parallel.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{What can happen}Two lines, a line and a plane, or two planes can \textbf{intersect} in a point or line, be \textbf{parallel}, or (lines only) be \textbf{skew}.
\lh{Solving}Substitute the line's parametric form into the plane's equation, or solve the system of equations, to find any intersection.
\lh{Interpreting the outcome}No solution means parallel or skew; a unique solution gives a point; infinitely many means the objects coincide or a line lies in a plane.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Line meets a plane}
Where does $\vec r=(1,0,0)+t(0,0,1)$ meet $z=5$?\soln $z=t=5$; $(1,0,5)$.
\end{example}

\begin{example}{ex1}{Example 2 --- Substitute into the plane}
Where does $x=t,y=t,z=t$ meet $x+y+z=6$?\soln $3t=6\Rightarrow t=2$; $(2,2,2)$.
\end{example}

\begin{example}{ex2}{Example 3 --- Line parallel to a plane}
Is $\vec r=(0,0,1)+t(1,1,0)$ parallel to $z=5$?\soln $\vec d\cdot\vec n=(1,1,0)\cdot(0,0,1)=0$ and $(0,0,1)$ not on $z=5$: parallel.
\end{example}

\begin{example}{ex3}{Example 4 --- Parallel lines}
Are $\vec r=(0,0)+t(2,4)$ and $\vec r=(1,1)+s(1,2)$ parallel?\soln $(2,4)=2(1,2)$: yes.
\end{example}

\begin{example}{ex4}{Example 5 --- Parallel planes}
Are $2x+y-z=1$ and $4x+2y-2z=9$ parallel?\soln Normals $(2,1,-1)$ and $2(2,1,-1)$; equations not multiples ⇒ parallel \& distinct.
\end{example}

\begin{example}{ex5}{Example 6 --- Line meets a plane}
Where does $x=t,y=2t,z=0$ meet $z=0$?\soln Every point — the line lies in the plane.
\end{example}

\begin{example}{ex6}{Example 7 --- Substitute}
Where does $x=t,y=t,z=t$ meet $x+y+z=9$?\soln $3t=9\Rightarrow t=3$; $(3,3,3)$.
\end{example}

\begin{example}{ex7}{Example 8 --- Line crosses a plane}
Is direction $(1,1,1)$ parallel to $x+y+z=4$?\soln $\vec d\cdot\vec n=3\ne0$: no, it crosses.
\end{example}

\begin{example}{ex8}{Example 9 --- Simple plane}
Where does $\vec r=(0,0,0)+t(1,0,0)$ meet $x=4$?\soln $t=4$; $(4,0,0)$.
\end{example}
