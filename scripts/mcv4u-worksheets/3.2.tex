{\LARGE\bfseries\color{titleblue}Worksheet 3.2 --- Derivatives of Exponential Functions}\\[2pt]
{\small Grade 12 Calculus and Vectors (MCV4U) \textperiodcentered{} Unit 3: Derivatives of Transcendental Functions}
\end{center}

\begin{ideabox}
$\dfrac{d}{dx}e^x=e^x$ (its own derivative); $\dfrac{d}{dx}b^x=b^x\ln b$; with the chain rule $\dfrac{d}{dx}e^{g}=e^{g}g'$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $\dfrac{d}{dx}e^x=e^x$.
  \item $\dfrac{d}{dx}b^x=b^x\ln b$.
  \item $\dfrac{d}{dx}e^{g(x)}=e^{g(x)}g'(x)$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The special base $e$}The exponential $e^x$ is its own derivative: $\dfrac{d}{dx}e^x=e^x$, which is why $e$ is calculus's natural base.
\lh{Other bases}For a general base, $\dfrac{d}{dx}b^x=b^x\ln b$ — the extra factor $\ln b$ appears.
\lh{With the chain rule}$\dfrac{d}{dx}e^{g(x)}=e^{g(x)}\,g'(x)$, multiplying by the derivative of the exponent.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Base e}
Differentiate $f(x)=e^x$.\soln $f'(x)=e^x$ — the slope equals the height at every point:\eplot{-2}{2}{-1}{8}{\addplot[exblue,very thick,domain=-2:2,samples=80]{exp(x)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Chain rule}
Differentiate $f(x)=e^{3x}$.\soln $e^{3x}\cdot3=3e^{3x}$.
\end{example}

\begin{example}{ex2}{Example 3 --- Other base}
Differentiate $f(x)=2^x$.\soln $2^x\ln 2$.
\end{example}

\begin{example}{ex3}{Example 4 --- Product rule}
Differentiate $f(x)=x e^x$.\soln $e^x+x e^x=e^x(1+x)$.
\end{example}

\begin{example}{ex4}{Example 5 --- Chain (square inside)}
Differentiate $f(x)=e^{x^2}$.\soln $2x e^{x^2}$.
\end{example}

\begin{example}{ex5}{Example 6 --- Chain rule}
Differentiate $f(x)=e^{5x}$.\soln $5e^{5x}$.
\end{example}

\begin{example}{ex6}{Example 7 --- Other base}
Differentiate $f(x)=3^x$.\soln $3^x\ln 3$.
\end{example}

\begin{example}{ex7}{Example 8 --- Negative exponent}
Differentiate $f(x)=e^{-x}$.\soln $-e^{-x}$.
\end{example}

\begin{example}{ex8}{Example 9 --- Product rule}
Differentiate $f(x)=x^2 e^x$.\soln $2x e^x+x^2 e^x=e^x(x^2+2x)$.
\end{example}
