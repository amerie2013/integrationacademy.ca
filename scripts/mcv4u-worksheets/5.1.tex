\begin{ideabox}
Write the quantity to optimize, reduce it to one variable using the constraint, solve $Q'=0$, and verify.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Define a variable; write $Q$.
  \item Use the constraint to make $Q$ one-variable; state the domain.
  \item Solve $Q'=0$; verify with the 1st/2nd derivative test.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Best possible value}\textbf{Optimization} finds a maximum or minimum in a real situation — largest area, least cost, shortest time.
\lh{The strategy}Write the quantity to optimize as a function of one variable (using any constraint), then find its critical points on the valid domain.
\lh{Confirm the extreme}Use the first or second derivative test, and check the endpoints, to confirm you found the true maximum or minimum.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Max area, fixed perimeter}
A pen uses 40 m of fence. Maximize the area.\soln $A=x(20-x)$; $A'=20-2x=0\Rightarrow x=10$; $A''=-2<0$. Square $10\times10$, area $100$ m²:\eplot{0}{20}{0}{120}{\addplot[exblue,very thick,domain=0:20]{x*(20-x)};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(10,100)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Wall on one side}
Only 3 sides fenced with 20 m. Maximize the area.\soln $A=x(20-2x)$; $A'=20-4x=0\Rightarrow x=5$, length $10$; area $50$ m².
\end{example}

\begin{example}{ex2}{Example 3 --- Minimize a sum}
For $x>0$, minimize $f(x)=x+\dfrac{16}{x}$.\soln $f'=1-\dfrac{16}{x^2}=0\Rightarrow x=4$; $f''>0$; min value $8$.
\end{example}

\begin{example}{ex3}{Example 4 --- Max-volume open box}
Cut side-$x$ squares from a $12\times12$ sheet; fold into an open box. Maximize the volume.\soln $V=x(12-2x)^2$; $V'=(12-2x)(12-6x)=0\Rightarrow x=2$; $V=128$ cm³.
\end{example}

\begin{example}{ex4}{Example 5 --- Maximum profit}
Maximize $P(x)=-2x^2+40x-100$.\soln $P'=-4x+40=0\Rightarrow x=10$; $P(10)=100$.
\end{example}

\begin{example}{ex5}{Example 6 --- Max product}
Two positive numbers sum to 12. Maximize the product.\soln $P=x(12-x)$; $P'=12-2x=0\Rightarrow x=6$; product $36$.
\end{example}

\begin{example}{ex6}{Example 7 --- Max area}
Maximize the area of a rectangle with 60 m of fence.\soln $15\times15$, area $225$ m².
\end{example}

\begin{example}{ex7}{Example 8 --- Minimize a sum}
For $x>0$, minimize $f(x)=x+\dfrac{9}{x}$.\soln $1-\dfrac9{x^2}=0\Rightarrow x=3$; min value $6$.
\end{example}

\begin{example}{ex8}{Example 9 --- Maximum value}
Maximize $P(x)=-x^2+10x$.\soln $P'=-2x+10=0\Rightarrow x=5$; $P(5)=25$.
\end{example}
