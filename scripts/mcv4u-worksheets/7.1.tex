\begin{ideabox}
A line needs a point and a direction vector $\vec d$. Vector: $\vec r=\vec r_0+t\vec d$; parametric: $x=x_0+ta,\dots$; symmetric: $\tfrac{x-x_0}{a}=\tfrac{y-y_0}{b}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Vector: $\vec r=\vec r_0+t\vec d$.
  \item Parametric: $x=x_0+ta,\ y=y_0+tb\ (,z=z_0+tc)$.
  \item Direction through two points: $\vec d=B-A$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Point and direction}A line is fixed by a point on it and a \textbf{direction vector}; the \textbf{vector equation} is $\vec r=\vec r_0+t\,\vec d$.
\lh{Parametric form}Splitting into components gives the \textbf{parametric equations}, each coordinate written in terms of the parameter $t$.
\lh{2-D vs 3-D}In 2-D a line also has scalar and symmetric forms; in 3-D the vector and parametric forms are the natural description.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Vector equation}
Line through $(1,2)$ with direction $(3,4)$.\soln $\vec r=(1,2)+t(3,4)$.
\end{example}

\begin{example}{ex1}{Example 2 --- Parametric}
Parametric form of that line.\soln $x=1+3t,\ y=2+4t$.
\end{example}

\begin{example}{ex2}{Example 3 --- Through two points}
Vector equation through $A(1,0,2),B(4,6,8)$.\soln $\vec d=B-A=(3,6,6)$; $\vec r=(1,0,2)+t(3,6,6)$.
\end{example}

\begin{example}{ex3}{Example 4 --- Symmetric}
Symmetric equation through $(1,2)$ with direction $(3,4)$.\soln $\dfrac{x-1}{3}=\dfrac{y-2}{4}$.
\end{example}

\begin{example}{ex4}{Example 5 --- Point on the line?}
Is $(4,6)$ on $\vec r=(1,2)+t(3,4)$?\soln $4=1+3t\Rightarrow t=1$; $2+4(1)=6$ ✓: yes.
\end{example}

\begin{example}{ex5}{Example 6 --- Vector equation}
Line through $(0,1)$ with direction $(2,5)$.\soln $\vec r=(0,1)+t(2,5)$.
\end{example}

\begin{example}{ex6}{Example 7 --- Parametric}
Parametric form of $\vec r=(3,-1)+t(1,2)$.\soln $x=3+t,\ y=-1+2t$.
\end{example}

\begin{example}{ex7}{Example 8 --- Direction vector}
Direction through $A(2,1,0),B(5,7,6)$.\soln $(3,6,6)$.
\end{example}

\begin{example}{ex8}{Example 9 --- Point on the line?}
Is $(7,11)$ on $\vec r=(1,2)+t(3,3)$?\soln $7=1+3t\Rightarrow t=2$; $2+3(2)=8\ne11$: no.
\end{example}
