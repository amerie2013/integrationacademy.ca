\begin{ideabox}
Exponential models describe anything multiplying by a fixed factor each period.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Growth/decay: $A=A_0(1\pm r)^{t}$.
  \item Half-life / doubling: $A=A_0\left(\tfrac12\right)^{t/h}$ or $A_0\,(2)^{t/d}$.
  \item Compound interest: $A=P(1+i)^{n}$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Percentage-change models}Constant-percentage change uses $A=A_0 b^{\,t}$: growth takes $b=1+r$ and decay $b=1-r$, with $r$ the rate as a decimal.
\lh{Half-life and doubling}Halving models use base $\tfrac12$ with exponent $\dfrac{t}{h}$; doubling uses base $2$ — each captures change over a set period.
\lh{Compound interest}Money compounding each period grows by $A=P(1+i)^n$, where $i$ is the rate per period and $n$ the number of periods.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Growth model}
A town of $600$ grows $5\%$ per year.\soln Growth factor $1+0.05=1.05$: $P=600(1.05)^{t}$.
\end{example}

\begin{example}{ex1}{Example 2 --- Decay model}
A $\$5000$ value falls $12\%$ per year.\soln Decay factor $1-0.12=0.88$: $V=5000(0.88)^{t}$.
\end{example}

\begin{example}{ex2}{Example 3 --- Compound interest}
$\$1000$ at $8\%$ compounded annually for $3$ years.\soln $A=1000(1.08)^{3}=1000(1.259712)\approx\$1259.71$.
\end{example}

\begin{example}{ex3}{Example 4 --- Half-life}
$40$ mg, half-life $10$ years; after $30$ years.\soln $30\div10=3$ half-lives: $40\left(\tfrac12\right)^{3}=40\cdot\tfrac18=5$ mg.
\end{example}

\begin{example}{ex4}{Example 5 --- Doubling}
$50$ bacteria double every $2$ h; after $8$ h.\soln $8\div2=4$ doublings: $50\cdot2^{4}=50\cdot16=800$.
\end{example}

\begin{example}{ex5}{Example 6 --- Depreciation}
A $\$30000$ asset drops $18\%$ per year; after $2$ years.\soln $30000(0.82)^{2}=30000(0.6724)\approx\$20172$.
\end{example}

\begin{example}{ex6}{Example 7 --- Population}
$2000$ grows $3\%$ per year; after $5$ years.\soln $2000(1.03)^{5}\approx2000(1.159274)\approx2319$.
\end{example}

\begin{example}{ex7}{Example 8 --- Compound}
$\$1500$ at $4\%$ compounded annually for $10$ years.\soln $1500(1.04)^{10}\approx\$2220.37$.
\end{example}

\begin{example}{ex8}{Example 9 --- Half-life amount}
$200$ mg, half-life $4$ days; after $8$ days.\soln $8\div4=2$ half-lives: $200\left(\tfrac12\right)^{2}=50$ mg.
\end{example}
