\begin{ideabox}
Factoring is expansion in reverse: rewrite a sum as a product.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Always remove a \textbf{common factor} first.
  \item $x^2+bx+c$: find two numbers that multiply to $c$ and add to $b$.
  \item $ax^2+bx+c$: decomposition (split the middle term, then group).
  \item Patterns: $a^2-b^2=(a+b)(a-b)$ and $a^2\pm2ab+b^2=(a\pm b)^2$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{A toolbox of methods}Factoring reverses expansion. Try in order: a \textbf{common factor}, a \textbf{difference of squares}, a \textbf{trinomial} pattern, or \textbf{grouping} for four terms.
\lh{Trinomials}For $ax^2+bx+c$, find factors of $a\cdot c$ that add to $b$, split the middle term, and group.
\lh{Factor fully}Keep going until nothing more factors, and verify by expanding your result.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Common factor}
Factor $6x+9$.\soln The GCF is $3$: $3(2x+3)$.
\end{example}

\begin{example}{ex1}{Example 2 --- Common factor with a variable}
Factor $4x^2-8x$.\soln The GCF is $4x$: $4x(x-2)$.
\end{example}

\begin{example}{ex2}{Example 3 --- Simple trinomial}
Factor $x^2+8x+15$.\soln Two numbers with product $15$, sum $8$: $3$ and $5$. \\ $(x+3)(x+5)$ â€” the factors are the x-intercepts $-3,-5$:\eplot{-7}{1}{-3}{8}{\addplot[exblue,very thick,domain=-6.2:0.2]{x^2+8*x+15};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(-3,0) (-5,0)};}
\end{example}

\begin{example}{ex3}{Example 4 --- Negative constant}
Factor $x^2-5x-14$.\soln Product $-14$, sum $-5$: $-7$ and $2$. \\ $(x-7)(x+2)$.
\end{example}

\begin{example}{ex4}{Example 5 --- Negative middle}
Factor $x^2-9x+20$.\soln Product $20$, sum $-9$: $-4$ and $-5$. \\ $(x-4)(x-5)$.
\end{example}

\begin{example}{ex5}{Example 6 --- Complex trinomial}
Factor $2x^2+7x+3$.\soln $ac=6$; split $7x=6x+x$: $2x^2+6x+x+3$. \\ Group: $2x(x+3)+1(x+3)=(x+3)(2x+1)$.
\end{example}

\begin{example}{ex6}{Example 7 --- Complex trinomial}
Factor $3x^2-10x-8$.\soln $ac=-24$; split $-12x+2x$: $3x^2-12x+2x-8$. \\ Group: $3x(x-4)+2(x-4)=(3x+2)(x-4)$.
\end{example}

\begin{example}{ex7}{Example 8 --- Difference of squares}
Factor $25x^2-16$.\soln $(5x)^2-(4)^2=(5x-4)(5x+4)$.
\end{example}

\begin{example}{ex8}{Example 9 --- Factor fully}
Factor $2x^2-50$.\soln Common factor 2: $2(x^2-25)$. \\ Difference of squares: $2(x-5)(x+5)$.
\end{example}
