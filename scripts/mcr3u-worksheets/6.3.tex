\begin{ideabox}
The sum of an arithmetic sequence: $S_n=\tfrac{n}{2}(a+t_n)=\tfrac{n}{2}\big(2a+(n-1)d\big)$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $S_n=\dfrac{n}{2}(a+t_n)$ when you know the last term.
  \item $S_n=\dfrac{n}{2}\big(2a+(n-1)d\big)$ otherwise.
  \item Find $n$ first if needed.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Summing a sequence}A \textbf{series} is the sum of a sequence's terms; an arithmetic series adds terms with a common difference.
\lh{The sum formula}$S_n=\dfrac{n}{2}(a+t_n)$ — the number of terms times the average of the first and last (equivalently $\dfrac{n}{2}\left[2a+(n-1)d\right]$).
\lh{Why it works}Pairing first with last, second with second-last, and so on gives equal sums — the idea behind the formula.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Gauss' sum}
Find $1+2+\dots+100$.\soln $S_{100}=\dfrac{100}{2}(1+100)=50\cdot101=5050$.
\end{example}

\begin{example}{ex1}{Example 2 --- First 10}
Sum the first $10$ of $3,7,11,\dots$\soln $t_{10}=3+9(4)=39$; $S_{10}=\dfrac{10}{2}(3+39)=5\cdot42=210$.
\end{example}

\begin{example}{ex2}{Example 3 --- Using 2a+(n-1)d}
Sum the first $20$ of $5,8,11,\dots$\soln $S_{20}=\dfrac{20}{2}\big(2(5)+19(3)\big)=10(10+57)=670$.
\end{example}

\begin{example}{ex3}{Example 4 --- Given first and last}
$a=4,\ t_n=44,\ n=11$. Find $S_{11}$.\soln $S_{11}=\dfrac{11}{2}(4+44)=\dfrac{11}{2}(48)=264$.
\end{example}

\begin{example}{ex4}{Example 5 --- Even numbers}
Find $2+4+\dots+50$.\soln $25$ terms; $S=\dfrac{25}{2}(2+50)=25\cdot26=650$.
\end{example}

\begin{example}{ex5}{Example 6 --- First 50}
Find $1+2+\dots+50$.\soln $\dfrac{50}{2}(1+50)=25(51)=1275$.
\end{example}

\begin{example}{ex6}{Example 7 --- First 10}
Sum the first $10$ of $2,5,8,\dots$\soln $t_{10}=29$; $S_{10}=\dfrac{10}{2}(2+29)=155$.
\end{example}

\begin{example}{ex7}{Example 8 --- First 15}
Sum the first $15$ of $1,4,7,\dots$\soln $t_{15}=43$; $S_{15}=\dfrac{15}{2}(1+43)=330$.
\end{example}

\begin{example}{ex8}{Example 9 --- Multiples of 5}
Find $5+10+\dots+100$.\soln $20$ terms; $S=\dfrac{20}{2}(5+100)=10(105)=1050$.
\end{example}
