\begin{ideabox}
Simple interest is paid only on the principal: $I=Prt$ and $A=P(1+rt)$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $I=Prt$ ($r$ as a decimal, $t$ in years).
  \item Amount $A=P+I$.
  \item Simple interest grows linearly.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Interest on the principal}\textbf{Simple interest} is earned only on the original \textbf{principal}: $I=Prt$, with rate $r$ per year and time $t$ in years.
\lh{The total amount}The final amount is $A=P+I=P(1+rt)$ — the principal plus all the interest.
\lh{Linear growth}Because the interest each period is the same, simple interest grows the balance in a straight line.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Interest}
Interest on $\$500$ at $4\%$ for $3$ years.\soln $I=Prt=500(0.04)(3)=\$60$.
\end{example}

\begin{example}{ex1}{Example 2 --- Amount}
Total amount for the above.\soln $A=P+I=500+60=\$560$.
\end{example}

\begin{example}{ex2}{Example 3 --- Interest}
Interest on $\$1000$ at $5\%$ for $2$ years.\soln $1000(0.05)(2)=\$100$.
\end{example}

\begin{example}{ex3}{Example 4 --- Find time}
$\$800$ at $5\%$ earns $\$120$. Find $t$.\soln Yearly interest $=800(0.05)=40$; $t=\dfrac{120}{40}=3$ years.
\end{example}

\begin{example}{ex4}{Example 5 --- Find principal}
At $3\%$ for $2$ years an investment earns $\$90$. Find $P$.\soln $P=\dfrac{I}{rt}=\dfrac{90}{0.06}=\$1500$.
\end{example}

\begin{example}{ex5}{Example 6 --- Find rate}
$\$2000$ earns $\$160$ in $2$ years. Find $r$.\soln $r=\dfrac{I}{Pt}=\dfrac{160}{4000}=0.04=4\%$.
\end{example}

\begin{example}{ex6}{Example 7 --- Interest}
Interest on $\$1200$ at $6\%$ for $5$ years.\soln $1200(0.06)(5)=\$360$.
\end{example}

\begin{example}{ex7}{Example 8 --- Amount}
Amount for $\$1200$ at $6\%$ for $5$ years.\soln $1200+360=\$1560$.
\end{example}

\begin{example}{ex8}{Example 9 --- Rate as decimal}
Write $7\%$ as a decimal.\soln Divide by 100: $0.07$.
\end{example}
