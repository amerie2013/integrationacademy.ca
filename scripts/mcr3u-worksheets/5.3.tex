{\LARGE\bfseries\color{titleblue}Worksheet 5.3 --- Transformations of Sinusoidal Functions}\\[2pt]
{\small Grade 11 Functions, University (MCR3U) \textperiodcentered{} Unit 5: Sinusoidal Functions}
\end{center}

\begin{ideabox}
For y = a sin(k(x − d)) + c: a is amplitude, period = 360°/k, c is the midline, d is the phase shift.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Amplitude $=|a|$; period $=\dfrac{360^\circ}{k}$.
  \item Midline $y=c$; range from $c-|a|$ to $c+|a|$.
  \item $d$ shifts horizontally (phase shift).
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The parameters}In $y=a\sin\!\left(k(x-d)\right)+c$: $|a|$ is the amplitude, $\dfrac{360^\circ}{|k|}$ the period, $d$ the phase shift, and $c$ the midline.
\lh{Factor first}Before reading the phase shift, factor out $k$ so the inside is $k(x-d)$; then $d$ is the shift.
\lh{Building the graph}Set the midline from $c$, height from $a$, width from $k$, and slide by $d$ — draw one cycle, then repeat.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Amplitude \& period}
For $y=3\sin(2x)$: amplitude and period.\soln Amplitude $|3|=3$; period $\dfrac{360^\circ}{2}=180^\circ$ — two cycles in $360^\circ$:\splot{-3.5}{3.5}{\addplot[exblue,very thick,domain=0:360]{3*sin(2*x)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Midline}
Midline of $y=\sin x+4$?\soln The $+4$ raises the midline to $y=4$.
\end{example}

\begin{example}{ex2}{Example 3 --- Period}
Period of $y=\cos(4x)$?\soln $\dfrac{360^\circ}{4}=90^\circ$.
\end{example}

\begin{example}{ex3}{Example 4 --- Write an equation}
Sine with amplitude $2$, period $360^\circ$, midline $0$.\soln $a=2$, $k=1$, $c=0$: $y=2\sin x$.
\end{example}

\begin{example}{ex4}{Example 5 --- Phase shift}
Phase shift of $y=\sin(x-30^\circ)$?\soln $(x-30^\circ)$ shifts the graph right $30^\circ$.
\end{example}

\begin{example}{ex5}{Example 6 --- Range}
Range of $y=3\sin x+1$?\soln Midline $1$, amplitude $3$: from $1-3$ to $1+3$, i.e. $-2\le y\le4$.
\end{example}

\begin{example}{ex6}{Example 7 --- Amplitude}
Amplitude of $y=5\sin x$?\soln $|5|=5$.
\end{example}

\begin{example}{ex7}{Example 8 --- Period}
Period of $y=\sin(3x)$?\soln $\dfrac{360^\circ}{3}=120^\circ$.
\end{example}

\begin{example}{ex8}{Example 9 --- Midline}
Midline of $y=2\cos x-5$?\soln The $-5$ gives midline $y=-5$.
\end{example}
