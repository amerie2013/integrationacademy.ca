\begin{ideabox}
Break a 3-D or multi-step problem into right and oblique triangles and solve them in turn.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Draw and label a clear diagram first.
  \item Solve the triangle you can, then carry a shared side into the next.
  \item Angles of elevation and depression between two points are equal.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Breaking 3-D into triangles}A three-dimensional problem is solved by finding right or oblique triangles \emph{within} the figure and applying trig to each.
\lh{Work in stages}Often one triangle produces a length needed as a side of a second triangle — solve them in sequence.
\lh{Draw carefully}A clear diagram that separates each triangle from the solid is the key to setting up the correct ratios.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Elevation}
From $50$ m away the elevation to a tower top is $38^\circ$. Find the height.\soln $\tan38^\circ=\dfrac{h}{50}\Rightarrow h=50\tan38^\circ\approx39.1$ m.\rtri{5}{3.9}{$h$}{$50$ m}{}
\end{example}

\begin{example}{ex1}{Example 2 --- Ladder}
A $6$ m ladder makes $70^\circ$ with the ground. How high does it reach?\soln $h=6\sin70^\circ\approx5.6$ m.
\end{example}

\begin{example}{ex2}{Example 3 --- Space diagonal}
A box is $3\times4\times12$. Find the space diagonal.\soln Base diagonal $\sqrt{3^2+4^2}=5$; space diagonal $\sqrt{5^2+12^2}=\sqrt{169}=13$.
\end{example}

\begin{example}{ex3}{Example 4 --- Two paths}
Two roads leave a point $65^\circ$ apart; cars travel $40$ km and $55$ km. Distance apart?\soln Cosine law: $d^2=40^2+55^2-2(40)(55)\cos65^\circ\approx2765\Rightarrow d\approx52.6$ km.
\end{example}

\begin{example}{ex4}{Example 5 --- Ramp angle}
A ramp rises $2$ m over a $10$ m run. Find the angle.\soln $\tan\theta=\dfrac{2}{10}=0.2\Rightarrow\theta\approx11.3^\circ$.
\end{example}

\begin{example}{ex5}{Example 6 --- Wire}
A $10$ m wire makes $55^\circ$ with the ground. Attachment height?\soln $h=10\sin55^\circ\approx8.2$ m.
\end{example}

\begin{example}{ex6}{Example 7 --- Elevation}
From $80$ m away the elevation to a roof is $25^\circ$. Find the height.\soln $h=80\tan25^\circ\approx37.3$ m.
\end{example}

\begin{example}{ex7}{Example 8 --- Space diagonal}
A box is $6\times8\times24$. Find the space diagonal.\soln Base $\sqrt{6^2+8^2}=10$; diagonal $\sqrt{10^2+24^2}=\sqrt{676}=26$.
\end{example}

\begin{example}{ex8}{Example 9 --- Two ships}
Two ships leave port $48^\circ$ apart at $20$ km and $30$ km. Distance apart?\soln $d^2=400+900-2(20)(30)\cos48^\circ\approx497\Rightarrow d\approx22.3$ km.
\end{example}
