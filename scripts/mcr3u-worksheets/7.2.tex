\begin{ideabox}
Compound interest pays interest on interest: $A=P(1+i)^{n}$ ($i=$ rate per period, $n=$ periods).
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $A=P(1+i)^{n}$.
  \item Monthly: $i=\dfrac{r}{12}$, $n=12\times$years.
  \item It grows exponentially.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Interest on interest}\textbf{Compound interest} adds each period's interest to the balance, so later interest is earned on a larger amount: $A=P(1+i)^n$.
\lh{Rate and periods}$i$ is the rate \emph{per compounding period} and $n$ the number of periods; more frequent compounding earns slightly more.
\lh{Why it outpaces simple}Because the base keeps growing, compound interest curves upward and overtakes simple interest over time.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Annual}
$\$1000$ at $5\%$ compounded annually for $3$ years.\soln $A=1000(1.05)^{3}=1000(1.157625)\approx\$1157.63$.
\end{example}

\begin{example}{ex1}{Example 2 --- Interest earned}
Interest in the above.\soln $A-P=1157.63-1000\approx\$157.63$.
\end{example}

\begin{example}{ex2}{Example 3 --- Monthly}
$\$2000$ at $6\%$ compounded monthly for $2$ years.\soln $i=\dfrac{0.06}{12}=0.005,\ n=24$: $2000(1.005)^{24}\approx\$2254.32$.
\end{example}

\begin{example}{ex3}{Example 4 --- Five years}
$\$2000$ at $4\%$ compounded annually for $5$ years.\soln $2000(1.04)^{5}\approx\$2433.31$.
\end{example}

\begin{example}{ex4}{Example 5 --- Beats simple}
$\$1000$ at $6\%$ for $10$ years.\soln Simple $1000(1+0.6)=\$1600$; compound $1000(1.06)^{10}\approx\$1790.85$ — the curve outpaces the line:\eplot{0}{20}{0}{4}{\addplot[qorange,very thick,domain=0:20]{1+0.06*x};\addplot[exblue,very thick,domain=0:20]{1.06^x};}
\end{example}

\begin{example}{ex5}{Example 6 --- Annual}
$\$500$ at $4\%$ compounded annually for $3$ years.\soln $500(1.04)^{3}\approx\$562.43$.
\end{example}

\begin{example}{ex6}{Example 7 --- Annual}
$\$1500$ at $5\%$ compounded annually for $4$ years.\soln $1500(1.05)^{4}\approx\$1823.26$.
\end{example}

\begin{example}{ex7}{Example 8 --- Annual}
$\$1000$ at $8\%$ compounded annually for $5$ years.\soln $1000(1.08)^{5}\approx\$1469.33$.
\end{example}

\begin{example}{ex8}{Example 9 --- Rate per period}
Find $i$ for $6\%$ compounded monthly.\soln $i=\dfrac{0.06}{12}=0.005$.
\end{example}
