\begin{ideabox}
Periodic functions repeat. Read off the period, amplitude, and axis (midline).
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Period: the horizontal length of one cycle.
  \item Amplitude $=\dfrac{\text{max}-\text{min}}{2}$.
  \item Axis (midline) $=\dfrac{\text{max}+\text{min}}{2}$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{What periodic means}A \textbf{periodic} function repeats its values at regular intervals; the \textbf{period} is the length of one full cycle.
\lh{Key measurements}Describe a periodic graph by its \textbf{period}, \textbf{amplitude} (half the max-to-min distance), and \textbf{midline} (the central horizontal line).
\lh{Real examples}Tides, sound waves, and a Ferris wheel's height are all periodic and modelled with these features.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Period}
A graph repeats every $5$ units. State the period.\soln One full cycle spans $5$ units, so the period is $5$.
\end{example}

\begin{example}{ex1}{Example 2 --- Amplitude}
Max $8$, min $2$. Find the amplitude.\soln $\dfrac{\text{max}-\text{min}}{2}=\dfrac{8-2}{2}=3$.
\end{example}

\begin{example}{ex2}{Example 3 --- Axis}
Max $8$, min $2$. Find the axis.\soln $\dfrac{\text{max}+\text{min}}{2}=\dfrac{8+2}{2}=5$.
\end{example}

\begin{example}{ex3}{Example 4 --- Period of sine}
State the period of $y=\sin x$.\soln The sine curve repeats once per full turn: $360^\circ$:\splot{-1.5}{1.5}{\addplot[exblue,very thick,domain=0:360]{sin(x)};}
\end{example}

\begin{example}{ex4}{Example 5 --- Max/min}
State the max and min of $y=\sin x$.\soln The peak is $1$ and the trough is $-1$.
\end{example}

\begin{example}{ex5}{Example 6 --- Amplitude (negatives)}
Max $10$, min $-2$. Amplitude?\soln $\dfrac{10-(-2)}{2}=\dfrac{12}{2}=6$.
\end{example}

\begin{example}{ex6}{Example 7 --- Axis (negatives)}
Max $10$, min $-2$. Axis?\soln $\dfrac{10+(-2)}{2}=4$.
\end{example}

\begin{example}{ex7}{Example 8 --- Range}
Amplitude $3$, axis $5$. Range?\soln From $5-3$ to $5+3$: $2$ to $8$.
\end{example}

\begin{example}{ex8}{Example 9 --- Cycle}
What is one "cycle"?\soln One complete repeat of the pattern; its length is the period.
\end{example}
