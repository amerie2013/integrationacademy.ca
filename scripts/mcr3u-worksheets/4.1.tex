\begin{ideabox}
Use SOH CAH TOA in right triangles and memorize the exact values of the special angles.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $\sin\theta=\dfrac{\text{opp}}{\text{hyp}}$, $\cos\theta=\dfrac{\text{adj}}{\text{hyp}}$, $\tan\theta=\dfrac{\text{opp}}{\text{adj}}$.
  \item Special angles come from the $45^\circ$-$45^\circ$-$90^\circ$ ($1,1,\sqrt2$) and $30^\circ$-$60^\circ$-$90^\circ$ ($1,\sqrt3,2$) triangles.
  \item Set your calculator to DEGREE mode.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The primary ratios}In a right triangle $\sin=\dfrac{\text{opp}}{\text{hyp}}$, $\cos=\dfrac{\text{adj}}{\text{hyp}}$, $\tan=\dfrac{\text{opp}}{\text{adj}}$ — recall SOH-CAH-TOA.
\lh{Special angles}The angles $30^\circ$, $45^\circ$, and $60^\circ$ have exact ratios from the $30$-$60$-$90$ and $45$-$45$-$90$ triangles, worth memorizing.
\lh{Exact vs decimal}Special angles give exact surds like $\sin60^\circ=\dfrac{\sqrt3}{2}$; other angles need a calculator.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Find a ratio}
Opposite $3$, hypotenuse $5$. Find $\sin\theta$.\soln $\sin\theta=\dfrac{\text{opp}}{\text{hyp}}=\dfrac{3}{5}$.\rtri{4}{3}{$3$}{$4$}{$5$}
\end{example}

\begin{example}{ex1}{Example 2 --- Cosine}
Adjacent $4$, hypotenuse $5$. Find $\cos\theta$.\soln $\cos\theta=\dfrac{\text{adj}}{\text{hyp}}=\dfrac{4}{5}$.
\end{example}

\begin{example}{ex2}{Example 3 --- Tangent}
Opposite $3$, adjacent $4$. Find $\tan\theta$.\soln $\tan\theta=\dfrac{\text{opp}}{\text{adj}}=\dfrac{3}{4}$.
\end{example}

\begin{example}{ex3}{Example 4 --- Special angle}
State $\sin30^\circ$.\soln From the $30$-$60$-$90$ triangle: $\dfrac12$.
\end{example}

\begin{example}{ex4}{Example 5 --- Special angle}
State $\cos45^\circ$.\soln From the $45$-$45$-$90$ triangle: $\dfrac{\sqrt2}{2}$.
\end{example}

\begin{example}{ex5}{Example 6 --- Special angle}
State $\tan60^\circ$.\soln $\dfrac{\sqrt3}{1}=\sqrt3$.
\end{example}

\begin{example}{ex6}{Example 7 --- Find a side}
Hypotenuse $10$, angle $30^\circ$. Find the opposite side.\soln $\sin30^\circ=\dfrac{\text{opp}}{10}\Rightarrow\text{opp}=10\sin30^\circ=5$.
\end{example}

\begin{example}{ex7}{Example 8 --- Find a side}
Hypotenuse $8$, angle $60^\circ$. Find the adjacent side.\soln $\cos60^\circ=\dfrac{\text{adj}}{8}\Rightarrow\text{adj}=8\cos60^\circ=4$.
\end{example}

\begin{example}{ex8}{Example 9 --- Find an angle}
Opposite $5$, hypotenuse $10$. Find $\theta$.\soln $\sin\theta=\dfrac{5}{10}=\tfrac12\Rightarrow\theta=30^\circ$.
\end{example}
