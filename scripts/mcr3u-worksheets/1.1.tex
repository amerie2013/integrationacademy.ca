\begin{ideabox}
A function gives exactly one output for each input; $f(x)$ is the output at $x$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Vertical-line test: a vertical line may meet the graph only once.
  \item A function is one-to-one or many-to-one, never one-to-many.
  \item Evaluate by substituting for every $x$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Relations and functions}A \textbf{relation} pairs inputs with outputs. It is a \textbf{function} when every input gives exactly \emph{one} output — no $x$-value maps to two different $y$-values.
\lh{The vertical line test}On a graph a relation is a function if no vertical line crosses it more than once, since each $x$ then has a single $y$.
\lh{Function notation}$f(x)$ names the output for input $x$. Writing $f(3)=7$ means input $3$ gives output $7$ — a compact way to describe and evaluate a function.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Function or not}
Is $x=y^2$ a function?\soln Solve for $y$: $y=\pm\sqrt{x}$. Then $x=4$ gives $y=2$ \emph{and} $y=-2$ — one input, two outputs. \\ \textbf{Not} a function.
\end{example}

\begin{example}{ex1}{Example 2 --- Evaluate}
For $f(x)=2x^2+3x-1$, find $f(2)$.\soln Substitute $x=2$: $2(2)^2+3(2)-1=8+6-1=13$.
\end{example}

\begin{example}{ex2}{Example 3 --- Negative input}
For $f(x)=2x^2+3x-1$, find $f(-1)$.\soln $2(-1)^2+3(-1)-1=2-3-1=-2$.
\end{example}

\begin{example}{ex3}{Example 4 --- From a set}
Is $\{(1,2),(1,3)\}$ a function?\soln Input $1$ maps to both $2$ and $3$ — one-to-many, so \textbf{not} a function.
\end{example}

\begin{example}{ex4}{Example 5 --- Evaluate}
For $f(x)=x^2-4$, find $f(3)$.\soln $(3)^2-4=9-4=5$.
\end{example}

\begin{example}{ex5}{Example 6 --- Evaluate}
For $f(x)=5-2x$, find $f(4)$.\soln $5-2(4)=5-8=-3$.
\end{example}

\begin{example}{ex6}{Example 7 --- Vertical-line test}
Is $y=3x+1$ a function?\soln A non-vertical line meets every vertical line once, so \textbf{yes}.
\end{example}

\begin{example}{ex7}{Example 8 --- Many-to-one}
Is $f(x)=x^2$ a function even though $f(2)=f(-2)$?\soln Two inputs sharing one output (many-to-one) is allowed; it still passes the vertical-line test, so \textbf{yes}:\eplot{-3}{3}{-1}{8}{\addplot[exblue,very thick,domain=-2.6:2.6]{x^2};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(2,4) (-2,4)};}
\end{example}

\begin{example}{ex8}{Example 9 --- Solve}
For $f(x)=2x-6$, solve $f(x)=0$.\soln $2x-6=0\Rightarrow x=3$.
\end{example}
