{\LARGE\bfseries\color{titleblue}Worksheet 1.6 --- Solving Quadratics \& Linear-Quadratic Systems}\\[2pt]
{\small Grade 11 Functions, University (MCR3U) \textperiodcentered{} Unit 1: Characteristics of Functions}
\end{center}

\begin{ideabox}
Solve quadratics by factoring or the formula $x=\dfrac{-b\pm\sqrt{b^2-4ac}}{2a}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Try factoring first; use the formula when it won't factor.
  \item Discriminant $b^2-4ac$: $>0$ two roots, $=0$ one, $<0$ none.
  \item A linear-quadratic system: set the expressions equal.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Solving a quadratic}Set the equation to $0$, then factor, complete the square, or use the quadratic formula. The discriminant $b^2-4ac$ tells the number of real roots.
\lh{Linear--quadratic systems}A line and a parabola meet at their shared points. Substitute the line into the parabola to get a single quadratic, then solve.
\lh{Counting intersections}That quadratic's discriminant tells whether the line cuts the parabola twice, touches it once (tangent), or misses it entirely.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Factor}
Solve $x^2-5x+6=0$.\soln $(x-2)(x-3)=0\Rightarrow x=2$ or $x=3$.
\end{example}

\begin{example}{ex1}{Example 2 --- System}
Where do $y=x^2-2$ and $y=x$ meet?\soln Set equal: $x^2-2=x\Rightarrow x^2-x-2=0\Rightarrow(x-2)(x+1)=0$. \\ $x=2,-1$, giving $(2,2)$ and $(-1,-1)$:\eplot{-3}{4}{-3}{6}{\addplot[exblue,very thick,domain=-2.5:3]{x^2-2};\addplot[qorange,very thick,domain=-3:4]{x};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(2,2) (-1,-1)};}
\end{example}

\begin{example}{ex2}{Example 3 --- Factor}
Solve $x^2+2x-8=0$.\soln $(x+4)(x-2)=0\Rightarrow x=-4,2$.
\end{example}

\begin{example}{ex3}{Example 4 --- Quadratic formula}
Solve $x^2-4x+1=0$.\soln $x=\dfrac{4\pm\sqrt{16-4}}{2}=\dfrac{4\pm\sqrt{12}}{2}=2\pm\sqrt3$.
\end{example}

\begin{example}{ex4}{Example 5 --- Discriminant}
How many real solutions has $x^2+x+1=0$?\soln $b^2-4ac=1-4=-3<0$: no real solutions.
\end{example}

\begin{example}{ex5}{Example 6 --- Factor}
Solve $x^2-7x+10=0$.\soln $(x-2)(x-5)=0\Rightarrow x=2,5$.
\end{example}

\begin{example}{ex6}{Example 7 --- Factor}
Solve $x^2-x-6=0$.\soln $(x-3)(x+2)=0\Rightarrow x=3,-2$.
\end{example}

\begin{example}{ex7}{Example 8 --- System}
Where do $y=x^2$ and $y=x+2$ meet?\soln $x^2=x+2\Rightarrow x^2-x-2=0\Rightarrow(x-2)(x+1)=0$. \\ Points $(2,4)$ and $(-1,1)$:\eplot{-3}{4}{-2}{8}{\addplot[exblue,very thick,domain=-2.5:3]{x^2};\addplot[qorange,very thick,domain=-3:4]{x+2};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(2,4) (-1,1)};}
\end{example}

\begin{example}{ex8}{Example 9 --- Quadratic formula}
Solve $x^2-2x-1=0$.\soln $x=\dfrac{2\pm\sqrt{4+4}}{2}=\dfrac{2\pm2\sqrt2}{2}=1\pm\sqrt2$.
\end{example}
