\begin{ideabox}
Vertex form $a(x-h)^2+k$ shows the vertex $(h,k)$; complete the square to find it.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Vertex is the max ($a<0$) or min ($a>0$).
  \item Zeros are where the parabola crosses the $x$-axis.
  \item Axis of symmetry: $x=h$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Three forms, three views}Standard form shows the $y$-intercept, factored form shows the \textbf{zeros}, and vertex form shows the \textbf{vertex} — the maximum or minimum point.
\lh{The vertex is the extreme}A parabola turns at its vertex, so the vertex's $y$-value is the function's maximum (opens down) or minimum (opens up).
\lh{Finding key features}Get zeros by factoring or the quadratic formula; the axis of symmetry is $x=-\dfrac{b}{2a}$, and substituting it gives the vertex.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Complete the square}
Vertex form of $x^2-6x+5$.\soln Half of $-6$ is $-3$; $(-3)^2=9$: $x^2-6x+9-9+5=(x-3)^2-4$. \\ Vertex $(3,-4)$; zeros $1$ and $5$:\eplot{-1}{7}{-6}{8}{\addplot[exblue,very thick,domain=-0.5:6.5]{x^2-6*x+5};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(1,0) (5,0) (3,-4)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Minimum}
Minimum of $x^2-6x+5$?\soln From $(x-3)^2-4$, the minimum value is $-4$.
\end{example}

\begin{example}{ex2}{Example 3 --- Zeros}
Zeros of $x^2-6x+5$?\soln Factor $(x-1)(x-5)=0\Rightarrow x=1,5$.
\end{example}

\begin{example}{ex3}{Example 4 --- Another vertex}
Vertex of $x^2+4x+1$?\soln $x^2+4x+4-4+1=(x+2)^2-3$; vertex $(-2,-3)$.
\end{example}

\begin{example}{ex4}{Example 5 --- Axis of symmetry}
Axis of symmetry of $x^2-6x+5$?\soln It passes through the vertex: $x=3$.
\end{example}

\begin{example}{ex5}{Example 6 --- Vertex form given}
Vertex of $y=(x-3)^2-4$?\soln Read off $(h,k)=(3,-4)$.
\end{example}

\begin{example}{ex6}{Example 7 --- Complete the square}
Complete the square: $x^2+6x+5$.\soln Half of $6$ is $3$: $(x+3)^2-9+5=(x+3)^2-4$.
\end{example}

\begin{example}{ex7}{Example 8 --- Minimum}
Minimum of $x^2-2x+5$?\soln $(x-1)^2-1+5=(x-1)^2+4$; minimum $4$.
\end{example}

\begin{example}{ex8}{Example 9 --- Maximum}
Max of $y=-x^2+4$?\soln Opens down, vertex $(0,4)$: maximum $4$.
\end{example}
