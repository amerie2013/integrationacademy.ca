{\LARGE\bfseries\color{titleblue}Worksheet 1.1 --- Power Functions \& End Behaviour}\\[2pt]
{\small Grade 12 Advanced Functions (MHF4U) \textperiodcentered{} Unit 1: Polynomial Functions}
\end{center}

\begin{ideabox}
Every polynomial behaves like its leading term at the ends. Degree parity and the sign of $a$ decide the end behaviour.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Even degree: both ends the same (up–up if $a>0$, down–down if $a<0$).
  \item Odd degree: ends opposite (down–up if $a>0$, up–down if $a<0$).
  \item Write it with limits, e.g. as $x\to+\infty,\ y\to+\infty$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Power functions}A \textbf{power function} has the form $y=ax^n$ with $n$ a whole number; its degree and the sign of $a$ dictate the overall shape.
\lh{End behaviour}As $x\to\pm\infty$, the ends are set by the degree's parity and the sign of $a$: even degrees send both ends the same way, odd degrees send them opposite ways.
\lh{Even vs odd symmetry}Even-degree power functions are symmetric about the $y$-axis; odd-degree ones have rotational symmetry about the origin.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Odd, positive}
End behaviour of $f(x)=x^3$?\soln Odd degree, $a>0$: as $x\to-\infty,\ y\to-\infty$; as $x\to+\infty,\ y\to+\infty$ (down–up):\eplot{-3}{3}{-9}{9}{\addplot[exblue,very thick,domain=-2.1:2.1]{x^3};}
\end{example}

\begin{example}{ex1}{Example 2 --- Even, positive}
End behaviour of $f(x)=x^4$?\soln Even degree, $a>0$: both ends $y\to+\infty$ (up–up):\eplot{-2}{2}{-1}{9}{\addplot[exblue,very thick,domain=-1.7:1.7]{x^4};}
\end{example}

\begin{example}{ex2}{Example 3 --- Odd, negative}
End behaviour of $f(x)=-2x^3$?\soln Odd, $a<0$ flips it: as $x\to-\infty,\ y\to+\infty$; as $x\to+\infty,\ y\to-\infty$ (up–down).
\end{example}

\begin{example}{ex3}{Example 4 --- Even, negative}
End behaviour of $f(x)=-x^4$?\soln Even, $a<0$: both ends $y\to-\infty$ (down–down).
\end{example}

\begin{example}{ex4}{Example 5 --- Read a graph}
A graph falls on the left and rises on the right.\soln Opposite ends ⇒ odd degree; rising on the right ⇒ $a>0$.
\end{example}

\begin{example}{ex5}{Example 6 --- Odd, positive}
End behaviour of $f(x)=x^5$?\soln Down–up (odd, $a>0$).
\end{example}

\begin{example}{ex6}{Example 7 --- Even, negative}
End behaviour of $f(x)=-x^2$?\soln Down–down (even, $a<0$).
\end{example}

\begin{example}{ex7}{Example 8 --- Even, positive}
End behaviour of $f(x)=3x^6$?\soln Up–up (even, $a>0$).
\end{example}

\begin{example}{ex8}{Example 9 --- Compare shapes}
How do $y=x^2$ and $y=x^4$ compare?\soln Both up–up; $x^4$ is flatter near 0 and steeper outside $[-1,1]$:\eplot{-2}{2}{-1}{9}{\addplot[exblue,very thick,domain=-2.9:2.9]{x^2};\addplot[qorange,very thick,domain=-1.7:1.7]{x^4};}
\end{example}
