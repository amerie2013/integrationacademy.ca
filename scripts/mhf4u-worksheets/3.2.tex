\begin{ideabox}
Sketch from the skeleton: intercepts, vertical and horizontal asymptotes, and the behaviour on each side of a VA.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item x-intercept: numerator $=0$. y-intercept: evaluate $f(0)$.
  \item VA: denominator $=0$. HA: from the degrees.
  \item Near a VA the curve rushes to $\pm\infty$ — test each side.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Locate asymptotes first}Find vertical asymptotes from the denominator's zeros, and the end-behaviour asymptote by comparing the degrees of top and bottom.
\lh{Intercepts and holes}Numerator zeros give $x$-intercepts; a factor shared by top and bottom creates a \textbf{hole} rather than an asymptote.
\lh{Sketching}Plot the intercepts, draw the asymptotes as guides, and check the sign in each region to finish the graph.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- The reciprocal}
Sketch $y=\dfrac1x$.\soln VA $x=0$, HA $y=0$, no intercepts:\begin{center}\begin{tikzpicture}\begin{axis}[width=7.6cm,height=5cm,axis lines=middle,xlabel={\small$x$},ylabel={\small$y$},xmin=-5,xmax=5,ymin=-5,ymax=5,samples=160,restrict y to domain=-6:6]\addplot[exblue,very thick,domain=-5:-0.18]{1/x};\addplot[exblue,very thick,domain=0.18:5]{1/x};\draw[dashed,gray] (axis cs:0,-5)--(axis cs:0,5);\end{axis}\end{tikzpicture}\end{center}
\end{example}

\begin{example}{ex1}{Example 2 --- Full skeleton}
For $y=\dfrac{x-2}{x+1}$, find intercepts and asymptotes.\soln x-int $(2,0)$; y-int $f(0)=-2$; VA $x=-1$; HA $y=1$:\begin{center}\begin{tikzpicture}\begin{axis}[width=7.6cm,height=5cm,axis lines=middle,xlabel={\small$x$},ylabel={\small$y$},xmin=-7,xmax=5,ymin=-5,ymax=7,samples=160,restrict y to domain=-6:8]\addplot[exblue,very thick,domain=-7:-1.18]{(x-2)/(x+1)};\addplot[exblue,very thick,domain=-0.82:5]{(x-2)/(x+1)};\draw[dashed,gray] (axis cs:-1,-5)--(axis cs:-1,7);\draw[dashed,gray] (axis cs:-7,1)--(axis cs:5,1);\end{axis}\end{tikzpicture}\end{center}
\end{example}

\begin{example}{ex2}{Example 3 --- Behaviour near a VA}
How does $y=\dfrac{1}{x-2}$ behave near $x=2$?\soln Just left, $y\to-\infty$; just right, $y\to+\infty$:\begin{center}\begin{tikzpicture}\begin{axis}[width=7.6cm,height=5cm,axis lines=middle,xlabel={\small$x$},ylabel={\small$y$},xmin=-3,xmax=7,ymin=-5,ymax=5,samples=160,restrict y to domain=-6:6]\addplot[exblue,very thick,domain=-3:1.82]{1/(x-2)};\addplot[exblue,very thick,domain=2.18:7]{1/(x-2)};\draw[dashed,gray] (axis cs:2,-5)--(axis cs:2,5);\draw[dashed,gray] (axis cs:-3,0)--(axis cs:7,0);\end{axis}\end{tikzpicture}\end{center}
\end{example}

\begin{example}{ex3}{Example 4 --- y-intercept}
y-intercept of $y=\dfrac{x+3}{x-1}$?\soln $f(0)=\dfrac{3}{-1}=-3$, so $(0,-3)$.
\end{example}

\begin{example}{ex4}{Example 5 --- End behaviour}
As $x\to\infty$, $y=\dfrac{2x+1}{x-3}\to$?\soln Equal degrees → $\tfrac21$, so $y\to2$.
\end{example}

\begin{example}{ex5}{Example 6 --- VA and HA}
VA and HA of $y=\dfrac{1}{x-4}$?\soln VA $x=4$, HA $y=0$.
\end{example}

\begin{example}{ex6}{Example 7 --- x-intercept}
x-intercept of $y=\dfrac{x-5}{x+2}$?\soln Numerator $0$ at $x=5$: $(5,0)$.
\end{example}

\begin{example}{ex7}{Example 8 --- y-intercept}
y-intercept of $y=\dfrac{x+4}{x-2}$?\soln $f(0)=\dfrac{4}{-2}=-2$: $(0,-2)$.
\end{example}

\begin{example}{ex8}{Example 9 --- Equal degrees}
HA of $y=\dfrac{3x}{x+5}$?\soln $y=3$.
\end{example}
