\begin{ideabox}
Find the zeros, then test a point on each interval. The sign flips at odd multiplicity but not at even multiplicity.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Move to one side, factor, find the zeros.
  \item Test a point in each interval for the sign.
  \item Include endpoints for $\ge$ or $\le$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Beyond equations}A polynomial \textbf{inequality} asks where the graph lies above or below the axis, not just where it crosses.
\lh{Use the zeros}The zeros split the number line into intervals; the polynomial keeps one sign on each, so test a single point in each interval.
\lh{Reading the sign}A sign chart across the intervals gives the full solution — remember an even-multiplicity zero does not change the sign.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Product positive}
Solve $(x-1)(x+2)>0$.\soln Zeros $1,-2$; positive outside $[-2,1]$. So $x<-2$ or $x>1$. The graph is above the axis there:\eplot{-4}{3}{-3}{6}{\addplot[exblue,very thick,domain=-3.4:2.4]{(x-1)*(x+2)};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(-2,0) (1,0)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Product negative}
Solve $(x-1)(x+2)<0$.\soln Negative on the middle interval: $-2<x<1$.
\end{example}

\begin{example}{ex2}{Example 3 --- Factor first}
Solve $x^2-x-6>0$.\soln $(x-3)(x+2)>0$; zeros $3,-2$: $x<-2$ or $x>3$.
\end{example}

\begin{example}{ex3}{Example 4 --- Three factors}
Solve $x(x-2)(x+1)<0$.\soln Zeros $-1,0,2$; signs $-,+,-,+$. So $x<-1$ or $0<x<2$.
\end{example}

\begin{example}{ex4}{Example 5 --- Even multiplicity}
Solve $(x-1)^2(x+3)>0$.\soln $(x-1)^2\ge0$; sign from $x+3$ (and $x\ne1$): $x>-3,\ x\ne1$.
\end{example}

\begin{example}{ex5}{Example 6 --- Difference of squares}
Solve $x^2-4<0$.\soln $(x-2)(x+2)<0\Rightarrow-2<x<2$.
\end{example}

\begin{example}{ex6}{Example 7 --- With equality}
Solve $x^2-2x-3\ge0$.\soln $(x-3)(x+1)\ge0\Rightarrow x\le-1$ or $x\ge3$.
\end{example}

\begin{example}{ex7}{Example 8 --- Three factors}
Solve $x(x-1)(x+2)>0$.\soln Zeros $-2,0,1$; signs $-,+,-,+$. So $-2<x<0$ or $x>1$.
\end{example}

\begin{example}{ex8}{Example 9 --- Cubic}
Solve $x^3-x<0$.\soln $x(x-1)(x+1)<0$; signs $-,+,-,+$. So $x<-1$ or $0<x<1$.
\end{example}
