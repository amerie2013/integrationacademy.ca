\begin{ideabox}
Apply $g(x)=a\,f\big(k(x-d)\big)+c$ to the parent $f(x)=x^n$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $a$: vertical stretch/reflection. $k$: horizontal stretch/reflection.
  \item $d$: horizontal shift (opposite to the sign). $c$: vertical shift.
  \item Stretches/reflections first, then translations.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{A single framework}Any base function $f$ transforms through $y=af\!\left(k(x-d)\right)+c$, where each constant stretches, reflects, or shifts the graph.
\lh{Vertical vs horizontal}$a$ and $c$ act vertically (stretch by $a$, shift by $c$); $k$ and $d$ act horizontally (compress by $k$, shift by $d$).
\lh{Order and mapping}Apply horizontal changes to $x$ and vertical changes to $y$: each point maps as $(x,y)\to\left(\tfrac{x}{k}+d,\ ay+c\right)$.
\end{learnbox}

\end{example}

\begin{example}{ex1}{Example 2 --- Reflection}
Describe $y=-x^4$.\soln $a=-1$: reflection in the x-axis (opens down).
\end{example}

\begin{example}{ex2}{Example 3 --- Horizontal stretch}
What does $k=2$ do in $y=(2x)^3$?\soln Horizontal compression by $\tfrac12$ (squeezed toward the y-axis).
\end{example}

\begin{example}{ex3}{Example 4 --- Map a point}
$(1,1)$ on $y=x^3$ maps where on $y=(x-2)^3+4$?\soln Right 2, up 4: $(3,5)$.
\end{example}

\begin{example}{ex4}{Example 5 --- Write the equation}
Write $y=x^4$ reflected, left 1, down 2.\soln $a=-1,\ d=-1,\ c=-2$: $y=-(x+1)^4-2$.
\end{example}

\begin{example}{ex5}{Example 6 --- Describe}
Describe $y=(x+3)^3-1$.\soln Left 3, down 1.
\end{example}

\begin{example}{ex6}{Example 7 --- Map a point}
$(1,1)$ on $y=x^3$ maps where on $y=x^3-5$?\soln $(1,-4)$.
\end{example}

\begin{example}{ex7}{Example 8 --- Stretch \& reflect}
Describe $y=-3x^4+2$.\soln Reflect, stretch by 3, up 2.
\end{example}

\begin{example}{ex8}{Example 9 --- Write the equation}
Write $y=x^3$ shifted right 4 and up 1.\soln $y=(x-4)^3+1$.
\end{example}
