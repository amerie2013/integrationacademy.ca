\begin{ideabox}
Combine functions point by point: $(f\pm g)(x)=f(x)\pm g(x)$, $(fg)(x)=f(x)g(x)$, $\left(\tfrac fg\right)(x)=\dfrac{f(x)}{g(x)}$ (with $g\ne0$).
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $(f\pm g)(x)=f(x)\pm g(x)$; $(fg)(x)=f(x)g(x)$.
  \item Domain $=$ intersection of the two domains.
  \item For $\tfrac fg$, also exclude $g(x)=0$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Arithmetic of functions}Functions can be added, subtracted, multiplied, and divided point by point: $(f+g)(x)=f(x)+g(x)$, and so on.
\lh{Domain of the result}The combined function is defined only where \emph{both} pieces are defined, and division also excludes where the divisor is zero.
\lh{Reading combined graphs}Add or subtract the $y$-values at each $x$ to picture the sum or difference of two graphs.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Sum}
$f=x^2,\ g=x$. Find $(f+g)(x)$.\soln $x^2+x$. Graphs add point by point:\eplot{-4}{3}{-3}{9}{\addplot[gray,thick,domain=-3:3]{x^2};\addplot[gray,thick,domain=-3:3]{x};\addplot[exblue,very thick,domain=-3.5:2.6]{x^2+x};}
\end{example}

\begin{example}{ex1}{Example 2 --- Difference}
Find $(f-g)(x)$.\soln $x^2-x$.
\end{example}

\begin{example}{ex2}{Example 3 --- Product}
Find $(fg)(x)$.\soln $x^2\cdot x=x^3$.
\end{example}

\begin{example}{ex3}{Example 4 --- Quotient}
Find $\left(\tfrac fg\right)(x)$ and its domain.\soln $\dfrac{x^2}{x}=x$, with $x\ne0$.
\end{example}

\begin{example}{ex4}{Example 5 --- Evaluate}
Find $(f+g)(2)$.\soln $f(2)+g(2)=4+2=6$.
\end{example}

\begin{example}{ex5}{Example 6 --- Sum}
$f=x+1,\ g=x-1$. Find $(f+g)(x)$.\soln $2x$.
\end{example}

\begin{example}{ex6}{Example 7 --- Product}
Same $f,g$: find $(fg)(x)$.\soln $(x+1)(x-1)=x^2-1$.
\end{example}

\begin{example}{ex7}{Example 8 --- Difference}
$f=x^2,\ g=3$. Find $(f-g)(x)$.\soln $x^2-3$.
\end{example}

\begin{example}{ex8}{Example 9 --- Evaluate}
$f=x,\ g=x+2$. Find $(f+g)(3)$.\soln $3+5=8$.
\end{example}
