{\small Grade 12 Advanced Functions (MHF4U) \textperiodcentered{} Unit 6: Trigonometric Identities and Equations}

\begin{ideabox}
Setting $B=A$: $\sin2\theta=2\sin\theta\cos\theta$ and $\cos2\theta=\cos^2\theta-\sin^2\theta=1-2\sin^2\theta=2\cos^2\theta-1$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $\sin2\theta=2\sin\theta\cos\theta$.
  \item $\cos2\theta=1-2\sin^2\theta=2\cos^2\theta-1$.
  \item $\tan2\theta=\tfrac{2\tan\theta}{1-\tan^2\theta}$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Doubling an angle}Setting $B=A$ in the compound formulas yields the \textbf{double angle} identities, such as $\sin2A=2\sin A\cos A$.
\lh{Cosine's three forms}$\cos2A=\cos^2A-\sin^2A=2\cos^2A-1=1-2\sin^2A$ — pick the version that fits your information.
\lh{Simplifying}Double angle formulas condense expressions and are essential for proving identities and solving equations.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Sine double angle}
$\sin\theta=\tfrac35,\cos\theta=\tfrac45$. Find $\sin2\theta$.\soln $2\cdot\tfrac35\cdot\tfrac45=\dfrac{24}{25}$. The graph $y=\sin2x$ completes two cycles in $2\pi$:\begin{center}\begin{tikzpicture}\begin{axis}[width=9.2cm,height=4.6cm,axis lines=middle,xlabel={\small$x$},ylabel={\small$y$},xmin=-0.4,xmax=6.9,ymin=-1.6,ymax=1.6,samples=220,xtick={1.5708,3.1416,4.7124,6.2832},xticklabels={$\frac{\pi}{2}$,$\pi$,$\frac{3\pi}{2}$,$2\pi$}]\addplot[exblue,very thick,domain=0:6.4]{sin(deg(2*x))};\end{axis}\end{tikzpicture}\end{center}
\end{example}

\begin{example}{ex1}{Example 2 --- Cosine double angle}
With $\sin\theta=\tfrac35$, find $\cos2\theta$.\soln $1-2\sin^2\theta=1-\tfrac{18}{25}=\dfrac7{25}$.
\end{example}

\begin{example}{ex2}{Example 3 --- Recognize}
Simplify $2\sin x\cos x$.\soln $\sin2x$.
\end{example}

\begin{example}{ex3}{Example 4 --- Cosine form}
Write $\cos2\theta$ using only $\cos\theta$.\soln $2\cos^2\theta-1$.
\end{example}

\begin{example}{ex4}{Example 5 --- Recognize}
Simplify $1-2\sin^2 x$.\soln $\cos2x$.
\end{example}

\begin{example}{ex5}{Example 6 --- Sine double angle}
$\sin\theta=\tfrac5{13},\cos\theta=\tfrac{12}{13}$. Find $\sin2\theta$.\soln $2\cdot\tfrac5{13}\cdot\tfrac{12}{13}=\dfrac{120}{169}$.
\end{example}

\begin{example}{ex6}{Example 7 --- Recognize}
Simplify $2\cos^2 x-1$.\soln $\cos2x$.
\end{example}

\begin{example}{ex7}{Example 8 --- Cosine double angle}
With $\cos\theta=\tfrac35$, find $\cos2\theta$.\soln $2\cdot\tfrac9{25}-1=-\dfrac7{25}$.
\end{example}

\begin{example}{ex8}{Example 9 --- Simplify}
Simplify $\dfrac{\sin2x}{2\cos x}$.\soln $\dfrac{2\sin x\cos x}{2\cos x}=\sin x$.
\end{example}
