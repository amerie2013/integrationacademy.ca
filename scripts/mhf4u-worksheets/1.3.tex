{\LARGE\bfseries\color{titleblue}Worksheet 1.3 --- Equations \& Graphs of Polynomial Functions}\\[2pt]
{\small Grade 12 Advanced Functions (MHF4U) \textperiodcentered{} Unit 1: Polynomial Functions}
\end{center}

\begin{ideabox}
In factored form $f(x)=a(x-r_1)^{m_1}\cdots$, the zeros are $r_i$ and the multiplicities $m_i$ control how the graph meets the x-axis.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Zeros: set each factor to $0$.
  \item Odd multiplicity crosses; even multiplicity touches and turns.
  \item y-intercept $=f(0)$; find $a$ from one extra point.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{From zeros to equation}Given the zeros, write the factored form $y=a(x-r_1)(x-r_2)\cdots$; one extra point then fixes the leading coefficient $a$.
\lh{Multiplicity and shape}A factor's multiplicity shapes the graph at its zero: odd crosses the axis, even touches it, and higher multiplicity flattens the curve there.
\lh{Sketching}Combine end behaviour, the zeros with their multiplicities, and the $y$-intercept to sketch the whole polynomial.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Zeros}
Zeros of $f(x)=(x-3)(x+1)(x-2)$?\soln Set each factor to 0: $x=3,-1,2$.
\end{example}

\begin{example}{ex1}{Example 2 --- Multiplicity}
How does $f(x)=(x+2)(x-1)^2$ meet the x-axis?\soln At $x=-2$ (mult 1) it crosses; at $x=1$ (mult 2) it touches:\eplot{-4}{3}{-6}{6}{\addplot[exblue,very thick,domain=-2.4:1.9]{(x+2)*(x-1)^2};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(-2,0) (1,0)};}
\end{example}

\begin{example}{ex2}{Example 3 --- y-intercept}
y-intercept of $f(x)=(x-1)(x+2)(x-3)$?\soln $f(0)=(-1)(2)(-3)=6$, so $(0,6)$.
\end{example}

\begin{example}{ex3}{Example 4 --- Build an equation}
Write a cubic with zeros $-1,0,2$.\soln $f(x)=a\,x(x+1)(x-2)$; e.g. $y=x(x+1)(x-2)$.
\end{example}

\begin{example}{ex4}{Example 5 --- Find a}
A cubic has zeros $1,2,3$ and passes through $(0,12)$. Find $a$.\soln $12=a(0-1)(0-2)(0-3)=a(-6)\Rightarrow a=-2$.
\end{example}

\begin{example}{ex5}{Example 6 --- Zeros with multiplicity}
Zeros of $f(x)=(x-4)(x+2)^2$ and behaviour?\soln $x=4$ (cross), $x=-2$ (touch).
\end{example}

\begin{example}{ex6}{Example 7 --- Cross or touch}
At $x=3$ in $(x-3)^2(x+1)$?\soln Even multiplicity (2): touches.
\end{example}

\begin{example}{ex7}{Example 8 --- y-intercept}
y-intercept of $(x-2)(x+1)(x-4)$?\soln $(-2)(1)(-4)=8$.
\end{example}

\begin{example}{ex8}{Example 9 --- Build an equation}
Write a polynomial with zeros $-3$ and $2$.\soln $y=a(x+3)(x-2)$.
\end{example}
