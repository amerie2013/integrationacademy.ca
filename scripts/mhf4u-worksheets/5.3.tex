\begin{ideabox}
$y=a\sin(k(x-d))+c$: amplitude $|a|$, period $\tfrac{2\pi}{k}$, phase shift $d$, midline $y=c$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Amplitude $=|a|$; midline $y=c$.
  \item Period $=\tfrac{2\pi}{k}$; phase shift $=d$.
  \item Read these four to sketch, or read the graph to write the equation.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Sine and cosine in radians}Graphed over radians, $y=\sin x$ and $y=\cos x$ have period $2\pi$, amplitude $1$, and midline $y=0$.
\lh{The transformed form}$y=a\sin\!\left(k(x-d)\right)+c$ gives amplitude $|a|$, period $\dfrac{2\pi}{|k|}$, phase shift $d$, and midline $c$.
\lh{Key points}Plot the maxima, minima, and midline crossings for one period, then repeat to build the full wave.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Amplitude}
Amplitude of $y=3\sin x$?\soln $|a|=3$. Compare $y=2\sin x$ (taller wave):\begin{center}\begin{tikzpicture}\begin{axis}[width=9.2cm,height=4.6cm,axis lines=middle,xlabel={\small$x$},ylabel={\small$y$},xmin=-0.4,xmax=6.9,ymin=-2.4,ymax=2.4,samples=220,xtick={1.5708,3.1416,4.7124,6.2832},xticklabels={$\frac{\pi}{2}$,$\pi$,$\frac{3\pi}{2}$,$2\pi$}]\addplot[exblue,very thick,domain=0:6.4]{2*sin(deg(x))};\end{axis}\end{tikzpicture}\end{center}
\end{example}

\begin{example}{ex1}{Example 2 --- Period}
Period of $y=\sin(2x)$?\soln $\tfrac{2\pi}{2}=\pi$ — two cycles in $2\pi$:\begin{center}\begin{tikzpicture}\begin{axis}[width=9.2cm,height=4.6cm,axis lines=middle,xlabel={\small$x$},ylabel={\small$y$},xmin=-0.4,xmax=6.9,ymin=-1.6,ymax=1.6,samples=220,xtick={1.5708,3.1416,4.7124,6.2832},xticklabels={$\frac{\pi}{2}$,$\pi$,$\frac{3\pi}{2}$,$2\pi$}]\addplot[exblue,very thick,domain=0:6.4]{sin(deg(2*x))};\end{axis}\end{tikzpicture}\end{center}
\end{example}

\begin{example}{ex2}{Example 3 --- Midline}
Midline of $y=\sin x+2$?\soln $c=2$, so $y=2$.
\end{example}

\begin{example}{ex3}{Example 4 --- Longer period}
Period of $y=\cos(\tfrac{x}{2})$?\soln $k=\tfrac12\Rightarrow\tfrac{2\pi}{1/2}=4\pi$.
\end{example}

\begin{example}{ex4}{Example 5 --- Both}
Amplitude and period of $y=2\sin(3x)$?\soln $|a|=2$; period $\tfrac{2\pi}{3}$.
\end{example}

\begin{example}{ex5}{Example 6 --- Amplitude}
Amplitude of $y=5\sin x$?\soln $5$.
\end{example}

\begin{example}{ex6}{Example 7 --- Period}
Period of $y=\sin(4x)$?\soln $\tfrac{2\pi}{4}=\tfrac{\pi}{2}$.
\end{example}

\begin{example}{ex7}{Example 8 --- Midline}
Midline of $y=\cos x-3$?\soln $y=-3$.
\end{example}

\begin{example}{ex8}{Example 9 --- Both}
Amplitude and period of $y=4\cos(2x)$?\soln Amplitude $4$, period $\pi$.
\end{example}
