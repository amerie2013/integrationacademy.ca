\begin{ideabox}
$y=\log_b x$ is the reflection of $y=b^x$ in $y=x$: vertical asymptote $x=0$, domain $x>0$, range all reals, through $(1,0)$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item VA at $x=0$; domain $x>0$; range all reals.
  \item Passes through $(1,0)$; increasing for $b>1$.
  \item $y=\log_b(x-d)$ shifts the VA to $x=d$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Inverse of the exponential}$y=\log_b x$ is the reflection of $y=b^x$ in the line $y=x$, with a \textbf{vertical} asymptote at $x=0$.
\lh{Domain and range}A logarithmic function is defined only for $x>0$, and its range is all real numbers — the mirror of the exponential.
\lh{Transformations}The framework $y=a\log_b\!\left(k(x-d)\right)+c$ shifts and stretches the curve and moves the vertical asymptote to $x=d$.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- The log graph}
Sketch $y=\log_2 x$.\soln VA $x=0$, through $(1,0)$, increasing:\eplot{0}{8}{-4}{4}{\addplot[exblue,very thick,domain=0.08:8,samples=140]{ln(x)/ln(2)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Domain}
Domain of $y=\log x$?\soln Argument $>0$: $x>0$.
\end{example}

\begin{example}{ex2}{Example 3 --- Inverse pair}
How are $y=2^x$ and $y=\log_2 x$ related?\soln They are inverses — mirror images across $y=x$:\eplot{-4}{8}{-4}{8}{\addplot[exblue,very thick,domain=-4:3,samples=80]{2^x};\addplot[qorange,very thick,domain=0.08:8,samples=140]{ln(x)/ln(2)};\addplot[gray,dashed,domain=-4:8]{x};}
\end{example}

\begin{example}{ex3}{Example 4 --- Vertical asymptote}
VA of $y=\log x$?\soln As $x\to0^+,\ y\to-\infty$: $x=0$.
\end{example}

\begin{example}{ex4}{Example 5 --- Key point}
What point is on every $y=\log_b x$?\soln $\log_b 1=0$, so $(1,0)$.
\end{example}

\begin{example}{ex5}{Example 6 --- Transformation}
VA of $y=\log(x-2)$?\soln $x-2>0\Rightarrow x>2$, so VA $x=2$.
\end{example}

\begin{example}{ex6}{Example 7 --- Range}
Range of $y=\log x$?\soln All real numbers.
\end{example}

\begin{example}{ex7}{Example 8 --- Inverse}
Inverse of $y=3^x$?\soln $y=\log_3 x$.
\end{example}

\begin{example}{ex8}{Example 9 --- Transformation}
VA of $y=\log(x+3)$?\soln $x>-3$, so VA $x=-3$.
\end{example}
