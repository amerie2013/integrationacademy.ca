\begin{ideabox}
The remainder of $P(x)\div(x-a)$ is $P(a)$; and $x-a$ is a factor exactly when $P(a)=0$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Remainder Theorem: remainder $=P(a)$.
  \item Factor Theorem: $x-a$ is a factor $\iff P(a)=0$.
  \item Rational Root Theorem: $\tfrac{p}{q}$ with $p\mid$ constant, $q\mid$ leading coeff.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The remainder theorem}The remainder of $P(x)\div(x-a)$ is exactly $P(a)$, so one substitution replaces a full division.
\lh{The factor theorem}When $P(a)=0$ the remainder is zero and $x-a$ is a \textbf{factor} â€” the key tool for factoring higher-degree polynomials.
\lh{Finding zeros}Test factors of the constant term as candidate zeros, then divide out each factor you confirm.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Find a remainder}
Remainder of $P(x)=x^3-2x+1$ divided by $x-2$.\soln $P(2)=8-4+1=5$.
\end{example}

\begin{example}{ex1}{Example 2 --- Test a factor}
Is $x-1$ a factor of $x^3-1$?\soln $P(1)=1-1=0$: yes. The graph crosses at $x=1$:\eplot{-2}{2}{-4}{4}{\addplot[exblue,very thick,domain=-1.6:1.6]{x^3-1};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(1,0)};}
\end{example}

\begin{example}{ex2}{Example 3 --- Sum of cubes}
Is $x+2$ a factor of $x^3+8$?\soln $P(-2)=-8+8=0$: yes.
\end{example}

\begin{example}{ex3}{Example 4 --- Solve for k}
Find $k$ so $x-1$ is a factor of $x^3+kx-2$.\soln $P(1)=1+k-2=0\Rightarrow k=1$.
\end{example}

\begin{example}{ex4}{Example 5 --- Find a zero}
A rational zero of $P(x)=x^3-2x^2-x+2$?\soln Candidates $\pm1,\pm2$. $P(1)=1-2-1+2=0$, so $x=1$.
\end{example}

\begin{example}{ex5}{Example 6 --- Remainder}
Remainder of $x^3+1$ divided by $x-1$.\soln $P(1)=2$.
\end{example}

\begin{example}{ex6}{Example 7 --- Difference of cubes}
Is $x-2$ a factor of $x^3-8$?\soln $P(2)=8-8=0$: yes.
\end{example}

\begin{example}{ex7}{Example 8 --- Solve for k}
Find $k$ so $x+1$ is a factor of $x^3+kx+4$.\soln $P(-1)=-1-k+4=0\Rightarrow k=3$.
\end{example}

\begin{example}{ex8}{Example 9 --- Find a zero}
A rational zero of $x^3-x^2-4x+4$?\soln $P(1)=1-1-4+4=0$, so $x=1$.
\end{example}
