\begin{ideabox}
$\csc\theta=\tfrac1{\sin\theta}$, $\sec\theta=\tfrac1{\cos\theta}$, $\cot\theta=\tfrac{\cos\theta}{\sin\theta}$ — each undefined where its base function is zero.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $\csc=\tfrac1{\sin}$ (undefined where $\sin=0$); $\sec=\tfrac1{\cos}$.
  \item $\cot=\tfrac{\cos}{\sin}$ (undefined where $\sin=0$).
  \item Evaluate the base ratio first, then reciprocate.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Three more ratios}The reciprocals $\csc\theta=\dfrac{1}{\sin\theta}$, $\sec\theta=\dfrac{1}{\cos\theta}$, and $\cot\theta=\dfrac{1}{\tan\theta}$ complete the six trig functions.
\lh{Their graphs}Each reciprocal graph has vertical asymptotes wherever the original ratio is zero; cosecant and secant never take values between $-1$ and $1$.
\lh{Where they help}Reciprocal ratios appear in identities and in problems where a side ratio is naturally inverted.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Cosecant}
Find $\csc\tfrac{\pi}{2}$.\soln $\sin\tfrac{\pi}{2}=1\Rightarrow\csc=1$. The reciprocal blows up wherever $\sin$ (below) hits 0:\begin{center}\begin{tikzpicture}\begin{axis}[width=9.2cm,height=4.6cm,axis lines=middle,xlabel={\small$x$},ylabel={\small$y$},xmin=-0.4,xmax=6.9,ymin=-1.6,ymax=1.6,samples=220,xtick={1.5708,3.1416,4.7124,6.2832},xticklabels={$\frac{\pi}{2}$,$\pi$,$\frac{3\pi}{2}$,$2\pi$}]\addplot[exblue,very thick,domain=0:6.4]{sin(deg(x))};\end{axis}\end{tikzpicture}\end{center}
\end{example}

\begin{example}{ex1}{Example 2 --- Secant}
Find $\sec 0$.\soln $\cos 0=1\Rightarrow\sec=1$.
\end{example}

\begin{example}{ex2}{Example 3 --- Cotangent}
Find $\cot\tfrac{\pi}{4}$.\soln $\tan\tfrac{\pi}{4}=1\Rightarrow\cot=1$.
\end{example}

\begin{example}{ex3}{Example 4 --- Cosecant}
Find $\csc\tfrac{\pi}{6}$.\soln $\sin\tfrac{\pi}{6}=\tfrac12\Rightarrow\csc=2$.
\end{example}

\begin{example}{ex4}{Example 5 --- Secant}
Find $\sec\tfrac{\pi}{3}$.\soln $\cos\tfrac{\pi}{3}=\tfrac12\Rightarrow\sec=2$.
\end{example}

\begin{example}{ex5}{Example 6 --- Cotangent}
Find $\cot\tfrac{\pi}{2}$.\soln $\tfrac{\cos(\pi/2)}{\sin(\pi/2)}=\tfrac01=0$.
\end{example}

\begin{example}{ex6}{Example 7 --- Secant of π}
Find $\sec\pi$.\soln $\tfrac1{\cos\pi}=\tfrac1{-1}=-1$.
\end{example}

\begin{example}{ex7}{Example 8 --- Cosecant}
Find $\csc\tfrac{3\pi}{2}$.\soln $\sin\tfrac{3\pi}{2}=-1\Rightarrow\csc=-1$.
\end{example}

\begin{example}{ex8}{Example 9 --- Secant of 2π}
Find $\sec 2\pi$.\soln $\cos 2\pi=1\Rightarrow\sec=1$.
\end{example}
