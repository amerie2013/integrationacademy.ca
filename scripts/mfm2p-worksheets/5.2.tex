\begin{ideabox}
\textbf{Similar} triangles ($\sim$) have equal angles and proportional sides; \textbf{congruent} ($\cong$) triangles are identical (scale factor $1$).
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Similar triangles: corresponding angles equal, corresponding sides in a constant ratio (the \textbf{scale factor}).
  \item Prove similar by \textbf{AA} (two equal angles), \textbf{SSS} (sides proportional), or \textbf{SAS}.
  \item Set up a proportion $\dfrac{\text{side}}{\text{matching side}}=\dfrac{\text{side}}{\text{matching side}}$ to find a missing length.
  \item Equal ratios of perimeters equal the scale factor.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Congruent vs similar}\textbf{Congruent} triangles are identical; \textbf{similar} triangles share the same shape but may differ in size.
\lh{Proportional sides}In similar triangles, matching angles are equal and matching sides share one scale factor.
\lh{Why it helps}Similarity lets you find an unknown length by comparing a small and a large copy.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- AA similarity}
Two triangles have angles $40^\circ,60^\circ,80^\circ$ each. Are they similar?
\soln Yes --- all corresponding angles are equal (AA), so the triangles are similar.
\end{example}

\begin{example}{ex1}{Example 2 --- SSS similarity}
Are triangles with sides $3,4,5$ and $6,8,10$ similar?
\soln $\tfrac{3}{6}=\tfrac{4}{8}=\tfrac{5}{10}=\tfrac12$ --- all ratios equal, so similar (scale factor $2$).
\end{example}

\begin{example}{ex2}{Example 3 --- Finding a side}
$\triangle ABC\sim\triangle DEF$ with $AB=4$, $DE=6$, $DF=9$. Find $AC$.
\soln $\dfrac{AC}{DF}=\dfrac{AB}{DE}\Rightarrow \dfrac{AC}{9}=\dfrac{4}{6}\Rightarrow AC=6.$
\end{example}

\begin{example}{ex3}{Example 4 --- Scale factor}
Find the scale factor from $5,12,13$ to $10,24,26$.
\soln Each side doubles, so the scale factor is $2$.
\end{example}

\begin{example}{ex4}{Example 5 --- Proportion}
In similar triangles, $\dfrac{2}{5}=\dfrac{8}{x}$. Find $x$.
\soln Cross-multiply: $2x=40\Rightarrow x=20.$
\end{example}

\begin{example}{ex5}{Example 6 --- Shadow problem}
A $1.5$ m person casts a $2$ m shadow; a tree casts a $16$ m shadow. Find the tree's height.
\soln $\dfrac{1.5}{2}=\dfrac{h}{16}\Rightarrow h=12$ m.
\end{example}

\begin{example}{ex6}{Example 7 --- Overlapping triangles}
Two nested similar triangles give $\dfrac{6}{9}=\dfrac{8}{x}$. Find $x$.
\soln $6x=72\Rightarrow x=12.$
\end{example}

\begin{example}{ex7}{Example 8 --- Checking similarity}
Are triangles $3,4,6$ and $6,8,12$ similar?
\soln $\tfrac36=\tfrac48=\tfrac{6}{12}=\tfrac12$ --- yes, similar.
\end{example}

\begin{example}{ex8}{Example 9 --- Perimeter ratio (harder)}
$\triangle ABC\sim\triangle DEF$ with scale factor $3:5$. If the perimeter of $\triangle ABC$ is $24$, find the perimeter of $\triangle DEF$.
\soln Perimeters share the scale factor: $24\times\tfrac53=40.$
\end{example}
