\begin{ideabox}
Expand, collect, gather the variable on one side, then isolate it.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Expand any brackets first.
  \item Collect like terms on each side.
  \item Move the variable to one side, numbers to the other.
  \item Divide to isolate the variable, then check.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{More steps, same idea}Multi-step equations may need you to expand brackets and collect like terms before isolating the variable.
\lh{Variables on both sides}Move all variable terms to one side and all numbers to the other, then solve as usual.
\lh{Stay organized}Work one step at a time and track your signs — most errors come from a dropped negative.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Variables both sides}
Solve $2x+3=x+8$.\soln Subtract $x$: $x+3=8$, so $x=5$.
\end{example}

\begin{example}{ex1}{Example 2 --- Brackets}
Solve $3(x-2)=12$.\soln $3x-6=12$, $3x=18$, $x=6$.
\end{example}

\begin{example}{ex2}{Example 3 --- Both sides + collect}
Solve $4x-7=2x+5$.\soln $2x=12$, $x=6$.
\end{example}

\begin{example}{ex3}{Example 4 --- A fraction}
Solve $\dfrac{x}{3}-2=4$.\soln $\dfrac{x}{3}=6$, $x=18$.
\end{example}

\begin{example}{ex4}{Example 5 --- Brackets both sides}
Solve $2(x+1)=3(x-2)$.\soln $2x+2=3x-6$, $8=x$.
\end{example}

\begin{example}{ex5}{Example 6 --- Collect first}
Solve $5x-3=2x+9$.\soln $3x=12$, $x=4$.
\end{example}

\begin{example}{ex6}{Example 7 --- Expand then solve}
Solve $2(x+4)=14$.\soln $2x+8=14$, $x=3$.
\end{example}

\begin{example}{ex7}{Example 8 --- Both sides}
Solve $6x-5=4x+7$.\soln $2x=12$, $x=6$.
\end{example}

\begin{example}{ex8}{Example 9 --- Brackets both sides (harder)}
Solve $3(x-1)=2(x+4)$.\soln $3x-3=2x+8$, $x=11$.
\end{example}
