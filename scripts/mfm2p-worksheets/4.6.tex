{\LARGE\bfseries\color{titleblue}Worksheet 4.6 --- Composite Figures \& Measurement Problems}\\[2pt]
{\small Grade 10 Applied (MFM2P) \textperiodcentered{} Unit 4: Measurement}
\end{center}

\begin{ideabox}
Break a composite figure into simple solids, then add (or subtract) their measurements.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Identify the simple solids (prism, cylinder, cone, hemisphere).
  \item Add the parts; subtract a hollowed-out part.
  \item For surface area, count only exposed faces (drop hidden joins).
  \item Keep $\pi$ exact until the end, then round if asked.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Break it into parts}A composite solid is built from simpler shapes; measure each part, then add — or subtract a hole.
\lh{Watch shared faces}When solids join, ignore the faces hidden where they meet when finding surface area.
\lh{Solve in context}Real problems ask for capacity, material, or cost — translate the measurement into what is requested.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Silo (cylinder + hemisphere)}
Cylinder $r=3,\ h=10$ topped by a hemisphere $r=3$.\soln $90\pi+18\pi=108\pi\approx339.3$.
\end{example}

\begin{example}{ex1}{Example 2 --- House (box + roof)}
Box $6\times4\times3$ + triangular-prism roof (triangle base $6$, height $2$, length $4$).\soln $72+\tfrac12(6)(2)(4)=72+24=96$.
\end{example}

\begin{example}{ex2}{Example 3 --- Pipe (subtract a hole)}
Cylinder $R=5$ with a hole $r=3$, length $10$.\soln $250\pi-90\pi=160\pi\approx502.7$.
\end{example}

\begin{example}{ex3}{Example 4 --- Cone + scoop}
Cone $r=3,\ h=8$ with a hemisphere $r=3$.\soln $24\pi+18\pi=42\pi\approx131.9$.
\end{example}

\begin{example}{ex4}{Example 5 --- Painting (surface area)}
A cube side $4$ on the ground (no bottom painted).\soln $5(16)=80$ units$^2$.
\end{example}

\begin{example}{ex5}{Example 6 --- Box + pyramid}
Box $5\times5\times2$ + pyramid (base $5\times5$, height $6$).\soln $50+\tfrac13(25)(6)=50+50=100$.
\end{example}

\begin{example}{ex6}{Example 7 --- Washer}
Disc $R=4$ with hole $r=2$, thickness $1$.\soln $16\pi-4\pi=12\pi$.
\end{example}

\begin{example}{ex7}{Example 8 --- Drilled cube}
Cube side $3$ with a $1\times1\times3$ hole.\soln $27-3=24$ units$^3$.
\end{example}

\begin{example}{ex8}{Example 9 --- Silo with cone top (harder)}
Cylinder $r=4,\ h=12$ + cone top $r=4,\ h=3$.\soln $192\pi+16\pi=208\pi\approx653.5$.
\end{example}
