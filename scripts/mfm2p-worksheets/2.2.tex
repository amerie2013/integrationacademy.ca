{\small Grade 10 Applied (MFM2P) \textperiodcentered{} Unit 2: Linear Relations}
\end{center}

\begin{ideabox}
Direct variation $y=kx$ passes through the origin; partial variation $y=mx+b$ has a starting value.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Direct: $y=kx$, constant of variation $k=\dfrac{y}{x}$, graph through $(0,0)$.
  \item Partial: $y=mx+b$, initial value $b$ plus rate $m$.
  \item A fixed fee plus a per-unit rate is partial variation.
  \item Find $k$ from one point; find a partial model from the fixed value and rate.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Direct variation}In \textbf{direct variation} $y=mx$, the line passes through the origin, so doubling $x$ doubles $y$.
\lh{Partial variation}In \textbf{partial variation} $y=mx+b$, there is a fixed starting value $b$ plus a rate $m$.
\lh{Spotting them}A base fee plus a per-unit charge is partial variation; a pure per-unit rate with no fee is direct.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Identify (direct)}
Is $y=4x$ direct or partial?\soln No constant term $\Rightarrow$ direct, $k=4$.
\end{example}

\begin{example}{ex1}{Example 2 --- Identify (partial)}
Is $y=3x+5$ direct or partial?\soln Starting value $5\neq0\Rightarrow$ partial.
\plot{-3}{3}{-2}{12}{\addplot[exblue,very thick,domain=-3:2.3,samples=2]{3*x+5};\addplot[mark=*,only marks,black]coordinates{(0,5)};}
\end{example}

\begin{example}{ex2}{Example 3 --- Find k}
$y$ varies directly with $x$; $y=12$ when $x=3$. Find $k$ and the equation.\soln $k=\dfrac{12}{3}=4$, so $y=4x$.
\end{example}

\begin{example}{ex3}{Example 4 --- Write a partial model}
A gym charges $20$ dollars to join plus $5$ dollars/visit. Write the equation.\soln $C=5v+20$.
\end{example}

\begin{example}{ex4}{Example 5 --- Direct vs partial graph}
Compare $y=2x$ and $y=x+10$ at $x=5$.\soln $2(5)=10$ vs $5+10=15$ — direct is cheaper here.
\plot{-1}{12}{-1}{22}{\addplot[exblue,very thick,domain=0:11,samples=2]{2*x};\addplot[qorange,very thick,domain=0:11,samples=2]{x+10};\addplot[mark=*,only marks,black]coordinates{(10,20)};}
\end{example}

\begin{example}{ex5}{Example 6 --- Cost as direct variation}
Cost varies directly with litres: $25$ dollars for $10$ L. Find the cost of $16$ L.\soln $k=2.5$; $C=2.5(16)=40$ dollars.
\end{example}

\begin{example}{ex6}{Example 7 --- Read a partial model}
For $C=4n+8$, state the fixed cost and the rate.\soln Fixed $8$, rate $4$ per unit.
\end{example}

\begin{example}{ex7}{Example 8 --- Through the origin?}
Does $y=2x-1$ pass through the origin?\soln At $x=0$, $y=-1\neq0$ — no (partial).
\end{example}

\begin{example}{ex8}{Example 9 --- Build and use (harder)}
A printer costs $60$ dollars plus $0.10$ dollars/page. Find the cost of $300$ pages.\soln $C=0.10(300)+60=90$ dollars.
\end{example}
