\begin{ideabox}
A ratio compares quantities; a proportion sets two ratios equal; a scale shrinks or enlarges by a constant factor.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Simplify a ratio like a fraction: $12:18=2:3$.
  \item Solve a proportion by cross-multiplying.
  \item A scale $1:n$ means $1$ drawing unit $=n$ real units; real $=$ drawing $\times n$.
  \item Ratios power similar triangles (proportional sides).
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Ratios and proportions}A \textbf{ratio} compares quantities; a \textbf{proportion} says two ratios are equal, solved by cross-multiplying.
\lh{Scale}A scale is a ratio between a drawing or model and the real object, letting you compute actual sizes.
\lh{Setting it up}Match corresponding parts in the correct order before cross-multiplying.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Simplify a ratio}
Simplify $12:18$.\soln Divide both by $6$: $2:3$.
\end{example}

\begin{example}{ex1}{Example 2 --- Solve a proportion}
Solve $\dfrac34=\dfrac{x}{12}$.\soln Cross-multiply: $4x=36$, $x=9$.
\end{example}

\begin{example}{ex2}{Example 3 --- Scale drawing}
On a $1:50$ plan a wall is $4$ cm. Find the real length.\soln $4\times50=200$ cm $=2$ m.
\end{example}

\begin{example}{ex3}{Example 4 --- Recipe ratio}
Flour : sugar $=3:1$. With $9$ cups flour, how much sugar?\soln $\dfrac31=\dfrac{9}{s}\Rightarrow s=3$ cups.
\end{example}

\begin{example}{ex4}{Example 5 --- Map scale}
A $1:100000$ map: two towns are $7.5$ cm apart. Real distance?\soln $7.5\times100000=750000$ cm $=7.5$ km.
\end{example}

\begin{example}{ex5}{Example 6 --- Simplify}
Simplify $20:35$.\soln Divide by $5$: $4:7$.
\end{example}

\begin{example}{ex6}{Example 7 --- Solve a proportion}
Solve $\dfrac58=\dfrac{x}{24}$.\soln $8x=120$, $x=15$.
\end{example}

\begin{example}{ex7}{Example 8 --- Model scale}
A $1:25$ model car is $18$ cm. Real length?\soln $18\times25=450$ cm.
\end{example}

\begin{example}{ex8}{Example 9 --- Map scale (harder)}
A $1:50000$ map: two points are $6$ cm apart. Real distance in km?\soln $6\times50000=300000$ cm $=3$ km.
\end{example}
