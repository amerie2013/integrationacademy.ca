\begin{ideabox}
Pyramids and cones taper to a point, so their volume is one-third of the matching prism or cylinder.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Volume: $V=\tfrac13(\text{base area})h$; cone $V=\tfrac13\pi r^2h$.
  \item Cone surface area: $SA=\pi r^2+\pi r s$ ($s=$ slant height).
  \item Square pyramid $SA=\text{base}+\tfrac12(\text{perimeter})s$.
  \item Slant height: $s=\sqrt{r^2+h^2}$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{One-third the prism}A pyramid or cone holds one-third of the matching prism or cylinder: $V=\tfrac13(\text{base})h$.
\lh{Slant height}For surface area, a cone's slant height comes from $\ell=\sqrt{r^2+h^2}$.
\lh{Match the formula}Choose the formula that fits the solid, keeping the height perpendicular to the base.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Cone volume}
$V$ of a cone $r=3,\ h=4$.\soln $\tfrac13\pi(9)(4)=12\pi\approx37.7$.
\end{example}

\begin{example}{ex1}{Example 2 --- Pyramid volume}
Square pyramid base $6\times6$, height $10$.\soln $\tfrac13(36)(10)=120$.
\end{example}

\begin{example}{ex2}{Example 3 --- Slant height}
Cone $r=3,\ h=4$. Find $s$.\soln $s=\sqrt{9+16}=5$.
\end{example}

\begin{example}{ex3}{Example 4 --- Cone surface area}
Cone $r=3,\ s=5$.\soln $\pi(9)+\pi(3)(5)=24\pi\approx75.4$.
\end{example}

\begin{example}{ex4}{Example 5 --- Pyramid surface area}
Square pyramid base $6$, slant $5$.\soln $36+\tfrac12(24)(5)=96$.
\end{example}

\begin{example}{ex5}{Example 6 --- Bigger cone}
$V$ of a cone $r=6,\ h=10$.\soln $\tfrac13\pi(36)(10)=120\pi\approx377.0$.
\end{example}

\begin{example}{ex6}{Example 7 --- Pyramid volume}
Square pyramid base $4\times4$, height $9$.\soln $\tfrac13(16)(9)=48$.
\end{example}

\begin{example}{ex7}{Example 8 --- Slant height}
Cone $r=6,\ h=8$. Find $s$.\soln $\sqrt{36+64}=10$.
\end{example}

\begin{example}{ex8}{Example 9 --- Total surface area (harder)}
Cone $r=3,\ h=4$. Find the total $SA$.\soln $s=5$; $SA=9\pi+15\pi=24\pi\approx75.4$.
\end{example}
