\begin{ideabox}
Algebra rests on confident integer arithmetic and careful evaluation.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Adding: same signs add and keep the sign; different signs subtract and keep the larger size's sign.
  \item Subtracting means adding the opposite: $a-(-b)=a+b$.
  \item Multiplying/dividing: same signs give positive, different signs give negative.
  \item Substitute negatives in brackets, then follow BEDMAS (brackets, exponents, $\times\div$, $+-$).
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Expressions and integers}An \textbf{expression} combines numbers and variables with operations, while \textbf{integers} include negatives whose signs follow fixed rules.
\lh{Integer rules}Same signs multiply or divide to a positive, different signs to a negative; subtracting is adding the opposite, so $5-(-3)=5+3=8$.
\lh{Order of operations}Follow BEDMAS — brackets, exponents, division and multiplication, then addition and subtraction — to evaluate correctly.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Adding integers}
Evaluate $-3+7-5$.\soln $-3+7=4$, then $4-5=-1$.
\end{example}

\begin{example}{ex1}{Example 2 --- Subtracting a negative}
Evaluate $4-(-6)$.\soln Add the opposite: $4+6=10$.
\end{example}

\begin{example}{ex2}{Example 3 --- Multiplying}
Evaluate $(-4)(3)$.\soln Different signs $\Rightarrow$ negative: $-12$.
\end{example}

\begin{example}{ex3}{Example 4 --- Dividing}
Evaluate $-20\div(-5)$.\soln Same signs $\Rightarrow$ positive: $4$.
\end{example}

\begin{example}{ex4}{Example 5 --- Evaluate by substitution}
Evaluate $2x^2-3x$ at $x=-2$.\soln $2(-2)^2-3(-2)=2(4)+6=14$.
\end{example}

\begin{example}{ex5}{Example 6 --- Order of operations}
Evaluate $5+2\times3^2-4$.\soln $3^2=9$; $2\times9=18$; $5+18-4=19$.
\end{example}

\begin{example}{ex6}{Example 7 --- Two variables}
Evaluate $\dfrac{a-b}{c}$ at $a=7,\ b=-3,\ c=2$.\soln $\dfrac{7-(-3)}{2}=\dfrac{10}{2}=5$.
\end{example}

\begin{example}{ex7}{Example 8 --- Sign and exponent}
Evaluate $-3^2$ and $(-3)^2$.\soln $-3^2=-(9)=-9$; $(-3)^2=9$.
\end{example}

\begin{example}{ex8}{Example 9 --- Putting it together}
Evaluate $\dfrac{2a+b}{a-b}$ at $a=4,\ b=-2$.\soln $\dfrac{8-2}{4+2}=\dfrac{6}{6}=1$.
\end{example}
