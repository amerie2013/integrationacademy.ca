{\LARGE\bfseries\color{titleblue}Worksheet 6.2 --- Common Factoring \& Factoring by Grouping}\\[2pt]
{\small Grade 10 Applied (MFM2P) \textperiodcentered{} Unit 6: Quadratic Expressions}
\end{center}

\begin{ideabox}
\textbf{Factoring} reverses expanding. Always start by removing the greatest common factor (GCF).
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Find the \textbf{GCF} of the coefficients and the lowest power of each variable.
  \item $ab+ac=a(b+c)$; check by expanding.
  \item Four terms? Try \textbf{grouping} in pairs and factor each pair.
  \item A negative leading term: factor out the negative too.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Common factor first}\textbf{Factoring} reverses expanding, so always pull out the greatest common factor first.
\lh{Grouping}With four terms, group them in pairs, factor each pair, then factor out the shared binomial.
\lh{Check}Expand your factors to confirm you recover the original expression.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Numeric GCF}
Factor $6x+9$.
\soln GCF $=3$: $6x+9=3(2x+3).$
\end{example}

\begin{example}{ex1}{Example 2 --- Variable GCF}
Factor $4x^2-8x$.
\soln GCF $=4x$: $4x^2-8x=4x(x-2).$
\end{example}

\begin{example}{ex2}{Example 3 --- Higher powers}
Factor $12x^3+18x^2$.
\soln GCF $=6x^2$: $12x^3+18x^2=6x^2(2x+3).$
\end{example}

\begin{example}{ex3}{Example 4 --- Two variables}
Factor $5x^2y+10xy^2$.
\soln GCF $=5xy$: $5xy(x+2y).$
\end{example}

\begin{example}{ex4}{Example 5 --- Negative GCF}
Factor $-3x^2-6x$.
\soln $-3x(x+2).$
\end{example}

\begin{example}{ex5}{Example 6 --- Grouping I}
Factor $x^3+2x^2+3x+6$.
\soln $x^2(x+2)+3(x+2)=(x+2)(x^2+3).$
\end{example}

\begin{example}{ex6}{Example 7 --- Grouping II}
Factor $2x^3-x^2+6x-3$.
\soln $x^2(2x-1)+3(2x-1)=(2x-1)(x^2+3).$
\end{example}

\begin{example}{ex7}{Example 8 --- Grouping with two variables}
Factor $xy+2x+3y+6$.
\soln $x(y+2)+3(y+2)=(x+3)(y+2).$
\end{example}

\begin{example}{ex8}{Example 9 --- Grouping (harder)}
Factor $6x^3-9x^2-4x+6$.
\soln $3x^2(2x-3)-2(2x-3)=(2x-3)(3x^2-2).$
\end{example}
