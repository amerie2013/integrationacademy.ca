{\LARGE\bfseries\color{titleblue}Worksheet 4.3 --- Volume of Prisms \& Cylinders}\\[2pt]
{\small Grade 10 Applied (MFM2P) \textperiodcentered{} Unit 4: Measurement}
\end{center}

\begin{ideabox}
Volume is the space inside: $V=(\text{base area})\times\text{height}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Rectangular prism: $V=lwh$.
  \item Cylinder: $V=\pi r^2h$.
  \item Triangular prism: $V=(\text{triangle area})\times\text{length}$.
  \item Volume units are cubic; use the radius for cylinders.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Base times height}The volume of a prism or cylinder is the \textbf{base area times the height}: box $V=lwh$, cylinder $V=\pi r^2h$.
\lh{Keep units consistent}Use the same unit for every length, and give the answer in cubic units.
\lh{What it means}Volume measures how much a solid holds or fills.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Rectangular prism}
$V$ of a $4\times3\times2$ box.\soln $4\times3\times2=24$ units$^3$.
\end{example}

\begin{example}{ex1}{Example 2 --- Cube}
$V$ of a cube, side $5$.\soln $5^3=125$ units$^3$.
\end{example}

\begin{example}{ex2}{Example 3 --- Cylinder}
$V$ of a cylinder $r=3,\ h=10$.\soln $\pi(9)(10)=90\pi\approx282.7$.
\end{example}

\begin{example}{ex3}{Example 4 --- Triangular prism}
Triangle area $6$, length $10$.\soln $6\times10=60$ units$^3$.
\end{example}

\begin{example}{ex4}{Example 5 --- Water tank}
Cylinder $r=2$ m, $h=5$ m.\soln $\pi(4)(5)=20\pi\approx62.8$ m$^3$.
\end{example}

\begin{example}{ex5}{Example 6 --- Box}
$V$ of a $6\times2\times3$ prism.\soln $36$ units$^3$.
\end{example}

\begin{example}{ex6}{Example 7 --- Smaller cube}
$V$ of a cube, side $4$.\soln $64$ units$^3$.
\end{example}

\begin{example}{ex7}{Example 8 --- Cylinder (exact)}
$V$ of a cylinder $r=5,\ h=4$.\soln $\pi(25)(4)=100\pi\approx314.2$.
\end{example}

\begin{example}{ex8}{Example 9 --- Find a height (harder)}
A cylinder holds $48\pi$ cm$^3$ with $r=4$. Find $h$.\soln $\pi(16)h=48\pi\Rightarrow h=3$ cm.
\end{example}
