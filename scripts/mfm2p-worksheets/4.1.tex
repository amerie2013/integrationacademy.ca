\begin{ideabox}
Applied math uses metric (powers of ten) and imperial units; convert within and between them.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Metric: $1$ m $=100$ cm, $1$ km $=1000$ m, $1$ L $=1000$ mL.
  \item Imperial: $1$ ft $=12$ in, $1$ yd $=3$ ft, $1$ lb $=16$ oz.
  \item Between: $1$ in $=2.54$ cm, $1$ mile $\approx1.6$ km, $1$ kg $\approx2.2$ lb.
  \item Going to a larger unit, divide; to a smaller unit, multiply.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Units matter}Every measurement needs a unit, and choosing a sensible one keeps the numbers manageable.
\lh{Metric conversions}Metric units scale by powers of ten: $1$ m $=100$ cm, $1$ km $=1000$ m. Moving to a bigger unit divides, to a smaller one multiplies.
\lh{Convert carefully}Multiply by a fraction equal to $1$ that cancels the old unit, and check the unwanted unit cancels.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Metre to cm}
Convert $3.5$ m to cm.\soln $3.5\times100=350$ cm.
\end{example}

\begin{example}{ex1}{Example 2 --- mL to L}
Convert $2500$ mL to L.\soln $2500\div1000=2.5$ L.
\end{example}

\begin{example}{ex2}{Example 3 --- Feet to inches}
Convert $5$ ft to inches.\soln $5\times12=60$ in.
\end{example}

\begin{example}{ex3}{Example 4 --- Inches to cm}
Convert $10$ in to cm.\soln $10\times2.54=25.4$ cm.
\end{example}

\begin{example}{ex4}{Example 5 --- km to miles}
Convert $8$ km to miles.\soln $8\div1.6=5$ miles.
\end{example}

\begin{example}{ex5}{Example 6 --- km to m}
Convert $4.2$ km to m.\soln $4.2\times1000=4200$ m.
\end{example}

\begin{example}{ex6}{Example 7 --- Yards to feet}
Convert $3$ yd to feet.\soln $3\times3=9$ ft.
\end{example}

\begin{example}{ex7}{Example 8 --- kg to pounds}
Convert $5$ kg to pounds.\soln $5\times2.2=11$ lb.
\end{example}

\begin{example}{ex8}{Example 9 --- Miles to km (harder)}
Convert $12$ miles to km.\soln $12\times1.6=19.2$ km.
\end{example}
