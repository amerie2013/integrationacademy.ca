\begin{ideabox}
Turn a situation into an equation, solve it, and answer in context.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Define a variable for the unknown.
  \item Write an equation from the words.
  \item Solve it, then answer in a sentence and check.
  \item Cost models are linear: $C=(\text{rate})x+(\text{fixed fee})$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{From words to an equation}Translate a problem into an equation by choosing a variable for the unknown and turning each condition into symbols.
\lh{Solve and interpret}Solve the equation, then state the answer in the words of the original problem.
\lh{Check for sense}Make sure the answer is realistic — a length or a price should be positive and sensible.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Number problem}
Five more than twice a number is $17$. Find it.\soln $2x+5=17\Rightarrow x=6$.
\end{example}

\begin{example}{ex1}{Example 2 --- Perimeter}
A rectangle's length is $3$ m more than its width; perimeter $26$ m. Find the width.\soln $2(w+3)+2w=26\Rightarrow w=5$ m.
\end{example}

\begin{example}{ex2}{Example 3 --- Cost model}
A plumber charges $50$ dollars plus $40$ dollars/hour. How long is a $210$-dollar job?\soln $40x+50=210\Rightarrow x=4$ h.
\end{example}

\begin{example}{ex3}{Example 4 --- Consecutive integers}
Three consecutive integers sum to $48$. Find them.\soln $3x+3=48\Rightarrow x=15$: $15,16,17$.
\end{example}

\begin{example}{ex4}{Example 5 --- Comparing plans}
Plan A: $5$ dollars/visit. Plan B: $3$ dollars/visit plus $4$ dollars. When equal?\soln $5x=3x+4\Rightarrow x=2$ visits.
\end{example}

\begin{example}{ex5}{Example 6 --- Less than}
Seven less than three times a number is $11$. Find it.\soln $3x-7=11\Rightarrow x=6$.
\end{example}

\begin{example}{ex6}{Example 7 --- Taxi}
A taxi charges $3$ dollars plus $2$ dollars/km. What distance costs $15$ dollars?\soln $2d+3=15\Rightarrow d=6$ km.
\end{example}

\begin{example}{ex7}{Example 8 --- Rectangle (length twice width)}
Length is twice the width; perimeter $36$. Find the dimensions.\soln $2(2w)+2w=36\Rightarrow w=6$, length $12$.
\end{example}

\begin{example}{ex8}{Example 9 --- Two gyms (harder)}
Gym A: $8$ dollars/visit. Gym B: $5$ dollars/visit plus $12$ dollars. When equal?\soln $8x=5x+12\Rightarrow x=4$ visits.
\end{example}

  \item $2x+1=35\Rightarrow 17,18$.
