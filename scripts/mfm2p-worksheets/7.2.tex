\begin{ideabox}
Every parabola has a \textbf{vertex}, an \textbf{axis of symmetry}, a \textbf{max or min}, possibly \textbf{zeros}, and a \textbf{$y$-intercept}.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item \textbf{Vertex:} the turning point; the \textbf{axis of symmetry (AOS)} is $x=h$ through it.
  \item \textbf{Zeros} ($x$-intercepts): where $y=0$; the \textbf{$y$-intercept} is where $x=0$.
  \item The AOS is the midpoint of the two zeros.
  \item $a>0$ gives a minimum at the vertex; $a<0$ gives a maximum.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The landmarks}A parabola has a \textbf{vertex} (highest or lowest point), an \textbf{axis of symmetry}, a $y$-intercept, and often $x$-intercepts.
\lh{Opening}A positive leading coefficient opens the parabola up; a negative one opens it down.
\lh{Symmetry}The curve mirrors itself across the axis of symmetry through the vertex.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Reading a simple parabola}
State the features of $y=x^2-4$.
\soln Vertex $(0,-4)$, AOS $x=0$, minimum $-4$, zeros $\pm2$, $y$-intercept $-4$, opens up.
\plot{-4}{4}{-5}{5}{\addplot[exblue,very thick,domain=-3:3,samples=60]{x^2-4};\addplot[mark=*,only marks,black]coordinates{(-2,0)(2,0)(0,-4)};}
\end{example}

\begin{example}{ex1}{Example 2 --- From vertex form}
State the features of $y=-(x-2)^2+9$.
\soln Vertex $(2,9)$, AOS $x=2$, maximum $9$, opens down.
\end{example}

\begin{example}{ex2}{Example 3 --- Finding zeros}
Find the zeros of $y=x^2-9$.
\soln $x^2-9=0\Rightarrow x=\pm3.$
\end{example}

\begin{example}{ex3}{Example 4 --- y-intercept}
Find the $y$-intercept of $y=x^2-2x-3$.
\soln Set $x=0$: $y=-3.$
\end{example}

\begin{example}{ex4}{Example 5 --- AOS from zeros}
A parabola has zeros at $1$ and $5$. Find its axis of symmetry.
\soln AOS is the midpoint: $x=\tfrac{1+5}{2}=3.$
\end{example}

\begin{example}{ex5}{Example 6 --- Vertex from the AOS}
Find the vertex of $y=x^2-6x+5$.
\soln AOS $x=\tfrac{6}{2}=3$; $y=9-18+5=-4$. Vertex $(3,-4)$.
\plot{-1}{7}{-5}{6}{\addplot[exblue,very thick,domain=0.2:5.8,samples=60]{x^2-6*x+5};\addplot[mark=*,only marks,black]coordinates{(1,0)(5,0)(3,-4)};}
\end{example}

\begin{example}{ex6}{Example 7 --- Maximum value}
Find the maximum value of $y=-2(x-1)^2+8$.
\soln Maximum $8$ at $x=1$ (since $a<0$).
\end{example}

\begin{example}{ex7}{Example 8 --- All features (vertex form)}
Describe $y=(x-3)^2-4$.
\soln Vertex $(3,-4)$, AOS $x=3$, opens up, minimum $-4$.
\end{example}

\begin{example}{ex8}{Example 9 --- Full analysis (harder)}
For $y=x^2-4x-5$, find the vertex, zeros, and $y$-intercept.
\soln AOS $x=2$, $y=4-8-5=-9$: vertex $(2,-9)$. Factor: $(x-5)(x+1)$, zeros $5,-1$. $y$-intercept $-5$.
\plot{-3}{6}{-10}{6}{\addplot[exblue,very thick,domain=-1.5:5.5,samples=60]{x^2-4*x-5};\addplot[mark=*,only marks,black]coordinates{(-1,0)(5,0)(2,-9)};}
\end{example}
