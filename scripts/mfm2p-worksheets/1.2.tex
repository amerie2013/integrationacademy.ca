{\LARGE\bfseries\color{titleblue}Worksheet 1.2 --- Like Terms \& the Distributive Property}\\[2pt]
{\small Grade 10 Applied (MFM2P) \textperiodcentered{} Unit 1: Algebra \& Solving Equations}
\end{center}

\begin{ideabox}
Simplify before you solve: collect like terms and remove brackets.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Like terms have the same variable and power; combine only their coefficients.
  \item Distributive property: $a(b+c)=ab+ac$ — multiply every term inside.
  \item A negative in front flips every sign inside the brackets.
  \item Expand brackets first, then collect like terms.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Like terms}\textbf{Like terms} share the same variable part, such as $3x$ and $5x$; only like terms combine, by adding their coefficients.
\lh{Distributive property}$a(b+c)=ab+ac$ lets you expand brackets, so $3(x+4)=3x+12$.
\lh{Simplify fully}Expand any brackets first, then collect like terms so the expression is as short as possible.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Collect like terms}
Simplify $4x+7-2x+3$.\soln $(4x-2x)+(7+3)=2x+10$.
\end{example}

\begin{example}{ex1}{Example 2 --- Distribute}
Expand $3(x+4)$.\soln $3x+12$.
\end{example}

\begin{example}{ex2}{Example 3 --- Negative factor}
Expand $-2(2x-5)$.\soln $-4x+10$.
\end{example}

\begin{example}{ex3}{Example 4 --- Expand then collect}
Simplify $2(x+3)+4x$.\soln $2x+6+4x=6x+6$.
\end{example}

\begin{example}{ex4}{Example 5 --- Two brackets}
Simplify $3(x-1)-2(x+4)$.\soln $3x-3-2x-8=x-11$.
\end{example}

\begin{example}{ex5}{Example 6 --- Monomial times binomial}
Expand $x(x+5)$.\soln $x^2+5x$.
\end{example}

\begin{example}{ex6}{Example 7 --- Two variables}
Simplify $5a-3b+2a+b$.\soln $7a-2b$.
\end{example}

\begin{example}{ex7}{Example 8 --- Distribute a negative one}
Expand $-(x-6)$.\soln $-x+6$.
\end{example}

\begin{example}{ex8}{Example 9 --- Putting it together}
Simplify $2(3x-1)-4(x-2)$.\soln $6x-2-4x+8=2x+6$.
\end{example}
