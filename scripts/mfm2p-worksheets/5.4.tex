{\small Grade 10 Applied (MFM2P) \textperiodcentered{} Unit 5: Similar Triangles \& Trigonometry}
\end{center}

\begin{ideabox}
In a right triangle, $a^2+b^2=c^2$ where $c$ is the hypotenuse (opposite the right angle).
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Hypotenuse: $c=\sqrt{a^2+b^2}$.
  \item Leg: $a=\sqrt{c^2-b^2}$.
  \item Only right triangles; $c$ is the longest side.
  \item Common triples: 3-4-5, 5-12-13, 8-15-17.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The right-triangle rule}In a right triangle $a^2+b^2=c^2$, where $c$ is the hypotenuse — the longest side, opposite the right angle.
\lh{Finding a side}Add the squares of the legs for the hypotenuse, or subtract for a leg.
\lh{Only right triangles}The theorem applies only when there is a $90^\circ$ angle.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Find the hypotenuse}
Legs $3$ and $4$. Find $c$.\soln $c=\sqrt{9+16}=\sqrt{25}=5$.
\plot{-1}{6}{-1}{5}{\addplot[exblue,very thick]coordinates{(0,0)(4,0)(4,3)(0,0)};\addplot[mark=*,only marks,black]coordinates{(0,0)(4,0)(4,3)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Bigger triangle}
Legs $6$ and $8$. Find $c$.\soln $\sqrt{36+64}=\sqrt{100}=10$.
\end{example}

\begin{example}{ex2}{Example 3 --- Find a leg}
Hypotenuse $13$, leg $5$. Find the other leg.\soln $\sqrt{169-25}=\sqrt{144}=12$.
\end{example}

\begin{example}{ex3}{Example 4 --- Decimal answer}
Legs $5$ and $7$. Find $c$ (1 dp).\soln $\sqrt{25+49}=\sqrt{74}\approx8.6$.
\end{example}

\begin{example}{ex4}{Example 5 --- Ladder}
A $5$ m ladder's base is $3$ m from a wall. How high does it reach?\soln $\sqrt{25-9}=\sqrt{16}=4$ m.
\end{example}

\begin{example}{ex5}{Example 6 --- Triple}
Legs $8$ and $15$. Find $c$.\soln $\sqrt{64+225}=\sqrt{289}=17$.
\end{example}

\begin{example}{ex6}{Example 7 --- Scaled triple}
Legs $9$ and $12$. Find $c$.\soln $\sqrt{81+144}=\sqrt{225}=15$.
\end{example}

\begin{example}{ex7}{Example 8 --- Find a leg}
Hypotenuse $25$, leg $7$. Find the other leg.\soln $\sqrt{625-49}=\sqrt{576}=24$.
\end{example}

\begin{example}{ex8}{Example 9 --- TV diagonal (harder)}
A TV is $40$ in wide and $30$ in tall. Find the diagonal.\soln $\sqrt{1600+900}=\sqrt{2500}=50$ in.
\end{example}
