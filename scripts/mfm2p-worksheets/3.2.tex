\begin{ideabox}
\textbf{Substitution} solves a system algebraically: isolate one variable, then replace it in the other equation.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item \textbf{Isolate} one variable (easiest when a coefficient is $1$).
  \item \textbf{Substitute} that expression into the \emph{other} equation and solve for one variable.
  \item \textbf{Back-substitute} to find the other variable; write the answer as $(x,y)$.
  \item A true statement like $2=2$ means \textbf{infinitely many} solutions; a false one like $0=3$ means \textbf{none}.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The idea}\textbf{Substitution} solves one equation for a variable, then replaces that variable in the other equation.
\lh{The steps}Isolate a variable, substitute into the second equation, solve, then back-substitute for the other.
\lh{When it fits}Substitution is easiest when a variable already has a coefficient of $1$.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- A variable already isolated}
Solve: $y=2x$ and $x+y=9$.
\soln Substitute $y=2x$: $x+2x=9\Rightarrow 3x=9\Rightarrow x=3$, so $y=6$. Solution $(3,6)$.
\end{example}

\begin{example}{ex1}{Example 2 --- Substitute and simplify}
Solve: $y=x+1$ and $2x+y=7$.
\soln $2x+(x+1)=7\Rightarrow 3x+1=7\Rightarrow x=2$, $y=3$. Solution $(2,3)$.
\end{example}

\begin{example}{ex2}{Example 3 --- Isolating from $x=$}
Solve: $x=y-2$ and $3x+y=10$.
\soln $3(y-2)+y=10\Rightarrow 4y-6=10\Rightarrow y=4$, $x=2$. Solution $(2,4)$.
\end{example}

\begin{example}{ex3}{Example 4 --- Isolate first}
Solve: $x+y=5$ and $2x-y=4$.
\soln From the first, $y=5-x$. Then $2x-(5-x)=4\Rightarrow 3x-5=4\Rightarrow x=3$, $y=2$. Solution $(3,2)$.
\end{example}

\begin{example}{ex4}{Example 5 --- Larger coefficients}
Solve: $2x+y=7$ and $3x-2y=0$.
\soln $y=7-2x$. Then $3x-2(7-2x)=0\Rightarrow 3x-14+4x=0\Rightarrow 7x=14\Rightarrow x=2$, $y=3$. Solution $(2,3)$.
\end{example}

\begin{example}{ex5}{Example 6 --- Infinitely many}
Solve: $y=2x-1$ and $4x-2y=2$.
\soln $4x-2(2x-1)=2\Rightarrow 4x-4x+2=2\Rightarrow 2=2$. Always true $\Rightarrow$ \textbf{infinitely many solutions}.
\end{example}

\begin{example}{ex6}{Example 7 --- No solution}
Solve: $y=3x+2$ and $6x-2y=1$.
\soln $6x-2(3x+2)=1\Rightarrow 6x-6x-4=1\Rightarrow -4=1$. False $\Rightarrow$ \textbf{no solution}.
\end{example}

\begin{example}{ex7}{Example 8 --- Word problem}
Two numbers have a sum of $20$ and a difference of $4$. Find them.
\soln $x+y=20$, $x-y=4$. From the second, $x=y+4$; then $(y+4)+y=20\Rightarrow 2y=16\Rightarrow y=8$, $x=12$. The numbers are $12$ and $8$.
\end{example}

\begin{example}{ex8}{Example 9 --- Multi-step with check (harder)}
Solve: $x-3y=1$ and $2x+y=9$.
\soln $x=1+3y$. Then $2(1+3y)+y=9\Rightarrow 2+7y=9\Rightarrow y=1$, $x=4$. Check: $4-3(1)=1\ \checkmark$, $2(4)+1=9\ \checkmark$. Solution $(4,1)$.
\end{example}

  \item $6y-y=10\Rightarrow y=2$, $x=4$: $(4,2)$.
