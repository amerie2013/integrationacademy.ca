\begin{ideabox}
Slope measures a constant rate of change: $m=\dfrac{\text{rise}}{\text{run}}=\dfrac{y_2-y_1}{x_2-x_1}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Subtract the $y$'s over the $x$'s, in the same order.
  \item Positive rises, negative falls, $0$ is horizontal, vertical is undefined.
  \item From a table with even $x$-steps: $m=\dfrac{\Delta y}{\Delta x}$.
  \item Slope has units in context (e.g. cm/hour).
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Rate of change}A \textbf{rate of change} tells how one quantity changes per unit of another, like cost per item or distance per hour.
\lh{Slope}On a graph, this rate is the \textbf{slope}: rise over run, or $\dfrac{\text{change in }y}{\text{change in }x}$.
\lh{Reading direction}A positive slope rises, a negative slope falls, and a zero slope is flat.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Slope from two points}
Find the slope through $(1,2)$ and $(4,8)$.\soln $m=\dfrac{8-2}{4-1}=2$.
\plot{-1}{6}{-1}{10}{\addplot[exblue,very thick,domain=-0.5:5,samples=2]{2*x};\addplot[mark=*,only marks,black]coordinates{(1,2)(4,8)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Negative slope}
Find the slope through $(-2,5)$ and $(2,-3)$.\soln $m=\dfrac{-3-5}{2-(-2)}=\dfrac{-8}{4}=-2$.
\end{example}

\begin{example}{ex2}{Example 3 --- From a table}
$x:0,1,2,3$, $y:1,4,7,10$. Find the rate of change.\soln Each $\Delta x=1$ gives $\Delta y=3$, so $m=3$.
\end{example}

\begin{example}{ex3}{Example 4 --- Horizontal line}
Find the slope through $(1,4)$ and $(5,4)$.\soln $m=\dfrac{0}{4}=0$ (horizontal).
\end{example}

\begin{example}{ex4}{Example 5 --- A rate in context}
A candle burns from $20$ cm to $8$ cm in $4$ h. Find the rate.\soln $\dfrac{8-20}{4}=-3$ cm/h.
\end{example}

\begin{example}{ex5}{Example 6 --- Through the origin}
Find the slope through $(0,1)$ and $(3,7)$.\soln $m=\dfrac{7-1}{3-0}=2$.
\plot{-1}{5}{-1}{9}{\addplot[exblue,very thick,domain=-0.5:4,samples=2]{2*x+1};\addplot[mark=*,only marks,black]coordinates{(0,1)(3,7)};}
\end{example}

\begin{example}{ex6}{Example 7 --- Vertical line}
Find the slope through $(3,2)$ and $(3,9)$.\soln Run $=0$, so the slope is undefined.
\end{example}

\begin{example}{ex7}{Example 8 --- Filling a pool}
A pool fills from $30$ L to $150$ L in $8$ min. Find the rate.\soln $\dfrac{150-30}{8}=15$ L/min.
\end{example}

\begin{example}{ex8}{Example 9 --- Fractional slope (harder)}
Find the slope through $(-4,3)$ and $(2,0)$.\soln $m=\dfrac{0-3}{2-(-4)}=\dfrac{-3}{6}=-\tfrac12$.
\plot{-5}{4}{-2}{5}{\addplot[exblue,very thick,domain=-5:4,samples=2]{-0.5*x+1};\addplot[mark=*,only marks,black]coordinates{(-4,3)(2,0)};}
\end{example}
