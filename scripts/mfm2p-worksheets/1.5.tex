\begin{ideabox}
Solve a formula for any of its variables using inverse operations.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Treat the wanted variable as the unknown; isolate it.
  \item Do the same operation to both sides.
  \item Undo the outermost operation on the target variable first.
  \item The answer is often a fraction — that's fine.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Solving for a variable}Rearranging a formula means isolating a chosen variable using the same inverse operations as solving an equation.
\lh{Treat others as numbers}The letters you are not solving for behave like constants while you rearrange.
\lh{Why it helps}A rearranged formula lets you compute the variable you want directly, without re-solving each time.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Area of a rectangle}
Solve $A=lw$ for $w$.\soln Divide by $l$: $w=\dfrac{A}{l}$.
\end{example}

\begin{example}{ex1}{Example 2 --- Perimeter}
Solve $P=2(l+w)$ for $w$.\soln $\dfrac{P}{2}=l+w$, so $w=\dfrac{P}{2}-l$.
\end{example}

\begin{example}{ex2}{Example 3 --- Circumference}
Solve $C=2\pi r$ for $r$.\soln $r=\dfrac{C}{2\pi}$.
\end{example}

\begin{example}{ex3}{Example 4 --- Distance}
Solve $d=vt$ for $t$.\soln $t=\dfrac{d}{v}$.
\end{example}

\begin{example}{ex4}{Example 5 --- Line equation}
Solve $y=mx+b$ for $x$.\soln $y-b=mx$, so $x=\dfrac{y-b}{m}$.
\end{example}

\begin{example}{ex5}{Example 6 --- Triangle area}
Solve $A=\tfrac12 bh$ for $h$.\soln Multiply by $2$, divide by $b$: $h=\dfrac{2A}{b}$.
\end{example}

\begin{example}{ex6}{Example 7 --- Square perimeter}
Solve $P=4s$ for $s$.\soln $s=\dfrac{P}{4}$.
\end{example}

\begin{example}{ex7}{Example 8 --- Force}
Solve $F=ma$ for $a$.\soln $a=\dfrac{F}{m}$.
\end{example}

\begin{example}{ex8}{Example 9 --- Motion (harder)}
Solve $v=u+at$ for $t$.\soln $v-u=at$, so $t=\dfrac{v-u}{a}$.
\end{example}
