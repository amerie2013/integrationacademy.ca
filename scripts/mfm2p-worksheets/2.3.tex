\begin{center}
{\LARGE\bfseries\color{titleblue}Worksheet 2.3 --- Slope-Intercept Form (y = mx + b)}\\[2pt]
{\small Grade 10 Applied (MFM2P) \textperiodcentered{} Unit 2: Linear Relations}
\end{center}

\begin{ideabox}
In $y=mx+b$, $m$ is the slope and $b$ is the $y$-intercept — read them straight off.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $m$ = steepness/direction; $b$ = where the line crosses the $y$-axis.
  \item To write a line, place the slope as $m$ and the intercept as $b$.
  \item From standard form, isolate $y$ (divide every term).
  \item In context, $m$ is the rate and $b$ is the fixed/initial value.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Reading the equation}In $y=mx+b$, the slope $m$ is the rate of change and $b$ is the $y$-intercept, where the line crosses the $y$-axis.
\lh{Building it}Read the starting value ($b$) and the step size ($m$) from a table or graph to write the equation.
\lh{A quick model}This form is the most common way to describe a linear relationship at a glance.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Read m and b}
State the slope and $y$-intercept of $y=-3x+5$.\soln $m=-3$, $b=5$.
\end{example}

\begin{example}{ex1}{Example 2 --- Build from words}
Slope $4$, $y$-intercept $-2$. Write the equation.\soln $y=4x-2$.
\end{example}

\begin{example}{ex2}{Example 3 --- From standard form}
Write $2x+y=5$ as $y=mx+b$.\soln $y=-2x+5$ (slope $-2$, $b=5$).
\plot{-2}{5}{-3}{7}{\addplot[exblue,very thick,domain=-0.5:4,samples=2]{-2*x+5};\addplot[mark=*,only marks,black]coordinates{(0,5)};}
\end{example}

\begin{example}{ex3}{Example 4 --- Divide to isolate y}
Write $4x-2y=6$ as $y=mx+b$.\soln $-2y=-4x+6$, so $y=2x-3$.
\end{example}

\begin{example}{ex4}{Example 5 --- Interpret in context}
A plan is $C=0.10t+15$. What do $0.10$ and $15$ mean?\soln $0.10$ = cost per minute; $15$ = fixed monthly fee.
\end{example}

\begin{example}{ex5}{Example 6 --- Negative intercept}
Write the line with slope $\tfrac12$ and $y$-intercept $-4$.\soln $y=\tfrac12x-4$.
\end{example}

\begin{example}{ex6}{Example 7 --- From standard (divide)}
Write $6x-3y=9$ as $y=mx+b$.\soln $-3y=-6x+9$, so $y=2x-3$.
\end{example}

\begin{example}{ex7}{Example 8 --- Find y at a value}
For $y=2x-4$, find $y$ at $x=3$.\soln $y=2(3)-4=2$.
\end{example}

\begin{example}{ex8}{Example 9 --- Read a graph (harder)}
A line crosses $(0,4)$ and falls $1$ for every $2$ across. Write it.\soln $m=-\tfrac12$, $b=4$: $y=-\tfrac12x+4$.
\plot{-2}{8}{-2}{6}{\addplot[exblue,very thick,domain=-1:8,samples=2]{-0.5*x+4};\addplot[mark=*,only marks,black]coordinates{(0,4)(8,0)};}
\end{example}
