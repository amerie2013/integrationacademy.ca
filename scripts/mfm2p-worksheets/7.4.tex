\begin{ideabox}
Factored form $y=a(x-r)(x-s)$ shows the \textbf{zeros} $r$ and $s$ directly.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item The zeros are $x=r$ and $x=s$ (set each factor to $0$).
  \item The AOS is the midpoint $x=\tfrac{r+s}{2}$; substitute to get the vertex.
  \item $a$ controls width/direction; find it from a known point such as the $y$-intercept.
  \item If the quadratic isn't factored yet, factor it first.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Factored form}$y=a(x-r)(x-s)$ shows the \textbf{zeros} $r$ and $s$ — the $x$-intercepts of the parabola.
\lh{Finding the vertex}The axis of symmetry sits halfway between the zeros, at $x=\dfrac{r+s}{2}$.
\lh{Why zeros matter}Zeros answer real questions like when a ball lands or when a quantity is zero.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Read the zeros}
State the zeros of $y=(x-2)(x+3)$.
\soln $x=2$ and $x=-3.$
\end{example}

\begin{example}{ex1}{Example 2 --- Zeros and AOS}
State the zeros and AOS of $y=(x-5)(x-1)$.
\soln Zeros $5,1$; AOS $x=\tfrac{5+1}{2}=3.$
\end{example}

\begin{example}{ex2}{Example 3 --- Find the vertex}
Find the vertex of $y=2(x+1)(x-3)$.
\soln Zeros $-1,3$, AOS $x=1$; $y=2(2)(-2)=-8$. Vertex $(1,-8)$.
\plot{-3}{5}{-9}{6}{\addplot[exblue,very thick,domain=-2:4,samples=60]{2*(x+1)*(x-3)};\addplot[mark=*,only marks,black]coordinates{(-1,0)(3,0)(1,-8)};}
\end{example}

\begin{example}{ex3}{Example 4 --- Build from zeros}
Write a quadratic with zeros $-2$ and $4$ and $a=1$.
\soln $y=(x+2)(x-4).$
\end{example}

\begin{example}{ex4}{Example 5 --- Factor then read zeros}
Find the zeros of $y=x^2-x-6$.
\soln $x^2-x-6=(x-3)(x+2)$, so zeros $3,-2.$
\end{example}

\begin{example}{ex5}{Example 6 --- Vertex via factoring}
Find the vertex of $y=x^2-6x+8$.
\soln $(x-2)(x-4)$, zeros $2,4$, AOS $x=3$, $y=9-18+8=-1$: vertex $(3,-1)$.
\plot{-1}{7}{-2}{6}{\addplot[exblue,very thick,domain=0.3:5.7,samples=60]{x^2-6*x+8};\addplot[mark=*,only marks,black]coordinates{(2,0)(4,0)(3,-1)};}
\end{example}

\begin{example}{ex6}{Example 7 --- Find $a$ from a point}
A parabola has zeros $1$ and $5$ and passes through $(3,-4)$. Find its equation.
\soln $-4=a(3-1)(3-5)=a(2)(-2)=-4a\Rightarrow a=1$, so $y=(x-1)(x-5).$
\end{example}

\begin{example}{ex7}{Example 8 --- Opens down}
Find the vertex of $y=-(x+2)(x-4)$.
\soln Zeros $-2,4$, AOS $x=1$; $y=-(3)(-3)=9$. Vertex $(1,9)$, opens down.
\end{example}

\begin{example}{ex8}{Example 9 --- From intercepts (harder)}
A parabola has $x$-intercepts $-3$ and $5$ and $y$-intercept $-30$. Find its equation.
\soln $y=a(x+3)(x-5)$; at $x=0$: $a(3)(-5)=-15a=-30\Rightarrow a=2$, so $y=2(x+3)(x-5).$
\end{example}
