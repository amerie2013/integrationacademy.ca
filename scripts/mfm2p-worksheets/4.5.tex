\begin{ideabox}
A sphere has volume $V=\tfrac43\pi r^3$ and surface area $SA=4\pi r^2$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Volume: $V=\tfrac43\pi r^3$.
  \item Surface area: $SA=4\pi r^2$.
  \item Hemisphere volume is half a sphere's; given a diameter, halve it for $r$.
  \item Find $r$ from $SA$: divide by $4\pi$, then square-root.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Sphere formulas}A sphere has volume $V=\tfrac43\pi r^3$ and surface area $SA=4\pi r^2$ — both depend only on the radius.
\lh{Half a sphere}A hemisphere is half a sphere, so its volume is $\tfrac23\pi r^3$.
\lh{Everyday spheres}Balls, tanks, and domes are modelled as spheres or hemispheres.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Volume}
$V$ of a sphere $r=3$.\soln $\tfrac43\pi(27)=36\pi\approx113.1$.
\end{example}

\begin{example}{ex1}{Example 2 --- Surface area}
$SA$ of a sphere $r=3$.\soln $4\pi(9)=36\pi\approx113.1$.
\end{example}

\begin{example}{ex2}{Example 3 --- Larger sphere}
$V$ of a sphere $r=6$.\soln $\tfrac43\pi(216)=288\pi\approx904.8$.
\end{example}

\begin{example}{ex3}{Example 4 --- From diameter}
$SA$ of a ball with diameter $10$.\soln $r=5$: $4\pi(25)=100\pi\approx314.2$.
\end{example}

\begin{example}{ex4}{Example 5 --- Hemisphere}
Volume of a hemisphere $r=3$.\soln $\tfrac12(36\pi)=18\pi\approx56.5$.
\end{example}

\begin{example}{ex5}{Example 6 --- Small sphere}
$V$ of a sphere $r=2$.\soln $\tfrac43\pi(8)=\tfrac{32}{3}\pi\approx33.5$.
\end{example}

\begin{example}{ex6}{Example 7 --- Surface area}
$SA$ of a sphere $r=5$.\soln $4\pi(25)=100\pi\approx314.2$.
\end{example}

\begin{example}{ex7}{Example 8 --- From diameter}
$SA$ of a sphere with diameter $8$.\soln $r=4$: $64\pi\approx201.1$.
\end{example}

\begin{example}{ex8}{Example 9 --- Find the radius (harder)}
A sphere has $SA=36\pi$. Find $r$.\soln $4\pi r^2=36\pi\Rightarrow r^2=9\Rightarrow r=3$.
\end{example}
