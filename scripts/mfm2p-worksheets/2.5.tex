{\small Grade 10 Applied (MFM2P) \textperiodcentered{} Unit 2: Linear Relations}
\end{center}

\begin{ideabox}
From a slope and a point, two points, or a graph, write $y=mx+b$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item From slope + point: substitute into $y=mx+b$ to find $b$.
  \item From two points: find the slope first, then $b$.
  \item If the point is on the $y$-axis, its $y$ is $b$.
  \item Check: both given points should satisfy your equation.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{From a graph or table}Read the $y$-intercept where the line crosses the axis and the slope between two clear points, then write $y=mx+b$.
\lh{From slope and a point}Substitute the slope and a known point into $y=mx+b$ and solve for $b$.
\lh{From two points}Find the slope with rise over run first, then use either point to find $b$.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Slope + intercept point}
Slope $4$ through $(0,-2)$.\soln $b=-2$: $y=4x-2$.
\end{example}

\begin{example}{ex1}{Example 2 --- Slope + point}
Slope $-2$ through $(3,1)$.\soln $1=-2(3)+b\Rightarrow b=7$: $y=-2x+7$.
\plot{-1}{6}{-3}{8}{\addplot[exblue,very thick,domain=-0.5:5.5,samples=2]{-2*x+7};\addplot[mark=*,only marks,black]coordinates{(3,1)};}
\end{example}

\begin{example}{ex2}{Example 3 --- Two points}
Through $(1,2)$ and $(3,8)$.\soln $m=3$; $2=3+b\Rightarrow b=-1$: $y=3x-1$.
\end{example}

\begin{example}{ex3}{Example 4 --- From a graph}
A line crosses $(0,4)$ and falls $1$ per $2$ across.\soln $m=-\tfrac12$, $b=4$: $y=-\tfrac12x+4$.
\end{example}

\begin{example}{ex4}{Example 5 --- Context model}
A candle is $20$ cm and burns $3$ cm/h. Write $h$ vs $t$.\soln $h=-3t+20$.
\end{example}

\begin{example}{ex5}{Example 6 --- Two points (negative slope)}
Through $(-1,2)$ and $(2,-7)$.\soln $m=\dfrac{-7-2}{2-(-1)}=-3$; $2=-3(-1)+b\Rightarrow b=-1$: $y=-3x-1$.
\end{example}

\begin{example}{ex6}{Example 7 --- Horizontal}
Through $(1,4)$ and $(3,4)$.\soln Slope $0$: $y=4$.
\end{example}

\begin{example}{ex7}{Example 8 --- Slope + point (fraction)}
Slope $\tfrac23$ through $(3,0)$.\soln $0=\tfrac23(3)+b\Rightarrow b=-2$: $y=\tfrac23x-2$.
\end{example}

\begin{example}{ex8}{Example 9 --- Two points (harder)}
Through $(2,5)$ and $(6,13)$.\soln $m=2$; $5=2(2)+b\Rightarrow b=1$: $y=2x+1$.
\plot{-1}{7}{-1}{14}{\addplot[exblue,very thick,domain=-0.5:6.3,samples=2]{2*x+1};\addplot[mark=*,only marks,black]coordinates{(2,5)(6,13)};}
\end{example}
