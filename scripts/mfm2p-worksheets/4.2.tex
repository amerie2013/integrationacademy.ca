{\LARGE\bfseries\color{titleblue}Worksheet 4.2 --- Surface Area of Prisms \& Cylinders}\\[2pt]
{\small Grade 10 Applied (MFM2P) \textperiodcentered{} Unit 4: Measurement}
\end{center}

\begin{ideabox}
Surface area is the total area of all faces.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Rectangular prism: $SA=2(lw+lh+wh)$.
  \item Cylinder: $SA=2\pi r^2+2\pi rh$ (two circles + the wrap).
  \item Triangular prism: $SA=(\text{perimeter})\times\text{length}+2\times\text{triangle area}$.
  \item Use the radius (not the diameter); area units are squared.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Add up the faces}\textbf{Surface area} totals all the outside faces; a box has three matching pairs, $SA=2(lw+lh+wh)$.
\lh{Cylinders}A cylinder is two circles plus a wrapped rectangle: $SA=2\pi r^2+2\pi rh$.
\lh{Square units}Surface area is always measured in square units.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Rectangular prism}
$SA$ of a $4\times3\times2$ box.\soln $2(12+8+6)=52$ units$^2$.
\end{example}

\begin{example}{ex1}{Example 2 --- Cube}
$SA$ of a cube, side $5$.\soln $6\times25=150$ units$^2$.
\end{example}

\begin{example}{ex2}{Example 3 --- Cylinder}
$SA$ of a cylinder $r=3,\ h=10$.\soln $2\pi(9)+2\pi(3)(10)=78\pi\approx245.0$.
\end{example}

\begin{example}{ex3}{Example 4 --- Triangular prism}
Triangle legs $3,4$ (hyp $5$), length $10$.\soln Lateral $(3+4+5)(10)=120$; ends $2(\tfrac12\cdot3\cdot4)=12$. Total $132$.
\end{example}

\begin{example}{ex4}{Example 5 --- Open-top box}
$SA$ of a $4\times4\times3$ box with no top.\soln Base $16$ + $4$ sides $4(12)=48$. Total $64$.
\end{example}

\begin{example}{ex5}{Example 6 --- Smaller cube}
$SA$ of a cube, side $4$.\soln $6\times16=96$ units$^2$.
\end{example}

\begin{example}{ex6}{Example 7 --- Cylinder (exact)}
$SA$ of a cylinder $r=2,\ h=5$.\soln $8\pi+20\pi=28\pi\approx88.0$.
\end{example}

\begin{example}{ex7}{Example 8 --- Lateral area}
Lateral area of a cylinder $r=3,\ h=7$.\soln $2\pi(3)(7)=42\pi$.
\end{example}

\begin{example}{ex8}{Example 9 --- Triangular prism (harder)}
Triangle base $6$, height $4$ (equal sides $5$), length $12$.\soln Lateral $(6+5+5)(12)=192$; ends $2(\tfrac12\cdot6\cdot4)=24$. Total $216$.
\end{example}
