\begin{ideabox}
\textbf{Expanding} means removing brackets using the distributive property, then collecting like terms.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Distribute: $a(b+c)=ab+ac$.
  \item Multiply two binomials with \textbf{FOIL} (First, Outer, Inner, Last).
  \item Multiply every term in one bracket by every term in the other.
  \item Finish by \textbf{collecting like terms}.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Distribute everything}To expand, multiply every term of one factor by every term of the other, then collect like terms.
\lh{Two binomials}FOIL — First, Outer, Inner, Last — is the shortcut for two binomials.
\lh{Simplify}Combine like terms so the product is written cleanly.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Distribute a constant}
Expand $3(x+4)$.
\soln $3(x+4)=3x+12.$
\end{example}

\begin{example}{ex1}{Example 2 --- Negative factor}
Expand $-2(3x-5)$.
\soln $-2(3x-5)=-6x+10.$
\end{example}

\begin{example}{ex2}{Example 3 --- Distribute a monomial}
Expand $x(2x+3)$.
\soln $x(2x+3)=2x^2+3x.$
\end{example}

\begin{example}{ex3}{Example 4 --- FOIL}
Expand $(x+2)(x+5)$.
\soln $x^2+5x+2x+10=x^2+7x+10.$
\end{example}

\begin{example}{ex4}{Example 5 --- Mixed signs}
Expand $(x-3)(x+4)$.
\soln $x^2+4x-3x-12=x^2+x-12.$
\end{example}

\begin{example}{ex5}{Example 6 --- Two leading coefficients}
Expand $(2x+1)(3x-2)$.
\soln $6x^2-4x+3x-2=6x^2-x-2.$
\end{example}

\begin{example}{ex6}{Example 7 --- A perfect square}
Expand $(x+3)^2$.
\soln $(x+3)(x+3)=x^2+3x+3x+9=x^2+6x+9.$
\end{example}

\begin{example}{ex7}{Example 8 --- With a common factor}
Expand and simplify $2(x+1)(x-4)$.
\soln $2(x^2-3x-4)=2x^2-6x-8.$
\end{example}

\begin{example}{ex8}{Example 9 --- Binomial times trinomial (harder)}
Expand $(2x-3)(x^2+x-4)$.
\soln $2x^3+2x^2-8x-3x^2-3x+12=2x^3-x^2-11x+12.$
\end{example}
