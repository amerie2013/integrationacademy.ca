\begin{ideabox}
Two patterns appear constantly: the \textbf{perfect square} $(a\pm b)^2=a^2\pm2ab+b^2$ and the \textbf{difference of squares} $a^2-b^2=(a-b)(a+b)$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $(a+b)^2=a^2+2ab+b^2$ and $(a-b)^2=a^2-2ab+b^2$ --- don't forget the middle term.
  \item $(a-b)(a+b)=a^2-b^2$ --- the middle terms cancel.
  \item To factor a difference of squares, take the square root of each term.
  \item Recognise perfect-square trinomials: first and last terms are squares and the middle is $2ab$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Handy patterns}Two products recur: the perfect square $(a\pm b)^2=a^2\pm2ab+b^2$ and the difference of squares $(a+b)(a-b)=a^2-b^2$.
\lh{Why memorize them}Recognizing them speeds up both expanding and factoring.
\lh{A common trap}$(a+b)^2\ne a^2+b^2$: the middle term $2ab$ must not be dropped.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Square of a sum}
Expand $(x+5)^2$.
\soln $x^2+2(5)x+25=x^2+10x+25.$
\end{example}

\begin{example}{ex1}{Example 2 --- Square of a difference}
Expand $(x-4)^2$.
\soln $x^2-8x+16.$
\end{example}

\begin{example}{ex2}{Example 3 --- Square with a coefficient}
Expand $(2x+3)^2$.
\soln $(2x)^2+2(2x)(3)+9=4x^2+12x+9.$
\end{example}

\begin{example}{ex3}{Example 4 --- Difference of squares (expand)}
Expand $(x-7)(x+7)$.
\soln $x^2-49.$
\end{example}

\begin{example}{ex4}{Example 5 --- With a coefficient}
Expand $(3x-2)(3x+2)$.
\soln $9x^2-4.$
\end{example}

\begin{example}{ex5}{Example 6 --- Factor a difference of squares}
Factor $x^2-16$.
\soln $(x-4)(x+4).$
\end{example}

\begin{example}{ex6}{Example 7 --- Factor with a coefficient}
Factor $9x^2-25$.
\soln $(3x-5)(3x+5).$
\end{example}

\begin{example}{ex7}{Example 8 --- Factor a perfect square}
Factor $x^2+10x+25$.
\soln $(x+5)^2.$
\end{example}

\begin{example}{ex8}{Example 9 --- Combine the tools (harder)}
Factor (a) $50x^2-8$ and (b) $16x^2-24x+9$.
\soln (a) $2(25x^2-4)=2(5x-2)(5x+2)$. (b) $(4x-3)^2.$
\end{example}
