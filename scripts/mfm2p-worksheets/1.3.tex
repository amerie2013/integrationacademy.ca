{\LARGE\bfseries\color{titleblue}Worksheet 1.3 --- Solving One- \& Two-Step Equations}\\[2pt]
{\small Grade 10 Applied (MFM2P) \textperiodcentered{} Unit 1: Algebra \& Solving Equations}
\end{center}

\begin{ideabox}
Solve an equation by isolating the variable with inverse operations.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Undo $+$ with $-$, and $\times$ with $\div$ — do the same to both sides.
  \item Two-step: undo addition/subtraction first, then multiplication/division.
  \item Check by substituting your answer back into the original equation.
  \item If the variable is negative ($-x=k$), multiply both sides by $-1$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Keep it balanced}An equation is a balance: whatever you do to one side, do to the other to keep them equal.
\lh{Undo in reverse}Isolate the variable with inverse operations — subtract to undo addition, divide to undo multiplication.
\lh{Check your answer}Substitute your solution into the original equation; both sides should match.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- One step (add)}
Solve $x+5=12$.\soln Subtract $5$: $x=7$.
\end{example}

\begin{example}{ex1}{Example 2 --- One step (multiply)}
Solve $3x=21$.\soln Divide by $3$: $x=7$.
\end{example}

\begin{example}{ex2}{Example 3 --- Subtraction}
Solve $x-4=-2$.\soln Add $4$: $x=2$.
\end{example}

\begin{example}{ex3}{Example 4 --- Two steps}
Solve $2x+1=9$.\soln $2x=8$, then $x=4$.
\end{example}

\begin{example}{ex4}{Example 5 --- Division}
Solve $\dfrac{x}{4}=3$.\soln Multiply by $4$: $x=12$.
\end{example}

\begin{example}{ex5}{Example 6 --- Negative variable}
Solve $5-x=2$.\soln $-x=-3$, so $x=3$.
\end{example}

\begin{example}{ex6}{Example 7 --- Two steps again}
Solve $3x-2=10$.\soln $3x=12$, then $x=4$.
\end{example}

\begin{example}{ex7}{Example 8 --- Two-step with fraction}
Solve $\dfrac{x}{2}+1=5$.\soln $\dfrac{x}{2}=4$, then $x=8$.
\end{example}

\begin{example}{ex8}{Example 9 --- Negative coefficient}
Solve $7-2x=1$.\soln $-2x=-6$, so $x=3$.
\end{example}
