\begin{ideabox}
$P(A\mid B)=\dfrac{P(A\cap B)}{P(B)}$ restricts to the outcomes where $B$ happened. The product rule $P(A\cap B)=P(B)\,P(A\mid B)$ and two-way tables make it concrete.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $P(A\mid B)=\dfrac{P(A\cap B)}{P(B)}$ --- a reduced sample space.
  \item Product rule: $P(A\cap B)=P(B)P(A\mid B)$.
  \item Read conditions directly off a two-way table.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Probability given information}$P(A\mid B)$ is the probability of $A$ \emph{given} that $B$ has already happened. Knowing $B$ shrinks the sample space to the outcomes inside $B$.
\lh{The formula}$P(A\mid B)=\dfrac{P(A\cap B)}{P(B)}$. Rearranged, this is the \textbf{product rule} $P(A\cap B)=P(B)\,P(A\mid B)$.
\lh{Reading a two-way table}In a two-way table a conditional probability uses just one row or column as the new total — the condition tells you which one.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Reduced sample space (cards)}
Find $P(\text{king}\mid\text{face card})$.\soln $\dfrac{4/52}{12/52}=\dfrac{4}{12}=\dfrac13$.
\end{example}

\begin{example}{ex1}{Example 2 --- Two-way table}
Of 100 people, 60 are female and 36 of them own a pet. Find $P(\text{owns pet}\mid\text{female})$.\soln $\dfrac{36}{60}=0.6$.
\end{example}

\begin{example}{ex2}{Example 3 --- Dependent second draw}
Drawing without replacement, find $P(\text{2nd is an ace}\mid\text{1st was an ace})$.\soln $\dfrac{3}{51}=\dfrac{1}{17}$.
\end{example}

\begin{example}{ex3}{Example 4 --- Product rule}
Find $P(\text{two kings in a row})$ without replacement.\soln $P(K_1)P(K_2\mid K_1)=\dfrac{4}{52}\cdot\dfrac{3}{51}=\dfrac{1}{221}$.
\end{example}

\begin{example}{ex4}{Example 5 --- Condition on one die}
Two dice. Find $P(\text{sum}=8\mid\text{first die}=5)$.\soln Given first $=5$, need second $=3$: $\dfrac16$.
\end{example}

\begin{example}{ex5}{Example 6 --- Reduced space with counting}
Two dice. Find $P(\text{sum}\ge10\mid\text{at least one 6})$.\soln The event \emph{at least one 6} has 11 outcomes; those with sum $\ge10$ are $(6,4),(4,6),(6,5),(5,6),(6,6)$: $\dfrac{5}{11}$.
\end{example}

\begin{example}{ex6}{Example 7 --- Natural frequencies (Bayes-lite)}
1\% of people have a condition. A test is 90\% sensitive with a 5\% false-positive rate. Estimate $P(\text{condition}\mid\text{positive})$.\soln Per 1000: $10$ sick $\to9$ positive; $990$ healthy $\to\approx50$ positive. $P\approx\dfrac{9}{9+50}\approx0.15$.
\end{example}

\begin{example}{ex7}{Example 8 --- Reverse the condition}
In a survey, 60 people own a pet and 36 of them are female. Find $P(\text{female}\mid\text{owns pet})$.\soln $\dfrac{36}{60}=0.6$.
\end{example}

\begin{example}{ex8}{Example 9 --- The two-child subtlety}
A family has two children (each B/G equally likely). Find \\ \textbf{(a)} $P(\text{both girls}\mid\text{at least one girl})$; \\ \textbf{(b)} $P(\text{both girls}\mid\text{elder is a girl})$.\soln (a) $\{BG,GB,GG\}$: $\dfrac13$. (b) $\{GB,GG\}$: $\dfrac12$.
\end{example}
