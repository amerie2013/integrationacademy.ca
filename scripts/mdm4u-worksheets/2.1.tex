\begin{ideabox}
For equally likely outcomes, $P(A)=\dfrac{n(A)}{n(S)}$, a number in $[0,1]$. \emph{Theoretical} probability is counted, \emph{experimental} is measured from trials, and \emph{subjective} is judged.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Sample space $S$ lists all outcomes; an event $A$ is a subset.
  \item $P(A)=\dfrac{n(A)}{n(S)}$ (equally likely); complement $P(A')=1-P(A)$.
  \item Theoretical vs. experimental vs. subjective probability; $0\le P\le1$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Defining probability}For equally likely outcomes, $P(A)=\dfrac{\text{favourable}}{\text{total}}$, a number between $0$ (impossible) and $1$ (certain). Always list the \textbf{sample space} of outcomes first.
\lh{Three kinds}\textbf{Theoretical} probability comes from counting, \textbf{experimental} from repeated trials, and \textbf{subjective} from informed judgement. Experimental results drift toward the theoretical value as trials pile up.
\lh{A worked idea}Two dice have $36$ equally likely outcomes; six of them sum to $7$, so $P(\text{sum}=7)=\dfrac{6}{36}=\dfrac16$.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Sample space of two dice}
Two dice are rolled. Find $P(\text{sum}=7)$.\soln $n(S)=36$. Sums of 7: $(1,6),(2,5),(3,4),(4,3),(5,2),(6,1)$, so $n(A)=6$. $P=\dfrac{6}{36}=\dfrac16$.
\end{example}

\begin{example}{ex1}{Example 2 --- An event on one die}
A die is rolled. Find $P(\text{prime})$.\soln Primes are $\{2,3,5\}$: $P=\dfrac36=\dfrac12$.
\end{example}

\begin{example}{ex2}{Example 3 --- A card event}
One card is drawn from a standard deck. Find $P(\text{face card})$.\soln $12$ face cards: $P=\dfrac{12}{52}=\dfrac{3}{13}$.
\end{example}

\begin{example}{ex3}{Example 4 --- Experimental vs. theoretical}
A die is rolled 200 times and shows a 3 on 44 rolls. Compare the experimental and theoretical probabilities.\soln Experimental $=\dfrac{44}{200}=0.22$; theoretical $=\dfrac16\approx0.167$. Close, as expected for many trials.
\end{example}

\begin{example}{ex4}{Example 5 --- Complement}
Find $P(\text{not a heart})$ for one card.\soln $1-\dfrac{13}{52}=\dfrac{39}{52}=\dfrac34$.
\end{example}

\begin{example}{ex5}{Example 6 --- Two coins}
Two coins are tossed. Find $P(\text{exactly one head})$.\soln $S=\{HH,HT,TH,TT\}$; favourable $\{HT,TH\}$: $P=\dfrac24=\dfrac12$.
\end{example}

\begin{example}{ex6}{Example 7 --- Unequal spinner}
A spinner has 8 equal wedges: 4 red, 3 blue, 1 green. Find $P(\text{blue})$.\soln $P=\dfrac38$.
\end{example}

\begin{example}{ex7}{Example 8 --- Classify the probability}
Classify each as theoretical, experimental, or subjective: \\ \textbf{(a)} $P(\text{ace})=\tfrac{1}{13}$; \\ \textbf{(b)} 63 of 150 tacks land point-up; \\ \textbf{(c)} a forecaster's 70\% chance of rain.\soln (a) theoretical; (b) experimental; (c) subjective.
\end{example}

\begin{example}{ex8}{Example 9 --- Valid probabilities}
Which are valid probabilities: $-0.2,\ 0.5,\ 1.3,\ \tfrac34$?\soln Only $0.5$ and $\tfrac34$: a probability must lie in $[0,1]$.
\end{example}
