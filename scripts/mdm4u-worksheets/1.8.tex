{\LARGE\bfseries\color{titleblue}Worksheet 1.8 --- The Binomial Theorem}\\[2pt]
{\small Grade 12 Mathematics of Data Management (MDM4U) \textperiodcentered{} Unit 1: Counting, Permutations \& Combinations}
\end{center}

\begin{ideabox}
$(a+b)^n=\displaystyle\sum_{k=0}^{n}\dbinom{n}{k}a^{n-k}b^{k}$. The general term is $T_{k+1}=\dbinom{n}{k}a^{n-k}b^{k}$ --- set the exponent of $x$ to pull out any single term.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $(a+b)^n=\sum_{k=0}^{n}\dbinom{n}{k}a^{n-k}b^{k}$; the expansion has $n+1$ terms.
  \item General term $T_{k+1}=\dbinom{n}{k}a^{n-k}b^{k}$.
  \item For a specific power (or the constant term), solve the exponent equation for $k$; set $x=1$ for the sum of coefficients.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Expanding a power}$(a+b)^n=\displaystyle\sum_{r=0}^{n}\dbinom{n}{r}a^{n-r}b^{r}$. The coefficients are exactly row $n$ of Pascal's triangle, and the exponents of $a$ and $b$ always sum to $n$.
\lh{The general term}The term containing $b^{r}$ is $\dbinom{n}{r}a^{n-r}b^{r}$. This lets you pick out a single term — such as the coefficient of $x^3$ — without expanding everything.
\lh{Reading the pattern}As $r$ climbs, the power of $a$ falls while the power of $b$ rises. There are $n+1$ terms in all, with symmetric coefficients.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Full expansion}
Expand $(x+2)^4$.\soln $x^4+8x^3+24x^2+32x+16$.
\end{example}

\begin{example}{ex1}{Example 2 --- Specific term}
Find the term in $x^2$ of $(x+3)^5$.\soln $T=\dbinom{5}{k}x^{5-k}3^{k}$; $5-k=2\Rightarrow k=3$: $\dbinom{5}{3}3^3x^2=270x^2$.
\end{example}

\begin{example}{ex2}{Example 3 --- Coefficient with a factor}
Coefficient of $x^3$ in $(2x-1)^5$.\soln $5-k=3\Rightarrow k=2$: $\dbinom{5}{2}(2)^3(-1)^2=80$.
\end{example}

\begin{example}{ex3}{Example 4 --- Constant term}
Constant term of $\left(x+\dfrac1x\right)^6$.\soln $\dbinom{6}{k}x^{6-2k}$; $6-2k=0\Rightarrow k=3$: $\dbinom{6}{3}=20$.
\end{example}

\begin{example}{ex4}{Example 5 --- Middle term}
Middle term of $(a+b)^8$.\soln $k=4$: $\dbinom{8}{4}a^4b^4=70a^4b^4$.
\end{example}

\begin{example}{ex5}{Example 6 --- Sum of coefficients}
Sum of the coefficients of $(2x+3)^4$.\soln Set $x=1$: $5^4=625$.
\end{example}

\begin{example}{ex6}{Example 7 --- Negative sign}
Expand $(1-x)^3$.\soln $1-3x+3x^2-x^3$.
\end{example}

\begin{example}{ex7}{Example 8 --- Power inside}
Coefficient of $x^4$ in $(x^2+2)^5$.\soln $\dbinom{5}{k}2^{k}x^{10-2k}$; $10-2k=4\Rightarrow k=3$: $\dbinom{5}{3}2^3=80$.
\end{example}

\begin{example}{ex8}{Example 9 --- Mixed powers}
Term containing $x^5$ in $\left(x-\dfrac2x\right)^7$.\soln $\dbinom{7}{k}(-2)^{k}x^{7-2k}$; $7-2k=5\Rightarrow k=1$: $\dbinom{7}{1}(-2)x^5=-14x^5$.
\end{example}
