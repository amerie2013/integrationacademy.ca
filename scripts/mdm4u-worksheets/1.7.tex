{\LARGE\bfseries\color{titleblue}Worksheet 1.7 --- Pascal's Triangle \& Combinatorial Identities}\\[2pt]
{\small Grade 12 Mathematics of Data Management (MDM4U) \textperiodcentered{} Unit 1: Counting, Permutations \& Combinations}
\end{center}

\begin{ideabox}
Each entry of Pascal's triangle is $\dbinom{n}{r}$. Pascal's identity $\dbinom{n}{r}=\dbinom{n-1}{r-1}+\dbinom{n-1}{r}$ builds each from the two above; row $n$ sums to $2^n$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Entry in row $n$, position $r$ (from 0) is $\dbinom{n}{r}$.
  \item Pascal's identity: $\dbinom{n}{r}=\dbinom{n-1}{r-1}+\dbinom{n-1}{r}$.
  \item Row sum $=2^n$; alternating row sum $=0$; hockey-stick along a diagonal.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Building the triangle}Each entry is the sum of the two directly above it. Row $n$ (counting from row 0) lists $\dbinom{n}{0},\dbinom{n}{1},\dots,\dbinom{n}{n}$ — the combination counts themselves.
\lh{Pascal's identity}$\dbinom{n}{r}=\dbinom{n-1}{r-1}+\dbinom{n-1}{r}$: choosing $r$ from $n$ either includes a fixed item (choose $r-1$ from the rest) or excludes it (choose $r$ from the rest).
\lh{Row sums are powers of 2}The entries in row $n$ add to $2^n$, because that is the number of subsets of an $n$-element set — a bridge to the binomial theorem.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Write a row}
Write row 5 of Pascal's triangle.\soln $1,\,5,\,10,\,10,\,5,\,1$ --- these are $\dbinom{5}{0}\dots\dbinom{5}{5}$.
\end{example}

\begin{example}{ex1}{Example 2 --- Pascal's identity}
Verify $\dbinom{7}{3}=\dbinom{6}{2}+\dbinom{6}{3}$.\soln $15+20=35=\dbinom{7}{3}$.
\end{example}

\begin{example}{ex2}{Example 3 --- Row sum}
What is the sum of the entries in row 6?\soln $2^6=64$ --- the number of subsets of a 6-set.
\end{example}

\begin{example}{ex3}{Example 4 --- Build a large entry}
Use Pascal's identity for $\dbinom{10}{4}$.\soln $\dbinom{9}{3}+\dbinom{9}{4}=84+126=210$.
\end{example}

\begin{example}{ex4}{Example 5 --- Hockey-stick (diagonal sum)}
Evaluate $\dbinom{2}{2}+\dbinom{3}{2}+\dbinom{4}{2}$.\soln $1+3+6=10=\dbinom{5}{3}$.
\end{example}

\begin{example}{ex5}{Example 6 --- Symmetry}
Simplify $\dbinom{9}{7}$.\soln $\dbinom{9}{7}=\dbinom{9}{2}=36$.
\end{example}

\begin{example}{ex6}{Example 7 --- Alternating sum}
Evaluate $\dbinom{4}{0}-\dbinom{4}{1}+\dbinom{4}{2}-\dbinom{4}{3}+\dbinom{4}{4}$.\soln $1-4+6-4+1=0$.
\end{example}

\begin{example}{ex7}{Example 8 --- Paths as Pascal entries}
How many shortest grid routes reach the cell 3 right and 2 up?\soln $\dbinom{5}{2}=10$ --- exactly a Pascal entry.
\end{example}

\begin{example}{ex8}{Example 9 --- Sum of squares (Vandermonde)}
Evaluate $\dbinom{4}{0}^2+\dbinom{4}{1}^2+\dbinom{4}{2}^2+\dbinom{4}{3}^2+\dbinom{4}{4}^2$.\soln $1+16+36+16+1=70=\dbinom{8}{4}$.
\end{example}
