{\LARGE\bfseries\color{titleblue}Worksheet 4.3 --- Normal Probabilities \& Percentiles}\\[2pt]
{\small Grade 12 Mathematics of Data Management (MDM4U) \textperiodcentered{} Unit 4: Continuous \& Normal Distributions}

\begin{ideabox}
Standardize, then read the table value $\Phi(z)=P(Z<z)$. Then $P(a<X<b)=\Phi(z_b)-\Phi(z_a)$, and percentiles invert the table: $x=\mu+z\sigma$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $P(X<x)=\Phi(z)$ after standardizing.
  \item $P(a<X<b)=\Phi(z_b)-\Phi(z_a)$; upper tail $=1-\Phi(z)$.
  \item Percentile $\to$ $z$ $\to$ $x=\mu+z\sigma$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Finding a probability}Convert the value to a z-score, then use a z-table or technology to read the area to its left, which is $P(X<x)$.
\lh{Between and above}For $P(X>x)$ subtract the left area from $1$; for $P(a<X<b)$ subtract the two left areas. A quick sketch of the shaded region prevents errors.
\lh{Percentiles run backwards}For a percentile, start from the area, look up the matching z-score, then reverse the formula: $x=\mu+z\sigma$.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Left area}
$X\sim N(100,15)$. Find $P(X<120)$.\soln $z=\tfrac{120-100}{15}=1.33$; $\Phi(1.33)=0.9082$.\ncurve{-3.8}{1.33}{}
\end{example}

\begin{example}{ex1}{Example 2 --- Upper tail}
$X\sim N(100,15)$. Find $P(X>85)$.\soln $z=-1$; $P=1-\Phi(-1)=\Phi(1)=0.8413$.
\end{example}

\begin{example}{ex2}{Example 3 --- Between two values}
$X\sim N(50,5)$. Find $P(45<X<55)$.\soln $\Phi(1)-\Phi(-1)=0.8413-0.1587=0.6826$.\ncurve{-1}{1}{}
\end{example}

\begin{example}{ex3}{Example 4 --- Percentile to value}
Find the $90$th percentile of $N(500,100)$.\soln $z=1.28$; $x=500+128=628$.
\end{example}

\begin{example}{ex4}{Example 5 --- Target probability to value}
$X\sim N(70,10)$. Find $x$ with $P(X>x)=0.05$.\soln $z=1.645$; $x=70+16.45=86.45$.
\end{example}

\begin{example}{ex5}{Example 6 --- Middle 95\% interval}
Find the central $95\%$ interval of $N(100,15)$.\soln $z=\pm1.96$: $100\pm29.4$, i.e. $70.6$ to $129.4$.
\end{example}

\begin{example}{ex6}{Example 7 --- Two-tailed area}
Find $P(|Z|>2)$.\soln $2\bigl(1-\Phi(2)\bigr)=2(0.0228)=0.0456$.
\end{example}

\begin{example}{ex7}{Example 8 --- Find sigma from a probability}
$X\sim N(200,\sigma)$ with $P(X<230)=0.84$. Find $\sigma$.\soln $\Phi(1)=0.84$, so $230$ is $+1\sigma$: $\sigma=30$.
\end{example}

\begin{example}{ex8}{Example 9 --- Exam scores (three parts)}
Scores are $N(72,12)$. Find \\ \textbf{(a)} $P(X>90)$; \\ \textbf{(b)} $P(60<X<84)$; \\ \textbf{(c)} the $95$th percentile.\soln (a) $z=1.5$, $1-\Phi(1.5)=0.0668$. (b) $\Phi(1)-\Phi(-1)=0.6826$. (c) $z=1.645$, $72+19.74=91.74$.
\end{example}
