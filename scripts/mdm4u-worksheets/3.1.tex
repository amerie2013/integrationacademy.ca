{\small Grade 12 Mathematics of Data Management (MDM4U) \textperiodcentered{} Unit 3: Discrete Probability Distributions}

\begin{ideabox}
A random variable assigns a number to each outcome. Its \emph{distribution} lists every value with its probability, and those probabilities must sum to $1$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Discrete random variable $X$ takes separate values.
  \item A distribution is valid when every $P(x)\ge0$ and $\sum P(x)=1$.
  \item Cumulative events: $P(X\le k)$, $P(X\ge k)$ add the relevant rows.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{What a random variable is}A \textbf{random variable} $X$ attaches a number to each outcome (like the number of heads in three flips). \emph{Discrete} means it takes separate, countable values.
\lh{Its distribution}A probability distribution pairs each value of $X$ with its probability. Because some outcome must occur, those probabilities must \textbf{sum to 1}.
\lh{Building one}Count the outcomes for each value of $X$ and divide by the total. If your probabilities do not add to $1$, an outcome is missing.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Build a distribution}
Two coins are tossed; $X=$ number of heads. Give the distribution.\soln $X\in\{0,1,2\}$ with $P=\tfrac14,\tfrac12,\tfrac14$; sum $=1$. \checkmark
\end{example}

\begin{example}{ex1}{Example 2 --- Distribution of a dice sum}
Two dice; $X=$ the sum. Find $P(X=5)$ and describe the shape.\soln $P(5)=\dfrac{4}{36}=\dfrac19$; the distribution rises to a peak at $7$, then falls (triangular).
\end{example}

\begin{example}{ex2}{Example 3 --- Find a missing probability}
A table gives $P(0)=0.1,\ P(1)=0.3,\ P(3)=0.2$. Find $P(2)$.\soln $\sum=1\Rightarrow P(2)=1-(0.1+0.3+0.2)=0.4$.
\end{example}

\begin{example}{ex3}{Example 4 --- Cumulative probabilities}
For $P(0)=0.1,\ P(1)=0.4,\ P(2)=0.3,\ P(3)=0.2$, find $P(X\ge2)$.\soln $0.3+0.2=0.5$.
\end{example}

\begin{example}{ex4}{Example 5 --- Solve for a constant}
$P(x)=kx$ for $x=1,2,3,4$. Find $k$.\soln $k(1+2+3+4)=1\Rightarrow 10k=1\Rightarrow k=\tfrac{1}{10}$.
\end{example}

\begin{example}{ex5}{Example 6 --- An indicator variable}
$X=1$ if a drawn card is a heart, else $0$. Give the distribution.\soln $P(1)=\tfrac14,\ P(0)=\tfrac34$.
\end{example}

\begin{example}{ex6}{Example 7 --- Distribution from a draw}
From 3 red and 2 blue, draw 2 without replacement; $X=$ number of red. Give the distribution.\soln $P(0)=\dfrac{\binom{2}{2}}{\binom{5}{2}}=\tfrac{1}{10}$, $P(1)=\dfrac{\binom{3}{1}\binom{2}{1}}{10}=\tfrac{6}{10}$, $P(2)=\dfrac{\binom{3}{2}}{10}=\tfrac{3}{10}$; sum $=1$. \checkmark
\end{example}

\begin{example}{ex7}{Example 8 --- Read a histogram}
A probability histogram has bars $P(1)=0.2,\ P(2)=0.5,\ P(3)=0.3$. Find $P(X\le2)$.\soln $0.2+0.5=0.7$.
\end{example}

\begin{example}{ex8}{Example 9 --- Full distribution, three parts}
Roll one die; $X=$ the value. Find \\ \textbf{(a)} the distribution; \\ \textbf{(b)} $P(X>4)$; \\ \textbf{(c)} $P(X\text{ even})$.\soln (a) uniform, $\tfrac16$ each. (b) $\tfrac26=\tfrac13$. (c) $\tfrac12$.
\end{example}
