\begin{center}
{\LARGE\bfseries\color{titleblue}Worksheet 2.5 --- Independent \& Dependent Events}\\[2pt]
{\small Grade 12 Mathematics of Data Management (MDM4U) \textperiodcentered{} Unit 2: Probability}
\end{center}

\begin{ideabox}
\emph{Independent} events multiply: $P(A\cap B)=P(A)P(B)$. \emph{Dependent} events (drawing without replacement) update the counts. Independence is \emph{not} the same as mutual exclusivity.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Independent \emph{and}: $P(A\cap B)=P(A)P(B)$.
  \item Dependent (no replacement): adjust the pool for the next draw.
  \item $P(\text{at least one})=1-P(\text{none})$ (independent trials).
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Independent events}Two events are \textbf{independent} when one has no effect on the other, like separate coin flips. Then $P(A\cap B)=P(A)\cdot P(B)$.
\lh{Dependent events}Drawing \emph{without replacement} makes events \textbf{dependent}: the first draw changes the second's odds. Two kings in a row: $\dfrac{4}{52}\times\dfrac{3}{51}$.
\lh{Not the same as exclusive}Independent is not mutually exclusive — independent events can both happen, whereas mutually exclusive events never can.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Independent coin and die}
Find $P(\text{head and a 6})$.\soln $\dfrac12\cdot\dfrac16=\dfrac{1}{12}$.
\end{example}

\begin{example}{ex1}{Example 2 --- Dependent draw (no replacement)}
Draw 2 cards without replacement. Find $P(\text{two aces})$.\soln $\dfrac{4}{52}\cdot\dfrac{3}{51}=\dfrac{12}{2652}=\dfrac{1}{221}$.
\end{example}

\begin{example}{ex2}{Example 3 --- With vs. without replacement}
From 5 red and 3 blue, draw 2. Find $P(\text{both red})$ \\ \textbf{(a)} with replacement; \\ \textbf{(b)} without.\soln (a) $\left(\dfrac58\right)^2=\dfrac{25}{64}$. (b) $\dfrac58\cdot\dfrac47=\dfrac{20}{56}=\dfrac{5}{14}$.
\end{example}

\begin{example}{ex3}{Example 4 --- Independent chain}
Find $P(\text{3 heads in a row})$.\soln $\left(\dfrac12\right)^3=\dfrac18$.
\end{example}

\begin{example}{ex4}{Example 5 --- One of each (no replacement)}
A bag has 4 red and 6 blue. Draw 2 without replacement. Find $P(\text{one of each colour})$.\soln $\dfrac{4}{10}\cdot\dfrac69+\dfrac{6}{10}\cdot\dfrac49=\dfrac{24}{90}+\dfrac{24}{90}=\dfrac{48}{90}=\dfrac{8}{15}$.
\end{example}

\begin{example}{ex5}{Example 6 --- Test for independence}
On one die, are \emph{even} and \emph{greater than 3} independent?\soln $P(\text{even}\cap{>}3)=P(\{4,6\})=\dfrac26=\dfrac13$, but $P(\text{even})P({>}3)=\dfrac12\cdot\dfrac12=\dfrac14\ne\dfrac13$. \emph{Not} independent.
\end{example}

\begin{example}{ex6}{Example 7 --- At least one via complement}
Find $P(\text{at least one 6 in 3 rolls})$.\soln $1-\left(\dfrac56\right)^3=1-\dfrac{125}{216}=\dfrac{91}{216}$.
\end{example}

\begin{example}{ex7}{Example 8 --- System reliability}
Two independent components each work with $P=0.9$. Find $P(\text{both work})$ and $P(\text{at least one works})$.\soln Both: $0.81$. At least one: $1-0.1^2=0.99$.
\end{example}

\begin{example}{ex8}{Example 9 --- Three-part jar}
A jar has 3 green and 7 yellow; draw 2 without replacement. Find \\ \textbf{(a)} $P(\text{both green})$; \\ \textbf{(b)} $P(\text{no green})$; \\ \textbf{(c)} $P(\text{at least one green})$.\soln (a) $\dfrac{3}{10}\cdot\dfrac29=\dfrac{1}{15}$. (b) $\dfrac{7}{10}\cdot\dfrac69=\dfrac{7}{15}$. (c) $1-\dfrac{7}{15}=\dfrac{8}{15}$.
\end{example}
