\begin{ideabox}
Multiply probabilities \emph{along} the branches of a tree and add \emph{across} disjoint paths. Trees handle multi-stage experiments; two-way tables organize joint outcomes.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Multiply along a branch (successive stages).
  \item Add the probabilities of disjoint paths to one outcome.
  \item Total probability: $P(E)=\sum P(\text{path to }E)$; reverse with a tree for $P(\text{cause}\mid E)$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Trees show sequences}A \textbf{tree diagram} maps a multi-stage experiment: every branch is an outcome, and the branch probabilities leaving any point add to $1$.
\lh{Multiply along, add across}Multiply the probabilities \emph{along} a path to get that outcome, then add the probabilities of the \emph{different} paths that satisfy your event.
\lh{Tables organize outcomes}A probability table lays every combined outcome in a grid, so totals and conditional probabilities can be read straight off.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Two-stage tree}
Flip a coin; if heads, roll a die. Find $P(\text{heads and a 6})$.\soln Along the branch: $\dfrac12\cdot\dfrac16=\dfrac{1}{12}$.
\end{example}

\begin{example}{ex1}{Example 2 --- Without replacement}
Draw 2 from 4 red and 3 blue. Find $P(\text{RR})$ and $P(\text{one of each})$.\soln $P(RR)=\dfrac47\cdot\dfrac36=\dfrac27$; one of each $=\dfrac47\cdot\dfrac36+\dfrac37\cdot\dfrac46=\dfrac{24}{42}=\dfrac47$.
\end{example}

\begin{example}{ex2}{Example 3 --- Total probability}
60\% of parts come from factory A (2\% defective), 40\% from B (5\% defective). Find $P(\text{defective})$.\soln $0.6(0.02)+0.4(0.05)=0.012+0.020=0.032$.
\end{example}

\begin{example}{ex3}{Example 4 --- Reverse the tree (Bayes)}
From the factories above, find $P(\text{from A}\mid\text{defective})$.\soln $\dfrac{0.012}{0.032}=\dfrac38$.
\end{example}

\begin{example}{ex4}{Example 5 --- Multi-stage independent}
Three traffic lights are each green with $P=0.4$. Find $P(\text{all green})$ and $P(\text{exactly one green})$.\soln All: $0.4^3=0.064$. Exactly one: $\dbinom{3}{1}(0.4)(0.6)^2=3(0.4)(0.36)=0.432$.
\end{example}

\begin{example}{ex5}{Example 6 --- Weighted paths}
70\% of students study (90\% pass); 30\% don't (40\% pass). Find $P(\text{pass})$.\soln $0.7(0.9)+0.3(0.4)=0.63+0.12=0.75$.
\end{example}

\begin{example}{ex6}{Example 7 --- Reverse the condition}
From that quiz model, find $P(\text{studied}\mid\text{passed})$.\soln $\dfrac{0.63}{0.75}=0.84$.
\end{example}

\begin{example}{ex7}{Example 8 --- Table to joint probability}
Of 100 people, 40 exercise and diet, 20 exercise only, 25 diet only, 15 neither. Find $P(\text{exercise or diet})$.\soln $\dfrac{40+20+25}{100}=0.85$.
\end{example}

\begin{example}{ex8}{Example 9 --- Three-draw urn}
An urn has 5 white and 3 black; draw 3 without replacement. Find \\ \textbf{(a)} $P(\text{WWW})$; \\ \textbf{(b)} $P(\text{exactly 2 white})$.\soln (a) $\dfrac58\cdot\dfrac47\cdot\dfrac36=\dfrac{5}{28}$. (b) $\dfrac{\dbinom{5}{2}\dbinom{3}{1}}{\dbinom{8}{3}}=\dfrac{30}{56}=\dfrac{15}{28}$.
\end{example}
