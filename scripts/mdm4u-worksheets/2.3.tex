\begin{ideabox}
Odds in favour $a:b$ correspond to $P=\dfrac{a}{a+b}$. The complement $P(A')=1-P(A)$ turns hard \emph{at least} events into easy \emph{none} events.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Odds in favour $a:b\Leftrightarrow P=\dfrac{a}{a+b}$; odds against $=b:a$.
  \item $P(A')=1-P(A)$.
  \item $P(\text{at least one})=1-P(\text{none})$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Odds versus probability}\textbf{Odds in favour} compare favourable to unfavourable outcomes, written $a:b$, while probability compares favourable to total. From odds $a:b$ you get $P=\dfrac{a}{a+b}$.
\lh{The complement}$P(A')=1-P(A)$. The complement — \emph{not} $A$ — is often far easier to count, especially for \emph{at least one} events.
\lh{Same information, two forms}If $P=0.25=\tfrac14$, the odds in favour are $1:3$ and the odds against are $3:1$; each conveys the same fact.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Odds to probability}
The odds in favour of an event are $3:2$. Find $P$.\soln $P=\dfrac{3}{3+2}=\dfrac35$.
\end{example}

\begin{example}{ex1}{Example 2 --- Probability to odds}
If $P(A)=\dfrac14$, state the odds in favour and against.\soln In favour $1:3$; against $3:1$.
\end{example}

\begin{example}{ex2}{Example 3 --- At least one (complement)}
Three coins are tossed. Find $P(\text{at least one head})$.\soln $1-P(\text{no heads})=1-\dfrac18=\dfrac78$.
\end{example}

\begin{example}{ex3}{Example 4 --- Odds against from a deck}
State the odds against drawing a heart.\soln $P(\text{not heart}):P(\text{heart})=\dfrac34:\dfrac14=3:1$.
\end{example}

\begin{example}{ex4}{Example 5 --- At least one 6}
Two dice are rolled. Find $P(\text{at least one 6})$.\soln $1-\left(\dfrac56\right)^2=1-\dfrac{25}{36}=\dfrac{11}{36}$.
\end{example}

\begin{example}{ex5}{Example 6 --- Implied probability from payout odds}
A horse has odds against of $5:1$. What is the implied probability of winning?\soln $P=\dfrac{1}{5+1}=\dfrac16$.
\end{example}

\begin{example}{ex6}{Example 7 --- Complement with combinations}
4 items are drawn from 20, of which 3 are defective. Find $P(\text{at least one defective})$.\soln $1-\dfrac{\dbinom{17}{4}}{\dbinom{20}{4}}=1-\dfrac{2380}{4845}\approx0.509$.
\end{example}

\begin{example}{ex7}{Example 8 --- Compare two chances}
Which is more likely: odds $4:3$ in favour, or $P=0.55$?\soln $4:3\Rightarrow\dfrac47\approx0.571>0.55$, so the odds are more likely.
\end{example}

\begin{example}{ex8}{Example 9 --- Two-part spinner}
A spinner has $P(\text{win})=0.3$. \\ \textbf{(a)} odds in favour of winning; \\ \textbf{(b)} $P(\text{win at least once in 2 spins})$.\soln (a) $3:7$. (b) $1-0.7^2=0.51$.
\end{example}
