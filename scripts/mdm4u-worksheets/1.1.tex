\begin{center}
{\LARGE\bfseries\color{titleblue}Worksheet 1.1 --- The Fundamental Counting Principle}\\[2pt]
{\small Grade 12 Mathematics of Data Management (MDM4U) \textperiodcentered{} Unit 1: Counting, Permutations \& Combinations}
\end{center}

\begin{ideabox}
Sequential (\emph{and}) choices \emph{multiply}; disjoint (\emph{or}) cases \emph{add}. Fill the most-restricted stage first, and count \emph{at least} with the complement (total $-$ unwanted).
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Multiplicative principle: independent stages multiply.
  \item Additive principle: mutually exclusive cases add --- watch for AND nested inside OR.
  \item Restrictions first; \emph{at least one} $=$ total $-$ none.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Why we multiply}When a task is built from several independent stages, the number of ways to finish it is the \textbf{product} of the choices at each stage. Choosing 1 of 3 shirts \emph{and} 1 of 4 pants gives $3\times4=12$ outfits, because every shirt pairs with every pair of pants.
\lh{AND multiplies, OR adds}Use the \textbf{multiplicative} principle when stages happen together (this \emph{and} then that). Use the \textbf{additive} principle for separate cases that cannot both occur (a book \emph{or} a movie), where you add the counts instead.
\lh{Repetition changes everything}Always ask whether choices may repeat. A 4-digit PIN allowing repeats has $10^4$ possibilities; if no digit may repeat, the choices shrink each stage: $10\times9\times8\times7$.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Product rule, with a follow-up}
A diner special is 1 of 3 egg styles, 1 of 4 breads, and 1 of 5 drinks. \\ \textbf{(a)} How many specials? \\ \textbf{(b)} How many if the drink must be coffee or tea (2 of the 5)?\soln (a) $3\cdot4\cdot5=60$. (b) restrict the last stage: $3\cdot4\cdot2=24$.
\end{example}

\begin{example}{ex1}{Example 2 --- Repeats vs. distinct (and the ratio)}
A 3-character wristband code uses letters A--Z. \\ \textbf{(a)} repeats allowed; \\ \textbf{(b)} all different; \\ \textbf{(c)} what fraction are all different?\soln (a) $26^3=17\,576$. (b) $26\cdot25\cdot24=15\,600$. (c) $\dfrac{15\,600}{17\,576}\approx0.89$.
\end{example}

\begin{example}{ex2}{Example 3 --- Product nested in cases (AND in OR)}
A café combo is either a hot drink with a pastry ($6\times4$) or a cold drink with a cookie ($5\times3$). How many combos?\soln Multiply within each case, then add the disjoint cases: $6\cdot4+5\cdot3=24+15=39$.
\end{example}

\begin{example}{ex3}{Example 4 --- Most-restricted stage first}
A 4-digit ID has first digit $1$--$9$, a distinct set of digits, and an even last digit. How many?\soln Split on the units. \emph{Units $=0$:} $9\cdot8\cdot7=504$. \emph{Units $\in\{2,4,6,8\}$:} first digit avoids $0$ and the units ($8$), then $8,7$: $4\cdot8\cdot8\cdot7=1792$. Total $504+1792=2296$.
\end{example}

\begin{example}{ex4}{Example 5 --- Complement (at least one)}
A 6-question true/false quiz. How many answer keys have at least one \emph{False}?\soln Total $-$ all-True: $2^6-1=63$.
\end{example}

\begin{example}{ex5}{Example 6 --- Adjacency restriction}
A 4-note ringtone uses 7 distinct notes; no two consecutive notes are equal. How many ringtones?\soln First note $7$; each later note only avoids the one before it ($6$): $7\cdot6\cdot6\cdot6=1512$.
\end{example}

\begin{example}{ex6}{Example 7 --- Ordered selection}
10 sprinters race. How many gold--silver--bronze podiums are possible?\soln Ordered, no repeats: $10\cdot9\cdot8=720$.
\end{example}

\begin{example}{ex7}{Example 8 --- Subsets via yes/no}
A frozen-yogurt bar has 8 toppings; take any subset. How many topping combinations, including none?\soln Each topping in or out: $2^8=256$.
\end{example}

\begin{example}{ex8}{Example 9 --- Complement in an ordered choice}
A 4-leg relay order is chosen from 6 runners. In how many orders does the fastest runner, Maya, run a leg?\soln Total $-$ none: $6\cdot5\cdot4\cdot3-5\cdot4\cdot3\cdot2=360-120=240$.
\end{example}
