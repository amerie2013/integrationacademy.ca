{\LARGE\bfseries\color{titleblue}Worksheet 5.3 --- Measures of Spread}\\[2pt]
{\small Grade 12 Mathematics of Data Management (MDM4U) \textperiodcentered{} Unit 5: One-Variable Statistics}
\end{center}

\begin{ideabox}
Two data sets can share a mean yet differ wildly. The \emph{range} ($\max-\min$), \emph{IQR} ($Q_3-Q_1$), and \emph{standard deviation} $\sigma=\sqrt{\dfrac{\sum(x-\bar x)^2}{n}}$ measure how spread out the data are.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Range $=\max-\min$; IQR $=Q_3-Q_1$.
  \item Variance $=$ mean of squared deviations; $\sigma=\sqrt{\text{variance}}$.
  \item Adding a constant leaves $\sigma$ unchanged; scaling by $k$ multiplies $\sigma$ by $|k|$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Why spread matters}Two data sets can share a mean yet differ wildly in consistency. \textbf{Spread} measures how far the values fall from the centre.
\lh{Range and IQR}The \textbf{range} is max minus min; the \textbf{interquartile range} $\mathrm{IQR}=Q_3-Q_1$ describes the middle $50\%$ and shrugs off outliers.
\lh{Standard deviation}The \textbf{standard deviation} $\sigma$ is the typical distance of a value from the mean; a larger $\sigma$ means more spread-out, less consistent data.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Range}
Find the range of $\{5,8,12,20\}$.\soln $20-5=15$.
\end{example}

\begin{example}{ex1}{Example 2 --- Standard deviation}
Find the (population) standard deviation of $\{2,4,6,8,10\}$.\soln $\bar x=6$; squared deviations $16,4,0,4,16$ sum to $40$; variance $=\tfrac{40}{5}=8$; $\sigma=\sqrt{8}\approx2.83$.
\end{example}

\begin{example}{ex2}{Example 3 --- Same mean, different spread}
Sets A and B both average $50$, with $\sigma_A=2$ and $\sigma_B=10$. Which is more consistent?\soln A --- its smaller $\sigma$ means values cluster tightly.
\end{example}

\begin{example}{ex3}{Example 4 --- IQR}
Find the IQR of $\{3,5,7,9,11,13,15\}$.\soln Median $9$; $Q_1=5$, $Q_3=13$; IQR $=8$.
\end{example}

\begin{example}{ex4}{Example 5 --- Add a constant}
Every value gains $5$. How does $\sigma$ change?\soln Unchanged --- shifting all data does not change the spread.
\end{example}

\begin{example}{ex5}{Example 6 --- Scale the data}
Every value is tripled. How does $\sigma$ change?\soln It triples ($\sigma\to3\sigma$).
\end{example}

\begin{example}{ex6}{Example 7 --- Zero spread}
Find $\sigma$ for $\{10,10,10,10\}$.\soln $0$ --- all values equal the mean.
\end{example}

\begin{example}{ex7}{Example 8 --- Variance}
Find the (population) variance of $\{1,3,5,7,9\}$.\soln $\bar x=5$; squared deviations $16,4,0,4,16$ sum to $40$; variance $=8$.
\end{example}

\begin{example}{ex8}{Example 9 --- Full spread (multi-part)}
For $\{2,4,4,4,5,5,7,9\}$ find \\ \textbf{(a)} mean; \\ \textbf{(b)} population $\sigma$; \\ \textbf{(c)} range.\soln (a) $\tfrac{40}{8}=5$. (b) squared deviations sum to $32$, variance $4$, $\sigma=2$. (c) $9-2=7$.
\end{example}
