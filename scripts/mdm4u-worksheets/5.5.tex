\begin{ideabox}
A histogram shows the \emph{shape} of a distribution. Describe it by symmetry (symmetric, left- or right-skewed) and by the number of peaks (unimodal, bimodal), then compare centre and spread.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Frequency vs. relative-frequency histograms.
  \item Shape: symmetric, right-skew (tail right), left-skew (tail left), unimodal/bimodal.
  \item Right-skew $\Rightarrow$ mean $>$ median; left-skew $\Rightarrow$ mean $<$ median.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Choosing a graph}\textbf{Histograms} show the shape of continuous data, \textbf{bar graphs} compare categories, and \textbf{stem-and-leaf} plots keep the original values visible.
\lh{Describe the shape}Summarize a distribution by its \textbf{shape} (symmetric, skewed, uniform), \textbf{centre}, and \textbf{spread}, noting any gaps or clusters.
\lh{Skew direction}A \textbf{right-skewed} distribution has a long right tail with mean greater than median; left-skew is its mirror image.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Build a histogram}
Frequencies for classes $1$--$5$ are $4,8,10,6,2$. Sketch the histogram.\soln Bars of those heights; the tallest is at class $3$.\hist{(1,4) (2,8) (3,10) (4,6) (5,2)}{12}
\end{example}

\begin{example}{ex1}{Example 2 --- Relative frequency}
Convert the frequency $8$ (of a total $30$) to a relative frequency.\soln $\tfrac{8}{30}\approx0.27$, or $27\%$.
\end{example}

\begin{example}{ex2}{Example 3 --- Symmetric shape}
A histogram rises to a central peak and falls evenly. Describe it.\soln Symmetric and unimodal --- mean $\approx$ median.
\end{example}

\begin{example}{ex3}{Example 4 --- Right skew}
Most incomes are low with a long high tail. Describe the shape and compare mean/median.\soln Right- (positively) skewed; mean $>$ median.
\end{example}

\begin{example}{ex4}{Example 5 --- Left skew}
Test scores bunch near the top with a low tail. Describe it.\soln Left- (negatively) skewed; mean $<$ median.
\end{example}

\begin{example}{ex5}{Example 6 --- Bimodal}
A histogram has two separate peaks. What is it called, and what might it mean?\soln Bimodal --- often two subgroups mixed together.
\end{example}

\begin{example}{ex6}{Example 7 --- Read a histogram}
Using the frequencies $4,8,10,6,2$, how many values are in classes $3$ or higher?\soln $10+6+2=18$.
\end{example}

\begin{example}{ex7}{Example 8 --- Compare distributions}
Two classes have the same mean, but one histogram is wider. Compare.\soln Same centre, but the wider one has greater spread (larger $\sigma$).
\end{example}

\begin{example}{ex8}{Example 9 --- Interpret (multi-part)}
For frequencies $2,5,9,7,3$ over classes $1$--$5$, \\ \textbf{(a)} state the total; \\ \textbf{(b)} the modal class; \\ \textbf{(c)} the shape.\soln (a) $26$. (b) class $3$. (c) roughly symmetric, unimodal.
\end{example}
