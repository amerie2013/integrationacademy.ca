{\small Grade 12 Mathematics of Data Management (MDM4U) \textperiodcentered{} Unit 1: Counting, Permutations \& Combinations}

\begin{ideabox}
When order does \emph{not} matter, count selections with $_nC_r=\dbinom{n}{r}=\dfrac{n!}{r!\,(n-r)!}$. Contrast with $_nP_r=r!\dbinom{n}{r}$ when order \emph{does} matter.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Combination $=$ unordered selection: $\dbinom{n}{r}$.
  \item Multiply the choices when selecting from several groups at once.
  \item Symmetry $\dbinom{n}{r}=\dbinom{n}{n-r}$; and $_nP_r=r!\dbinom{n}{r}$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Order does not matter}A \textbf{combination} is a selection in which arrangement is irrelevant — a committee of 3 is the same group no matter the order you name them. Look for \emph{choose}, \emph{select}, or \emph{group}.
\lh{The formula}$\dbinom{n}{r}=\dfrac{n!}{r!\,(n-r)!}$. It is the permutation count ${}_{n}P_{r}$ divided by $r!$, because each unordered group was counted $r!$ times when order mattered.
\lh{Use the symmetry}$\dbinom{n}{r}=\dbinom{n}{n-r}$: choosing which $r$ to keep is the same as choosing which $n-r$ to leave out, so $\dbinom{18}{16}=\dbinom{18}{2}=153$.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Basic combination}
Compute $\dbinom{8}{3}$.\soln $\dfrac{8\cdot7\cdot6}{3!}=56$.
\end{example}

\begin{example}{ex1}{Example 2 --- Choose a squad}
Choose 5 starters from 12 players.\soln $\dbinom{12}{5}=792$.
\end{example}

\begin{example}{ex2}{Example 3 --- Handshakes (pairs)}
How many handshakes among 8 people (each pair once)?\soln $\dbinom{8}{2}=28$ --- one per pair.
\end{example}

\begin{example}{ex3}{Example 4 --- Two groups at once}
Choose 2 of 6 men and 3 of 5 women.\soln $\dbinom{6}{2}\dbinom{5}{3}=15\cdot10=150$.
\end{example}

\begin{example}{ex4}{Example 5 --- Diagonals of a polygon}
How many diagonals does an octagon have?\soln $\dbinom{8}{2}-8=28-8=20$ (subtract the 8 sides).
\end{example}

\begin{example}{ex5}{Example 6 --- Triangles from points}
From 10 points, no 3 collinear, how many triangles?\soln Each set of 3 points makes one triangle: $\dbinom{10}{3}=120$.
\end{example}

\begin{example}{ex6}{Example 7 --- Symmetry shortcut}
Compute $\dbinom{15}{13}$.\soln $\dbinom{15}{13}=\dbinom{15}{2}=105$.
\end{example}

\begin{example}{ex7}{Example 8 --- Combination vs. permutation}
From 9 books, \\ \textbf{(a)} choose 4 for a bag; \\ \textbf{(b)} arrange 4 on a shelf.\soln (a) $\dbinom{9}{4}=126$. (b) $_9P_4=4!\dbinom{9}{4}=3024$.
\end{example}

\begin{example}{ex8}{Example 9 --- Any, then a condition}
A 4-person committee from 7 men and 5 women. \\ \textbf{(a)} any committee; \\ \textbf{(b)} exactly 2 women.\soln (a) $\dbinom{12}{4}=495$. (b) $\dbinom{5}{2}\dbinom{7}{2}=10\cdot21=210$.
\end{example}
