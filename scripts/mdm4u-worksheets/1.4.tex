{\LARGE\bfseries\color{titleblue}Worksheet 1.4 --- Permutations with Constraints}\\[2pt]
{\small Grade 12 Mathematics of Data Management (MDM4U) \textperiodcentered{} Unit 1: Counting, Permutations \& Combinations}
\end{center}

\begin{ideabox}
\emph{Together}: glue the block, arrange it, then arrange inside. \emph{Apart}: seat the rest, then slot into the gaps (or use the complement). \emph{Circular}: $(n-1)!$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item \emph{Together} --- one block: (block \& others)! $\times$ (inside)!.
  \item \emph{Apart} --- gap method, or total $-$ together.
  \item Circular arrangements of $n$: $(n-1)!$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Objects that must stay together}If certain objects must be adjacent, \textbf{glue} them into one block, arrange the blocks, then multiply by the arrangements \emph{inside} the block. Two people together among 5 seats: $4!\times2!$.
\lh{Objects that must stay apart}If objects must \emph{not} touch, first arrange the others, then drop the restricted objects into the \textbf{gaps} between them so they cannot be neighbours.
\lh{Around a circle}At a round table there is no fixed starting seat, so $n$ people arrange in $(n-1)!$ ways — fixing one person removes the rotations that would otherwise be counted repeatedly.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Two together (glue)}
5 people in a row; two specific people sit together.\soln Glue them: $4!\cdot2!=48$.
\end{example}

\begin{example}{ex1}{Example 2 --- Two apart (complement)}
5 people in a row; two specific people do \emph{not} sit together.\soln Total $-$ together: $5!-48=72$.
\end{example}

\begin{example}{ex2}{Example 3 --- Block of three}
6 people in a row; a group of 3 must stay together.\soln $4!\cdot3!=144$.
\end{example}

\begin{example}{ex3}{Example 4 --- Circular}
Seat 6 people around a round table.\soln $(6-1)!=120$.
\end{example}

\begin{example}{ex4}{Example 5 --- Circular with a pair}
Seat 6 around a round table with two specific people together.\soln Block $\Rightarrow5$ units: $(5-1)!\cdot2!=48$.
\end{example}

\begin{example}{ex5}{Example 6 --- None adjacent (gap method)}
4 boys and 3 girls in a row with no two girls adjacent.\soln Seat boys $4!$; gaps $=5$; place 3 girls ordered $_5P_3=60$: $4!\cdot60=1440$.
\end{example}

\begin{example}{ex6}{Example 7 --- Fixed ends}
Arrange PENCIL with P and L at the two ends.\soln Ends $2!$, middle $4!$: $2!\cdot4!=48$.
\end{example}

\begin{example}{ex7}{Example 8 --- Alternating}
4 boys and 4 girls in a row, seated alternately by gender.\soln Two gender patterns, each side arranged: $2\cdot4!\cdot4!=1152$.
\end{example}

\begin{example}{ex8}{Example 9 --- Two conditions}
8 books: 3 distinct math, 5 distinct novels. \\ \textbf{(a)} all math together? \\ \textbf{(b)} math together at the far left?\soln (a) $6!\cdot3!=4320$. (b) block fixed at left, arrange 5 novels and the 3 math inside: $5!\cdot3!=720$.
\end{example}
