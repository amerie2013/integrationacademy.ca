{\small Grade 10 Academic (MPM2D) \textperiodcentered{} Unit 7: Sine \& Cosine Laws}
\end{center}

\begin{ideabox}
Many real situations form non-right triangles --- surveying, navigation, and design. Sketch the triangle, label it, and pick a law.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Draw and label the triangle from the words; mark known sides and angles.
  \item SAS or SSS $\Rightarrow$ cosine law; a side with its opposite angle $\Rightarrow$ sine law.
  \item Bearings and directions become the angles of the triangle.
  \item State the final answer with units.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Modelling with triangles}Surveying, navigation, and structural problems become oblique triangles — sketch and label them, then choose a law.
\lh{Bearings and angles}Convert a \textbf{bearing} (measured clockwise from north) or an elevation angle into an interior angle of your triangle.
\lh{Finish with meaning}Solve, then translate the result back into the quantity asked for — a distance, a height, or a heading.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Two roads}
Two straight roads leave a point $70^\circ$ apart, running $5$ km and $8$ km. How far apart are their ends?
\soln $c^2=25+64-80\cos70^\circ\approx61.6\Rightarrow c\approx7.9$ km.
\end{example}

\begin{example}{ex1}{Example 2 --- Triangular field}
A field has sides $40$, $60$, $50$ m. Find the angle between the $40$ m and $60$ m sides.
\soln Opposite the $50$ m side: $\cos\theta=\dfrac{1600+3600-2500}{2(40)(60)}=0.5625\Rightarrow\theta\approx55.8^\circ.$
\end{example}

\begin{example}{ex2}{Example 3 --- Two ships}
Two ships leave port on courses $65^\circ$ apart. When they have sailed $30$ km and $45$ km, how far apart are they?
\soln $c^2=900+2025-2700\cos65^\circ\approx1784\Rightarrow c\approx42.2$ km.
\end{example}

\begin{example}{ex3}{Example 4 --- Surveying a baseline}
From the ends of a $100$ m baseline, a tree is sighted at $50^\circ$ and $60^\circ$. Find its distance from the first point.
\soln Third angle $=70^\circ$; distance $=\dfrac{100\sin60^\circ}{\sin70^\circ}\approx92.2$ m.
\end{example}

\begin{example}{ex4}{Example 5 --- Third side}
Two sides of a triangle are $7$ m and $9$ m with a $80^\circ$ angle between. Find the third side.
\soln $c^2=49+81-126\cos80^\circ\approx108.1\Rightarrow c\approx10.4$ m.
\end{example}

\begin{example}{ex5}{Example 6 --- Parallelogram diagonal}
A parallelogram has sides $6$ and $8$ with a $110^\circ$ angle. Find the longer diagonal.
\soln $d^2=36+64-96\cos110^\circ\approx132.8\Rightarrow d\approx11.5.$
\end{example}

\begin{example}{ex6}{Example 7 --- Flight path}
A plane flies $200$ km, turns $30^\circ$, and flies $150$ km. How far is it from the start?
\soln Interior angle $=150^\circ$: $c^2=200^2+150^2-2(200)(150)\cos150^\circ\approx114460\Rightarrow c\approx338.3$ km.
\end{example}

\begin{example}{ex7}{Example 8 --- Largest angle of a plot}
A triangular plot has sides $10$, $14$, $18$ m. Find its largest angle.
\soln Opposite $18$: $\cos\theta=\dfrac{100+196-324}{280}=-0.1\Rightarrow\theta\approx95.7^\circ.$
\end{example}

\begin{example}{ex8}{Example 9 --- Two-step survey (harder)}
A triangle has sides $12$ and $15$ with an included angle of $95^\circ$. Find the third side, then one of the remaining angles.
\soln $c^2=144+225-360\cos95^\circ\approx400.4\Rightarrow c\approx20.0$. Then $\sin A=\dfrac{12\sin95^\circ}{20}\approx0.598\Rightarrow A\approx36.7^\circ.$
\end{example}
