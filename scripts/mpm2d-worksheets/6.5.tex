\begin{ideabox}
Given two sides, use the \textbf{inverse} trig functions $\sin^{-1}$, $\cos^{-1}$, $\tan^{-1}$ to find the angle.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Choose the ratio matching the two known sides, then apply its inverse.
  \item $\theta=\sin^{-1}\!\left(\tfrac{\text{opp}}{\text{hyp}}\right)$, and similarly for $\cos^{-1}$ and $\tan^{-1}$.
  \item Keep the calculator in \textbf{degree} mode.
  \item The two acute angles add to $90^\circ$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Working backwards}When two sides are known, the angle comes from the \textbf{inverse} ratios $\sin^{-1}$, $\cos^{-1}$, and $\tan^{-1}$.
\lh{The steps}Build the correct ratio from the two known sides, then apply the matching inverse function to get the angle.
\lh{Sanity check}In a right triangle each other angle is between $0^\circ$ and $90^\circ$, and the larger side always faces the larger angle.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Inverse sine}
Find $\theta$ if $\sin\theta=0.5$.
\soln $\theta=\sin^{-1}(0.5)=30^\circ.$
\end{example}

\begin{example}{ex1}{Example 2 --- Inverse tangent}
Find $\theta$ if $\tan\theta=1$.
\soln $\theta=\tan^{-1}(1)=45^\circ.$
\end{example}

\begin{example}{ex2}{Example 3 --- Inverse cosine}
Find $\theta$ if $\cos\theta=0.6$.
\soln $\theta=\cos^{-1}(0.6)\approx53.1^\circ.$
\end{example}

\begin{example}{ex3}{Example 4 --- From opp and hyp}
Opp $=3$, hyp $=5$. Find $\theta$.
\soln $\theta=\sin^{-1}\!\left(\tfrac35\right)\approx36.9^\circ.$
\end{example}

\begin{example}{ex4}{Example 5 --- From opp and adj}
Opp $=4$, adj $=3$. Find $\theta$.
\soln $\theta=\tan^{-1}\!\left(\tfrac43\right)\approx53.1^\circ.$
\end{example}

\begin{example}{ex5}{Example 6 --- From adj and hyp}
Adj $=5$, hyp $=13$. Find $\theta$.
\soln $\theta=\cos^{-1}\!\left(\tfrac{5}{13}\right)\approx67.4^\circ.$
\end{example}

\begin{example}{ex6}{Example 7 --- Both acute angles}
Find both acute angles of a $5$-$12$-$13$ triangle.
\soln $\tan^{-1}\!\left(\tfrac{5}{12}\right)\approx22.6^\circ$ and $90-22.6=67.4^\circ.$
\end{example}

\begin{example}{ex7}{Example 8 --- Small angle}
Opp $=7$, adj $=24$. Find $\theta$.
\soln $\theta=\tan^{-1}\!\left(\tfrac{7}{24}\right)\approx16.3^\circ.$
\end{example}

\begin{example}{ex8}{Example 9 --- From a triangle (harder)}
Find both acute angles of an $8$-$15$-$17$ triangle.
\soln $\tan^{-1}\!\left(\tfrac{8}{15}\right)\approx28.1^\circ$ and $90-28.1=61.9^\circ.$
\end{example}
