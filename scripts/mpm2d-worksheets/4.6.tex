\begin{ideabox}
Standard form $y=ax^2+bx+c$ shows the $y$-intercept; convert to \textbf{vertex} form (complete the square) or \textbf{factored} form (factor) as needed.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item AOS $x=-\dfrac{b}{2a}$; substitute to get the vertex.
  \item The $y$-intercept is $c$.
  \item Standard $\to$ factored: factor the trinomial; the zeros pop out.
  \item Standard $\to$ vertex: complete the square, or use the AOS to find the vertex.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Standard form}$y=ax^2+bx+c$ shows the $y$-intercept $c$ at once, but hides the vertex and the zeros.
\lh{Switching forms}\textbf{Expand} vertex or factored form to reach standard form; \textbf{factor} or \textbf{complete the square} to go back. Each form answers a different question.
\lh{The axis shortcut}From standard form the axis of symmetry is $x=-\dfrac{b}{2a}$ — a quick route to the vertex.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Vertex from standard form}
Find the vertex of $y=x^2-6x+5$.
\soln AOS $x=-\tfrac{-6}{2}=3$; $y=9-18+5=-4$. Vertex $(3,-4)$.
\end{example}

\begin{example}{ex1}{Example 2 --- y-intercept}
State the $y$-intercept of $y=2x^2+3x-5$.
\soln $c=-5$, so $(0,-5)$.
\end{example}

\begin{example}{ex2}{Example 3 --- Standard to vertex}
Write $y=x^2-4x+1$ in vertex form.
\soln $x^2-4x+1=(x^2-4x+4)-4+1=(x-2)^2-3.$
\end{example}

\begin{example}{ex3}{Example 4 --- Standard to factored}
Write $y=x^2+6x+5$ in factored form.
\soln $(x+1)(x+5).$
\end{example}

\begin{example}{ex4}{Example 5 --- Vertex to standard}
Write $y=(x-2)^2-9$ in standard form.
\soln $x^2-4x+4-9=x^2-4x-5.$
\end{example}

\begin{example}{ex5}{Example 6 --- Factored to standard}
Write $y=(x+3)(x-1)$ in standard form.
\soln $x^2+2x-3.$
\end{example}

\begin{example}{ex6}{Example 7 --- Complete the square with $a
eq1$}
Write $y=2x^2-8x+1$ in vertex form.
\soln $2(x^2-4x)+1=2((x-2)^2-4)+1=2(x-2)^2-7.$
\end{example}

\begin{example}{ex7}{Example 8 --- Zeros via factoring}
Find the zeros of $y=x^2-2x-8$.
\soln $(x-4)(x+2)$, so $x=4,-2.$
\end{example}

\begin{example}{ex8}{Example 9 --- Full conversion (harder)}
For $y=3x^2-12x+5$, find the vertex, direction, min value, $y$-intercept, and vertex form.
\soln AOS $x=2$, $y=12-24+5=-7$: vertex $(2,-7)$, opens up, min $-7$, $y$-intercept $5$. Vertex form: $3(x-2)^2-7.$
\end{example}
