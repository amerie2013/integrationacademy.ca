\begin{ideabox}
Many problems --- areas, numbers, projectiles --- lead to a quadratic equation. Set it to $0$ and solve.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Define a variable, write an equation, and rearrange to $ax^2+bx+c=0$.
  \item Solve by factoring or the quadratic formula.
  \item \textbf{Projectiles:} ground level means height $=0$; max height is at the vertex.
  \item Reject answers that don't fit (e.g. negative lengths or times).
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Modelling with quadratics}Areas, projectile heights, and revenue problems often produce a quadratic; define the variable and write the relation.
\lh{Choosing what to solve}\emph{Maximum height} or \emph{best price} asks for the \textbf{vertex}; \emph{when it lands} or \emph{breaks even} asks for the \textbf{zeros}.
\lh{Keep valid answers}Discard negative lengths or times — only the solutions that fit the real situation count.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Rectangle area}
A rectangle's length is $3$ m more than its width and its area is $40$ m$^2$. Find its dimensions.
\soln $w(w+3)=40\Rightarrow w^2+3w-40=0\Rightarrow (w+8)(w-5)=0$. Take $w=5$, so length $8$ m.
\end{example}

\begin{example}{ex1}{Example 2 --- Consecutive integers}
Two consecutive integers have a product of $56$. Find them.
\soln $n(n+1)=56\Rightarrow n^2+n-56=0\Rightarrow (n+8)(n-7)=0\Rightarrow n=7$, giving $7$ and $8$.
\end{example}

\begin{example}{ex2}{Example 3 --- Projectile hits the ground}
A ball's height is $h=-5t^2+20t$. When does it land?
\soln $-5t^2+20t=0\Rightarrow -5t(t-4)=0\Rightarrow t=0$ or $t=4$. It lands at $t=4$ s.
\end{example}

\begin{example}{ex3}{Example 4 --- Maximum height}
For $h=-5t^2+20t$, find the maximum height.
\soln Vertex at $t=2$: $h=-20+40=20$ m.
\end{example}

\begin{example}{ex4}{Example 5 --- Number puzzle}
A number plus its square is $30$. Find it.
\soln $x^2+x=30\Rightarrow x^2+x-30=0\Rightarrow (x+6)(x-5)=0\Rightarrow x=5$ or $x=-6.$
\end{example}

\begin{example}{ex5}{Example 6 --- Projectile from a height}
A ball is thrown so $h=-5t^2+20t+25$. When does it land?
\soln $-5t^2+20t+25=0\Rightarrow t^2-4t-5=0\Rightarrow (t-5)(t+1)=0\Rightarrow t=5$ s.
\end{example}

\begin{example}{ex6}{Example 7 --- Product with a difference}
Two numbers differ by $4$ and have a product of $45$. Find them.
\soln $x(x+4)=45\Rightarrow x^2+4x-45=0\Rightarrow (x+9)(x-5)=0\Rightarrow x=5$, giving $5$ and $9.$
\end{example}

\begin{example}{ex7}{Example 8 --- Right triangle}
The legs of a right triangle differ by $7$ and the hypotenuse is $17$. Find the legs.
\soln $a^2+(a+7)^2=289\Rightarrow 2a^2+14a-240=0\Rightarrow a^2+7a-120=0\Rightarrow (a+15)(a-8)=0$. So legs $8$ and $15.$
\end{example}

\begin{example}{ex8}{Example 9 --- Full projectile (harder)}
For $h=-5t^2+25t+30$, find when the ball lands and its maximum height.
\soln Land: $-5t^2+25t+30=0\Rightarrow t^2-5t-6=0\Rightarrow (t-6)(t+1)=0\Rightarrow t=6$ s. Max at $t=2.5$: $h=-31.25+62.5+30=61.25$ m.
\end{example}
