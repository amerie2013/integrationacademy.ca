\begin{ideabox}
Use \textbf{slope}, \textbf{length}, and \textbf{midpoint} to prove facts about figures on the coordinate plane.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item \textbf{Parallel} sides $\Rightarrow$ equal slopes; \textbf{perpendicular} $\Rightarrow$ slopes multiply to $-1$.
  \item \textbf{Equal lengths} prove isosceles/equilateral/rhombus; \textbf{$90^\circ$} proves right angles/rectangles.
  \item A \textbf{parallelogram} has both pairs of opposite sides parallel \emph{or} diagonals that bisect each other.
  \item State the property, do the calculation, then write a concluding sentence.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Proof with coordinates}\textbf{Analytic geometry} proves geometric claims using algebra: place the figure on a grid, then compute lengths, midpoints, and slopes.
\lh{The toolkit}Equal \textbf{lengths} reveal an isosceles shape or a rhombus; equal \textbf{slopes} show parallel sides; \emph{negative-reciprocal} slopes show a right angle.
\lh{A tidy argument}State what you will show, compute the quantities you need, then conclude — a coordinate proof is simply organized calculation.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Right triangle by slopes}
Show $\triangle ABC$ with $A(1,1)$, $B(4,1)$, $C(4,5)$ is right-angled.
\soln Slope $AB=0$ (horizontal); slope $BC$ is undefined (vertical). Horizontal $\perp$ vertical, so the angle at $B$ is $90^\circ$ --- a right triangle.
\end{example}

\begin{example}{ex1}{Example 2 --- Isosceles by lengths}
Show $\triangle ABC$ with $A(0,0)$, $B(6,0)$, $C(3,4)$ is isosceles.
\soln $AB=6$; $AC=\sqrt{9+16}=5$; $BC=\sqrt{9+16}=5$. Since $AC=BC$, it is isosceles.
\end{example}

\begin{example}{ex2}{Example 3 --- Parallelogram by slopes}
Show $ABCD$ with $A(0,0)$, $B(4,0)$, $C(5,3)$, $D(1,3)$ is a parallelogram.
\soln Slope $AB=0=$ slope $DC$; slope $AD=3=$ slope $BC$. Both pairs of opposite sides are parallel.
\end{example}

\begin{example}{ex3}{Example 4 --- Diagonals bisect each other}
Show the diagonals of $ABCD$ with $A(0,0)$, $B(4,1)$, $C(6,5)$, $D(2,4)$ bisect each other.
\soln Midpoint of $AC=(3,2.5)$; midpoint of $BD=(3,2.5)$. They share a midpoint, so the diagonals bisect each other (a parallelogram).
\end{example}

\begin{example}{ex4}{Example 5 --- Right angle test}
Show $\triangle ABC$ with $A(0,0)$, $B(2,4)$, $C(6,2)$ has a right angle at $B$.
\soln Slope $AB=2$; slope $BC=\tfrac{2-4}{6-2}=-\tfrac12$. Product $=2\cdot(-\tfrac12)=-1$, so $AB\perp BC$: right angle at $B$.
\end{example}

\begin{example}{ex5}{Example 6 --- Classify a triangle}
Classify $\triangle ABC$ with $A(2,1)$, $B(5,1)$, $C(2,5)$.
\soln $AB=3$, $AC=4$, $BC=5$. Different lengths (scalene) and $3^2+4^2=5^2$, so it is a right scalene triangle.
\end{example}

\begin{example}{ex6}{Example 7 --- Midsegment property}
In $\triangle ABC$ with $A(0,0)$, $B(8,0)$, $C(2,6)$, show the segment joining the midpoints of $AC$ and $BC$ is parallel to $AB$ and half its length.
\soln Midpoints: $AC\to(1,3)$, $BC\to(5,3)$. The midsegment has slope $0=$ slope $AB$ (parallel) and length $4=\tfrac12(8)$.
\end{example}

\begin{example}{ex7}{Example 8 --- Square check}
Show $ABCD$ with $A(0,0)$, $B(3,0)$, $C(3,3)$, $D(0,3)$ is a square.
\soln All sides $=3$; slope $AB=0\perp$ slope $BC$ (undefined). Four equal sides and a right angle $\Rightarrow$ square.
\end{example}

\begin{example}{ex8}{Example 9 --- Rectangle proof (harder)}
Show $ABCD$ with $A(1,2)$, $B(5,4)$, $C(3,8)$, $D(-1,6)$ is a rectangle.
\soln Slope $AB=\tfrac12=$ slope $DC$; slope $AD=-2=$ slope $BC$ (parallelogram). Also $AB\cdot AD=\tfrac12\cdot(-2)=-1$, a right angle. A parallelogram with a right angle is a rectangle.
\end{example}

  \item $AC=\sqrt{9+16}=5=BC$ --- isosceles.
  \item Slope $AB=\tfrac13=$ slope $DC$; slope $AD=3=$ slope $BC$: yes, parallelogram.
  \item Mid $AC=(4,3)=$ mid $BD$ --- diagonals bisect each other.
  \item Slope $AB=\tfrac12$, slope $AC=-2$; product $-1$: yes, right angle at $A$.
  \item $3,4,5$ --- right scalene.
  \item All sides $4$; $AB\perp BC$: square.

  \item Distance to each $=\sqrt{9+16}=5$ --- equidistant.
  \item Slope $AB=\tfrac13=$ slope $DC$; slope $AD=3=$ slope $BC$: parallelogram.

\end{enumerate}
