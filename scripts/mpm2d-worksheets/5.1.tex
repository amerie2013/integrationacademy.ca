{\LARGE\bfseries\color{titleblue}Worksheet 5.1 --- Solving Quadratic Equations by Factoring}\\[2pt]
{\small Grade 10 Academic (MPM2D) \textperiodcentered{} Unit 5: Quadratic Equations}
\end{center}

\begin{ideabox}
If a product is zero, one of its factors is zero. So set the equation to $0$, factor, and solve each factor.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Rearrange to $ax^2+bx+c=0$ first.
  \item Factor fully, then apply the \textbf{zero product property}: if $PQ=0$ then $P=0$ or $Q=0$.
  \item A repeated factor gives a \textbf{double root}.
  \item The solutions are the $x$-intercepts of the parabola.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Set it to zero}To solve, first arrange the equation as $ax^2+bx+c=0$. A quadratic equation can have $0$, $1$, or $2$ real solutions.
\lh{Zero-product property}Factor, then use the fact that a product is $0$ only when a factor is $0$: from $(x-2)(x+5)=0$ you get $x=2$ or $x=-5$.
\lh{When factoring fails}Not every quadratic factors neatly — if it resists, switch to completing the square or the quadratic formula.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Simple trinomial}
Solve $x^2-5x+6=0$.
\soln $(x-2)(x-3)=0\Rightarrow x=2$ or $x=3.$
\plot{-1}{6}{-1}{6}{\addplot[exblue,very thick,domain=0.3:5.2,samples=60]{x^2-5*x+6};\addplot[mark=*,only marks,black]coordinates{(2,0)(3,0)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Negative constant}
Solve $x^2+x-12=0$.
\soln $(x+4)(x-3)=0\Rightarrow x=-4$ or $x=3.$
\end{example}

\begin{example}{ex2}{Example 3 --- Difference of squares}
Solve $x^2-9=0$.
\soln $(x-3)(x+3)=0\Rightarrow x=\pm3.$
\end{example}

\begin{example}{ex3}{Example 4 --- Common factor}
Solve $x^2-4x=0$.
\soln $x(x-4)=0\Rightarrow x=0$ or $x=4.$
\end{example}

\begin{example}{ex4}{Example 5 --- Leading coefficient}
Solve $2x^2+5x-3=0$.
\soln $(2x-1)(x+3)=0\Rightarrow x=\tfrac12$ or $x=-3.$
\end{example}

\begin{example}{ex5}{Example 6 --- Rearrange first}
Solve $x^2=2x+8$.
\soln $x^2-2x-8=0\Rightarrow (x-4)(x+2)=0\Rightarrow x=4$ or $x=-2.$
\end{example}

\begin{example}{ex6}{Example 7 --- Factor out first}
Solve $3x^2=12x$.
\soln $3x^2-12x=0\Rightarrow 3x(x-4)=0\Rightarrow x=0$ or $x=4.$
\end{example}

\begin{example}{ex7}{Example 8 --- Double root}
Solve $x^2-6x+9=0$.
\soln $(x-3)^2=0\Rightarrow x=3$ (double root).
\end{example}

\begin{example}{ex8}{Example 9 --- Expand then solve (harder)}
Solve $(x-1)(x+2)=4$.
\soln $x^2+x-2=4\Rightarrow x^2+x-6=0\Rightarrow (x+3)(x-2)=0\Rightarrow x=-3$ or $x=2.$
\end{example}
