{\LARGE\bfseries\color{titleblue}Worksheet 4.4 --- Graphing from Vertex Form}\\[2pt]
{\small Grade 10 Academic (MPM2D) \textperiodcentered{} Unit 4: Quadratic Relations}
\end{center}

\begin{ideabox}
From $y=a(x-h)^2+k$ you can graph quickly: plot the vertex, then use the step pattern and symmetry.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Plot the vertex $(h,k)$ and draw the AOS $x=h$.
  \item Step pattern from the vertex: over $1$/up $a$, over $2$/up $4a$ (mirror on both sides).
  \item Set $y=0$ and solve to find any $x$-intercepts.
  \item The range is $y\ge k$ (opens up) or $y\le k$ (opens down).
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Start at the vertex}Plot $(h,k)$ first and draw the axis of symmetry; everything else is measured out from there.
\lh{Use the step pattern}From the vertex, the curve rises by $a\cdot1,\ a\cdot4,\ a\cdot9$ as you step out $1,2,3$ units — then mirror those points on the other side.
\lh{Sketch smoothly}Join the points in a smooth curve, not straight segments, and label the vertex and intercepts.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Basic vertex-form graph}
Graph $y=(x-2)^2-1$.
\soln Vertex $(2,-1)$, opens up.
\plot{-2}{6}{-2}{8}{\addplot[exblue,very thick,domain=-0.8:4.8,samples=60]{(x-2)^2-1};\addplot[mark=*,only marks,black]coordinates{(2,-1)(1,0)(3,0)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Opens down}
Graph $y=-(x+1)^2+4$.
\soln Vertex $(-1,4)$, opens down; zeros at $x=1,-3$.
\plot{-5}{4}{-6}{5}{\addplot[exblue,very thick,domain=-4:2,samples=60]{-(x+1)^2+4};\addplot[mark=*,only marks,black]coordinates{(-1,4)(1,0)(-3,0)};}
\end{example}

\begin{example}{ex2}{Example 3 --- With a stretch}
Graph $y=2(x-1)^2-3$.
\soln Vertex $(1,-3)$, narrow (step $2,8,\dots$).
\plot{-3}{5}{-4}{9}{\addplot[exblue,very thick,domain=-1.2:3.2,samples=60]{2*(x-1)^2-3};\addplot[mark=*,only marks,black]coordinates{(1,-3)}; }
\end{example}

\begin{example}{ex3}{Example 4 --- With a compression}
Graph $y=\tfrac12(x+2)^2$.
\soln Vertex $(-2,0)$, wide.
\plot{-7}{3}{-1}{9}{\addplot[exblue,very thick,domain=-6:2,samples=60]{0.5*(x+2)^2};\addplot[mark=*,only marks,black]coordinates{(-2,0)}; }
\end{example}

\begin{example}{ex4}{Example 5 --- Another opens-down}
Graph $y=-(x-3)^2+5$.
\soln Vertex $(3,5)$, opens down.
\plot{-1}{7}{-4}{6}{\addplot[exblue,very thick,domain=0.4:5.6,samples=60]{-(x-3)^2+5};\addplot[mark=*,only marks,black]coordinates{(3,5)}; }
\end{example}

\begin{example}{ex5}{Example 6 --- Finding x-intercepts}
Find the $x$-intercepts of $y=(x-2)^2-9$.
\soln $(x-2)^2=9\Rightarrow x-2=\pm3\Rightarrow x=5$ or $x=-1.$
\end{example}

\begin{example}{ex6}{Example 7 --- Zeros from vertex form}
Graph $y=(x+1)^2-4$ and state its zeros.
\soln Vertex $(-1,-4)$; $(x+1)^2=4\Rightarrow x=1,-3.$
\plot{-5}{4}{-5}{6}{\addplot[exblue,very thick,domain=-3.4:1.4,samples=60]{(x+1)^2-4};\addplot[mark=*,only marks,black]coordinates{(-1,-4)(1,0)(-3,0)};}
\end{example}

\begin{example}{ex7}{Example 8 --- Range}
State the range of $y=2(x-1)^2+3$.
\soln Minimum $3$, opens up: range $y\ge3.$
\end{example}

\begin{example}{ex8}{Example 9 --- Full graph (harder)}
Graph $y=-2(x+2)^2+8$; give the zeros and range.
\soln Vertex $(-2,8)$, opens down. $-2(x+2)^2+8=0\Rightarrow (x+2)^2=4\Rightarrow x=0,-4$; range $y\le8.$
\plot{-6}{2}{-4}{9}{\addplot[exblue,very thick,domain=-4.5:0.5,samples=60]{-2*(x+2)^2+8};\addplot[mark=*,only marks,black]coordinates{(-2,8)(0,0)(-4,0)};}
\end{example}
