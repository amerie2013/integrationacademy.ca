\begin{ideabox}
The \textbf{distance} between two points comes from the Pythagorean theorem: $d=\sqrt{(x_2-x_1)^2+(y_2-y_1)^2}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Subtract the $x$'s and the $y$'s, square, add, then take the square root.
  \item Order of subtraction doesn't matter --- the squares remove the sign.
  \item Leave answers as exact radicals (e.g. $\sqrt{20}=2\sqrt5$) unless a decimal is asked.
  \item Horizontal/vertical segments: just count the change in one coordinate.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The idea}The \textbf{length} between two points comes straight from the Pythagorean theorem applied to the horizontal and vertical gaps.
\lh{The formula}$d=\sqrt{(x_2-x_1)^2+(y_2-y_1)^2}$: the coordinate differences are the legs of a right triangle, and the segment is its hypotenuse.
\lh{Uses}Distances let you classify triangles as isosceles or equilateral, and test whether points sit on a circle of a given radius.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- A 3-4-5 triangle}
Find the distance between $(0,0)$ and $(3,4)$.
\soln $d=\sqrt{3^2+4^2}=\sqrt{25}=5.$
\end{example}

\begin{example}{ex1}{Example 2 --- Shifted 3-4-5}
Find the distance between $(1,2)$ and $(4,6)$.
\soln $d=\sqrt{3^2+4^2}=\sqrt{25}=5.$
\end{example}

\begin{example}{ex2}{Example 3 --- With negatives}
Find the distance between $(-2,1)$ and $(2,4)$.
\soln $d=\sqrt{4^2+3^2}=\sqrt{25}=5.$
\end{example}

\begin{example}{ex3}{Example 4 --- Another}
Find the distance between $(-1,-1)$ and $(2,3)$.
\soln $d=\sqrt{3^2+4^2}=5.$
\end{example}

\begin{example}{ex4}{Example 5 --- Vertical segment}
Find the length of the segment from $(2,5)$ to $(2,-3)$.
\soln Same $x$, so $d=|5-(-3)|=8.$
\end{example}

\begin{example}{ex5}{Example 6 --- Exact radical}
Find the distance between $(1,1)$ and $(4,3)$.
\soln $d=\sqrt{3^2+2^2}=\sqrt{13}.$
\end{example}

\begin{example}{ex6}{Example 7 --- Simplifying a radical}
Find the distance between $(-3,2)$ and $(1,-2)$.
\soln $d=\sqrt{4^2+(-4)^2}=\sqrt{32}=4\sqrt2.$
\end{example}

\begin{example}{ex7}{Example 8 --- Perimeter of a triangle}
Find the perimeter of the triangle with vertices $(0,0)$, $(4,0)$, $(4,3)$.
\soln Sides: $4$, $3$, and $\sqrt{4^2+3^2}=5$. Perimeter $=4+3+5=12.$
\plot{-1}{6}{-1}{5}{\addplot[exblue,very thick]coordinates{(0,0)(4,0)(4,3)(0,0)};\addplot[mark=*,only marks,black]coordinates{(0,0)(4,0)(4,3)};}
\end{example}

\begin{example}{ex8}{Example 9 --- Proving isosceles (harder)}
Show that $\triangle ABC$ with $A(1,1)$, $B(5,1)$, $C(3,5)$ is isosceles.
\soln $AB=\sqrt{4^2+0^2}=4$; $AC=\sqrt{2^2+4^2}=\sqrt{20}$; $BC=\sqrt{(-2)^2+4^2}=\sqrt{20}$. Since $AC=BC$, the triangle is isosceles.
\end{example}
