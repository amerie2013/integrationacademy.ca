{\LARGE\bfseries\color{titleblue}Worksheet 6.2 --- Solving Problems with Similar Triangles}\\[2pt]
{\small Grade 10 Academic (MPM2D) \textperiodcentered{} Unit 6: Trigonometry}
\end{center}

\begin{ideabox}
Similar triangles let you measure things indirectly --- heights, distances, and map scales --- with a proportion.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Match corresponding sides and write a proportion.
  \item Shadow problems compare height to shadow length.
  \item Map/scale problems use the scale ratio as one fraction.
  \item Always include units and a concluding sentence.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Setting up proportions}Match corresponding sides of the similar triangles, write a \textbf{proportion}, then cross-multiply to solve for the unknown.
\lh{Real-world reach}Shadows, mirrors, and scale drawings all create similar triangles, letting you measure heights and distances indirectly.
\lh{Name sides carefully}Line up corresponding sides — those opposite equal angles — or the proportion will be set up wrong.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Building by shadows}
A $1.8$ m person casts a $3$ m shadow; a building casts a $30$ m shadow. Find its height.
\soln $\dfrac{1.8}{3}=\dfrac{h}{30}\Rightarrow h=18$ m.
\end{example}

\begin{example}{ex1}{Example 2 --- Pole by shadows}
A $1$ m stick casts a $1.5$ m shadow; a pole casts a $9$ m shadow. Find its height.
\soln $\dfrac{1}{1.5}=\dfrac{h}{9}\Rightarrow h=6$ m.
\end{example}

\begin{example}{ex2}{Example 3 --- Map scale}
On a $1:50000$ map, two towns are $4$ cm apart. Find the real distance.
\soln $4\text{ cm}\times50000=200000$ cm $=2$ km.
\end{example}

\begin{example}{ex3}{Example 4 --- Mirror method}
A person whose eyes are $1.6$ m up stands $2$ m from a mirror and sees the top of a tree $6$ m beyond the mirror. Find the tree's height.
\soln $\dfrac{1.6}{2}=\dfrac{h}{6}\Rightarrow h=4.8$ m.
\end{example}

\begin{example}{ex4}{Example 5 --- Nested triangles}
A $1.5$ m sign and a lamp post stand in line; the sign is $4$ m and the post $10$ m from a light. Find the post's height.
\soln $\dfrac{1.5}{4}=\dfrac{h}{10}\Rightarrow h=3.75$ m.
\end{example}

\begin{example}{ex5}{Example 6 --- River width}
Similar triangles give $\dfrac{w}{40}=\dfrac{15}{20}$ for a river width $w$. Find $w$.
\soln $w=40\times\tfrac{15}{20}=30$ m.
\end{example}

\begin{example}{ex6}{Example 7 --- Scale drawing}
A scale model uses $1:20$. A model car is $22$ cm long. Find the real length.
\soln $22\times20=440$ cm $=4.4$ m.
\end{example}

\begin{example}{ex7}{Example 8 --- Ramp}
A ramp and a support post are similar triangles: $\dfrac{1.2}{3}=\dfrac{h}{5}$. Find $h$.
\soln $h=5\times\tfrac{1.2}{3}=2$ m.
\end{example}

\begin{example}{ex8}{Example 9 --- Two-step (harder)}
A $2$ m person stands so their $2.5$ m shadow just reaches the same point as a tree's shadow; the person is $12$ m from the tree's base. Find the tree's height.
\soln Tree's shadow $=12+2.5=14.5$ m. $\dfrac{2}{2.5}=\dfrac{h}{14.5}\Rightarrow h=11.6$ m.
\end{example}
