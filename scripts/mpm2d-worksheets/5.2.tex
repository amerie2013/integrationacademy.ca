\begin{ideabox}
\textbf{Completing the square} turns $x^2+bx$ into a perfect square, letting you solve any quadratic with a square root.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Make the leading coefficient $1$ (divide if needed) and move the constant to the right.
  \item Add $\left(\tfrac{b}{2}\right)^2$ to \emph{both} sides to form $(x+\tfrac{b}{2})^2$.
  \item Take the square root of both sides --- remember the $\pm$.
  \item The same steps written without "$=0$" give \textbf{vertex form}.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The goal}\textbf{Completing the square} rewrites $ax^2+bx+c$ as $a(x-h)^2+k$, which reveals the vertex and lets you solve by taking a square root.
\lh{The move}Take half of the $x$-coefficient, square it, and add-and-subtract it to build a perfect-square trinomial.
\lh{Why it matters}It works on \emph{every} quadratic and is the reasoning behind the quadratic formula itself.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Perfect outcome}
Solve $x^2+6x+5=0$ by completing the square.
\soln $x^2+6x=-5\Rightarrow x^2+6x+9=4\Rightarrow (x+3)^2=4\Rightarrow x+3=\pm2\Rightarrow x=-1,-5.$
\end{example}

\begin{example}{ex1}{Example 2 --- Irrational roots}
Solve $x^2-4x-1=0$.
\soln $x^2-4x=1\Rightarrow (x-2)^2=5\Rightarrow x=2\pm\sqrt5.$
\end{example}

\begin{example}{ex2}{Example 3 --- Another irrational}
Solve $x^2+2x-5=0$.
\soln $(x+1)^2=6\Rightarrow x=-1\pm\sqrt6.$
\end{example}

\begin{example}{ex3}{Example 4 --- Nice roots}
Solve $x^2-8x+12=0$.
\soln $(x-4)^2=4\Rightarrow x=4\pm2\Rightarrow x=6,2.$
\end{example}

\begin{example}{ex4}{Example 5 --- Larger numbers}
Solve $x^2+10x+9=0$.
\soln $(x+5)^2=16\Rightarrow x=-5\pm4\Rightarrow x=-1,-9.$
\end{example}

\begin{example}{ex5}{Example 6 --- Divide first}
Solve $2x^2-12x+10=0$.
\soln Divide by $2$: $x^2-6x+5=0\Rightarrow (x-3)^2=4\Rightarrow x=5,1.$
\end{example}

\begin{example}{ex6}{Example 7 --- Irrational again}
Solve $x^2-2x-2=0$.
\soln $(x-1)^2=3\Rightarrow x=1\pm\sqrt3.$
\end{example}

\begin{example}{ex7}{Example 8 --- To vertex form}
Write $y=x^2+4x+1$ in vertex form by completing the square.
\soln $(x^2+4x+4)-4+1=(x+2)^2-3$; vertex $(-2,-3)$.
\end{example}

\begin{example}{ex8}{Example 9 --- Irrational roots (harder)}
Solve $x^2-6x+7=0$.
\soln $x^2-6x=-7\Rightarrow (x-3)^2=2\Rightarrow x=3\pm\sqrt2.$
\end{example}
