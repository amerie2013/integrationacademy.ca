{\small Grade 10 Academic (MPM2D) \textperiodcentered{} Unit 5: Quadratic Equations}

\begin{ideabox}
The \textbf{discriminant} $D=b^2-4ac$ tells how many real roots a quadratic has --- without solving it.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $D>0$: two distinct real roots; if $D$ is a perfect square they are rational.
  \item $D=0$: exactly one (double) real root.
  \item $D<0$: no real roots.
  \item Set $D$ to the right condition to find an unknown coefficient.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The discriminant}The quantity $D=b^2-4ac$ — the part under the square root — tells you the number of real roots \emph{before} you solve.
\lh{Three cases}$D>0$ gives two distinct real roots, $D=0$ gives one repeated root, and $D<0$ gives no real roots (the parabola misses the $x$-axis).
\lh{Why it is useful}It answers questions like \emph{does this parabola touch the axis?} without a full solution.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Two roots}
How many real roots does $x^2-5x+6=0$ have?
\soln $D=25-24=1>0$ --- two real roots (rational, since $1$ is a perfect square).
\end{example}

\begin{example}{ex1}{Example 2 --- One root}
How many real roots does $x^2-4x+4=0$ have?
\soln $D=16-16=0$ --- exactly one (double) root.
\end{example}

\begin{example}{ex2}{Example 3 --- No roots}
How many real roots does $x^2+x+1=0$ have?
\soln $D=1-4=-3<0$ --- no real roots.
\end{example}

\begin{example}{ex3}{Example 4 --- Rational vs irrational}
Describe the roots of $2x^2-3x-5=0$.
\soln $D=9+40=49$, a perfect square $\Rightarrow$ two \emph{rational} roots.
\end{example}

\begin{example}{ex4}{Example 5 --- Irrational roots}
Describe the roots of $x^2-2x-1=0$.
\soln $D=4+4=8>0$ but not a perfect square $\Rightarrow$ two \emph{irrational} roots.
\end{example}

\begin{example}{ex5}{Example 6 --- Solve for $k$ (one root)}
Find $k$ so $x^2+kx+9=0$ has exactly one root.
\soln $D=k^2-36=0\Rightarrow k=\pm6.$
\end{example}

\begin{example}{ex6}{Example 7 --- Solve for $k$ (two roots)}
Find $k$ so $x^2-4x+k=0$ has two real roots.
\soln $D=16-4k>0\Rightarrow k<4.$
\end{example}

\begin{example}{ex7}{Example 8 --- Solve for $k$ (no roots)}
Find $k$ so $x^2+2x+k=0$ has no real roots.
\soln $D=4-4k<0\Rightarrow k>1.$
\end{example}

\begin{example}{ex8}{Example 9 --- One root with $a
eq1$ (harder)}
Find $k$ so $2x^2+kx+8=0$ has exactly one root.
\soln $D=k^2-4(2)(8)=k^2-64=0\Rightarrow k=\pm8.$
\end{example}

\begin{keybox}

\small
\begin{enumerate}[leftmargin=*,itemsep=3pt]
  \item $D=16>0$: two real roots.
  \item $D=0$: one (double) root.
  \item $D=-16<0$: no real roots.
  \item $D=25$ (perfect square): two rational roots.
  \item $D=12$: two irrational roots.
  \item $k^2=16\Rightarrow k=\pm4$.
  \item $36-4k>0\Rightarrow k<9$.
  \item $16-4k<0\Rightarrow k>4$.
  \item $D=1-12<0$: no real roots.
  \item $D=0$: one root.
  \item $64-8k=0\Rightarrow k=8$.
  \item $D=13>0$: yes, two real roots.
  \item $k=\pm6$; the root is $x=-\tfrac{k}{2}=\mp3$.
\end{enumerate}
\end{multicols}
