\begin{ideabox}
The quadratic formula solves \emph{any} $ax^2+bx+c=0$: $x=\dfrac{-b\pm\sqrt{b^2-4ac}}{2a}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Identify $a$, $b$, $c$ from $ax^2+bx+c=0$ (watch the signs).
  \item Compute the discriminant $b^2-4ac$ first.
  \item Simplify the radical; leave exact answers unless a decimal is asked.
  \item A discriminant of $0$ gives one (double) root.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{A universal tool}The \textbf{quadratic formula} $x=\dfrac{-b\pm\sqrt{b^2-4ac}}{2a}$ solves any equation $ax^2+bx+c=0$, whether or not it factors.
\lh{Substitute carefully}Identify $a$, $b$, and $c$ with their signs, substitute, and simplify; the $\pm$ produces the two roots.
\lh{Reading the result}A whole-number answer means the quadratic could have been factored; a surd means it could not.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Rational roots}
Solve $x^2-5x+6=0$.
\soln $x=\dfrac{5\pm\sqrt{25-24}}{2}=\dfrac{5\pm1}{2}=3,2.$
\end{example}

\begin{example}{ex1}{Example 2 --- Mixed signs}
Solve $x^2+3x-10=0$.
\soln $x=\dfrac{-3\pm\sqrt{9+40}}{2}=\dfrac{-3\pm7}{2}=2,-5.$
\end{example}

\begin{example}{ex2}{Example 3 --- Leading coefficient}
Solve $2x^2-7x+3=0$.
\soln $x=\dfrac{7\pm\sqrt{49-24}}{4}=\dfrac{7\pm5}{4}=3,\tfrac12.$
\end{example}

\begin{example}{ex3}{Example 4 --- Irrational roots}
Solve $x^2-2x-1=0$.
\soln $x=\dfrac{2\pm\sqrt{8}}{2}=1\pm\sqrt2.$
\end{example}

\begin{example}{ex4}{Example 5 --- Another with $a
eq1$}
Solve $3x^2+5x-2=0$.
\soln $x=\dfrac{-5\pm\sqrt{25+24}}{6}=\dfrac{-5\pm7}{6}=\tfrac13,-2.$
\end{example}

\begin{example}{ex5}{Example 6 --- Simplify the radical}
Solve $x^2+4x+1=0$.
\soln $x=\dfrac{-4\pm\sqrt{12}}{2}=-2\pm\sqrt3.$
\end{example}

\begin{example}{ex6}{Example 7 --- Leave exact}
Solve $2x^2+3x-1=0$.
\soln $x=\dfrac{-3\pm\sqrt{17}}{4}.$
\end{example}

\begin{example}{ex7}{Example 8 --- Double root}
Solve $x^2-6x+9=0$.
\soln $x=\dfrac{6\pm\sqrt{0}}{2}=3$ (double).
\end{example}

\begin{example}{ex8}{Example 9 --- Irrational (harder)}
Solve $5x^2-3x-1=0$.
\soln $x=\dfrac{3\pm\sqrt{9+20}}{10}=\dfrac{3\pm\sqrt{29}}{10}.$
\end{example}
