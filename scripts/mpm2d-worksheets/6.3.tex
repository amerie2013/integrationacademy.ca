\begin{ideabox}
In a right triangle, the angle $\theta$ relates the sides through \textbf{SOH CAH TOA}: $\sin\theta=\tfrac{\text{opp}}{\text{hyp}}$, $\cos\theta=\tfrac{\text{adj}}{\text{hyp}}$, $\tan\theta=\tfrac{\text{opp}}{\text{adj}}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Label sides relative to $\theta$: \textbf{opposite}, \textbf{adjacent}, and the \textbf{hypotenuse} (across from the right angle).
  \item SOH CAH TOA gives the three ratios.
  \item Use the Pythagorean theorem to find a missing side first if needed.
  \item Ratios depend only on the angle, not the triangle's size.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Sine, cosine, tangent}In a right triangle the side ratios depend only on an angle: $\sin=\dfrac{\text{opp}}{\text{hyp}}$, $\cos=\dfrac{\text{adj}}{\text{hyp}}$, $\tan=\dfrac{\text{opp}}{\text{adj}}$.
\lh{Remember SOH-CAH-TOA}The mnemonic \textbf{SOH-CAH-TOA} packs all three definitions — pair each ratio with the sides it uses.
\lh{Label first}Relative to the angle you are using, mark the \emph{opposite}, \emph{adjacent}, and \emph{hypotenuse}, then pick the matching ratio.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- The 3-4-5 triangle}
State $\sin\theta$, $\cos\theta$, $\tan\theta$ for this triangle.
\rtri{4}{3}{$3$}{$4$}{$5$}
\soln $\sin\theta=\tfrac35$, $\cos\theta=\tfrac45$, $\tan\theta=\tfrac34.$
\end{example}

\begin{example}{ex1}{Example 2 --- Sine from two sides}
If the opposite is $5$ and the hypotenuse is $13$, find $\sin\theta$.
\soln $\sin\theta=\tfrac{5}{13}.$
\end{example}

\begin{example}{ex2}{Example 3 --- Cosine}
If the adjacent is $8$ and the hypotenuse is $17$, find $\cos\theta$.
\soln $\cos\theta=\tfrac{8}{17}.$
\end{example}

\begin{example}{ex3}{Example 4 --- Tangent}
If the opposite is $7$ and the adjacent is $24$, find $\tan\theta$.
\soln $\tan\theta=\tfrac{7}{24}.$
\end{example}

\begin{example}{ex4}{Example 5 --- All three (6-8-10)}
For a $6$-$8$-$10$ right triangle with $\theta$ opposite the $6$, find all three ratios.
\soln $\sin\theta=\tfrac{6}{10}=\tfrac35$, $\cos\theta=\tfrac{8}{10}=\tfrac45$, $\tan\theta=\tfrac68=\tfrac34.$
\end{example}

\begin{example}{ex5}{Example 6 --- Find the hypotenuse first}
The legs are $3$ and $4$; find $\sin\theta$ for $\theta$ opposite the $3$.
\soln Hyp $=\sqrt{3^2+4^2}=5$, so $\sin\theta=\tfrac35.$
\end{example}

\begin{example}{ex6}{Example 7 --- Identifying the ratio}
Which ratio uses the opposite and adjacent sides only?
\soln $\tan\theta=\dfrac{\text{opp}}{\text{adj}}.$
\end{example}

\begin{example}{ex7}{Example 8 --- 5-12-13}
For a $5$-$12$-$13$ triangle with $\theta$ opposite the $5$, find the three ratios.
\soln $\sin\theta=\tfrac{5}{13}$, $\cos\theta=\tfrac{12}{13}$, $\tan\theta=\tfrac{5}{12}.$
\end{example}

\begin{example}{ex8}{Example 9 --- Both acute angles (harder)}
For an $8$-$15$-$17$ triangle, give the three ratios for each acute angle $\alpha$ (opp $8$) and $\beta$ (opp $15$).
\soln $\alpha$: $\sin=\tfrac{8}{17}$, $\cos=\tfrac{15}{17}$, $\tan=\tfrac{8}{15}$. $\beta$: $\sin=\tfrac{15}{17}$, $\cos=\tfrac{8}{17}$, $\tan=\tfrac{15}{8}.$
\end{example}
