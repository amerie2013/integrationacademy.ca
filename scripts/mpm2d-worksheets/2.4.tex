\begin{ideabox}
Three special lines in a triangle: a \textbf{median} joins a vertex to the midpoint of the opposite side; a \textbf{right (perpendicular) bisector} passes through a side's midpoint at $90^\circ$; an \textbf{altitude} runs from a vertex perpendicular to the opposite side.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item \textbf{Median:} find the opposite side's midpoint, then the line through it and the vertex.
  \item \textbf{Right bisector:} find the side's midpoint and the \emph{negative reciprocal} of the side's slope.
  \item \textbf{Altitude:} use the \emph{negative reciprocal} of the opposite side's slope, through the vertex.
  \item Horizontal side $\Rightarrow$ vertical perpendicular ($x=$const), and vice-versa.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Three special lines}In a triangle, a \textbf{median} joins a vertex to the midpoint of the opposite side, a \textbf{right bisector} crosses a side at $90^\circ$ through its midpoint, and an \textbf{altitude} runs from a vertex perpendicular to the opposite side.
\lh{Building each}Use the midpoint formula for medians and bisectors, and slopes for perpendicularity: an altitude or bisector takes the \emph{negative reciprocal} of the side's slope.
\lh{Then the equation}With a point and a slope in hand, write each line as $y=mx+b$ — the tools are just midpoint, slope, and the equation of a line.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Median through the origin}
Find the median from $A(0,0)$ in $\triangle ABC$ with $B(4,2)$, $C(2,6)$.
\soln Midpoint of $BC=(3,4)$. Slope $=\tfrac{4}{3}$, through the origin: $y=\tfrac43x.$
\end{example}

\begin{example}{ex1}{Example 2 --- Median (general)}
Find the median from $A(2,1)$, where $B(6,3)$, $C(0,5)$.
\soln Midpoint $BC=(3,4)$. Slope $=\tfrac{4-1}{3-2}=3$: $y-1=3(x-2)\Rightarrow y=3x-5.$
\end{example}

\begin{example}{ex2}{Example 3 --- Right bisector}
Find the right bisector of $AB$ with $A(1,2)$, $B(5,6)$.
\soln Midpoint $(3,4)$; slope $AB=1$, so perpendicular slope $=-1$: $y-4=-1(x-3)\Rightarrow y=-x+7.$
\end{example}

\begin{example}{ex3}{Example 4 --- Right bisector (fractions)}
Find the right bisector of $AB$ with $A(-2,1)$, $B(4,3)$.
\soln Midpoint $(1,2)$; slope $AB=\tfrac{2}{6}=\tfrac13$, perpendicular $=-3$: $y-2=-3(x-1)\Rightarrow y=-3x+5.$
\end{example}

\begin{example}{ex4}{Example 5 --- Altitude to a horizontal side}
Find the altitude from $A(0,5)$ to $BC$, where $B(-2,1)$, $C(4,1)$.
\soln $BC$ is horizontal ($y=1$), so the altitude is vertical through $A$: $x=0.$
\end{example}

\begin{example}{ex5}{Example 6 --- Altitude (general)}
Find the altitude from $A(1,6)$ to $BC$, where $B(0,0)$, $C(4,2)$.
\soln Slope $BC=\tfrac12$, so altitude slope $=-2$: $y-6=-2(x-1)\Rightarrow y=-2x+8.$
\end{example}

\begin{example}{ex6}{Example 7 --- Length of a median}
Find the length of the median from $A(1,1)$ in $\triangle ABC$, $B(5,1)$, $C(3,7)$.
\soln Midpoint $BC=(4,4)$; length $=\sqrt{3^2+3^2}=\sqrt{18}=3\sqrt2.$
\end{example}

\begin{example}{ex7}{Example 8 --- Right bisector of a vertical side}
Find the right bisector of the segment from $(2,-1)$ to $(2,5)$.
\soln The segment is vertical; midpoint $(2,2)$. The perpendicular is horizontal: $y=2.$
\end{example}

\begin{example}{ex8}{Example 9 --- Two cevians at once (harder)}
For $\triangle PQR$ with $P(-2,2)$, $Q(4,2)$, $R(1,7)$, find the median from $R$ and the altitude from $R$.
\soln Midpoint $PQ=(1,2)$, so the median from $R(1,7)$ is the vertical line $x=1$. Since $PQ$ is horizontal, the altitude from $R$ is also vertical: $x=1$ (the triangle is isosceles).
\end{example}
