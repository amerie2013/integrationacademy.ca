\begin{center}
{\LARGE\bfseries\color{titleblue}Worksheet 2.3 --- Equation of a Circle (centre at the origin)}\\[2pt]
{\small Grade 10 Academic (MPM2D) \textperiodcentered{} Unit 2: Analytic Geometry}
\end{center}

\begin{ideabox}
A circle centred at the origin with radius $r$ has equation $x^2+y^2=r^2$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $r^2$ is the right side; the radius is $r=\sqrt{r^2}$.
  \item A point $(x,y)$ is \textbf{on} the circle if $x^2+y^2=r^2$, \textbf{inside} if $<r^2$, \textbf{outside} if $>r^2$.
  \item If a point is on the circle, $r^2=x^2+y^2$ gives the equation directly.
  \item The radius equals the distance from the origin to any point on the circle.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The idea}A circle centred at the origin is the set of all points a fixed distance $r$ — the radius — from $(0,0)$.
\lh{The equation}By the distance formula that condition becomes $x^2+y^2=r^2$; the radius is the square root of the right side.
\lh{Reading it}From $x^2+y^2=25$ the radius is $5$. To test a point, substitute it: equal to $r^2$ is on the circle, less is inside, greater is outside.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- From the radius}
Write the equation of the circle centred at the origin with radius $5$.
\soln $x^2+y^2=25.$
\end{example}

\begin{example}{ex1}{Example 2 --- Smaller radius}
Write the equation with radius $3$.
\soln $x^2+y^2=9.$
\end{example}

\begin{example}{ex2}{Example 3 --- Through a point}
A circle centred at the origin passes through $(3,4)$. Find its equation.
\soln $r^2=3^2+4^2=25$, so $x^2+y^2=25.$
\end{example}

\begin{example}{ex3}{Example 4 --- Radius from equation}
State the radius of $x^2+y^2=49$.
\soln $r=\sqrt{49}=7.$
\end{example}

\begin{example}{ex4}{Example 5 --- Point on the circle?}
Is $(2,3)$ on $x^2+y^2=13$?
\soln $2^2+3^2=13$, which equals $13$, so \textbf{yes}, it is on the circle.
\end{example}

\begin{example}{ex5}{Example 6 --- Inside or outside?}
Is $(1,2)$ inside or outside $x^2+y^2=16$?
\soln $1^2+2^2=5<16$, so the point is \textbf{inside}.
\end{example}

\begin{example}{ex6}{Example 7 --- Through a negative point}
A circle centred at the origin passes through $(-6,8)$. Find its equation.
\soln $r^2=(-6)^2+8^2=100$, so $x^2+y^2=100$ (radius $10$).
\end{example}

\begin{example}{ex7}{Example 8 --- Graphing a circle}
Graph $x^2+y^2=16$.
\soln Radius $4$, centred at the origin.
\plot{-6}{6}{-6}{6}{\addplot[exblue,very thick,domain=0:360,samples=100]({4*cos(x)},{4*sin(x)});\addplot[mark=*,only marks,black]coordinates{(4,0)(0,4)(-4,0)(0,-4)};}
\end{example}

\begin{example}{ex8}{Example 9 --- Full problem (harder)}
A circle centred at the origin passes through $(5,12)$. Find its radius and equation, and check whether $(-13,0)$ is on it.
\soln $r=\sqrt{5^2+12^2}=\sqrt{169}=13$, so $x^2+y^2=169$. For $(-13,0)$: $(-13)^2+0^2=169\ \checkmark$ --- yes, it is on the circle.
\end{example}
