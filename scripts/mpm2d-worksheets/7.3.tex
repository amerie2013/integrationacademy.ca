\begin{center}
{\LARGE\bfseries\color{titleblue}Worksheet 7.3 --- Choosing the Right Law}\\[2pt]
{\small Grade 10 Academic (MPM2D) \textperiodcentered{} Unit 7: Sine \& Cosine Laws}
\end{center}

\begin{ideabox}
Pick the law from the information you're given.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item \textbf{Sine law:} you have a side with its opposite angle (AAS, ASA, SSA).
  \item \textbf{Cosine law:} you have SAS (two sides + included angle) or SSS (three sides).
  \item Right triangle? Plain SOH CAH TOA is simplest.
  \item After finding one unknown, the sine law is often easiest for the rest.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Match the given information}Choose the law from what you know: a matched side--angle pair points to the \textbf{sine law}; SAS or SSS points to the \textbf{cosine law}.
\lh{A quick decision}If you can write a complete $\dfrac{\text{side}}{\sin(\text{opposite angle})}$ ratio, use the sine law; otherwise begin with the cosine law.
\lh{Then continue}After one law gives a new side or angle, the sine law is usually easiest for the remaining unknowns.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Identify (AAS)}
Which law: given $A$, $B$, and $a$?
\soln A side with its opposite angle $\Rightarrow$ \textbf{sine law}.
\end{example}

\begin{example}{ex1}{Example 2 --- Identify (SAS)}
Which law: given $a$, $b$, and $C$?
\soln Two sides and the included angle $\Rightarrow$ \textbf{cosine law}.
\end{example}

\begin{example}{ex2}{Example 3 --- Identify (SSS)}
Which law: given $a$, $b$, $c$?
\soln Three sides $\Rightarrow$ \textbf{cosine law} (to find an angle).
\end{example}

\begin{example}{ex3}{Example 4 --- Identify (SSA)}
Which law: given $A$, $a$, $b$?
\soln A side with its opposite angle $\Rightarrow$ \textbf{sine law} (find $B$).
\end{example}

\begin{example}{ex4}{Example 5 --- Solve with cosine}
$a=6$, $b=8$, $C=60^\circ$. Find $c$.
\soln SAS: $c^2=36+64-96\cos60^\circ=52\Rightarrow c\approx7.2.$
\end{example}

\begin{example}{ex5}{Example 6 --- Solve with sine}
$A=40^\circ$, $B=60^\circ$, $a=10$. Find $b$.
\soln Sine law: $b=\dfrac{10\sin60^\circ}{\sin40^\circ}\approx13.5.$
\end{example}

\begin{example}{ex6}{Example 7 --- Identify (ASA)}
Which law: given $A$, $B$, $c$?
\soln \textbf{Sine law} (after finding $C=180^\circ-A-B$).
\end{example}

\begin{example}{ex7}{Example 8 --- Solve SSS angle}
Find the largest angle of a $5$-$6$-$7$ triangle.
\soln Cosine law, opposite $7$: $\cos\theta=\dfrac{25+36-49}{60}=0.2\Rightarrow\theta\approx78.5^\circ.$
\end{example}

\begin{example}{ex8}{Example 9 --- Two-step (harder)}
$a=9$, $b=7$, $C=110^\circ$. Find $c$, then $A$.
\soln Cosine: $c^2=81+49-126\cos110^\circ\approx173.1\Rightarrow c\approx13.2$. Then sine: $\sin A=\dfrac{9\sin110^\circ}{13.2}\approx0.641\Rightarrow A\approx39.9^\circ.$
\end{example}
