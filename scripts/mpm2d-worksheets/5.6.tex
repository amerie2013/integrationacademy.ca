{\LARGE\bfseries\color{titleblue}Worksheet 5.6 --- Linear--Quadratic Systems}\\[2pt]
{\small Grade 10 Academic (MPM2D) \textperiodcentered{} Unit 5: Quadratic Equations}
\end{center}

\begin{ideabox}
A line and a parabola can meet at $2$, $1$, or $0$ points. Solve by substitution and a quadratic equation.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Set the expressions equal (substitute the line into the parabola).
  \item Solve the resulting quadratic; each solution is an $x$ of an intersection point.
  \item Find each $y$ by substituting back into either equation.
  \item The discriminant tells you whether the line is a secant ($2$), tangent ($1$), or misses ($0$).
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Two curves meeting}A \textbf{linear--quadratic system} pairs a line with a parabola; the solutions are the points they share.
\lh{Solve by substitution}Substitute the line's expression for $y$ into the parabola to get a quadratic; each root gives one intersection point.
\lh{Count the meetings}Such a system can meet in $2$, $1$, or $0$ points, matching the discriminant of the resulting quadratic.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Two intersections}
Solve the system $y=x^2$ and $y=x+2$.
\soln $x^2=x+2\Rightarrow x^2-x-2=0\Rightarrow (x-2)(x+1)=0\Rightarrow x=2,-1$. Points $(2,4)$ and $(-1,1)$.
\plot{-3}{4}{-1}{8}{\addplot[exblue,very thick,domain=-2.6:2.6,samples=60]{x^2};\addplot[qorange,very thick,domain=-3:3,samples=2]{x+2};\addplot[mark=*,only marks,black]coordinates{(2,4)(-1,1)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Tangent line}
Solve $y=x^2$ and $y=2x-1$.
\soln $x^2=2x-1\Rightarrow x^2-2x+1=0\Rightarrow (x-1)^2=0\Rightarrow x=1$ (tangent). Point $(1,1)$.
\end{example}

\begin{example}{ex2}{Example 3 --- Shifted parabola}
Solve $y=x^2-4$ and $y=x-2$.
\soln $x^2-4=x-2\Rightarrow x^2-x-2=0\Rightarrow (x-2)(x+1)=0\Rightarrow x=2,-1$. Points $(2,0)$ and $(-1,-3)$.
\end{example}

\begin{example}{ex3}{Example 4 --- Negative slope line}
Solve $y=x^2+1$ and $y=-x+3$.
\soln $x^2+1=-x+3\Rightarrow x^2+x-2=0\Rightarrow (x+2)(x-1)=0\Rightarrow x=-2,1$. Points $(-2,5)$ and $(1,2)$.
\end{example}

\begin{example}{ex4}{Example 5 --- No intersection}
Solve $y=x^2+2$ and $y=x-1$.
\soln $x^2+2=x-1\Rightarrow x^2-x+3=0$; $D=1-12<0$, so there is \textbf{no intersection}.
\end{example}

\begin{example}{ex5}{Example 6 --- Parabola with stretch}
Solve $y=2x^2$ and $y=x+1$.
\soln $2x^2=x+1\Rightarrow 2x^2-x-1=0\Rightarrow (2x+1)(x-1)=0\Rightarrow x=-\tfrac12,1$. Points $\left(-\tfrac12,\tfrac12\right)$ and $(1,2)$.
\end{example}

\begin{example}{ex6}{Example 7 --- Horizontal line}
Solve $y=x^2-2x$ and $y=3$.
\soln $x^2-2x=3\Rightarrow x^2-2x-3=0\Rightarrow (x-3)(x+1)=0\Rightarrow x=3,-1$. Points $(3,3)$ and $(-1,3)$.
\end{example}

\begin{example}{ex7}{Example 8 --- Finding a tangent point}
Show $y=-x-1$ is tangent to $y=x^2+x$ and find the point.
\soln $x^2+x=-x-1\Rightarrow x^2+2x+1=0\Rightarrow (x+1)^2=0\Rightarrow x=-1$ (one solution). Point $(-1,0)$.
\end{example}

\begin{example}{ex8}{Example 9 --- Full system (harder)}
Solve $y=x^2-3x+2$ and $y=x-1$.
\soln $x^2-3x+2=x-1\Rightarrow x^2-4x+3=0\Rightarrow (x-1)(x-3)=0\Rightarrow x=1,3$. Points $(1,0)$ and $(3,2)$.
\plot{-1}{5}{-2}{6}{\addplot[exblue,very thick,domain=-0.3:4,samples=60]{x^2-3*x+2};\addplot[qorange,very thick,domain=-1:5,samples=2]{x-1};\addplot[mark=*,only marks,black]coordinates{(1,0)(3,2)};}
\end{example}

\begin{keybox}
\begin{multicols}{2}
\small
\begin{enumerate}[leftmargin=*,itemsep=3pt]
  \item $x^2-x-6=0\Rightarrow (3,9),(-2,4)$.
  \item $x^2-x-2=0\Rightarrow (2,2),(-1,-1)$.
  \item $(x-1)^2=0\Rightarrow (1,2)$ --- tangent.
  \item $x=\pm2$: $(2,4),(-2,4)$.
  \item $x^2-4x+3=0\Rightarrow (1,-3),(3,-3)$.
  \item $x^2-x+3=0$, $D<0$: no intersection.
  \item $2x^2-x-3=0\Rightarrow \left(\tfrac32,\tfrac92\right),(-1,2)$.
  \item $x^2+3x+2=0\Rightarrow (-1,-2),(-2,-2)$.
  \item $x^2-3x+2=0\Rightarrow (1,0),(2,3)$.
  \item $x^2+x-2=0\Rightarrow (-2,0),(1,3)$.
  \item $x^2-x+5=0$, $D=1-20<0$: never.
  \item $x^2-3x+2=0\Rightarrow (1,0),(2,1)$.
  \item $x^2-kx+2=0$ tangent $\Rightarrow k^2-8=0\Rightarrow k=\pm2\sqrt2$.
\end{enumerate}
\end{multicols}
\end{keybox}
