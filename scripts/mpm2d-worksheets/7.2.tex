{\LARGE\bfseries\color{titleblue}Worksheet 7.2 --- The Cosine Law}\\[2pt]
{\small Grade 10 Academic (MPM2D) \textperiodcentered{} Unit 7: Sine \& Cosine Laws}
\end{center}

\begin{ideabox}
The \textbf{cosine law} generalises the Pythagorean theorem: $c^2=a^2+b^2-2ab\cos C$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Use it for \textbf{SAS} (two sides and the included angle) to find the third side.
  \item Use the rearranged form $\cos C=\dfrac{a^2+b^2-c^2}{2ab}$ for \textbf{SSS} (find an angle).
  \item The angle and the side it helps find are always \emph{opposite} each other.
  \item A negative cosine means an obtuse angle.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{A generalized Pythagoras}The \textbf{cosine law} $a^2=b^2+c^2-2bc\cos A$ handles triangles where the sine law cannot even start.
\lh{When to use it}Reach for it with two sides and the angle between them (\textbf{SAS}), or with all three sides (\textbf{SSS}) to find an angle.
\lh{Right-angle check}When $A=90^\circ$, $\cos A=0$ and the law collapses to $a^2=b^2+c^2$ — ordinary Pythagoras.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- SAS find a side}
In $\triangle ABC$, $a=5$, $b=7$, $C=60^\circ$. Find $c$.
\gtri{$A$}{$B$}{$C$}{$a$}{$b$}{$c$}
\soln $c^2=25+49-2(5)(7)\cos60^\circ=74-35=39\Rightarrow c\approx6.2.$
\end{example}

\begin{example}{ex1}{Example 2 --- Obtuse not yet}
$a=8$, $b=6$, $C=40^\circ$. Find $c$.
\soln $c^2=64+36-96\cos40^\circ\approx26.5\Rightarrow c\approx5.1.$
\end{example}

\begin{example}{ex2}{Example 3 --- SSS find an angle}
Find $C$ in a triangle with $a=5$, $b=6$, $c=7$.
\soln $\cos C=\dfrac{25+36-49}{2(5)(6)}=\dfrac{12}{60}=0.2\Rightarrow C\approx78.5^\circ.$
\end{example}

\begin{example}{ex3}{Example 4 --- Largest angle}
Find the largest angle of a $4$-$5$-$6$ triangle.
\soln Largest is opposite $6$: $\cos\theta=\dfrac{16+25-36}{40}=0.125\Rightarrow\theta\approx82.8^\circ.$
\end{example}

\begin{example}{ex4}{Example 5 --- Obtuse included angle}
$a=10$, $b=12$, $C=100^\circ$. Find $c$.
\soln $c^2=100+144-240\cos100^\circ\approx285.7\Rightarrow c\approx16.9.$
\end{example}

\begin{example}{ex5}{Example 6 --- Find an angle}
Find $A$ in a triangle with $a=7$, $b=8$, $c=9$.
\soln $\cos A=\dfrac{64+81-49}{2(8)(9)}=\dfrac{96}{144}=0.667\Rightarrow A\approx48.2^\circ.$
\end{example}

\begin{example}{ex6}{Example 7 --- SAS again}
$a=11$, $b=13$, $C=72^\circ$. Find $c$.
\soln $c^2=121+169-286\cos72^\circ\approx201.6\Rightarrow c\approx14.2.$
\end{example}

\begin{example}{ex7}{Example 8 --- Smallest angle}
Find the smallest angle of a $6$-$8$-$9$ triangle.
\soln Smallest is opposite $6$: $\cos\theta=\dfrac{64+81-36}{144}\approx0.757\Rightarrow\theta\approx40.8^\circ.$
\end{example}

\begin{example}{ex8}{Example 9 --- Recognising a right angle (harder)}
Use the cosine law to find $C$ in a $9$-$12$-$15$ triangle ($c=15$).
\soln $\cos C=\dfrac{81+144-225}{2(9)(12)}=\dfrac{0}{216}=0\Rightarrow C=90^\circ$ --- it's a right triangle.
\end{example}
