\begin{ideabox}
To find an unknown side, choose the ratio that links the known angle, the known side, and the unknown side.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Opposite from hyp: $\text{opp}=\text{hyp}\sin\theta$. Adjacent from hyp: $\text{adj}=\text{hyp}\cos\theta$.
  \item Opposite from adj: $\text{opp}=\text{adj}\tan\theta$.
  \item If the unknown is in the denominator, divide: $\text{hyp}=\dfrac{\text{opp}}{\sin\theta}$.
  \item Set the calculator to \textbf{degrees}; round sensibly (e.g. $1$ decimal).
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Pick the right ratio}To find a missing side, choose the ratio that involves the known angle, the known side, and the side you want.
\lh{Solve the equation}Set up the ratio, then multiply or divide to isolate the unknown side, remembering the hypotenuse is the longest side.
\lh{Estimate to check}A quick sketch tells you whether an answer is reasonable — the side opposite a small angle should come out short.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Opposite from hypotenuse}
Find the opposite side: hyp $=10$, $\theta=30^\circ$.
\soln $\text{opp}=10\sin30^\circ=5.$
\end{example}

\begin{example}{ex1}{Example 2 --- Adjacent from hypotenuse}
Find the adjacent side: hyp $=20$, $\theta=60^\circ$.
\soln $\text{adj}=20\cos60^\circ=10.$
\end{example}

\begin{example}{ex2}{Example 3 --- Opposite from adjacent}
Find the opposite side: adj $=8$, $\theta=40^\circ$.
\soln $\text{opp}=8\tan40^\circ\approx6.7.$
\end{example}

\begin{example}{ex3}{Example 4 --- Hypotenuse from opposite}
Find the hypotenuse: opp $=6$, $\theta=30^\circ$.
\soln $\text{hyp}=\dfrac{6}{\sin30^\circ}=12.$
\end{example}

\begin{example}{ex4}{Example 5 --- Adjacent in denominator}
Find the hypotenuse: adj $=12$, $\theta=25^\circ$.
\soln $\text{hyp}=\dfrac{12}{\cos25^\circ}\approx13.2.$
\end{example}

\begin{example}{ex5}{Example 6 --- Opposite (calculator)}
Find the opposite side: hyp $=14$, $\theta=35^\circ$.
\soln $\text{opp}=14\sin35^\circ\approx8.0.$
\end{example}

\begin{example}{ex6}{Example 7 --- Using tangent}
A right triangle has $\theta=50^\circ$ and adjacent $9$. Find the opposite side $x$.
\soln $x=9\tan50^\circ\approx10.7.$
\end{example}

\begin{example}{ex7}{Example 8 --- Hypotenuse via cosine}
Find the hypotenuse: adj $=10$, $\theta=40^\circ$.
\soln $\text{hyp}=\dfrac{10}{\cos40^\circ}\approx13.1.$
\end{example}

\begin{example}{ex8}{Example 9 --- Two unknown sides (harder)}
A right triangle has hyp $=18$ and $\theta=28^\circ$. Find both legs.
\soln Opp $=18\sin28^\circ\approx8.5$; adj $=18\cos28^\circ\approx15.9.$
\end{example}
