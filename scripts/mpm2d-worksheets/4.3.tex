{\LARGE\bfseries\color{titleblue}Worksheet 4.3 --- Transformations \& Vertex Form}\\[2pt]
{\small Grade 10 Academic (MPM2D) \textperiodcentered{} Unit 4: Quadratic Relations}
\end{center}

\begin{ideabox}
Vertex form $y=a(x-h)^2+k$ shows every transformation of $y=x^2$ at a glance.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $h$ shifts \textbf{horizontally} (right if $h>0$); $k$ shifts \textbf{vertically} (up if $k>0$).
  \item The vertex is $(h,k)$ and the AOS is $x=h$.
  \item $a$ stretches ($|a|>1$, narrower) or compresses ($|a|<1$, wider); $a<0$ reflects (opens down).
  \item To find $a$, substitute another known point and solve.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Vertex form}$y=a(x-h)^2+k$ puts the vertex at $(h,k)$ and displays every transformation at a glance.
\lh{Reading the parameters}$a$ stretches and, if negative, flips the parabola; $h$ shifts it right; $k$ shifts it up. Mind the minus sign: $(x-3)$ means right $3$.
\lh{Building a graph}From vertex form, plot the vertex, then use $a$ and symmetry to place a few points on each side.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Describe a shift}
Describe how $y=(x-3)^2+2$ transforms $y=x^2$.
\soln Right $3$, up $2$ (vertex $(3,2)$).
\end{example}

\begin{example}{ex1}{Example 2 --- Reflection and shift}
Describe $y=-(x+1)^2$.
\soln Reflected in the $x$-axis (opens down), left $1$; vertex $(-1,0)$.
\end{example}

\begin{example}{ex2}{Example 3 --- Vertical stretch}
Describe $y=2x^2$.
\soln Vertical stretch by factor $2$ (narrower); vertex $(0,0)$.
\end{example}

\begin{example}{ex3}{Example 4 --- Vertical compression}
Describe $y=\tfrac12x^2$.
\soln Vertical compression by factor $\tfrac12$ (wider).
\end{example}

\begin{example}{ex4}{Example 5 --- Two shifts}
Describe $y=(x+4)^2-5$.
\soln Left $4$, down $5$; vertex $(-4,-5)$.
\end{example}

\begin{example}{ex5}{Example 6 --- Build from a vertex}
Write the equation with vertex $(2,-3)$ and $a=1$.
\soln $y=(x-2)^2-3.$
\end{example}

\begin{example}{ex6}{Example 7 --- Build with reflection}
Write the equation with vertex $(-1,4)$ opening down with $a=-2$.
\soln $y=-2(x+1)^2+4.$
\end{example}

\begin{example}{ex7}{Example 8 --- Read off features}
For $y=-3(x-2)^2+5$, state the vertex, direction, and stretch.
\soln Vertex $(2,5)$, opens down, vertical stretch by $3$.
\end{example}

\begin{example}{ex8}{Example 9 --- Finding $a$ (harder)}
A parabola has vertex $(1,2)$ and passes through $(3,10)$. Find its equation.
\soln $10=a(3-1)^2+2\Rightarrow 10=4a+2\Rightarrow a=2$, so $y=2(x-1)^2+2.$
\end{example}
