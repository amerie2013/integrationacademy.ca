{\LARGE\bfseries\color{titleblue}Worksheet 3.3 --- Factoring Simple Trinomials ($x^2+bx+c$)}\\[2pt]
{\small Grade 10 Academic (MPM2D) \textperiodcentered{} Unit 3: Algebraic Expressions}
\end{center}

\begin{ideabox}
To factor $x^2+bx+c$, find two numbers that \textbf{multiply to $c$} and \textbf{add to $b$}.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item List factor pairs of $c$; pick the pair that sums to $b$.
  \item $c>0$: both numbers share $b$'s sign. $c<0$: the numbers have opposite signs.
  \item Result: $x^2+bx+c=(x+m)(x+n)$ where $mn=c$, $m+n=b$.
  \item Always remove a common factor first.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The target}To factor $x^2+bx+c$, find two numbers that \textbf{multiply to $c$} and \textbf{add to $b$}; they become the constants in $(x+\ )(x+\ )$.
\lh{Signs guide the search}If $c>0$ the two numbers share $b$'s sign; if $c<0$ they have opposite signs. This quickly narrows the pairs to test.
\lh{Verify}Expand your answer with FOIL to make sure the middle term $bx$ comes back.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Both positive}
Factor $x^2+5x+6$.
\soln $2$ and $3$ multiply to $6$, add to $5$: $(x+2)(x+3)$. The zeros are $x=-2,-3$.
\plot{-5}{2}{-1}{7}{\addplot[exblue,very thick,domain=-4.2:0.7,samples=60]{x^2+5*x+6};\addplot[mark=*,only marks,black]coordinates{(-2,0)(-3,0)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Larger product}
Factor $x^2+7x+12$.
\soln $3,4$: $(x+3)(x+4).$
\end{example}

\begin{example}{ex2}{Example 3 --- Both negative}
Factor $x^2-5x+6$.
\soln $-2,-3$: $(x-2)(x-3).$
\end{example}

\begin{example}{ex3}{Example 4 --- Opposite signs I}
Factor $x^2+x-12$.
\soln $4,-3$: $(x+4)(x-3).$
\end{example}

\begin{example}{ex4}{Example 5 --- Opposite signs II}
Factor $x^2-2x-15$.
\soln $-5,3$: $(x-5)(x+3).$
\end{example}

\begin{example}{ex5}{Example 6 --- Both negative again}
Factor $x^2-9x+20$.
\soln $-4,-5$: $(x-4)(x-5).$
\end{example}

\begin{example}{ex6}{Example 7 --- Negative constant}
Factor $x^2+2x-8$.
\soln $4,-2$: $(x+4)(x-2).$
\end{example}

\begin{example}{ex7}{Example 8 --- Remove a common factor first}
Factor $2x^2+10x+12$.
\soln $2(x^2+5x+6)=2(x+2)(x+3).$
\end{example}

\begin{example}{ex8}{Example 9 --- Bigger numbers (harder)}
Factor $x^2-13x+36$.
\soln $-4,-9$: $(x-4)(x-9).$
\end{example}
