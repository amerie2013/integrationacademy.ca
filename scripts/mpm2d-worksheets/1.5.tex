\begin{ideabox}
Real problems become systems once you \textbf{define two variables} and write \textbf{two equations}.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Let each unknown be a variable; state clearly what each represents.
  \item Translate the two conditions into two equations.
  \item Solve by substitution or elimination --- whichever is cleaner.
  \item \textbf{Answer in words} and check the numbers against the problem.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Modelling with two equations}A word problem with two unknowns becomes a system: assign a variable to each unknown, then translate each condition into an equation.
\lh{Common setups}Mixture, distance, and cost problems usually give one equation for a total (quantity) and one for a value (cost, speed, or concentration).
\lh{Finish and check}Solve by substitution or elimination, then confirm the answer makes sense in the original words — not just in the algebra.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Sum and difference}
Two numbers have a sum of $25$ and a difference of $7$. Find them.
\soln $x+y=25$, $x-y=7$. Add: $2x=32\Rightarrow x=16$, $y=9$. The numbers are $16$ and $9$.
\end{example}

\begin{example}{ex1}{Example 2 --- Tickets}
Adult tickets cost \$10, child \$6. $200$ tickets sold for \$1640. How many of each?
\soln $a+c=200$, $10a+6c=1640$. From the first $c=200-a$: $10a+6(200-a)=1640\Rightarrow 4a=440\Rightarrow a=110$, $c=90$.
\end{example}

\begin{example}{ex2}{Example 3 --- Coins}
A box of $20$ coins (dimes and quarters) is worth \$3.65. How many of each?
\soln $d+q=20$, $0.10d+0.25q=3.65$. Sub $d=20-q$: $0.10(20-q)+0.25q=3.65\Rightarrow 0.15q=1.65\Rightarrow q=11$, $d=9$.
\end{example}

\begin{example}{ex3}{Example 4 --- Rectangle}
A rectangle's length is $3$ more than its width and its perimeter is $26$. Find its dimensions.
\soln $l=w+3$, $2(l+w)=26\Rightarrow l+w=13$. Then $(w+3)+w=13\Rightarrow w=5$, $l=8$.
\end{example}

\begin{example}{ex4}{Example 5 --- Meal deal}
Two burgers and three fries cost \$12; three burgers and one fry cost \$11. Find each price.
\soln $2b+3f=12$, $3b+f=11\Rightarrow f=11-3b$. Then $2b+3(11-3b)=12\Rightarrow -7b=-21\Rightarrow b=3$, $f=2$.
\end{example}

\begin{example}{ex5}{Example 6 --- Mixture}
How much $40\%$ and $70\%$ acid makes $30$ L of $50\%$ acid?
\soln $x+y=30$, $0.40x+0.70y=15$. Sub $x=30-y$: $0.40(30-y)+0.70y=15\Rightarrow 0.30y=3\Rightarrow y=10$, $x=20$. ($20$ L of $40\%$, $10$ L of $70\%$.)
\end{example}

\begin{example}{ex6}{Example 7 --- Ages}
Maria is four times as old as her son. In $5$ years she will be three times as old. Find their ages.
\soln $m=4s$ and $m+5=3(s+5)$. So $4s+5=3s+15\Rightarrow s=10$, $m=40$.
\end{example}

\begin{example}{ex7}{Example 8 --- Current}
A canoe goes $12$ km downstream in $2$ h and $12$ km upstream in $3$ h. Find the canoe's speed and the current.
\soln Downstream speed $=6=c+w$; upstream $=4=c-w$. Add: $2c=10\Rightarrow c=5$ km/h, $w=1$ km/h.
\end{example}

\begin{example}{ex8}{Example 9 --- Revenue (harder)}
A theatre charges \$12 (adult) and \$8 (student). One night $500$ people paid \$5200 total. How many of each?
\soln $a+s=500$, $12a+8s=5200$. Sub $s=500-a$: $12a+8(500-a)=5200\Rightarrow 4a=1200\Rightarrow a=300$, $s=200$.
\end{example}

\begin{keybox}
\begin{multicols}{2}
\small
\begin{enumerate}[leftmargin=*,itemsep=3pt]
  \item $x+y=40$, $x-y=6\Rightarrow x=23$, $y=17$.
  \item $a+c=150$, $9a+5c=1090\Rightarrow a=85$, $c=65$.
  \item $d+n=25$, $0.10d+0.05n=1.85\Rightarrow d=12$, $n=13$.
  \item $l=2w$, $2(l+w)=36\Rightarrow w=6$, $l=12$.
  \item Subtract: $2$ coffees $=6\Rightarrow$ coffee \$3, muffin \$2.
  \item $x+y=60$, $0.2x+0.5y=18\Rightarrow y=20$ (50\%), $x=40$ (20\%).
  \item $f=3d$, $f+10=2(d+10)\Rightarrow d=10$, $f=30$.
  \item $p+w=300$, $p-w=200\Rightarrow p=250$ km/h, $w=50$ km/h.
  \item $x+y=50$, $x-y=8\Rightarrow x=29$, $y=21$.
  \item Subtract: $2$ shirts $=20\Rightarrow$ shirt \$10, hat \$15.
  \item $x+y=20$, $x+3y=46\Rightarrow y=13$ (3-mark), $x=7$ (1-mark).
  \item $x+y=90$, $x-y=30\Rightarrow x=60^\circ$, $y=30^\circ$.
  \item $x+y=5000$, $0.03x+0.05y=210\Rightarrow y=3000$ (5\%), $x=2000$ (3\%).
\end{enumerate}
\end{multicols}
\end{keybox}
