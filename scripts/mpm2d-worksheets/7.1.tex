{\LARGE\bfseries\color{titleblue}Worksheet 7.1 --- The Sine Law}\\[2pt]
{\small Grade 10 Academic (MPM2D) \textperiodcentered{} Unit 7: Sine \& Cosine Laws}
\end{center}

\begin{ideabox}
The \textbf{sine law} relates sides to their opposite angles in any triangle: $\dfrac{a}{\sin A}=\dfrac{b}{\sin B}=\dfrac{c}{\sin C}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Each side pairs with the angle \emph{opposite} it.
  \item Use the sine law when you know a side and its opposite angle (plus one more piece): \textbf{AAS}, \textbf{ASA}, or \textbf{SSA}.
  \item To find a side, put sides on top; to find an angle, flip both fractions.
  \item Find a missing third angle with $A+B+C=180^\circ$ when needed.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Beyond right triangles}The \textbf{sine law} works in \emph{any} triangle: $\dfrac{a}{\sin A}=\dfrac{b}{\sin B}=\dfrac{c}{\sin C}$, tying each side to the sine of its opposite angle.
\lh{When to use it}Use it when you know a side and its opposite angle, plus one more fact — the \textbf{ASA}, \textbf{AAS}, or \textbf{SSA} cases.
\lh{Pair them up}Always match a side with the angle across from it; that opposite pairing is what keeps the ratios equal.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Finding a side}
In $\triangle ABC$, $A=40^\circ$, $B=60^\circ$, $b=10$. Find $a$.
\gtri{$A$}{$B$}{$C$}{$a$}{$b$}{$c$}
\soln $\dfrac{a}{\sin40^\circ}=\dfrac{10}{\sin60^\circ}\Rightarrow a=\dfrac{10\sin40^\circ}{\sin60^\circ}\approx7.4.$
\end{example}

\begin{example}{ex1}{Example 2 --- Another side}
$A=50^\circ$, $a=8$, $B=70^\circ$. Find $b$.
\soln $b=\dfrac{8\sin70^\circ}{\sin50^\circ}\approx9.8.$
\end{example}

\begin{example}{ex2}{Example 3 --- Finding an angle}
$a=7$, $A=40^\circ$, $b=10$. Find $B$.
\soln $\sin B=\dfrac{10\sin40^\circ}{7}\approx0.918\Rightarrow B\approx66.6^\circ.$
\end{example}

\begin{example}{ex3}{Example 4 --- ASA (find third angle first)}
$A=45^\circ$, $B=65^\circ$, $c=12$. Find $a$.
\soln $C=70^\circ$; $a=\dfrac{12\sin45^\circ}{\sin70^\circ}\approx9.0.$
\end{example}

\begin{example}{ex4}{Example 5 --- Mixed}
$B=30^\circ$, $A=80^\circ$, $a=15$. Find $b$.
\soln $b=\dfrac{15\sin30^\circ}{\sin80^\circ}\approx7.6.$
\end{example}

\begin{example}{ex5}{Example 6 --- Angle again}
$a=12$, $A=70^\circ$, $c=9$. Find $C$.
\soln $\sin C=\dfrac{9\sin70^\circ}{12}\approx0.705\Rightarrow C\approx44.8^\circ.$
\end{example}

\begin{example}{ex6}{Example 7 --- AAS}
$A=55^\circ$, $B=40^\circ$, $a=18$. Find $b$.
\soln $b=\dfrac{18\sin40^\circ}{\sin55^\circ}\approx14.1.$
\end{example}

\begin{example}{ex7}{Example 8 --- Find a side after the third angle}
$A=60^\circ$, $B=50^\circ$, $b=14$. Find $c$.
\soln $C=70^\circ$; $c=\dfrac{14\sin70^\circ}{\sin50^\circ}\approx17.2.$
\end{example}

\begin{example}{ex8}{Example 9 --- Solve two sides (harder)}
$A=35^\circ$, $B=85^\circ$, $a=20$. Find $b$ and $c$.
\soln $C=60^\circ$; $b=\dfrac{20\sin85^\circ}{\sin35^\circ}\approx34.7$; $c=\dfrac{20\sin60^\circ}{\sin35^\circ}\approx30.2.$
\end{example}
