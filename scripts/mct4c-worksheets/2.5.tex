\begin{ideabox}
The remainder of $P(x)\div(x-a)$ is $P(a)$ â€” one substitution instead of a full division. For a divisor $bx-a$, evaluate at $x=\tfrac{a}{b}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Remainder $=P(a)$.
  \item Set $P(a)$ equal to a given remainder to find an unknown.
  \item For $bx-a$, use $x=\tfrac{a}{b}$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{A shortcut for remainders}The \textbf{remainder theorem} says the remainder of $P(x)\div(x-a)$ is simply $P(a)$ â€” one substitution instead of a full division.
\lh{Finding unknowns}Setting $P(a)$ equal to a given remainder produces an equation you can solve for an unknown coefficient.
\lh{Divisors like $bx-a$}For a divisor $bx-a$, evaluate at $x=\dfrac{a}{b}$ to obtain the remainder.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- By substitution}
Remainder of $(x^3-4x+1)\div(x-2)$.\soln $P(2)=8-8+1=1$.
\end{example}

\begin{example}{ex1}{Example 2 --- Negative value}
Remainder of $(x^3-4x+1)\div(x+3)$.\soln $P(-3)=-27+12+1=-14$.
\end{example}

\begin{example}{ex2}{Example 3 --- Find a coefficient}
$P(x)=x^3+kx-6$ has remainder $4$ on $\div(x-2)$. Find $k$.\soln $8+2k-6=4\Rightarrow k=1$.
\end{example}

\begin{example}{ex3}{Example 4 --- Divisor $2x-1$}
Remainder of $(2x^3-3x^2+1)\div(2x-1)$.\soln $P(\tfrac12)=\tfrac14-\tfrac34+1=\tfrac12$.
\end{example}

\begin{example}{ex4}{Example 5 --- Fast evaluation}
Evaluate $P(-1)$ for $P(x)=x^3+2x^2-x+5$.\soln $-1+2+1+5=7$.
\end{example}

\begin{example}{ex5}{Example 6 --- Zero remainder}
Remainder of $(x^3-2x+4)\div(x+2)$.\soln $P(-2)=-8+4+4=0$.
\end{example}

\begin{example}{ex6}{Example 7 --- Small case}
Remainder of $(x^2+3x-2)\div(x-1)$.\soln $P(1)=2$.
\end{example}

\begin{example}{ex7}{Example 8 --- Another coefficient}
$P(x)=x^3+ax+2$ has remainder $8$ on $\div(x-1)$. Find $a$.\soln $1+a+2=8\Rightarrow a=5$.
\end{example}

\begin{example}{ex8}{Example 9 --- Quartic remainder}
Remainder of $(x^4-1)\div(x-1)$.\soln $P(1)=0$.
\end{example}
