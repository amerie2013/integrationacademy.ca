{\LARGE\bfseries\color{titleblue}Worksheet 4.6 --- Surface Area \& Volume}\\[2pt]
{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 4: Vectors \& Geometry}
\end{center}

\begin{ideabox}
Prism/cylinder volume $=(\text{base area})\times h$; pyramid/cone $=\tfrac13(\text{base area})h$; sphere $=\tfrac43\pi r^3$. Cylinder $SA=2\pi r^2+2\pi rh$, cone $SA=\pi r^2+\pi r\ell$, sphere $SA=4\pi r^2$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Prisms/cylinders: base $\times h$; cones/pyramids: $\tfrac13$ of that.
  \item Sphere $V=\tfrac43\pi r^3$, $SA=4\pi r^2$.
  \item Cone slant $\ell=\sqrt{r^2+h^2}$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Volume of common solids}Prisms and cylinders hold (base area)$\times$height; cones and pyramids hold one-third of that; a sphere holds $\tfrac43\pi r^3$.
\lh{Surface area}Add every outer face: a closed cylinder is $2\pi r^2+2\pi rh$, a cone is $\pi r^2+\pi r\ell$, and a sphere is $4\pi r^2$.
\lh{Slant height}A cone's slant height comes from its radius and height, $\ell=\sqrt{r^2+h^2}$, and is needed for the lateral surface.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Cylinder}
Radius $5$, height $10$: volume and surface area.\soln $V=250\pi\approx785.4$; $SA=50\pi+100\pi=150\pi\approx471.2$.
\end{example}

\begin{example}{ex1}{Example 2 --- Cone}
Radius $3$, height $4$ (slant $5$): volume and surface area.\soln $V=12\pi\approx37.7$; $SA=9\pi+15\pi=24\pi\approx75.4$.
\end{example}

\begin{example}{ex2}{Example 3 --- Sphere}
Radius $6$: volume and surface area.\soln $V=288\pi\approx904.8$; $SA=144\pi\approx452.4$.
\end{example}

\begin{example}{ex3}{Example 4 --- Rectangular prism}
$4\times5\times6$: volume and surface area.\soln $V=120$; $SA=2(20+24+30)=148$.
\end{example}

\begin{example}{ex4}{Example 5 --- Square pyramid}
Base $6$, height $4$, slant height $5$: volume and surface area.\soln $V=\tfrac13(36)(4)=48$; $SA=36+4(\tfrac12\cdot6\cdot5)=96$.
\end{example}

\begin{example}{ex5}{Example 6 --- Cylinder volume}
Radius $4$, height $9$: volume.\soln $144\pi\approx452.4$.
\end{example}

\begin{example}{ex6}{Example 7 --- Larger cone}
Radius $6$, height $8$ (slant $10$): volume and surface area.\soln $V=96\pi\approx301.6$; $SA=36\pi+60\pi=96\pi\approx301.6$.
\end{example}

\begin{example}{ex7}{Example 8 --- Small sphere}
Radius $3$: volume and surface area.\soln $V=36\pi\approx113.1$; $SA=36\pi\approx113.1$.
\end{example}

\begin{example}{ex8}{Example 9 --- Find slant first}
A cone has radius $9$, height $12$. Find slant height.\soln $\sqrt{81+144}=15$.
\end{example}
