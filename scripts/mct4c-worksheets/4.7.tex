\begin{ideabox}
Split a solid into standard pieces, compute each volume, and add — or subtract a hole. Common combos: cylinder $+$ hemisphere (silo/capsule), cone $+$ cylinder, or a prism with a drilled hole.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Add the piece volumes (or subtract a hole).
  \item Hemisphere $V=\tfrac23\pi r^3$.
  \item Drilled hole subtracts a cylinder $\pi r^2 h$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Solids built from parts}Real objects combine solids — a silo is a cylinder plus a hemisphere, a machined block a prism with a drilled hole.
\lh{Add or subtract volumes}Total the parts' volumes, or subtract the volume of a hole; a hemisphere contributes $\tfrac23\pi r^3$.
\lh{Mind the shared faces}When solids join, ignore the internal faces where they meet — only outer surfaces count toward surface area.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Cylinder + hemisphere}
A cylinder ($r=3,h=10$) topped by a hemisphere ($r=3$). Volume?\soln $90\pi+18\pi=108\pi\approx339.3$.
\end{example}

\begin{example}{ex1}{Example 2 --- Cone on a cylinder}
A cone ($r=3,h=4$) on a cylinder ($r=3,h=6$). Volume?\soln $12\pi+54\pi=66\pi\approx207.3$.
\end{example}

\begin{example}{ex2}{Example 3 --- Drilled block}
A $10\times8\times6$ box with a cylindrical hole ($r=2$) drilled through the height $6$. Volume?\soln $480-24\pi\approx404.6$.
\end{example}

\begin{example}{ex3}{Example 4 --- Dome on a cylinder}
A hemisphere ($r=6$) on a cylinder ($r=6,h=10$). Volume?\soln $360\pi+144\pi=504\pi\approx1583.4$.
\end{example}

\begin{example}{ex4}{Example 5 --- Capsule}
A cylinder ($r=2,h=10$) with a hemisphere on each end. Volume?\soln $40\pi+\tfrac43\pi(8)=40\pi+\tfrac{32}{3}\pi\approx159.2$.
\end{example}

\begin{example}{ex5}{Example 6 --- Two prisms}
A $5\times4\times3$ box stacked on a $5\times4\times2$ box. Volume?\soln $60+40=100$.
\end{example}

\begin{example}{ex6}{Example 7 --- Ice-cream cone}
A cone ($r=3,h=8$) filled and topped by a hemisphere ($r=3$). Volume?\soln $24\pi+18\pi=42\pi\approx131.9$.
\end{example}

\begin{example}{ex7}{Example 8 --- Block with a bore}
A $12\times10\times8$ box with a cylindrical hole ($r=3$) through the height $8$. Volume?\soln $960-72\pi\approx733.8$.
\end{example}

\begin{example}{ex8}{Example 9 --- Silo}
A grain silo: cylinder ($r=4,h=15$) with a hemispherical top ($r=4$). Volume?\soln $240\pi+\tfrac23\pi(64)=240\pi+\tfrac{128}{3}\pi\approx888.0$.
\end{example}
