\begin{ideabox}
For a central angle $\theta$ (degrees): arc length $=\tfrac{\theta}{360}\,2\pi r$ and sector area $=\tfrac{\theta}{360}\,\pi r^2$. An inscribed angle is half the central angle on the same arc.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Arc $=\tfrac{\theta}{360}(2\pi r)$.
  \item Sector area $=\tfrac{\theta}{360}(\pi r^2)$.
  \item Inscribed angle $=\tfrac12\times$ central angle.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Arcs and sectors}For a central angle $\theta$ (degrees), the arc length is $\tfrac{\theta}{360}\,2\pi r$ and the sector area is $\tfrac{\theta}{360}\,\pi r^2$ — fractions of the whole circle.
\lh{Inscribed angles}An inscribed angle is half its central angle on the same arc, so an angle inscribed in a semicircle is $90^\circ$.
\lh{Whole-circle basics}Circumference $2\pi r$ and area $\pi r^2$ anchor every arc and sector calculation.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Arc length}
Arc of a $72^\circ$ sector, radius $10$.\soln $\tfrac{72}{360}(2\pi\cdot10)=4\pi\approx12.57$.
\end{example}

\begin{example}{ex1}{Example 2 --- Sector area}
Area of that $72^\circ$ sector.\soln $\tfrac{72}{360}(\pi\cdot100)=20\pi\approx62.83$.
\end{example}

\begin{example}{ex2}{Example 3 --- Bigger angle}
Arc of a $120^\circ$ sector, radius $6$.\soln $\tfrac13(2\pi\cdot6)=4\pi\approx12.57$.
\end{example}

\begin{example}{ex3}{Example 4 --- Small sector}
Area of a $45^\circ$ sector, radius $8$.\soln $\tfrac18(\pi\cdot64)=8\pi\approx25.13$.
\end{example}

\begin{example}{ex4}{Example 5 --- Whole circle}
Circumference and area for radius $7$.\soln $14\pi\approx43.98$; $49\pi\approx153.94$.
\end{example}

\begin{example}{ex5}{Example 6 --- Quarter arc}
Arc of a $90^\circ$ sector, radius $12$.\soln $6\pi\approx18.85$.
\end{example}

\begin{example}{ex6}{Example 7 --- Reflex sector}
Area of a $216^\circ$ sector, radius $5$.\soln $\tfrac{216}{360}(25\pi)=15\pi\approx47.12$.
\end{example}

\begin{example}{ex7}{Example 8 --- Inscribed angle}
A central angle is $80^\circ$; find the inscribed angle on the same arc.\soln $40^\circ$.
\end{example}

\begin{example}{ex8}{Example 9 --- From inscribed to central}
An inscribed angle is $35^\circ$; find the central angle.\soln $70^\circ$.
\end{example}
