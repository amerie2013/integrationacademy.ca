\begin{ideabox}
Set the equation to zero, factor completely, and use the zero-product property: a product is zero only when a factor is. Repeated factors give repeated roots; watch for a quadratic in $x^2$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Move to one side, then factor.
  \item Zero-product property gives the roots.
  \item Quadratic-in-$x^2$ and grouping tricks.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Set to zero, then factor}Move everything to one side, factor completely, and apply the zero-product property: a product is $0$ only when a factor is.
\lh{Repeated and hidden roots}A repeated factor gives a repeated root, while an equation that is quadratic in $x^2$ yields to a substitution.
\lh{Real roots only}Some factors are irreducible over the reals, so a degree-$n$ equation can have fewer than $n$ real solutions.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Repeated root}
Solve $x^3-3x^2+4=0$.\soln $(x-2)^2(x+1)=0\Rightarrow x=2\ (\text{double}),\,-1$.
\end{example}

\begin{example}{ex1}{Example 2 --- Three roots}
Solve $x^3+2x^2-5x-6=0$.\soln $(x+1)(x+3)(x-2)=0\Rightarrow x=-1,-3,2$.
\end{example}

\begin{example}{ex2}{Example 3 --- Quadratic in $x^2$}
Solve $x^4-5x^2+4=0$.\soln $(x^2-1)(x^2-4)=0\Rightarrow x=\pm1,\pm2$.
\end{example}

\begin{example}{ex3}{Example 4 --- Grouping}
Solve $2x^3-x^2-8x+4=0$.\soln $(2x-1)(x-2)(x+2)=0\Rightarrow x=\tfrac12,2,-2$.
\end{example}

\begin{example}{ex4}{Example 5 --- Common factor}
Solve $x^3-4x=0$.\soln $x(x-2)(x+2)=0\Rightarrow x=0,\pm2$.
\end{example}

\begin{example}{ex5}{Example 6 --- Simple product}
Solve $(x-3)(x+4)=0$.\soln $x=3,-4$.
\end{example}

\begin{example}{ex6}{Example 7 --- Cube factor}
Solve $x^3-9x=0$.\soln $x(x-3)(x+3)=0\Rightarrow x=0,\pm3$.
\end{example}

\begin{example}{ex7}{Example 8 --- Biquadratic}
Solve $x^4-13x^2+36=0$.\soln $(x^2-4)(x^2-9)=0\Rightarrow x=\pm2,\pm3$.
\end{example}

\begin{example}{ex8}{Example 9 --- Group and factor}
Solve $x^3-x^2-4x+4=0$.\soln $(x-1)(x-2)(x+2)=0\Rightarrow x=1,2,-2$.
\end{example}
