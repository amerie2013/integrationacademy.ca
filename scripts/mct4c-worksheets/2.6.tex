{\LARGE\bfseries\color{titleblue}Worksheet 2.6 --- The Factor Theorem}\\[2pt]
{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 2: Polynomial Functions}
\end{center}

\begin{ideabox}
$x-a$ is a factor of $P(x)$ exactly when $P(a)=0$. Test the factors of the constant term to find a first zero, divide it out, then factor the leftover quadratic.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $x-a$ a factor $\Leftrightarrow P(a)=0$.
  \item Candidate zeros are factors of the constant term.
  \item Divide, then factor completely.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Zero remainder means a factor}$x-a$ is a factor of $P(x)$ exactly when $P(a)=0$ â€” the factor theorem is the remainder theorem with remainder zero.
\lh{Finding a first factor}Test the factors of the constant term as candidate zeros; a value giving $P(a)=0$ hands you a factor to divide out.
\lh{Factoring completely}After dividing out one factor, factor the leftover quotient â€” often a quadratic â€” to finish the job.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Test a factor}
Is $x-2$ a factor of $x^3-3x^2+4$?\soln $P(2)=8-12+4=0$: yes.
\end{example}

\begin{example}{ex1}{Example 2 --- Factor completely}
Factor $x^3-3x^2+4$.\soln Divide by $x-2$: $x^2-x-2=(x-2)(x+1)$; so $(x-2)^2(x+1)$.
\end{example}

\begin{example}{ex2}{Example 3 --- Find a zero, factor}
Factor $x^3+2x^2-5x-6$.\soln $P(-1)=0$; $(x+1)(x^2+x-6)=(x+1)(x+3)(x-2)$.
\end{example}

\begin{example}{ex3}{Example 4 --- Candidates}
Factor $x^3-4x^2+x+6$.\soln $P(-1)=0$; $(x+1)(x^2-5x+6)=(x+1)(x-2)(x-3)$.
\end{example}

\begin{example}{ex4}{Example 5 --- Quartic}
Factor $x^4-1$.\soln $(x^2-1)(x^2+1)=(x-1)(x+1)(x^2+1)$.
\end{example}

\begin{example}{ex5}{Example 6 --- Common factor first}
Factor $x^3-x$.\soln $x(x-1)(x+1)$.
\end{example}

\begin{example}{ex6}{Example 7 --- Another cubic}
Factor $x^3-2x^2-x+2$.\soln $P(1)=0$; $(x-1)(x^2-x-2)=(x-1)(x-2)(x+1)$.
\end{example}

\begin{example}{ex7}{Example 8 --- Repeated factor}
Factor $x^3+3x^2-4$.\soln $P(1)=0$; $(x-1)(x^2+4x+4)=(x-1)(x+2)^2$.
\end{example}

\begin{example}{ex8}{Example 9 --- Non-monic}
Factor $2x^3-3x^2-11x+6$ given $x=3$ is a zero.\soln $(x-3)(2x^2+3x-2)=(x-3)(2x-1)(x+2)$.
\end{example}
