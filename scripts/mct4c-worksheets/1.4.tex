\begin{center}
{\LARGE\bfseries\color{titleblue}Worksheet 1.4 --- Solving Exponential Equations by Common Bases}\\[2pt]
{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 1: Exponential Functions}
\end{center}

\begin{ideabox}
If both sides can be written as powers of the same base, set the exponents equal. Watch for hidden powers: $4=2^2,\ 8=2^3,\ 27=3^3,\ \tfrac14=2^{-2}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Rewrite each side with a common base.
  \item Equate exponents and solve.
  \item Distribute powers over exponent expressions.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Matching the bases}If both sides can be written as powers of the \emph{same} base, their exponents must be equal — turning an exponential equation into a linear one.
\lh{Spot the hidden powers}Rewrite numbers as powers first: $8=2^3$, $27=3^3$, $\tfrac14=2^{-2}$. Then equate the exponents and solve.
\lh{When it applies}This works only when a common base exists; otherwise logarithms are needed, which come up next.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Same base}
Solve $2^{x}=32$.\soln $2^x=2^5\Rightarrow x=5$.
\end{example}

\begin{example}{ex1}{Example 2 --- Rewrite one side}
Solve $9^{x}=27$.\soln $3^{2x}=3^3\Rightarrow x=\tfrac32$.
\end{example}

\begin{example}{ex2}{Example 3 --- Exponent expression}
Solve $8^{\,x-1}=16$.\soln $2^{3(x-1)}=2^4\Rightarrow 3x-3=4\Rightarrow x=\tfrac73$.
\end{example}

\begin{example}{ex3}{Example 4 --- Fractional base}
Solve $\left(\tfrac14\right)^{x}=8$.\soln $2^{-2x}=2^3\Rightarrow x=-\tfrac32$.
\end{example}

\begin{example}{ex4}{Example 5 --- Variable both sides}
Solve $4^{\,x+1}=8^{\,x-1}$.\soln $2^{2x+2}=2^{3x-3}\Rightarrow 2x+2=3x-3\Rightarrow x=5$.
\end{example}

\begin{example}{ex5}{Example 6 --- Base 5}
Solve $25^{x}=125$.\soln $5^{2x}=5^3\Rightarrow x=\tfrac32$.
\end{example}

\begin{example}{ex6}{Example 7 --- Negative solution}
Solve $\left(\tfrac19\right)^{x}=27$.\soln $3^{-2x}=3^3\Rightarrow x=-\tfrac32$.
\end{example}

\begin{example}{ex7}{Example 8 --- Coefficient in exponent}
Solve $2^{\,3x}=32$.\soln $2^{3x}=2^5\Rightarrow x=\tfrac53$.
\end{example}

\begin{example}{ex8}{Example 9 --- Both sides powers}
Solve $27^{\,x-1}=9^{\,2x+1}$.\soln $3^{3(x-1)}=3^{2(2x+1)}\Rightarrow 3x-3=4x+2\Rightarrow x=-5$.
\end{example}
