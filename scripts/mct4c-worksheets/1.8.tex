\begin{center}
{\LARGE\bfseries\color{titleblue}Worksheet 1.8 --- Growth, Decay \& Compound Interest}\\[2pt]
{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 1: Exponential Functions}
\end{center}

\begin{ideabox}
Model constant-percentage change with $A=A_0\,b^{t}$: growth $b=1+r$, decay $b=1-r$. Half-life uses $\left(\tfrac12\right)^{t/h}$; compound interest is $A=P(1+i)^{n}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Growth/decay factor $b=1\pm r$.
  \item Half-life / doubling models.
  \item Compound interest $A=P(1+i)^{n}$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Percentage change over time}Repeated percentage change follows $A=A_0 b^{\,t}$: use $b=1+r$ for growth and $b=1-r$ for decay.
\lh{Half-life and doubling}A half-life model uses $\left(\tfrac12\right)^{t/h}$ and a doubling model uses $2^{\,t/d}$ — each tracking change over a fixed interval.
\lh{Compound interest}Interest that compounds grows by $A=P(1+i)^n$; solving for the number of periods $n$ calls for logarithms.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Population growth}
$P=500(1.03)^{t}$; find $P$ after 10 years.\soln $500(1.03)^{10}\approx672$.
\end{example}

\begin{example}{ex1}{Example 2 --- Half-life}
$80$ mg, half-life $6$ h. Model it; find the amount after $18$ h.\soln $A=80\left(\tfrac12\right)^{t/6}$; at $18$: $80\cdot\tfrac18=10$ mg.
\end{example}

\begin{example}{ex2}{Example 3 --- Compound interest}
\$2000 at $4\%$/yr, compounded annually, $5$ years.\soln $2000(1.04)^{5}\approx\$2433.31$.
\end{example}

\begin{example}{ex3}{Example 4 --- Doubling time}
At $6\%$/yr, how long to double?\soln $1.06^t=2\Rightarrow t=\dfrac{\log 2}{\log 1.06}\approx11.9$ yr.
\end{example}

\begin{example}{ex4}{Example 5 --- Decay to half}
$A=A_0(0.95)^{t}$; when is $50\%$ left?\soln $0.95^t=0.5\Rightarrow t\approx13.5$ yr.
\end{example}

\begin{example}{ex5}{Example 6 --- Growth value}
$800(1.02)^{t}$; find after $15$ years.\soln $\approx1077$.
\end{example}

\begin{example}{ex6}{Example 7 --- Half-life count}
$60$ mg, half-life $4$ h; amount after $12$ h.\soln $60\left(\tfrac12\right)^{3}=7.5$ mg.
\end{example}

\begin{example}{ex7}{Example 8 --- Depreciation}
A \$30\,000 car depreciates $12\%$/yr. When worth \$15\,000?\soln $(0.88)^t=0.5\Rightarrow t\approx5.4$ yr.
\end{example}

\begin{example}{ex8}{Example 9 --- Interest amount}
\$1500 at $5\%$/yr, $4$ years.\soln $1500(1.05)^{4}\approx\$1823.26$.
\end{example}
