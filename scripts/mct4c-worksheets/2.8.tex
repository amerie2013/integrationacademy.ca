{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 2: Polynomial Functions}
\end{center}

\begin{ideabox}
Model a situation with a polynomial (usually a volume or height), set the equation, solve by factoring, and keep only physically sensible roots (positive lengths).
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Translate the context into a polynomial.
  \item Solve by factoring.
  \item Discard out-of-range roots.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Modelling with polynomials}Volumes, projectile heights, and design quantities often form a polynomial; define the variable and write the relation.
\lh{Solving in context}Factor and solve, then keep only the physically sensible roots â€” lengths and times must be positive.
\lh{Interpreting the answer}Match the question to the feature: a maximum points to a turning point, a landing or break-even to a zero.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Open box}
Squares of side $x$ are cut from a $12\times12$ sheet. For what $x$ is the volume $128$?\soln $V=x(12-2x)^2$; $x=2$ gives $2(8)^2=128$ ($0<x<6$).
\end{example}

\begin{example}{ex1}{Example 2 --- Projectile}
$h(t)=-5t^2+30t$. When does it land?\soln $t(-5t+30)=0\Rightarrow t=6$ s.
\end{example}

\begin{example}{ex2}{Example 3 --- Volume from factors}
$V=x^3+4x^2+3x$; factor and find $V$ at $x=2$.\soln $x(x+1)(x+3)$; at $2$: $2\cdot3\cdot5=30$.
\end{example}

\begin{example}{ex3}{Example 4 --- Break-even}
Solve $x^3-7x-6=0$ for the positive root.\soln $(x-3)(x+1)(x+2)=0\Rightarrow x=3$.
\end{example}

\begin{example}{ex4}{Example 5 --- Beam deflection}
$d(x)=x^3-12x$. Where is $d=0$?\soln $x(x^2-12)=0\Rightarrow x=0,\ 2\sqrt3$.
\end{example}

\begin{example}{ex5}{Example 6 --- Landing time}
$h(t)=-5t^2+20t$. When does it land?\soln $t(-5t+20)=0\Rightarrow t=4$ s.
\end{example}

\begin{example}{ex6}{Example 7 --- Box dimensions}
A box has volume $x(x-1)(x+2)$; find $V$ at $x=3$.\soln $3\cdot2\cdot5=30$.
\end{example}

\begin{example}{ex7}{Example 8 --- Design equation}
Solve $x^3-9x=0$ for the positive length.\soln $x=3$.
\end{example}

\begin{example}{ex8}{Example 9 --- Profit roots}
Solve $x^3-x^2-4x+4=0$; give the positive roots.\soln $x=1,2$.
\end{example}
