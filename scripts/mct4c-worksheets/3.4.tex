{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 3: Trigonometric Functions}
\end{center}

\begin{ideabox}
Real problems (surveying, navigation, structures) reduce to a triangle: sketch it, label the given parts, choose the sine or cosine law, then answer the question. Bearings are measured clockwise from north.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Sketch and label before computing.
  \item SAS/SSS $\to$ cosine law; matched pair $\to$ sine law.
  \item Convert bearings/elevations into interior angles.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{From words to a triangle}Surveying and navigation problems become oblique triangles: sketch, label the known parts, then choose the sine or cosine law.
\lh{Bearings and elevations}Convert a bearing (clockwise from north) or an angle of elevation into an interior angle of the triangle.
\lh{Area from two sides}The area of any triangle from two sides and their included angle is $\tfrac12ab\sin C$.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Two ships}
Two ships leave port with a $60^\circ$ angle between courses, sailing $20$ km and $30$ km. How far apart?\soln $d=\sqrt{400+900-1200\cos60^\circ}=\sqrt{700}\approx26.5$ km.
\end{example}

\begin{example}{ex1}{Example 2 --- Survey a distance}
From two points $100$ m apart, a rock is seen at $40^\circ$ and $75^\circ$. Find its distance from the first point.\soln Third angle $65^\circ$; $d=\dfrac{100\sin75^\circ}{\sin65^\circ}\approx106.6$ m.
\end{example}

\begin{example}{ex2}{Example 3 --- Leaning pole}
A $12$ m pole leans; a $15$ m wire from its top meets the ground $9$ m from the base. Find the angle the pole makes with the ground.\soln $\cos\theta=\dfrac{12^2+9^2-15^2}{2(12)(9)}=0\Rightarrow\theta=90^\circ$.
\end{example}

\begin{example}{ex3}{Example 4 --- Triangular lot}
A lot has sides $40$ m, $55$ m with a $80^\circ$ angle between. Find the third side.\soln $\sqrt{1600+3025-4400\cos80^\circ}\approx62.1$ m.
\end{example}

\begin{example}{ex4}{Example 5 --- Height by angles}
Two points $50$ m apart on level ground sight a tower top at $30^\circ$ and $50^\circ$. Find the tower's slant distance from the nearer point.\soln Angle at tower $=20^\circ$; $d=\dfrac{50\sin30^\circ}{\sin20^\circ}\approx73.1$ m.
\end{example}

\begin{example}{ex5}{Example 6 --- Bearing problem}
A plane flies $200$ km, turns through an interior angle of $110^\circ$, flies $150$ km. Distance from start?\soln $\sqrt{200^2+150^2-2(200)(150)\cos110^\circ}\approx288$ km.
\end{example}

\begin{example}{ex6}{Example 7 --- Area of a lot}
Find the area of a triangular lot with sides $40,55$ and included $80^\circ$.\soln $\tfrac12(40)(55)\sin80^\circ\approx1083$ m$^2$.
\end{example}

\begin{example}{ex7}{Example 8 --- Guy wires}
Two guy wires reach a tower top from points $30$ m apart, at $58^\circ$ and $49^\circ$. Find the longer wire.\soln Angle at top $=73^\circ$; longer wire opposite the larger ground angle: $\dfrac{30\sin58^\circ}{\sin73^\circ}\approx26.6$ m.
\end{example}

\begin{example}{ex8}{Example 9 --- Distance across a lake}
From one point, two landmarks are $500$ m and $650$ m away with a $72^\circ$ angle between. Find the distance between landmarks.\soln $\sqrt{500^2+650^2-2(500)(650)\cos72^\circ}\approx686.8$ m.
\end{example}
