\begin{ideabox}
A polynomial has real coefficients and whole-number exponents. The \emph{degree} is the highest exponent, the \emph{leading coefficient} multiplies it, and \emph{standard form} lists terms from highest to lowest degree.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Whole-number exponents only (no $\sqrt{x}$, no $\tfrac1x$).
  \item Degree $=$ highest exponent; leading coefficient multiplies it.
  \item Evaluate with function notation $P(a)$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{What counts as a polynomial}A \textbf{polynomial} uses only whole-number exponents and real coefficients — no variables under roots or in denominators.
\lh{Degree and leading coefficient}The \textbf{degree} is the highest exponent and the \textbf{leading coefficient} multiplies it; together they set much of the function's behaviour.
\lh{Evaluating}Function notation $P(a)$ means substitute $x=a$ — the basis for the remainder and factor theorems later in this unit.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Identify polynomials}
Which are polynomials: $x^3-5x^2+1$, $2^{x}$, $\sqrt{x}+3$, $4x^{-2}$?\soln Only $x^3-5x^2+1$ (the others have a variable, fractional, or negative exponent).
\end{example}

\begin{example}{ex1}{Example 2 --- Degree and leading coefficient}
State the degree and leading coefficient of $-4x^5+2x^3-7$.\soln Degree $5$, leading coefficient $-4$.
\end{example}

\begin{example}{ex2}{Example 3 --- Evaluate}
For $P(x)=2x^3-x+5$, find $P(2)$ and $P(-1)$.\soln $16-2+5=19$; $-2+1+5=4$.
\end{example}

\begin{example}{ex3}{Example 4 --- Standard form}
Write $3x-5x^4+2-x^2$ in standard form.\soln $-5x^4-x^2+3x+2$.
\end{example}

\begin{example}{ex4}{Example 5 --- Classify}
Classify $4x^3-2x+7$.\soln Cubic (degree 3) trinomial (3 terms).
\end{example}

\begin{example}{ex5}{Example 6 --- Evaluate a quartic}
For $P(x)=x^4-3x^2+2x$, find $P(-2)$.\soln $16-12-4=0$.
\end{example}

\begin{example}{ex6}{Example 7 --- Not a polynomial}
Is $3x^2-\tfrac1x$ a polynomial? Why?\soln No — $\tfrac1x=x^{-1}$ has a negative exponent.
\end{example}

\begin{example}{ex7}{Example 8 --- Degree of a product}
What is the degree of $(x^2+1)(x^3-x)$?\soln $2+3=5$.
\end{example}

\begin{example}{ex8}{Example 9 --- Constant and coefficient}
For $P(x)=6-2x^3+x^7$, give the degree, leading coefficient, and constant term.\soln Degree $7$, leading coefficient $1$, constant $6$.
\end{example}
