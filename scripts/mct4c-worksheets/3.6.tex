{\LARGE\bfseries\color{titleblue}Worksheet 3.6 --- Transformations of Sinusoidal Functions}\\[2pt]

\begin{ideabox}
In $y=a\sin\!\big(k(x-d)\big)+c$: $|a|$ is the amplitude, $\dfrac{360^\circ}{|k|}$ the period, $d$ the horizontal (phase) shift, and $c$ the vertical shift. Factor out $k$ before reading $d$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Amplitude $|a|$, period $\tfrac{360^\circ}{|k|}$.
  \item $d$: phase shift; $c$: vertical shift.
  \item Build an equation from amplitude, period, and midline.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The full form}$y=a\sin\!\left(k(x-d)\right)+c$ carries every feature: amplitude $|a|$, period $\tfrac{360^\circ}{|k|}$, phase shift $d$, and midline $c$.
\lh{Factor before reading}Factor out $k$ so the inside reads $k(x-d)$; only then does $d$ give the true horizontal shift.
\lh{Max, min, range}The maximum is $c+|a|$ and the minimum $c-|a|$, so the range runs between them.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Describe fully}
Describe $y=2\sin(x-30^\circ)+1$.\soln Amplitude $2$, right $30^\circ$, up $1$; midline $y=1$.
\end{example}

\begin{example}{ex1}{Example 2 --- Factor for phase}
Phase shift and period of $y=3\cos(2(x-45^\circ))$.\soln Right $45^\circ$; period $180^\circ$.
\end{example}

\begin{example}{ex2}{Example 3 --- Reflection + shift}
Describe $y=-\sin x+4$.\soln Amplitude $1$, reflected, midline $y=4$.
\end{example}

\begin{example}{ex3}{Example 4 --- Everything at once}
For $y=4\sin(3x)-2$: amplitude, period, midline, max, min.\soln $4;\ 120^\circ;\ y=-2;\ 2;\ -6$.
\end{example}

\begin{example}{ex4}{Example 5 --- Left shift}
Describe $y=2\cos(x+60^\circ)$.\soln Amplitude $2$, left $60^\circ$.
\end{example}

\begin{example}{ex5}{Example 6 --- Max/min}
Max and min of $y=-4\sin(2x)+1$.\soln Max $5$; min $-3$.
\end{example}

\begin{example}{ex6}{Example 7 --- Read phase from factored form}
Phase shift of $y=\sin(2(x-90^\circ))$.\soln Right $90^\circ$.
\end{example}

\begin{example}{ex7}{Example 8 --- Build an equation}
Write a sine function with amplitude $3$, period $180^\circ$, midline $y=2$.\soln $y=3\sin(2x)+2$ (since $k=\tfrac{360}{180}=2$).
\end{example}

\begin{example}{ex8}{Example 9 --- Range}
State the range of $y=5\sin x-1$.\soln $[-6,4]$.
\end{example}
