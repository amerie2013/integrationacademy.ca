{\LARGE\bfseries\color{titleblue}Worksheet 3.7 --- Modelling with Sinusoidal Functions}\\[2pt]
{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 3: Trigonometric Functions}
\end{center}

\begin{ideabox}
Turn repeating real data (tides, Ferris wheels, temperatures, daylight) into $y=a\sin\!\big(k(x-d)\big)+c$ or a cosine model: midline $c=\tfrac{\max+\min}{2}$, amplitude $a=\tfrac{\max-\min}{2}$, and $k=\tfrac{360^\circ}{\text{period}}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $c=\tfrac{\max+\min}{2}$, $a=\tfrac{\max-\min}{2}$.
  \item $k=\tfrac{360^\circ}{\text{period}}$ (time in the unit given).
  \item Use $-\cos$ to start at a minimum, $+\cos$ to start at a maximum.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Turning cycles into equations}Repeating data — tides, wheels, temperatures — fits $y=a\sin\!\left(k(x-d)\right)+c$ with $c=\tfrac{\max+\min}{2}$ and $a=\tfrac{\max-\min}{2}$.
\lh{Setting the period}Read $k$ from the cycle length: $k=\dfrac{360^\circ}{\text{period}}$, keeping time in the stated units.
\lh{Choosing the start}Use $-\cos$ to begin at a minimum and $+\cos$ to begin at a maximum, matching the model to the situation.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Ferris wheel model}
A wheel of radius $10$ m has its centre $12$ m up and one turn every $60$ s, starting at the bottom. Write $h(t)$.\soln $k=\tfrac{360}{60}=6$; starts at min: $h=12-10\cos(6t)$ ($t$ in s).
\end{example}

\begin{example}{ex1}{Example 2 --- Value from the model}
Using the model, find the height at $t=15$ s.\soln $12-10\cos(90^\circ)=12$ m.
\end{example}

\begin{example}{ex2}{Example 3 --- Max and min heights}
Give the max and min heights of that wheel.\soln $22$ m and $2$ m.
\end{example}

\begin{example}{ex3}{Example 4 --- Tide amplitude/midline}
High tide $8$ m, low tide $2$ m. Find the amplitude and midline.\soln Amplitude $3$; midline $5$.
\end{example}

\begin{example}{ex4}{Example 5 --- Tide period}
A tide cycle takes $12$ h. Find $k$ (degrees per hour).\soln $k=\tfrac{360}{12}=30$.
\end{example}

\begin{example}{ex5}{Example 6 --- Temperature model}
Daily high $25^\circ$, low $5^\circ$. Amplitude and midline?\soln Amplitude $10$; midline $15$.
\end{example}

\begin{example}{ex6}{Example 7 --- Write a tide model}
Tide: max $8$ m, min $2$ m, period $12$ h, starting at high tide. Write $h(t)$.\soln $h=5+3\cos(30t)$.
\end{example}

\begin{example}{ex7}{Example 8 --- Start-at-minimum model}
A quantity ranges $2$ to $8$, period $4$ s, starting at the minimum. Write it.\soln Midline $5$, amp $3$, $k=90$: $y=5-3\cos(90t)$.
\end{example}

\begin{example}{ex8}{Example 9 --- Interpret}
For $h=5+3\cos(30t)$, when is the first low tide?\soln $\cos(30t)=-1\Rightarrow 30t=180^\circ\Rightarrow t=6$ h.
\end{example}
