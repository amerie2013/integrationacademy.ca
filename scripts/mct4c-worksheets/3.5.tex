\begin{ideabox}
For $y=a\sin x+c$ (or $\cos$): amplitude $=|a|$, period $=360^\circ$, midline $y=c$, maximum $c+|a|$, minimum $c-|a|$. A coefficient $k$ on $x$ changes the period to $\dfrac{360^\circ}{|k|}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Amplitude $=|a|$; midline $y=c$.
  \item Period $=\dfrac{360^\circ}{|k|}$.
  \item Max $=c+|a|$, min $=c-|a|$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The wave shape}Sine and cosine graphs are smooth repeating waves; describe each by its \textbf{amplitude}, \textbf{period}, and \textbf{midline}.
\lh{Reading the numbers}In $y=a\sin x+c$, the amplitude is $|a|$, the midline is $y=c$, and in degrees the base period is $360^\circ$.
\lh{Effect of a coefficient}A coefficient $k$ on $x$ changes the period to $\dfrac{360^\circ}{|k|}$ — the larger $k$, the shorter the waves.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Basic sine}
State the amplitude, period, max, min of $y=3\sin x$.\soln $3,\ 360^\circ,\ 3,\ -3$.
\end{example}

\begin{example}{ex1}{Example 2 --- Vertical shift}
For $y=2\sin x+5$, give the midline, max, min.\soln Midline $y=5$; max $7$; min $3$.
\end{example}

\begin{example}{ex2}{Example 3 --- Changed period}
Find the period of $y=4\cos(2x)$.\soln $\dfrac{360^\circ}{2}=180^\circ$; amplitude $4$.
\end{example}

\begin{example}{ex3}{Example 4 --- Small $k$}
Find the period of $y=\sin(0.5x)$.\soln $\dfrac{360^\circ}{0.5}=720^\circ$.
\end{example}

\begin{example}{ex4}{Example 5 --- Reflection}
Describe $y=-3\sin x$.\soln Amplitude $3$, reflected in the $x$-axis.
\end{example}

\begin{example}{ex5}{Example 6 --- Combined}
For $y=5\cos x-2$, give max and min.\soln Max $3$; min $-7$.
\end{example}

\begin{example}{ex6}{Example 7 --- Period only}
Period of $y=\cos(3x)$?\soln $120^\circ$.
\end{example}

\begin{example}{ex7}{Example 8 --- Read both}
Amplitude and period of $y=6\sin(4x)$.\soln $6$ and $90^\circ$.
\end{example}

\begin{example}{ex8}{Example 9 --- Key point}
For $y=2\cos x$, where is the first maximum?\soln At $x=0^\circ$, $y=2$.
\end{example}
