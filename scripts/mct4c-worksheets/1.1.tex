{\LARGE\bfseries\color{titleblue}Worksheet 1.1 --- Exponent Laws}\\[2pt]
{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 1: Exponential Functions}
\end{center}

\begin{ideabox}
Same base: multiply $\Rightarrow$ add exponents; divide $\Rightarrow$ subtract; power of a power $\Rightarrow$ multiply. A negative exponent is a reciprocal ($a^{-n}=\tfrac{1}{a^n}$); a rational exponent is a root ($a^{m/n}=\sqrt[n]{a^m}$).
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Product/quotient/power laws (same base only).
  \item $a^{0}=1,\ a^{-n}=\dfrac{1}{a^n}$.
  \item $a^{m/n}=\left(\sqrt[n]{a}\right)^{m}$ â€” take the root first.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Powers as shorthand}A power packs repeated multiplication into one symbol: $a^n$ means $a$ used as a factor $n$ times. The exponent laws let you combine such powers without expanding them.
\lh{The core rules}For a shared base: multiply by adding exponents ($a^m a^n=a^{m+n}$), divide by subtracting ($\tfrac{a^m}{a^n}=a^{m-n}$), and raise a power by multiplying ($(a^m)^n=a^{mn}$).
\lh{Zero, negative, fractional}Three special cases finish the toolkit: $a^0=1$, $a^{-n}=\dfrac{1}{a^n}$, and $a^{m/n}=\sqrt[n]{a^m}$ â€” a root written as an exponent.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Multi-variable simplification}
Simplify $\dfrac{(3x^2y^3)^2(2xy^4)^3}{6x^3y^5}$.\soln Numerator $=9x^4y^6\cdot8x^3y^{12}=72x^7y^{18}$; divide: $12x^4y^{13}$.
\end{example}

\begin{example}{ex1}{Example 2 --- Negative exponents}
Simplify $(2a^{-3}b^2)^{-2}\cdot a^5b^{-1}$ with positive exponents.\soln $\tfrac14a^6b^{-4}\cdot a^5b^{-1}=\tfrac14a^{11}b^{-5}=\dfrac{a^{11}}{4b^5}$.
\end{example}

\begin{example}{ex2}{Example 3 --- Rational powers by hand}
Evaluate $27^{2/3}$, $16^{-3/4}$, $\left(\tfrac{8}{27}\right)^{-2/3}$.\soln $(\sqrt[3]{27})^2=9$; $\tfrac{1}{2^3}=\tfrac18$; $\left(\tfrac{27}{8}\right)^{2/3}=\left(\tfrac32\right)^2=\tfrac94$.
\end{example}

\begin{example}{ex3}{Example 4 --- Rational exponents, variables}
Simplify $\dfrac{(x^{1/2}y^{-2})^4}{x^3y^{-5}}$.\soln $\dfrac{x^2y^{-8}}{x^3y^{-5}}=x^{-1}y^{-3}=\dfrac{1}{xy^3}$.
\end{example}

\begin{example}{ex4}{Example 5 --- Quotient raised to a power}
Simplify $\left(\dfrac{a^{2/3}b^{-1/2}}{a^{-1/3}b^{1/2}}\right)^{6}$.\soln Inside: $a^{1}b^{-1}$; to the 6th: $a^6b^{-6}=\dfrac{a^6}{b^6}$.
\end{example}

\begin{example}{ex5}{Example 6 --- Mixed evaluation}
Evaluate $8^{2/3}\cdot 9^{-1/2}\cdot 5^{0}$.\soln $4\cdot\tfrac13\cdot1=\tfrac43$.
\end{example}

\begin{example}{ex6}{Example 7 --- Reciprocal of a product}
Simplify $(4x^{-2}y^5)^{-1}$ with positive exponents.\soln $4^{-1}x^2y^{-5}=\dfrac{x^2}{4y^5}$.
\end{example}

\begin{example}{ex7}{Example 8 --- Power then product}
Simplify $(2x^3)^4\cdot x^{-5}$.\soln $16x^{12}\cdot x^{-5}=16x^7$.
\end{example}

\begin{example}{ex8}{Example 9 --- Combine like bases}
Simplify $\dfrac{a^5b^2\cdot a^2b^4}{a^3b}$.\soln $\dfrac{a^7b^6}{a^3b}=a^4b^5$.
\end{example}
