{\LARGE\bfseries\color{titleblue}Worksheet 3.1 --- Trigonometric Ratios to 360°}\\[2pt]
{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 3: Trigonometric Functions}
\end{center}

\begin{ideabox}
Use \emph{CAST} for signs by quadrant and the \emph{reference angle} (the acute angle to the $x$-axis) to reduce any angle to a special-angle value. Exact values come from the $30^\circ$-$60^\circ$-$90^\circ$ and $45^\circ$-$45^\circ$-$90^\circ$ triangles.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item CAST: which of $\sin,\cos,\tan$ is positive in each quadrant.
  \item Reference angle to the $x$-axis reduces to a special angle.
  \item Solve equations by combining reference angle with quadrants.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Angles all the way round}Trig ratios extend past $90^\circ$ using a \textbf{reference angle} to a special value, plus a sign set by the quadrant.
\lh{The CAST rule}\textbf{CAST} shows which ratio is positive where: all in QI, sine in QII, tangent in QIII, cosine in QIV.
\lh{Solving for angles}To solve $\cos\theta=-\tfrac12$, find the reference angle, then list every quadrant whose sign matches.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Reference angles}
Find the reference angle of $150^\circ$, $210^\circ$, $300^\circ$.\soln $30^\circ,\ 30^\circ,\ 60^\circ$.
\end{example}

\begin{example}{ex1}{Example 2 --- Signs by quadrant}
In which quadrants is $\cos\theta<0$?\soln Quadrants II and III.
\end{example}

\begin{example}{ex2}{Example 3 --- Exact value}
Find $\sin150^\circ$, $\cos150^\circ$, $\tan150^\circ$ exactly.\soln $\tfrac12,\ -\tfrac{\sqrt3}{2},\ -\tfrac{\sqrt3}{3}$.
\end{example}

\begin{example}{ex3}{Example 4 --- From a ratio}
$\sin\theta=\tfrac35$ with $\theta$ in QII. Find $\cos\theta,\tan\theta$.\soln $\cos\theta=-\tfrac45,\ \tan\theta=-\tfrac34$.
\end{example}

\begin{example}{ex4}{Example 5 --- More exact values}
Find $\cos225^\circ$ and $\sin300^\circ$.\soln $-\tfrac{\sqrt2}{2}$ and $-\tfrac{\sqrt3}{2}$.
\end{example}

\begin{example}{ex5}{Example 6 --- Solve a sine equation}
Solve $\sin\theta=\tfrac12$ for $0^\circ\le\theta<360^\circ$.\soln Reference $30^\circ$; positive in QI, QII: $\theta=30^\circ,150^\circ$.
\end{example}

\begin{example}{ex6}{Example 7 --- Solve a cosine equation}
Solve $\cos\theta=-\tfrac12$.\soln Reference $60^\circ$; negative in QII, QIII: $\theta=120^\circ,240^\circ$.
\end{example}

\begin{example}{ex7}{Example 8 --- Solve a tangent equation}
Solve $\tan\theta=1$.\soln Reference $45^\circ$; positive in QI, QIII: $\theta=45^\circ,225^\circ$.
\end{example}

\begin{example}{ex8}{Example 9 --- From a ratio, QIII}
$\cos\theta=-\tfrac{5}{13}$ with $\theta$ in QIII. Find $\sin\theta,\tan\theta$.\soln $\sin\theta=-\tfrac{12}{13},\ \tan\theta=\tfrac{12}{5}$.
\end{example}
