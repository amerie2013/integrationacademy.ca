{\LARGE\bfseries\color{titleblue}Worksheet 2.3 --- Factored Form \& Zeros}\\[2pt]
{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 2: Polynomial Functions}
\end{center}

\begin{ideabox}
Each factor $(x-r)$ names a zero at $x=r$. Odd multiplicity crosses the axis; even multiplicity touches. Build an equation with one factor per zero and a leading coefficient $a$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $(x-r)$ $\Rightarrow$ zero at $x=r$.
  \item Odd multiplicity crosses; even touches.
  \item Find $a$ by substituting a known point.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Factors reveal zeros}In $y=a(x-r)(x-s)\cdots$, each factor $(x-r)$ marks a zero at $x=r$ — the values that make the polynomial $0$.
\lh{Multiplicity and shape}A repeated factor gives a repeated zero: odd multiplicity crosses the axis, even multiplicity only touches it.
\lh{Fixing the leading coefficient}One extra known point determines $a$, pinning down the exact polynomial through those zeros.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Read the zeros}
Zeros of $y=(x-2)(x+3)(x-1)$?\soln $x=2,-3,1$.
\end{example}

\begin{example}{ex1}{Example 2 --- Multiplicity}
Behaviour of $y=(x-1)^2(x+2)$ at each zero.\soln $x=1$ (even): touches; $x=-2$ (odd): crosses.
\end{example}

\begin{example}{ex2}{Example 3 --- Build from zeros}
Write a cubic with zeros $-1,2,3$.\soln $y=a(x+1)(x-2)(x-3)$.
\end{example}

\begin{example}{ex3}{Example 4 --- Find $a$}
The cubic above passes $(0,12)$. Find $a$.\soln $12=a(1)(-2)(-3)=6a\Rightarrow a=2$.
\end{example}

\begin{example}{ex4}{Example 5 --- Triple root}
Behaviour of $y=x(x-2)^3$ at $x=2$.\soln Multiplicity 3 (odd): crosses but flattens.
\end{example}

\begin{example}{ex5}{Example 6 --- Zeros with a quadratic factor}
Zeros of $y=(x-4)(x+1)^2$?\soln $x=4$ and $x=-1$ (double).
\end{example}

\begin{example}{ex6}{Example 7 --- Quartic zeros}
Zeros of $y=x^2(x-3)(x+2)$?\soln $x=0$ (double), $3,-2$.
\end{example}

\begin{example}{ex7}{Example 8 --- Write with a double}
Write a cubic with a double zero at $2$ and a zero at $-1$.\soln $y=a(x-2)^2(x+1)$.
\end{example}

\begin{example}{ex8}{Example 9 --- Find $a$ from a point}
A cubic with zeros $0,1,-2$ passes $(2,8)$. Find $a$.\soln $8=a(2)(1)(4)=8a\Rightarrow a=1$.
\end{example}

\begin{keybox}
\begin{multicols}{2}
\small
\begin{enumerate}[leftmargin=*,itemsep=3pt]
  \item $-4,5$
  \item touches
  \item $y=a(x-2)(x+5)$
  \item $a=1$
  \item touches (flattens)
  \item $0,4,-3$
  \item crosses (flattens)
  \item $y=a(x+3)x(x-3)$
  \item $2$ (double), $-5$
  \item $-16=a(-4)(4)=-16a\Rightarrow a=1$
  \item crosses (flattens)
  \item $y=a(x+1)^2(x-4)$
  \item $4=a(-1)(1)(-2)=2a\Rightarrow a=2$
\end{enumerate}
\end{multicols}
\end{keybox}
