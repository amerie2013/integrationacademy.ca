{\LARGE\bfseries\color{titleblue}Worksheet 4.2 --- Adding \& Subtracting Vectors}\\[2pt]
{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 4: Vectors \& Geometry}
\end{center}

\begin{ideabox}
Add or subtract vectors component-by-component, then take the magnitude. For two vectors with angle $\theta$ between them, the resultant magnitude is $\sqrt{a^2+b^2+2ab\cos\theta}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Add/subtract components, then find magnitude.
  \item Resultant $|\vec R|=\sqrt{a^2+b^2+2ab\cos\theta}$.
  \item Perpendicular vectors ($\theta=90^\circ$) reduce to $\sqrt{a^2+b^2}$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Combining component-wise}Add or subtract vectors by combining their $x$- and $y$-components separately, then take the magnitude of the result.
\lh{The resultant}Geometrically, vectors join tip-to-tail; the single vector from start to finish is the \textbf{resultant}.
\lh{Angle between them}For two vectors at angle $\theta$, the resultant magnitude is $\sqrt{a^2+b^2+2ab\cos\theta}$ — largest when they are parallel.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Add components}
Find $(3,4)+(1,2)$ and its magnitude.\soln $(4,6)$; $\sqrt{52}\approx7.21$.
\end{example}

\begin{example}{ex1}{Example 2 --- Subtract components}
Find $(5,2)-(1,6)$ and its magnitude.\soln $(4,-4)$; $\sqrt{32}\approx5.66$.
\end{example}

\begin{example}{ex2}{Example 3 --- Resultant at 60°}
Two vectors, magnitudes $8$ and $5$, have $60^\circ$ between them. Find the resultant.\soln $\sqrt{64+25+2(8)(5)\cos60^\circ}=\sqrt{129}\approx11.4$.
\end{example}

\begin{example}{ex3}{Example 4 --- Resultant at 120°}
Same magnitudes but $120^\circ$ apart.\soln $\sqrt{89+80\cos120^\circ}=\sqrt{49}=7$.
\end{example}

\begin{example}{ex4}{Example 5 --- Perpendicular}
Resultant of magnitudes $6$ and $8$ at right angles.\soln $\sqrt{36+64}=10$.
\end{example}

\begin{example}{ex5}{Example 6 --- Linear combination}
With $\vec u=(1,2),\vec v=(3,-1)$, find $2\vec u+\vec v$ and its magnitude.\soln $(5,3)$; $\sqrt{34}\approx5.83$.
\end{example}

\begin{example}{ex6}{Example 7 --- Difference vector}
With $\vec u=(2,5),\vec v=(4,1)$, find $\vec u-\vec v$ and its magnitude.\soln $(-2,4)$; $\sqrt{20}\approx4.47$.
\end{example}

\begin{example}{ex7}{Example 8 --- Sum in different quadrants}
Find $(-2,3)+(5,-1)$ and its magnitude.\soln $(3,2)$; $\sqrt{13}\approx3.61$.
\end{example}

\begin{example}{ex8}{Example 9 --- Equal magnitudes, right angle}
Resultant of two magnitude-$10$ vectors at $90^\circ$.\soln $\sqrt{200}\approx14.1$.
\end{example}
