\begin{ideabox}
Division gives $P(x)=D(x)Q(x)+R$. Synthetic division (for divisor $x-a$): write coefficients (zeros for missing terms), bring down, then multiply-by-$a$ and add.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $P(x)=D(x)Q(x)+R$.
  \item Synthetic division for $x-a$.
  \item Insert $0$ for missing terms.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The division statement}Dividing gives $P(x)=D(x)Q(x)+R$, where the remainder $R$ has a lower degree than the divisor $D$.
\lh{Synthetic division}For a divisor $x-a$, synthetic division is a fast shortcut: bring down, multiply by $a$, add, and repeat.
\lh{Watch the gaps}Insert a $0$ coefficient for any missing power so the columns stay aligned throughout the division.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Divide out a factor}
Divide $(x^3-2x^2-5x+6)\div(x-1)$.\soln Synthetic ($a=1$): quotient $x^2-x-6$, remainder $0$.
\end{example}

\begin{example}{ex1}{Example 2 --- Non-zero remainder}
Divide $(2x^3+3x^2-1)\div(x+2)$.\soln $a=-2$: quotient $2x^2-x+2$, remainder $-5$.
\end{example}

\begin{example}{ex2}{Example 3 --- Find a factor}
Divide $(x^3-7x-6)\div(x+1)$.\soln $a=-1$: quotient $x^2-x-6$, remainder $0$.
\end{example}

\begin{example}{ex3}{Example 4 --- Difference of cubes}
Divide $(x^3-1)\div(x-1)$.\soln $x^2+x+1$, remainder $0$.
\end{example}

\begin{example}{ex4}{Example 5 --- Higher degree}
Divide $(x^4-16)\div(x-2)$.\soln $x^3+2x^2+4x+8$, remainder $0$.
\end{example}

\begin{example}{ex5}{Example 6 --- Quadratic dividend}
Divide $(2x^2-3x+1)\div(x-2)$.\soln $2x+1$, remainder $3$.
\end{example}

\begin{example}{ex6}{Example 7 --- Missing term}
Divide $(x^3+2x^2-3)\div(x-1)$.\soln $x^2+3x+3$, remainder $0$.
\end{example}

\begin{example}{ex7}{Example 8 --- Write the identity}
Write $2x^3+3x^2-1$ using its division by $x+2$.\soln $(x+2)(2x^2-x+2)-5$.
\end{example}

\begin{example}{ex8}{Example 9 --- Simple factor}
Divide $(x^2+5x+6)\div(x+2)$.\soln $x+3$, remainder $0$.
\end{example}
