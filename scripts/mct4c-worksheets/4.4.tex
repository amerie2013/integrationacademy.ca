{\LARGE\bfseries\color{titleblue}Worksheet 4.4 --- Composite Two-Dimensional Areas}\\[2pt]
{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 4: Vectors \& Geometry}
\end{center}

\begin{ideabox}
Break a composite region into rectangles, triangles, trapezoids, and circles; add the pieces, or subtract a cut-out. Key areas: rectangle $lw$, triangle $\tfrac12bh$, trapezoid $\tfrac12(a+b)h$, circle $\pi r^2$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Decompose into standard shapes.
  \item Add pieces or subtract holes.
  \item Semicircle $=\tfrac12\pi r^2$; annulus $=\pi(R^2-r^2)$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Decompose the region}Break a complex shape into rectangles, triangles, trapezoids, and circles, then add the pieces — or subtract a cut-out.
\lh{Formulas to keep handy}Rectangle $lw$, triangle $\tfrac12bh$, trapezoid $\tfrac12(a+b)h$, and circle $\pi r^2$ cover most composite regions.
\lh{Add or subtract}Removed or hollow parts are subtracted, while a semicircular end or an attached triangle is added.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- L-shape (subtract)}
A $10\times8$ rectangle with a $4\times3$ corner removed. Area?\soln $80-12=68$.
\end{example}

\begin{example}{ex1}{Example 2 --- Rectangle + semicircle}
A $12\times7$ rectangle capped by a semicircle of diameter $7$. Area?\soln $84+\tfrac12\pi(3.5)^2\approx84+19.2=103.2$.
\end{example}

\begin{example}{ex2}{Example 3 --- Trapezoid}
Parallel sides $8$ and $14$, height $6$. Area?\soln $\tfrac12(8+14)(6)=66$.
\end{example}

\begin{example}{ex3}{Example 4 --- Annulus}
Region between circles of radius $6$ and $4$. Area?\soln $\pi(36-16)=20\pi\approx62.8$.
\end{example}

\begin{example}{ex4}{Example 5 --- Stadium (track)}
A $20\times12$ rectangle with a semicircle on each short end ($r=6$). Area?\soln $240+\pi(6)^2\approx240+113.1=353.1$.
\end{example}

\begin{example}{ex5}{Example 6 --- Triangle}
Base $10$, height $8$. Area?\soln $\tfrac12(10)(8)=40$.
\end{example}

\begin{example}{ex6}{Example 7 --- Two rectangles}
A $6\times4$ rectangle joined to a $3\times2$ rectangle. Area?\soln $24+6=30$.
\end{example}

\begin{example}{ex7}{Example 8 --- Parallelogram}
Base $9$, height $5$. Area?\soln $45$.
\end{example}

\begin{example}{ex8}{Example 9 --- Square minus circle}
A square of side $10$ with an inscribed circle ($r=5$) removed. Area?\soln $100-25\pi\approx100-78.5=21.5$.
\end{example}
