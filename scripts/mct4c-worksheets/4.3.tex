\begin{ideabox}
Forces and velocities add as vectors. Resolve into components or use the resultant law; perpendicular cases (crosswinds, currents) use $\sqrt{a^2+b^2}$ with direction $\tan^{-1}\tfrac{\text{cross}}{\text{main}}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Velocities/forces combine as vectors.
  \item Crosswind/current: $\sqrt{a^2+b^2}$ and a drift angle.
  \item Two equal forces at angle $2\alpha$: resultant $=2F\cos\alpha$ along the bisector.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Forces and velocities}Real problems combine vectors: a plane's velocity plus a wind, or several forces on an object, add to a single resultant.
\lh{Perpendicular components}Crosswinds and currents meet the main motion at right angles, so the resultant speed is $\sqrt{a^2+b^2}$ with drift angle $\tan^{-1}\tfrac{\text{cross}}{\text{main}}$.
\lh{Equilibrium}When forces balance, their resultant is zero; the \textbf{equilibrant} is the single force that produces that balance.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Crosswind}
A plane heads at $400$ km/h with a $60$ km/h perpendicular crosswind. Find the ground speed and drift.\soln $\sqrt{400^2+60^2}\approx404.5$ km/h; drift $\tan^{-1}\tfrac{60}{400}\approx8.5^\circ$.
\end{example}

\begin{example}{ex1}{Example 2 --- Perpendicular forces}
Two forces $10$ N and $15$ N act at right angles. Find the resultant.\soln $\sqrt{100+225}\approx18.0$ N.
\end{example}

\begin{example}{ex2}{Example 3 --- Forces at 60°}
Forces $12$ N and $18$ N have $60^\circ$ between. Resultant?\soln $\sqrt{144+324+2(12)(18)\cos60^\circ}\approx26.2$ N.
\end{example}

\begin{example}{ex3}{Example 4 --- River crossing}
A boat rows $8$ km/h straight across; the current is $3$ km/h. Find the actual speed and angle.\soln $\sqrt{64+9}\approx8.5$ km/h; $\tan^{-1}\tfrac38\approx20.6^\circ$ downstream.
\end{example}

\begin{example}{ex4}{Example 5 --- Wind from the side}
A plane flies $300$ km/h north into a $50$ km/h east wind. Ground speed?\soln $\sqrt{300^2+50^2}\approx304.1$ km/h.
\end{example}

\begin{example}{ex5}{Example 6 --- Resolve a force}
Resolve a $25$ N force at $40^\circ$ into horizontal and vertical parts.\soln $(25\cos40^\circ,25\sin40^\circ)\approx(19.2,16.1)$ N.
\end{example}

\begin{example}{ex6}{Example 7 --- Two tugboats}
Two tugboats each pull $5000$ N, at $30^\circ$ on either side of a barge's axis. Find the forward pull.\soln $2(5000)\cos30^\circ\approx8660$ N.
\end{example}

\begin{example}{ex7}{Example 8 --- Equilibrant}
If a resultant is $18$ N at $40^\circ$, what is the equilibrant?\soln Same magnitude, opposite direction: $18$ N at $220^\circ$.
\end{example}

\begin{example}{ex8}{Example 9 --- Two equal forces}
Two $20$ N forces have $60^\circ$ between. Resultant?\soln $2(20)\cos30^\circ\approx34.6$ N.
\end{example}
