\begin{ideabox}
A logarithm is an exponent: $\log_b x=y$ means $b^{y}=x$ (with $b>0,b\ne1,x>0$). $\log x$ is base $10$; $\ln x$ is base $e$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $\log_b x=y\iff b^{y}=x$.
  \item $\log_b 1=0$, $\log_b b=1$.
  \item Evaluate by asking "what exponent?".
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{A logarithm is an exponent}$\log_b x$ answers \emph{to what power must $b$ be raised to give $x$?} So $\log_2 8=3$ because $2^3=8$.
\lh{Switching forms}The statements $\log_b x=y$ and $b^{\,y}=x$ say the same thing; moving fluently between them is the key skill.
\lh{Common and natural logs}$\log x$ means base $10$ and $\ln x$ means base $e$ — the two logarithms a calculator gives directly.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Switch forms}
Rewrite $2^5=32$ in log form; $\log_3 81=4$ in exponential form.\soln $\log_2 32=5$; $3^4=81$.
\end{example}

\begin{example}{ex1}{Example 2 --- Evaluate several}
Evaluate $\log_2 16,\ \log_5\tfrac1{25},\ \log 1000,\ \log_9 3$.\soln $4,\ -2,\ 3,\ \tfrac12$.
\end{example}

\begin{example}{ex2}{Example 3 --- Solve for argument}
Solve $\log_2 x=5$.\soln $x=2^5=32$.
\end{example}

\begin{example}{ex3}{Example 4 --- Solve for base}
Solve $\log_x 64=3$.\soln $x^3=64\Rightarrow x=4$.
\end{example}

\begin{example}{ex4}{Example 5 --- Fractional value}
Evaluate $\log_4 8$.\soln $2^{2y}=2^3\Rightarrow y=\tfrac32$.
\end{example}

\begin{example}{ex5}{Example 6 --- Base 10}
Evaluate $\log 100000$.\soln $10^5\Rightarrow 5$.
\end{example}

\begin{example}{ex6}{Example 7 --- Negative log}
Evaluate $\log_3\tfrac{1}{27}$.\soln $3^{-3}\Rightarrow -3$.
\end{example}

\begin{example}{ex7}{Example 8 --- Solve for base again}
Solve $\log_x 81=4$.\soln $x^4=81\Rightarrow x=3$.
\end{example}

\begin{example}{ex8}{Example 9 --- Root value}
Evaluate $\log_8 32$.\soln $2^{3y}=2^5\Rightarrow y=\tfrac53$.
\end{example}
