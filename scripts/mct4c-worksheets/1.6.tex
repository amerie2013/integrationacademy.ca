{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 1: Exponential Functions}

\begin{ideabox}
Products become sums, quotients become differences, powers become multipliers: $\log_b(MN)=\log_b M+\log_b N$, $\log_b\tfrac{M}{N}=\log_b M-\log_b N$, $\log_b M^p=p\log_b M$. Change base with $\log_b x=\tfrac{\log x}{\log b}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Product/quotient/power laws.
  \item Expand or condense expressions.
  \item Change of base for any calculator.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Products, quotients, powers}Logarithms turn multiplication into addition: $\log_b(MN)=\log_b M+\log_b N$, $\log_b\tfrac{M}{N}=\log_b M-\log_b N$, and $\log_b M^p=p\log_b M$.
\lh{Expanding and condensing}Read the laws left-to-right to \emph{expand} one log into several, or right-to-left to \emph{condense} several into one.
\lh{Change of base}Any logarithm evaluates on a calculator through $\log_b x=\dfrac{\log x}{\log b}$.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Expand}
Expand $\log\!\left(\dfrac{x^2y}{z}\right)$.\soln $2\log x+\log y-\log z$.
\end{example}

\begin{example}{ex1}{Example 2 --- Condense}
Write $2\log a+\log b-3\log c$ as one log.\soln $\log\!\left(\dfrac{a^2b}{c^3}\right)$.
\end{example}

\begin{example}{ex2}{Example 3 --- Evaluate a sum}
Evaluate $\log_6 9+\log_6 4$.\soln $\log_6 36=2$.
\end{example}

\begin{example}{ex3}{Example 4 --- Power law}
Evaluate $\log_2 8^4$.\soln $4\log_2 8=4\cdot3=12$.
\end{example}

\begin{example}{ex4}{Example 5 --- Change of base}
Evaluate $\log_2 10$ to 3 decimals.\soln $\dfrac{\log 10}{\log 2}\approx3.322$.
\end{example}

\begin{example}{ex5}{Example 6 --- Evaluate a difference}
Evaluate $\log_5 100-\log_5 4$.\soln $\log_5 25=2$.
\end{example}

\begin{example}{ex6}{Example 7 --- Expand a root}
Expand $\log\sqrt{x}$.\soln $\tfrac12\log x$.
\end{example}

\begin{example}{ex7}{Example 8 --- Condense with coefficient}
Write $3\log x-\log y$ as one log.\soln $\log\!\left(\dfrac{x^3}{y}\right)$.
\end{example}

\begin{example}{ex8}{Example 9 --- Change of base 2}
Evaluate $\log_3 20$ to 3 decimals.\soln $\dfrac{\log 20}{\log 3}\approx2.727$.
\end{example}
