{\LARGE\bfseries\color{titleblue}Worksheet 1.2 --- Exponential Functions \& Their Graphs}\\[2pt]
{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 1: Exponential Functions}
\end{center}

\begin{ideabox}
An exponential function $y=a\,b^{x}$ multiplies by $b$ each unit step. It has $y$-intercept $(0,a)$, horizontal asymptote $y=0$, and grows if $b>1$ or decays if $0<b<1$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Form $y=a\,b^{x}$; $y$-intercept $(0,a)$.
  \item Horizontal asymptote $y=0$; range $y>0$ (for $a>0$).
  \item Base $b>1$ grows; $0<b<1$ decays.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{A constant multiplier}Where a line adds the same amount each step, an \textbf{exponential function} $y=a\,b^x$ \emph{multiplies} by $b$ each step, giving rapid growth or decay.
\lh{Reading the graph}Every such curve passes through $(0,a)$ and hugs the horizontal asymptote $y=0$ without ever crossing it.
\lh{Growth or decay}A base $b>1$ makes the curve climb; a base $0<b<1$ makes it fall. The further $b$ sits from $1$, the steeper the change.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Read the features}
For $y=3\cdot2^{x}$, give the $y$-intercept, asymptote, and behaviour.\soln $(0,3)$; asymptote $y=0$; growth.
\end{example}

\begin{example}{ex1}{Example 2 --- Growth vs. decay}
Compare $y=2^{x}$ and $y=\left(\tfrac12\right)^{x}$.\soln Both pass $(0,1)$ with asymptote $y=0$; the second decays (mirror of the first).
\end{example}

\begin{example}{ex2}{Example 3 --- Table and trend}
Tabulate $y=4\left(\tfrac12\right)^{x}$ at $x=0,1,2,3$.\soln $4,2,1,\tfrac12$ — halving; $y$-intercept $(0,4)$, decay.
\end{example}

\begin{example}{ex3}{Example 4 --- Find the equation}
$y=a\,b^{x}$ has $y$-intercept $5$ and passes $(2,45)$. Find it.\soln $a=5$; $45=5b^2\Rightarrow b=3$; $y=5\cdot3^{x}$.
\end{example}

\begin{example}{ex4}{Example 5 --- Domain and range}
State the domain, range, asymptote of $y=5\cdot3^{x}$ and find $y(-1)$.\soln Domain $\mathbb{R}$; range $y>0$; asymptote $y=0$; $y(-1)=\tfrac53$.
\end{example}

\begin{example}{ex5}{Example 6 --- Two points}
A function passes $(1,12)$ and $(3,48)$. Find $y=a\,b^{x}$.\soln $\tfrac{48}{12}=b^2=4\Rightarrow b=2$; $12=2a\Rightarrow a=6$; $y=6\cdot2^{x}$.
\end{example}

\begin{example}{ex6}{Example 7 --- Evaluate}
Evaluate $y=2\cdot5^{x}$ at $x=2$ and $x=-1$.\soln $50$ and $\tfrac25$.
\end{example}

\begin{example}{ex7}{Example 8 --- Decay factor}
A quantity halves each step from $80$. Write the model and give the value after 3 steps.\soln $y=80\left(\tfrac12\right)^{x}$; at $x=3$: $10$.
\end{example}

\begin{example}{ex8}{Example 9 --- Compare bases}
Which grows faster: $y=2^{x}$ or $y=3^{x}$? Compare at $x=3$.\soln $2^3=8$ vs. $3^3=27$; $3^x$ grows faster.
\end{example}
