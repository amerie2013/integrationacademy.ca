\begin{ideabox}
A vector has \emph{magnitude} and \emph{direction}. From components $(x,y)$: magnitude $=\sqrt{x^2+y^2}$ and direction is the angle to the positive $x$-axis. From a magnitude $m$ at angle $\theta$: components $(m\cos\theta,\ m\sin\theta)$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Magnitude $|\vec v|=\sqrt{x^2+y^2}$.
  \item Components $(m\cos\theta,\ m\sin\theta)$.
  \item Opposite vector negates both components; a unit vector has magnitude $1$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Size and direction}A \textbf{vector} carries both a magnitude and a direction, unlike a plain number (a scalar). Displacement, velocity, and force are all vectors.
\lh{Magnitude and components}From components $(x,y)$ the magnitude is $\sqrt{x^2+y^2}$; from a magnitude $m$ at angle $\theta$ the components are $(m\cos\theta,\ m\sin\theta)$.
\lh{Direction and unit vectors}The direction is the angle to the positive $x$-axis, and dividing a vector by its magnitude gives a \textbf{unit vector} that keeps only direction.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Magnitude and direction}
Find the magnitude and direction of $(3,4)$.\soln $\sqrt{9+16}=5$; $\theta=\tan^{-1}\tfrac43\approx53.1^\circ$.
\end{example}

\begin{example}{ex1}{Example 2 --- A 3-4-5 relative}
Find $|(5,-12)|$.\soln $\sqrt{25+144}=13$.
\end{example}

\begin{example}{ex2}{Example 3 --- Resolve into components}
A vector has magnitude $10$ at $30^\circ$. Find its components.\soln $(10\cos30^\circ,10\sin30^\circ)\approx(8.7,5)$.
\end{example}

\begin{example}{ex3}{Example 4 --- Larger magnitude}
Components of magnitude $20$ at $60^\circ$?\soln $(20\cos60^\circ,20\sin60^\circ)=(10,17.3)$.
\end{example}

\begin{example}{ex4}{Example 5 --- Opposite vector}
Write the opposite of $(2,3)$ and its magnitude.\soln $(-2,-3)$; magnitude $\sqrt{13}\approx3.6$.
\end{example}

\begin{example}{ex5}{Example 6 --- Scalar multiple}
Find $3(2,-1)$ and its magnitude.\soln $(6,-3)$; $\sqrt{45}=3\sqrt5\approx6.7$.
\end{example}

\begin{example}{ex6}{Example 7 --- Unit vector}
Find the unit vector in the direction of $(3,4)$.\soln $\tfrac{1}{5}(3,4)=(0.6,0.8)$.
\end{example}

\begin{example}{ex7}{Example 8 --- Another magnitude}
Find $|(6,8)|$.\soln $10$.
\end{example}

\begin{example}{ex8}{Example 9 --- Direction in QII}
Find the direction of $(-1,1)$.\soln Reference $45^\circ$ in QII: $135^\circ$.
\end{example}
