{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 3: Trigonometric Functions}

\begin{ideabox}
$\dfrac{a}{\sin A}=\dfrac{b}{\sin B}=\dfrac{c}{\sin C}$ works with any side–opposite-angle pair. Given two sides and a non-included angle (SSA), there may be \emph{two}, one, or no triangles.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Use a matched side/opposite-angle pair.
  \item SSA: check for a second angle ($180^\circ-B$).
  \item If $\sin B>1$, no triangle exists.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{A law for any triangle}The \textbf{sine law} $\dfrac{a}{\sin A}=\dfrac{b}{\sin B}=\dfrac{c}{\sin C}$ links each side to the sine of its opposite angle.
\lh{When to use it}It applies when you know a side and its opposite angle, plus one more piece of information.
\lh{The ambiguous case}With two sides and a non-included angle, there can be zero, one, or two triangles — always test for a second angle.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Find a side}
$A=40^\circ,B=60^\circ,a=10$. Find $b$.\soln $b=\dfrac{10\sin60^\circ}{\sin40^\circ}\approx13.5$.
\end{example}

\begin{example}{ex1}{Example 2 --- Find the third side}
Continue: $C=80^\circ$, find $c$.\soln $c=\dfrac{10\sin80^\circ}{\sin40^\circ}\approx15.3$.
\end{example}

\begin{example}{ex2}{Example 3 --- Find an angle}
$a=8,b=6,A=50^\circ$. Find $B$.\soln $\sin B=\dfrac{6\sin50^\circ}{8}\Rightarrow B\approx35.1^\circ$.
\end{example}

\begin{example}{ex3}{Example 4 --- Ambiguous — two triangles}
$a=7,b=10,A=40^\circ$. Find $B$.\soln $\sin B=\dfrac{10\sin40^\circ}{7}\approx0.92\Rightarrow B\approx66.7^\circ$ or $113.3^\circ$.
\end{example}

\begin{example}{ex4}{Example 5 --- No triangle}
$a=5,b=10,A=70^\circ$. Find $B$.\soln $\sin B=\dfrac{10\sin70^\circ}{5}\approx1.88>1$: no triangle.
\end{example}

\begin{example}{ex5}{Example 6 --- One triangle ($a>b$)}
$a=12,b=8,A=55^\circ$. Find $B$.\soln $\sin B=\dfrac{8\sin55^\circ}{12}\Rightarrow B\approx33.1^\circ$ (only, since $a>b$).
\end{example}

\begin{example}{ex6}{Example 7 --- Third angle}
From Example 3, find $C$.\soln $C=180^\circ-50^\circ-35.1^\circ=94.9^\circ$.
\end{example}

\begin{example}{ex7}{Example 8 --- Area follow-up}
For Example 1's triangle, find its area ($\tfrac12ab\sin C$).\soln $\tfrac12(10)(13.5)\sin80^\circ\approx66.5$.
\end{example}

\begin{example}{ex8}{Example 9 --- Set up a pair}
$A=25^\circ,C=100^\circ,c=15$. Find $a$.\soln $a=\dfrac{15\sin25^\circ}{\sin100^\circ}\approx6.4$.
\end{example}
