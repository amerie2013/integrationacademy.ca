{\LARGE\bfseries\color{titleblue}Worksheet 1.3 --- Transformations of Exponential Functions}\\[2pt]
{\small Grade 12 Mathematics for College Technology (MCT4C) \textperiodcentered{} Unit 1: Exponential Functions}

\begin{ideabox}
In $y=a\,b^{\,k(x-d)}+c$: $a$ stretches/reflects, $k$ stretches horizontally, $d$ shifts right, and $c$ shifts up — moving the asymptote to $y=c$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $a$: vertical stretch ($a<0$ reflects).
  \item $d$: right shift; $c$: up shift (asymptote $y=c$).
  \item $y$-intercept: substitute $x=0$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Moving the base curve}Writing $y=a\,b^{\,k(x-d)}+c$ shows how the standard exponential is shifted and stretched; the $+c$ lifts the asymptote to $y=c$.
\lh{Each letter's job}$a$ scales height (and flips the curve if negative), $k$ scales width, and $d$ slides it sideways — recall $(x-d)$ moves it right.
\lh{Intercepts and range}Substitute $x=0$ for the $y$-intercept; the range is $y>c$ when $a>0$ and $y<c$ when $a<0$.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Describe the shifts}
Describe $y=2^{\,x-3}+1$.\soln Right $3$, up $1$; asymptote $y=1$.
\end{example}

\begin{example}{ex1}{Example 2 --- Reflection + shift}
Find the $y$-intercept and asymptote of $y=-3\cdot2^{x}+4$.\soln $y(0)=-3+4=1$, so $(0,1)$; asymptote $y=4$.
\end{example}

\begin{example}{ex2}{Example 3 --- Graph features}
For $y=2^{\,x+1}-3$, give the asymptote and $y$-intercept.\soln Asymptote $y=-3$; $y(0)=2^1-3=-1$.
\end{example}

\begin{example}{ex3}{Example 4 --- Write the equation}
$y=3^{x}$ is stretched by $2$, reflected in the $x$-axis, shifted up $5$. Write it.\soln $y=-2\cdot3^{x}+5$; asymptote $y=5$.
\end{example}

\begin{example}{ex4}{Example 5 --- Range}
State the range of $y=-4\cdot2^{x}+10$.\soln Asymptote $y=10$, $a<0$: $y<10$.
\end{example}

\begin{example}{ex5}{Example 6 --- Left shift}
Describe $y=5^{\,x+2}$.\soln Left $2$; asymptote $y=0$.
\end{example}

\begin{example}{ex6}{Example 7 --- Down shift range}
Range of $y=3\cdot2^{\,x-1}-6$.\soln Asymptote $y=-6$, $a>0$: $y>-6$.
\end{example}

\begin{example}{ex7}{Example 8 --- Intercept after shift}
Find the $y$-intercept of $y=-2\cdot4^{x}+9$.\soln $-2+9=7$, so $(0,7)$.
\end{example}

\begin{example}{ex8}{Example 9 --- Combine}
Describe $y=2\cdot3^{\,x-1}+4$ fully.\soln Stretch $2$, right $1$, up $4$; asymptote $y=4$.
\end{example}
