\begin{ideabox}
SOH CAH TOA links an angle to two sides: $\sin\theta=\frac{\text{opp}}{\text{hyp}}$, $\cos\theta=\frac{\text{adj}}{\text{hyp}}$, $\tan\theta=\frac{\text{opp}}{\text{adj}}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Name the sides relative to the angle.
  \item Find a side: pick a ratio and multiply.
  \item Find an angle: use $\sin^{-1},\cos^{-1},\tan^{-1}$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The three ratios}In a right triangle, $\sin=\dfrac{\text{opp}}{\text{hyp}}$, $\cos=\dfrac{\text{adj}}{\text{hyp}}$, $\tan=\dfrac{\text{opp}}{\text{adj}}$ — remember SOH-CAH-TOA.
\lh{Finding sides and angles}Use a ratio to find a missing side, or an \emph{inverse} ratio ($\sin^{-1}$, and so on) to find a missing angle.
\lh{Label first}Relative to your chosen angle, identify the opposite, adjacent, and hypotenuse before picking a ratio.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- A ratio}
Opposite $=3$, hypotenuse $=5$. Find $\sin\theta$.\soln $\sin\theta=\dfrac{\text{opp}}{\text{hyp}}=\dfrac35=0.6$.\rtri{4}{3}{$3$}{$4$}{$5$}
\end{example}

\begin{example}{ex1}{Example 2 --- Find a side}
Hypotenuse $=10$, angle $=30^\circ$. Find the opposite side.\soln $\sin30^\circ=\dfrac{\text{opp}}{10}\Rightarrow \text{opp}=10\sin30^\circ=10(0.5)=5$.
\end{example}

\begin{example}{ex2}{Example 3 --- Find a side}
Adjacent $=8$, angle $=40^\circ$. Find the opposite side.\soln $\tan40^\circ=\dfrac{\text{opp}}{8}\Rightarrow \text{opp}=8\tan40^\circ\approx6.71$.
\end{example}

\begin{example}{ex3}{Example 4 --- Find an angle}
Opposite $=3$, adjacent $=4$. Find $\theta$.\soln $\tan\theta=\dfrac34\Rightarrow\theta=\tan^{-1}(0.75)\approx36.9^\circ$.\rtri{4}{3}{$3$}{$4$}{$5$}
\end{example}

\begin{example}{ex4}{Example 5 --- Ladder}
A 10 m ladder leans at $60^\circ$. How high does it reach?\soln height $=10\sin60^\circ\approx8.66$ m.
\end{example}

\begin{example}{ex5}{Example 6 --- Cosine ratio}
Adjacent $=4$, hypotenuse $=5$. Find $\cos\theta$.\soln $\cos\theta=\dfrac45=0.8$.
\end{example}

\begin{example}{ex6}{Example 7 --- Find a side}
Hypotenuse $=20$, angle $=30^\circ$. Find the opposite side.\soln $20\sin30^\circ=20(0.5)=10$.
\end{example}

\begin{example}{ex7}{Example 8 --- Find an angle}
Opposite $=7$, hypotenuse $=10$. Find $\theta$.\soln $\sin\theta=\dfrac{7}{10}=0.7\Rightarrow\theta=\sin^{-1}(0.7)\approx44.4^\circ$.
\end{example}

\begin{example}{ex8}{Example 9 --- Ladder}
An 8 m ladder leans at $70^\circ$. How high does it reach?\soln $8\sin70^\circ\approx7.52$ m.
\end{example}
