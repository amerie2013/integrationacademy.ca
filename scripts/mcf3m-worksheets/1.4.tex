\begin{ideabox}
Use $x=\dfrac{-b\pm\sqrt{b^2-4ac}}{2a}$; the discriminant $b^2-4ac$ gives the number of real roots.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item The formula solves any quadratic.
  \item Discriminant: $>0$ two roots, $=0$ one, $<0$ none.
  \item Use the formula when the quadratic will not factor over the integers.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{A formula for every quadratic}$x=\dfrac{-b\pm\sqrt{b^2-4ac}}{2a}$ solves any quadratic, whether or not it factors.
\lh{The discriminant}The value $b^2-4ac$ under the root reveals the number of real roots: positive gives two, zero gives one, negative gives none.
\lh{Substitute carefully}Identify $a$, $b$, $c$ with their signs before substituting, and simplify the surd fully.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Apply the formula}
Solve $x^2-4x+1=0$.\soln Here $a=1,\ b=-4,\ c=1$. \\ $x=\dfrac{4\pm\sqrt{16-4}}{2}=\dfrac{4\pm\sqrt{12}}{2}$. \\ Since $\sqrt{12}=2\sqrt3$: $\ x=2\pm\sqrt3$.
\end{example}

\begin{example}{ex1}{Example 2 --- Leading coefficient}
Solve $2x^2+3x-2=0$.\soln $a=2,\ b=3,\ c=-2$; discriminant $=9+16=25$. \\ $x=\dfrac{-3\pm5}{4}=\tfrac12$ or $-2$.
\end{example}

\begin{example}{ex2}{Example 3 --- No real roots}
How many real roots has $x^2+2x+5=0$?\soln Discriminant $=4-20=-16<0$. \\ No real roots — the parabola never reaches the x-axis:\eplot{-6}{4}{-1}{12}{\addplot[exblue,very thick,domain=-5:3]{x^2+2*x+5};}
\end{example}

\begin{example}{ex3}{Example 4 --- One (double) root}
How many real roots has $x^2-6x+9=0$?\soln Discriminant $=36-36=0$ — one (double) root. \\ $(x-3)^2=0\Rightarrow x=3$, where the vertex touches the axis:\eplot{-1}{7}{-1}{10}{\addplot[exblue,very thick,domain=0:6]{x^2-6*x+9};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(3,0)};}
\end{example}

\begin{example}{ex4}{Example 5 --- Irrational roots}
Solve $x^2+x-1=0$.\soln $a=1,\ b=1,\ c=-1$. \\ $x=\dfrac{-1\pm\sqrt{1+4}}{2}=\dfrac{-1\pm\sqrt5}{2}$.
\end{example}

\begin{example}{ex5}{Example 6 --- Irrational roots}
Solve $x^2-2x-1=0$.\soln $x=\dfrac{2\pm\sqrt{4+4}}{2}=\dfrac{2\pm2\sqrt2}{2}=1\pm\sqrt2$.
\end{example}

\begin{example}{ex6}{Example 7 --- Irrational roots}
Solve $x^2-6x+7=0$.\soln $x=\dfrac{6\pm\sqrt{36-28}}{2}=\dfrac{6\pm2\sqrt2}{2}=3\pm\sqrt2$.
\end{example}

\begin{example}{ex7}{Example 8 --- Discriminant}
Find the discriminant of $3x^2-5x+1$.\soln $b^2-4ac=25-12=13$. \\ Since $13>0$, there are two real roots.
\end{example}

\begin{example}{ex8}{Example 9 --- Simplify the root}
Solve $2x^2-4x+1=0$.\soln $x=\dfrac{4\pm\sqrt{16-8}}{4}=\dfrac{4\pm2\sqrt2}{4}=\dfrac{2\pm\sqrt2}{2}$. \\ Two x-intercepts near $0.29$ and $1.71$:\eplot{-1}{4}{-2}{6}{\addplot[exblue,very thick,domain=-0.3:2.3]{2*x^2-4*x+1};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(0.29,0) (1.71,0)};}
\end{example}
