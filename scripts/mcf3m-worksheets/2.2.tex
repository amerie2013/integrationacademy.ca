\begin{ideabox}
Vertex form $a(x-h)^2+k$ has vertex $(h,k)$; $a$ stretches/reflects, $h$ shifts horizontally, $k$ vertically.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $a$: stretch by $|a|$; reflect if $a<0$.
  \item $h$: horizontal shift (opposite to the sign). $k$: vertical shift.
  \item Vertex is $(h,k)$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Vertex form}$y=a(x-h)^2+k$ places the vertex at $(h,k)$ and shows every transformation of the base parabola at once.
\lh{Reading the parameters}$a$ stretches and (if negative) flips the parabola, $h$ shifts it right, and $k$ shifts it up — recall $(x-h)$ moves it right.
\lh{Sketching}Plot the vertex, use $a$ with the step pattern $1,4,9$, and mirror across the axis of symmetry.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Read the vertex}
State the vertex of $y=(x-3)^2+2$.\soln In $a(x-h)^2+k$, $h=3$ and $k=2$. \\ Vertex $(3,2)$ — shifted right 3 and up 2:\eplot{-1}{7}{-1}{10}{\addplot[exblue,very thick,domain=0.5:5.5]{(x-3)^2+2};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(3,2)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Describe}
Describe $y=-2(x+1)^2+5$ from $y=x^2$.\soln $a=-2$: reflect in the x-axis and stretch by 2. \\ $h=-1$: left 1. $k=5$: up 5.
\end{example}

\begin{example}{ex2}{Example 3 --- Negative h}
State the vertex of $y=(x+4)^2-2$.\soln $h=-4$, $k=-2$: vertex $(-4,-2)$.
\end{example}

\begin{example}{ex3}{Example 4 --- Direction \& max}
For $y=-(x-2)^2+3$, state the direction and max.\soln $a=-1<0$, so it opens down; the vertex $(2,3)$ is the highest point. \\ Maximum value $3$:\eplot{-2}{6}{-6}{5}{\addplot[exblue,very thick,domain=-0.5:4.5]{-(x-2)^2+3};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(2,3)};}
\end{example}

\begin{example}{ex4}{Example 5 --- Write the equation}
Write $y=x^2$ shifted right 5 and down 4.\soln Right 5 gives $h=5$; down 4 gives $k=-4$. \\ $y=(x-5)^2-4$.
\end{example}

\begin{example}{ex5}{Example 6 --- Read the vertex}
State the vertex of $y=(x-1)^2-7$.\soln $h=1$, $k=-7$: vertex $(1,-7)$.
\end{example}

\begin{example}{ex6}{Example 7 --- Stretch \& shift}
Describe $y=3(x-2)^2$ from $y=x^2$.\soln $a=3$: stretch by 3. $h=2$: right 2.
\end{example}

\begin{example}{ex7}{Example 8 --- Vertex \& direction}
State the vertex and direction of $y=-2(x+3)^2+1$.\soln $h=-3$, $k=1$: vertex $(-3,1)$; $a=-2<0$ opens down.
\end{example}

\begin{example}{ex8}{Example 9 --- Min value}
State the min of $y=(x-4)^2+6$.\soln $a=1>0$ opens up, so the vertex $(4,6)$ is the lowest point. \\ Minimum value $6$.
\end{example}
