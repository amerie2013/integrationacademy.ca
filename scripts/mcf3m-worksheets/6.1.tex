\begin{ideabox}
A periodic function repeats at regular intervals; the sine function $f(x)=\sin x$ has period $360^\circ$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Cycle: one complete repeat; period: the length of one cycle.
  \item Amplitude $=\dfrac{\max-\min}{2}$.
  \item Midline $=\dfrac{\max+\min}{2}$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Repeating behaviour}A \textbf{periodic} function repeats at regular intervals; the \textbf{period} is the length of one full cycle.
\lh{Key features}Describe such a function by its period, \textbf{amplitude} (half the max-to-min gap), and \textbf{midline}.
\lh{The sine function}$y=\sin x$ is the model periodic function, oscillating smoothly between $-1$ and $1$.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Period}
A wave repeats every $360^\circ$. State its period.\soln One full cycle spans $360^\circ$, so the period is $360^\circ$. The parent sine wave:\splot{-1.5}{1.5}{\addplot[exblue,very thick,domain=0:360]{sin(x)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Amplitude}
Max $5$, min $1$. Find the amplitude.\soln $\dfrac{\max-\min}{2}=\dfrac{5-1}{2}=2$.
\end{example}

\begin{example}{ex2}{Example 3 --- Midline}
Max $5$, min $1$. Find the midline.\soln $\dfrac{\max+\min}{2}=\dfrac{5+1}{2}=3$, so $y=3$.
\end{example}

\begin{example}{ex3}{Example 4 --- Function?}
Is $y=\sin x$ a function?\soln Each angle gives exactly one sine value, so \textbf{yes}.
\end{example}

\begin{example}{ex4}{Example 5 --- Parent period}
Period of $y=\sin x$?\soln The sine ratio repeats once per full turn: $360^\circ$.
\end{example}

\begin{example}{ex5}{Example 6 --- Amplitude}
Max $7$, min $1$. Find the amplitude.\soln $\dfrac{7-1}{2}=3$.
\end{example}

\begin{example}{ex6}{Example 7 --- Midline}
Max $7$, min $1$. Find the midline.\soln $\dfrac{7+1}{2}=4$, so $y=4$.
\end{example}

\begin{example}{ex7}{Example 8 --- Maximum}
Maximum of $y=\sin x$?\soln The peak of the parent sine curve is $1$.
\end{example}

\begin{example}{ex8}{Example 9 --- Minimum}
Minimum of $y=\sin x$?\soln The trough of the parent sine curve is $-1$.
\end{example}
