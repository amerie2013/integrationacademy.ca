{\small Grade 11 Functions \& Applications (MCF3M) \textperiodcentered{} Unit 5: Trigonometry}

\begin{ideabox}
Cosine law: $c^2=a^2+b^2-2ab\cos C$ (find a side); $\cos C=\dfrac{a^2+b^2-c^2}{2ab}$ (find an angle).
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Use for SAS (find a side) or SSS (find an angle).
  \item At $C=90^\circ$ it reduces to Pythagoras.
  \item The largest angle is opposite the longest side.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{A wider tool}The \textbf{cosine law} $a^2=b^2+c^2-2bc\cos A$ handles triangles the sine law cannot start.
\lh{Two cases}Use it with two sides and their included angle (SAS), or with all three sides (SSS) to find an angle.
\lh{A familiar limit}At $90^\circ$ the cosine term vanishes and the law becomes the Pythagorean theorem.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Find a side (SAS)}
$a=5,\ b=7,\ C=60^\circ$. Find $c$.\soln $c^2=5^2+7^2-2(5)(7)\cos60^\circ=25+49-70(0.5)=39$. \\ $c=\sqrt{39}\approx6.24$.\gtri{$A$}{$B$}{$60^\circ$}{$5$}{$7$}{$c$}
\end{example}

\begin{example}{ex1}{Example 2 --- Right angle}
$a=8,\ b=6,\ C=90^\circ$. Find $c$.\soln $\cos90^\circ=0$, so $c^2=64+36=100$. \\ $c=10$.
\end{example}

\begin{example}{ex2}{Example 3 --- Find an angle (SSS)}
Sides $6,8,10$; angle opposite $10$.\soln $\cos C=\dfrac{6^2+8^2-10^2}{2(6)(8)}=\dfrac{0}{96}=0$. \\ $C=\cos^{-1}(0)=90^\circ$.
\end{example}

\begin{example}{ex3}{Example 4 --- SAS}
$a=7,\ b=9,\ C=50^\circ$. Find $c$.\soln $c^2=49+81-126\cos50^\circ\approx130-81.0=49.0$. \\ $c\approx7.0$.
\end{example}

\begin{example}{ex4}{Example 5 --- Largest angle (SSS)}
Sides $5,6,7$; largest angle (opposite $7$).\soln $\cos C=\dfrac{25+36-49}{2(5)(6)}=\dfrac{12}{60}=0.2$. \\ $C\approx78.5^\circ$.\gtri{$A$}{$B$}{$C$}{$5$}{$6$}{$7$}
\end{example}

\begin{example}{ex5}{Example 6 --- Right angle}
$a=4,\ b=5,\ C=90^\circ$. Find $c$.\soln $c=\sqrt{16+25}=\sqrt{41}\approx6.40$.
\end{example}

\begin{example}{ex6}{Example 7 --- SAS}
$a=9,\ b=12,\ C=80^\circ$. Find $c$.\soln $c^2=81+144-216\cos80^\circ\approx187.5$. \\ $c\approx13.7$.
\end{example}

\begin{example}{ex7}{Example 8 --- Find an angle (SSS)}
Sides $7,8,9$; angle opposite $9$.\soln $\cos C=\dfrac{49+64-81}{2(7)(8)}=\dfrac{32}{112}\approx0.286$. \\ $C\approx73.4^\circ$.
\end{example}

\begin{example}{ex8}{Example 9 --- SAS}
$a=10,\ b=6,\ C=40^\circ$. Find $c$.\soln $c^2=100+36-120\cos40^\circ\approx44.1$. \\ $c\approx6.64$.
\end{example}
