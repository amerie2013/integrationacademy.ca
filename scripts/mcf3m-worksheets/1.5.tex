{\LARGE\bfseries\color{titleblue}Worksheet 1.5 --- Roots and x-Intercepts}\\[2pt]
{\small Grade 11 Functions \& Applications (MCF3M) \textperiodcentered{} Unit 1: Quadratic Expressions and Equations}
\end{center}

\begin{ideabox}
The real roots of $ax^2+bx+c=0$ are the x-intercepts of $y=ax^2+bx+c$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $D>0$ two x-intercepts, $D=0$ one, $D<0$ none.
  \item A double root means the vertex sits on the x-axis.
  \item Factoring and graphing agree because both find where $y=0$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Roots are x-intercepts}The \textbf{roots} of $0=ax^2+bx+c$ are exactly the \textbf{$x$-intercepts} of $y=ax^2+bx+c$ — where the parabola meets the axis.
\lh{How many}The discriminant also counts intercepts: two, one (a touch), or none, matching the number of roots.
\lh{Axis of symmetry}The axis of symmetry lies midway between the intercepts, at $x=-\dfrac{b}{2a}$, which locates the vertex.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Two intercepts}
Find the x-intercepts of $y=x^2-4x+3$.\soln Set $y=0$ and factor: $(x-1)(x-3)=0$. \\ $x=1$ and $x=3$:\eplot{-1}{6}{-2}{8}{\addplot[exblue,very thick,domain=0:5]{x^2-4*x+3};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(1,0) (3,0)};}
\end{example}

\begin{example}{ex1}{Example 2 --- One intercept}
How many x-intercepts has $y=x^2-6x+9$?\soln $(x-3)^2$ — a double root; discriminant $=0$. \\ One x-intercept, at $x=3$.
\end{example}

\begin{example}{ex2}{Example 3 --- No intercepts}
How many x-intercepts has $y=x^2+1$?\soln Discriminant $=0-4=-4<0$. \\ None — the parabola sits entirely above the x-axis:\eplot{-4}{4}{-1}{10}{\addplot[exblue,very thick,domain=-3:3]{x^2+1};}
\end{example}

\begin{example}{ex3}{Example 4 --- Count from D}
How many x-intercepts has $y=2x^2-3x-5$?\soln Discriminant $=9+40=49>0$. \\ Two x-intercepts.
\end{example}

\begin{example}{ex4}{Example 5 --- Application — profit}
Profit (in $\$1000$s) is $P=-5x^2+550x-5000$, where $x$ is advertising (in $\$1000$s). Where is $P=0$?\soln Factor out $-5$: $-5(x^2-110x+1000)$. \\ Factor: $-5(x-10)(x-100)=0$. \\ $x=10$ or $x=100$ (thousand dollars).
\end{example}

\begin{example}{ex5}{Example 6 --- Factor to intercepts}
Find the x-intercepts of $y=x^2-x-6$.\soln $(x-3)(x+2)=0$. \\ $x=3$ and $x=-2$.
\end{example}

\begin{example}{ex6}{Example 7 --- Difference of squares}
Find the x-intercepts of $y=x^2-9$.\soln $(x-3)(x+3)=0$. \\ $x=\pm3$.
\end{example}

\begin{example}{ex7}{Example 8 --- No intercepts}
How many x-intercepts has $y=x^2-2x+5$?\soln Discriminant $=4-20=-16<0$. \\ None.
\end{example}

\begin{example}{ex8}{Example 9 --- Double root}
How many x-intercepts has $y=4x^2-12x+9$?\soln $(2x-3)^2$, discriminant $=0$. \\ One, at $x=\tfrac32$:\eplot{-1}{4}{-1}{10}{\addplot[exblue,very thick,domain=0:3]{4*x^2-12*x+9};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(1.5,0)};}
\end{example}
