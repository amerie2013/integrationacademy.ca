\begin{ideabox}
The parent sine graph has five key points: $(0,0),(90,1),(180,0),(270,-1),(360,0)$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Amplitude $1$, period $360^\circ$, midline $y=0$.
  \item Max at $90^\circ$, min at $270^\circ$.
  \item Zeros at $0^\circ,180^\circ,360^\circ$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The base wave}$y=\sin x$ has period $360^\circ$, amplitude $1$, and midline $y=0$, starting at the midline and rising.
\lh{Five key points}Plot the start, maximum, midline, minimum, and return over one period, then repeat the shape.
\lh{Reading it}From the graph you can read the period, amplitude, and midline directly off the axes.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Key value}
$\sin0^\circ$.\soln The curve starts on the midline: $0$.
\end{example}

\begin{example}{ex1}{Example 2 --- Peak}
$\sin90^\circ$.\soln The maximum of the curve: $1$.
\end{example}

\begin{example}{ex2}{Example 3 --- Zero}
$\sin180^\circ$.\soln A zero crossing: $0$.
\end{example}

\begin{example}{ex3}{Example 4 --- Trough}
$\sin270^\circ$.\soln The minimum of the curve: $-1$.
\end{example}

\begin{example}{ex4}{Example 5 --- Five key points}
List one cycle's key points.\soln $(0,0),(90,1),(180,0),(270,-1),(360,0)$ — they split the cycle into quarters:\splot{-1.5}{1.5}{\addplot[exblue,very thick,domain=0:360]{sin(x)};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(0,0) (90,1) (180,0) (270,-1) (360,0)};}
\end{example}

\begin{example}{ex5}{Example 6 --- Full cycle}
$\sin360^\circ$.\soln Back to the start: $0$.
\end{example}

\begin{example}{ex6}{Example 7 --- Maximum}
Where is $y=\sin x$ maximum on $[0^\circ,360^\circ]$?\soln At the peak, $x=90^\circ$.
\end{example}

\begin{example}{ex7}{Example 8 --- Range}
Range of $y=\sin x$.\soln The curve stays between $-1$ and $1$: $[-1,1]$.
\end{example}

\begin{example}{ex8}{Example 9 --- Period}
Period of $y=\sin x$.\soln One cycle spans $360^\circ$.
\end{example}
