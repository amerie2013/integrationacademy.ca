\begin{ideabox}
Completing the square turns $ax^2+bx+c$ into vertex form $a(x-h)^2+k$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Add the square of half the $x$-coefficient (and subtract it).
  \item If $a\ne1$, factor $a$ from the $x^2$ and $x$ terms first.
  \item Vertex is $(h,k)$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Why complete the square}\textbf{Completing the square} converts $ax^2+bx+c$ into vertex form, revealing the maximum or minimum directly.
\lh{The technique}Factor out $a$, take half the $x$-coefficient and square it, then add and subtract that value to form a perfect square.
\lh{Payoff}The finished form $a(x-h)^2+k$ gives the vertex $(h,k)$ without any graphing.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Basic}
Complete the square: $x^2+6x+5$.\soln Half of $6$ is $3$; $3^2=9$: \\ $x^2+6x+9-9+5=(x+3)^2-4$. \\ Vertex $(-3,-4)$:\eplot{-7}{1}{-6}{8}{\addplot[exblue,very thick,domain=-6.5:0.5]{x^2+6*x+5};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(-3,-4)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Negative middle}
Complete the square: $x^2-4x+1$.\soln Half of $-4$ is $-2$; $(-2)^2=4$: \\ $x^2-4x+4-4+1=(x-2)^2-3$.
\end{example}

\begin{example}{ex2}{Example 3 --- Larger constant}
Complete the square: $x^2+8x+10$.\soln Half of $8$ is $4$; $4^2=16$: \\ $(x+4)^2-16+10=(x+4)^2-6$.
\end{example}

\begin{example}{ex3}{Example 4 --- Leading coefficient}
Complete the square: $2x^2-12x+5$.\soln Factor 2 from the $x$-terms: $2(x^2-6x)+5$. \\ Inside: $x^2-6x=(x-3)^2-9$. \\ $2((x-3)^2-9)+5=2(x-3)^2-13$. Vertex $(3,-13)$.
\end{example}

\begin{example}{ex4}{Example 5 --- Vertex to standard}
Expand $(x-1)^2+4$.\soln $(x-1)^2=x^2-2x+1$, so \\ $x^2-2x+1+4=x^2-2x+5$.
\end{example}

\begin{example}{ex5}{Example 6 --- Negative constant}
Complete the square: $x^2+2x-3$.\soln Half of $2$ is $1$: \\ $(x+1)^2-1-3=(x+1)^2-4$.
\end{example}

\begin{example}{ex6}{Example 7 --- No constant}
Complete the square: $x^2-10x$.\soln Half of $-10$ is $-5$; $25$: \\ $(x-5)^2-25$.
\end{example}

\begin{example}{ex7}{Example 8 --- Leading coefficient}
Complete the square: $2x^2+8x+3$.\soln $2(x^2+4x)+3=2((x+2)^2-4)+3=2(x+2)^2-5$.
\end{example}

\begin{example}{ex8}{Example 9 --- Find the vertex}
Find the vertex of $y=x^2-6x+5$.\soln $x^2-6x+9-9+5=(x-3)^2-4$. \\ Vertex $(3,-4)$:\eplot{-1}{7}{-6}{8}{\addplot[exblue,very thick,domain=-0.5:6.5]{x^2-6*x+5};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(3,-4)};}
\end{example}
