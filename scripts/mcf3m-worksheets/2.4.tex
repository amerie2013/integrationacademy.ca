\begin{ideabox}
In factored form $a(x-r)(x-s)$ the x-intercepts are $r,s$; the axis is $x=\dfrac{r+s}{2}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Set each factor to $0$ for the x-intercepts.
  \item The axis of symmetry is midway between them; substitute it back for the vertex.
  \item The y-intercept is found by setting $x=0$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Factored form}$y=a(x-r)(x-s)$ displays the $x$-intercepts $r$ and $s$ immediately.
\lh{Finding the vertex}The axis of symmetry is halfway between the intercepts, at $x=\dfrac{r+s}{2}$; substitute to find the vertex.
\lh{Graphing}Plot the intercepts, mark the vertex, and use $a$ to set the direction and width of the parabola.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Intercepts \& axis}
For $y=(x-1)(x-5)$, find the x-intercepts and axis.\soln Set each factor to $0$: $x=1$ and $x=5$. \\ Axis of symmetry: $x=\dfrac{1+5}{2}=3$:\eplot{-1}{7}{-6}{8}{\addplot[exblue,very thick,domain=0:6]{x^2-6*x+5};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(1,0) (5,0) (3,-4)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Vertex}
Find the vertex of $y=(x-1)(x-5)$.\soln Axis $x=3$. Substitute: $y=(3-1)(3-5)=(2)(-2)=-4$. \\ Vertex $(3,-4)$.
\end{example}

\begin{example}{ex2}{Example 3 --- Factor first}
Find the x-intercepts of $y=x^2-2x-3$.\soln Factor: $(x-3)(x+1)$. \\ $x=3$ and $x=-1$; axis $x=1$.
\end{example}

\begin{example}{ex3}{Example 4 --- Leading coefficient}
For $y=2(x+2)(x-4)$, find the x-intercepts and vertex.\soln Intercepts $x=-2,4$; axis $x=1$. \\ $y=2(1+2)(1-4)=2(3)(-3)=-18$. Vertex $(1,-18)$:\eplot{-4}{6}{-20}{12}{\addplot[exblue,very thick,domain=-3:5]{2*x^2-4*x-16};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(-2,0) (4,0) (1,-18)};}
\end{example}

\begin{example}{ex4}{Example 5 --- y-intercept}
Find the y-intercept of $y=(x-1)(x-5)$.\soln Set $x=0$: $(0-1)(0-5)=(-1)(-5)=5$. \\ y-intercept $(0,5)$.
\end{example}

\begin{example}{ex5}{Example 6 --- Intercepts \& axis}
For $y=(x+2)(x-6)$, find the intercepts and axis.\soln $x=-2,6$; axis $x=\dfrac{-2+6}{2}=2$.
\end{example}

\begin{example}{ex6}{Example 7 --- Vertex}
Find the vertex of $y=(x-2)(x-8)$.\soln Axis $x=5$; $y=(5-2)(5-8)=(3)(-3)=-9$. Vertex $(5,-9)$.
\end{example}

\begin{example}{ex7}{Example 8 --- Difference of squares}
Find the x-intercepts and axis of $y=x^2-9$.\soln $(x-3)(x+3)=0$: $x=\pm3$; axis $x=0$.
\end{example}

\begin{example}{ex8}{Example 9 --- y-intercept}
Find the y-intercept of $y=(x-3)(x+4)$.\soln Set $x=0$: $(-3)(4)=-12$. y-intercept $(0,-12)$.
\end{example}
