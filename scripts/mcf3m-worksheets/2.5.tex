\begin{ideabox}
The vertex gives the maximum (if $a<0$) or minimum (if $a>0$). Find it with $x=-\dfrac{b}{2a}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $a>0$ gives a minimum; $a<0$ gives a maximum.
  \item The max/min is the $y$-value of the vertex.
  \item Set up a quadratic model, then find the vertex.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Optimizing with quadratics}Many real quantities — area, revenue, height — are quadratic, so their maximum or minimum sits at the vertex.
\lh{Set up the model}Write the quantity as a quadratic in one variable, then find the vertex by completing the square or by $x=-\dfrac{b}{2a}$.
\lh{Interpret the answer}Report the vertex in context: the optimal input and the maximum or minimum value it produces.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Read a max}
Find the maximum of $y=-(x-3)^2+9$.\soln $a=-1<0$ opens down; the vertex $(3,9)$ is the highest point. \\ Maximum value $9$ at $x=3$:\eplot{-1}{7}{-2}{11}{\addplot[exblue,very thick,domain=0:6]{-(x-3)^2+9};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(3,9)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Find a min}
Find the minimum of $y=x^2-6x+5$.\soln Vertex $x=-\dfrac{-6}{2}=3$; $y=9-18+5=-4$. \\ Minimum value $-4$:\eplot{-1}{7}{-6}{8}{\addplot[exblue,very thick,domain=-0.5:6.5]{x^2-6*x+5};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(3,-4)};}
\end{example}

\begin{example}{ex2}{Example 3 --- Max height}
A ball's height is $h=-5t^2+20t$. Find the maximum height.\soln Vertex $t=-\dfrac{20}{2(-5)}=2$; $h=-5(4)+40=20$. \\ Maximum height $20$ m at $t=2$ s:\eplot{-0.5}{4.5}{-2}{24}{\addplot[exblue,very thick,domain=0:4]{-5*x^2+20*x};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(2,20)};}
\end{example}

\begin{example}{ex3}{Example 4 --- Max profit}
Profit is $P=-2x^2+12x-10$. Find the maximum profit.\soln Vertex $x=-\dfrac{12}{2(-2)}=3$; $P=-2(9)+36-10=8$. \\ Maximum profit $8$.
\end{example}

\begin{example}{ex4}{Example 5 --- Max area}
A rectangle has perimeter 20 m. What dimensions give the maximum area?\soln Width $w$, length $10-w$; $A=w(10-w)=10w-w^2$. \\ Vertex $w=5$, so $5\times5$, area $25$ m\textsuperscript{2}.
\end{example}

\begin{example}{ex5}{Example 6 --- Read a max}
Find the maximum of $y=-(x-2)^2+7$.\soln Vertex $(2,7)$, opens down: maximum $7$.
\end{example}

\begin{example}{ex6}{Example 7 --- Find a min}
Find the minimum of $y=x^2-4x+1$.\soln Vertex $x=2$; $y=4-8+1=-3$. Minimum $-3$.
\end{example}

\begin{example}{ex7}{Example 8 --- Max height}
$h=-5t^2+30t$. Find the maximum height.\soln $t=-\dfrac{30}{2(-5)}=3$; $h=-45+90=45$ m.
\end{example}

\begin{example}{ex8}{Example 9 --- Find a min}
Find the minimum of $y=2x^2-8x+1$.\soln $x=-\dfrac{-8}{2(2)}=2$; $y=8-16+1=-7$. Minimum $-7$.
\end{example}
