{\LARGE\bfseries\color{titleblue}Worksheet 5.2 --- Two Right Triangles in 2-D}\\[2pt]
{\small Grade 11 Functions \& Applications (MCF3M) \textperiodcentered{} Unit 5: Trigonometry}
\end{center}

\begin{ideabox}
Two right triangles often share a side; write a ratio in each, then combine the equations.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Identify the shared side (height or base).
  \item Elevation looks up, depression looks down; they are equal across parallels.
  \item One equation per triangle, then subtract or substitute.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Chaining triangles}Some problems hide two right triangles that share a side; solve one to get a length needed in the other.
\lh{Angles of elevation}Sightlines above the horizontal form angles of elevation (below, depression) that set up the triangles.
\lh{Work in order}Draw and label carefully, then solve the triangles in sequence, passing the shared side between them.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Helicopter \& trucks}
Helicopter 500 m up; depressions $60^\circ$, $20^\circ$ to two trucks ahead.\soln Each truck's distance from the point below: $\dfrac{500}{\tan60^\circ}\approx288.7$ and $\dfrac{500}{\tan20^\circ}\approx1373.7$. \\ Apart $=1373.7-288.7\approx1085$ m.
\end{example}

\begin{example}{ex1}{Example 2 --- Pole on a building}
From 50 m, elevations $30^\circ$ (base) and $40^\circ$ (top).\soln Top height $50\tan40^\circ\approx41.95$; base height $50\tan30^\circ\approx28.87$. \\ Pole $=41.95-28.87\approx13.1$ m.
\end{example}

\begin{example}{ex2}{Example 3 --- Two buoys}
From a 40 m cliff, depressions $50^\circ$, $30^\circ$.\soln $\dfrac{40}{\tan30^\circ}\approx69.28$ and $\dfrac{40}{\tan50^\circ}\approx33.56$. \\ Apart $\approx35.7$ m.
\end{example}

\begin{example}{ex3}{Example 4 --- Walking to a hill}
Elevation $20^\circ$ from A; 100 m closer it is $35^\circ$. Find the height $h$.\soln Each triangle: distance $=h\cot\theta$. \\ $h\cot20^\circ-h\cot35^\circ=100\Rightarrow h(2.747-1.428)=100$. \\ $h\approx75.8$ m.
\end{example}

\begin{example}{ex4}{Example 5 --- Approaching a tower}
Elevation $45^\circ$, then $60^\circ$ from 20 m closer.\soln $h(\cot45^\circ-\cot60^\circ)=20\Rightarrow h(1-0.5774)=20$. \\ $h\approx47.3$ m.
\end{example}

\begin{example}{ex5}{Example 6 --- One-triangle warm-up}
From 40 m, elevation to a roof is $35^\circ$. Find its height.\soln height $=40\tan35^\circ\approx28.0$ m.\rtri{5}{3.5}{$h$}{$40$ m}{}
\end{example}

\begin{example}{ex6}{Example 7 --- Boat}
From a 60 m cliff, depression to a boat is $25^\circ$.\soln distance $=\dfrac{60}{\tan25^\circ}\approx128.7$ m.
\end{example}

\begin{example}{ex7}{Example 8 --- Pole}
From 30 m, elevations $25^\circ$, $40^\circ$.\soln $30(\tan40^\circ-\tan25^\circ)\approx25.17-13.99\approx11.2$ m.
\end{example}

\begin{example}{ex8}{Example 9 --- Two trucks}
From 300 m up, depressions $45^\circ$, $30^\circ$.\soln $\dfrac{300}{\tan30^\circ}-\dfrac{300}{\tan45^\circ}\approx519.6-300\approx219.6$ m.
\end{example}
