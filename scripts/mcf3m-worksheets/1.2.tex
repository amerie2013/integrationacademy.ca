\begin{ideabox}
Factoring reverses expanding. Look for a common factor first, then a pattern.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Pull out the GCF first.
  \item $x^2+bx+c=(x+p)(x+q)$ with $p+q=b,\ pq=c$; use decomposition for $ax^2+bx+c$.
  \item Special: $a^2-b^2=(a-b)(a+b)$ and $a^2\pm2ab+b^2=(a\pm b)^2$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Factoring reverses expanding}Factoring rewrites $ax^2+bx+c$ as a product of two binomials â€” the reverse of expanding.
\lh{Common factor first}Always remove the greatest common factor before anything else; it simplifies whatever remains.
\lh{Trinomials}For $x^2+bx+c$, find two numbers that multiply to $c$ and add to $b$; with a leading coefficient, use decomposition (or the difference of squares where it fits).
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Common factor}
Factor $6x^2+9x$.\soln The GCF of $6x^2$ and $9x$ is $3x$: \\ $3x(2x+3)$. \\ Check: $3x\cdot2x+3x\cdot3=6x^2+9x$. \checkmark
\end{example}

\begin{example}{ex1}{Example 2 --- Simple trinomial}
Factor $x^2+7x+12$.\soln Find two numbers with product $12$ and sum $7$: $3$ and $4$. \\ $(x+3)(x+4)$.
\end{example}

\begin{example}{ex2}{Example 3 --- Negative constant}
Factor $x^2-x-6$.\soln Product $-6$, sum $-1$: $-3$ and $2$. \\ $(x-3)(x+2)$.
\end{example}

\begin{example}{ex3}{Example 4 --- Leading coefficient}
Factor $2x^2+7x+3$.\soln Decomposition: $a\cdot c=6$; two numbers with product $6$, sum $7$ are $6,1$. \\ Split: $2x^2+6x+x+3$. \\ Group: $2x(x+3)+1(x+3)=(2x+1)(x+3)$.
\end{example}

\begin{example}{ex4}{Example 5 --- Difference of squares}
Factor $x^2-25$.\soln Both terms are perfect squares: $x^2-5^2$. \\ $(x-5)(x+5)$. \\ The graph crosses the x-axis at $x=\pm5$:\eplot{-8}{8}{-28}{12}{\addplot[exblue,very thick,domain=-6:6]{x^2-25};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(5,0) (-5,0)};}
\end{example}

\begin{example}{ex5}{Example 6 --- Perfect square}
Factor $x^2-10x+25$.\soln First and last are squares ($x^2$, $5^2$); the middle is $2\cdot5\cdot x=10x$. \\ $(x-5)^2$.
\end{example}

\begin{example}{ex6}{Example 7 --- Decomposition}
Factor $3x^2+13x-10$.\soln $a\cdot c=-30$; product $-30$, sum $13$ are $15,-2$. \\ $3x^2+15x-2x-10=3x(x+5)-2(x+5)=(3x-2)(x+5)$.
\end{example}

\begin{example}{ex7}{Example 8 --- Perfect square (ax)}
Factor $9x^2+24x+16$.\soln $(3x)^2$, $(4)^2$, middle $2\cdot3x\cdot4=24x$. \\ $(3x+4)^2$.
\end{example}

\begin{example}{ex8}{Example 9 --- Difference of squares (ax)}
Factor $4x^2-9$.\soln $(2x)^2-(3)^2$. \\ $(2x-3)(2x+3)$.
\end{example}
