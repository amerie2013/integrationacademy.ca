{\LARGE\bfseries\color{titleblue}Worksheet 3.1 --- Exponent Laws \& Rational Exponents}\\[2pt]
{\small Grade 11 Functions \& Applications (MCF3M) \textperiodcentered{} Unit 3: Exponential Functions}
\end{center}

\begin{ideabox}
Exponent laws combine powers; extend them to zero, negative, and rational exponents.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Product $x^m x^n=x^{m+n}$; quotient $\dfrac{x^m}{x^n}=x^{m-n}$; power $(x^m)^n=x^{mn}$.
  \item $x^0=1$ and $x^{-n}=\dfrac{1}{x^n}$.
  \item $x^{m/n}=\left(\sqrt[n]{x}\right)^m$ — denominator is the root.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The laws}Powers with a shared base obey $a^m a^n=a^{m+n}$, $\dfrac{a^m}{a^n}=a^{m-n}$, and $(a^m)^n=a^{mn}$; also $a^0=1$ and $a^{-n}=\dfrac{1}{a^n}$.
\lh{Rational exponents}A fractional exponent is a root: $a^{m/n}=\sqrt[n]{a^m}$, so $8^{2/3}=4$.
\lh{Simplify in steps}Apply one law at a time, leaving negative or fractional exponents until last.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Product}
Simplify $x^5\cdot x^3$.\soln Product law — add the exponents: \\ $x^{5+3}=x^8$.
\end{example}

\begin{example}{ex1}{Example 2 --- Power of a power}
Simplify $(x^4)^3$.\soln Multiply the exponents: \\ $x^{4\cdot3}=x^{12}$.
\end{example}

\begin{example}{ex2}{Example 3 --- Quotient}
Simplify $\dfrac{x^9}{x^4}$.\soln Subtract the exponents: \\ $x^{9-4}=x^5$.
\end{example}

\begin{example}{ex3}{Example 4 --- Negative exponent}
Evaluate $2^{-3}$.\soln A negative exponent means the reciprocal: \\ $\dfrac{1}{2^3}=\dfrac18$.
\end{example}

\begin{example}{ex4}{Example 5 --- Rational exponent}
Evaluate $27^{2/3}$.\soln Root first, then power: \\ $\left(\sqrt[3]{27}\right)^2=3^2=9$.
\end{example}

\begin{example}{ex5}{Example 6 --- Power of a product}
Simplify $(2x^3)^2$.\soln The exponent reaches both factors: \\ $2^2\cdot x^{3\cdot2}=4x^6$.
\end{example}

\begin{example}{ex6}{Example 7 --- Rational exponent}
Evaluate $16^{3/4}$.\soln $\left(\sqrt[4]{16}\right)^3=2^3=8$.
\end{example}

\begin{example}{ex7}{Example 8 --- Zero exponent}
Evaluate $x^0$ (for $x\ne0$).\soln From the quotient law $\dfrac{x^n}{x^n}=x^{0}$, and anything over itself is $1$. \\ So $x^0=1$.
\end{example}

\begin{example}{ex8}{Example 9 --- Negative rational}
Evaluate $\left(\tfrac12\right)^{-2}$.\soln Reciprocal then square: $2^2=4$. \\ Bases $>1$ grow, bases between 0 and 1 decay:\eplot{-3}{3}{-1}{9}{\addplot[exblue,very thick,domain=-3:3]{2^x};\addplot[qorange,very thick,domain=-3:3]{0.5^x};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(0,1)};}
\end{example}
