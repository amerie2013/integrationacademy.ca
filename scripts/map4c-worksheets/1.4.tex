\begin{ideabox}
Model constant-percentage change with $A=A_0\,b^{t}$: growth $b=1+r$, decay $b=1-r$. Half-life uses $\left(\tfrac12\right)^{t/h}$; compound interest is $A=P(1+i)^n$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Growth/decay factor $b=1\pm r$.
  \item Half-life / doubling models.
  \item Compound interest $A=P(1+i)^n$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Percentage-change model}Constant-percentage change is $A=A_0\,b^{t}$: growth uses $b=1+r$, decay uses $b=1-r$.
\lh{Half-life and doubling}Halving uses base $\tfrac12$ with exponent $\tfrac{t}{h}$; doubling uses base $2$ with exponent $\tfrac{t}{d}$.
\lh{Find the time}Solve for the time with a table or graph (MAP4C is log-free).
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Population growth}
A town of $5000$ grows $3\%$/yr. Find the population after $10$ years.\soln $5000(1.03)^{10}\approx6720$.
\end{example}

\begin{example}{ex1}{Example 2 --- Half-life}
An $80$ mg sample has a $6$-hour half-life. Find the amount after $18$ h.\soln $80\left(\tfrac12\right)^{18/6}=80\left(\tfrac12\right)^{3}=10$ mg.
\end{example}

\begin{example}{ex2}{Example 3 --- Depreciation}
A $\$30\,000$ car loses $12\%$/yr. Value after $4$ years?\soln $30\,000(0.88)^{4}\approx\$17\,990$.
\end{example}

\begin{example}{ex3}{Example 4 --- Compound interest}
$\$2000$ at $4\%$/yr compounded annually, $5$ years.\soln $2000(1.04)^{5}\approx\$2433.31$.
\end{example}

\begin{example}{ex4}{Example 5 --- Doubling time (table)}
At $6\%$/yr, when does an investment double?\soln $1.06^{11}\approx1.90$, $1.06^{12}\approx2.01$; about $11.9$ years.
\end{example}

\begin{example}{ex5}{Example 6 --- Growth value}
$800(1.02)^{t}$; find the amount after $15$ years.\soln $\approx1077$.
\end{example}

\begin{example}{ex6}{Example 7 --- Half-life count}
$60$ mg, half-life $4$ h; amount after $12$ h.\soln $60\left(\tfrac12\right)^{3}=7.5$ mg.
\end{example}

\begin{example}{ex7}{Example 8 --- Decay to a fraction}
$A=A_0(0.85)^{t}$; when is $25\%$ left?\soln $0.85^{8}\approx0.272$, $0.85^{9}\approx0.232$; about $8.5$ years.
\end{example}

\begin{example}{ex8}{Example 9 --- Interest amount}
$\$1500$ at $5\%$/yr, $4$ years.\soln $1500(1.05)^{4}\approx\$1823.26$.
\end{example}
