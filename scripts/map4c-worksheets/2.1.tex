\begin{ideabox}
Simple interest is on the principal only ($I=Prt$); compound interest is on the growing balance ($A=P(1+i)^{n}$). For compounding $k$ times a year, use $i=\tfrac{r}{k}$ and $n=kt$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Simple: $I=Prt$, $A=P(1+rt)$.
  \item Compound: $A=P(1+i)^{n}$.
  \item $i=\dfrac{r}{k}$, $n=kt$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Simple vs compound}Simple interest $I=Prt$ is on the principal only; compound interest $A=P(1+i)^{n}$ is on the growing balance.
\lh{Match the period}For compounding $k$ times a year, use $i=\dfrac{r}{k}$ and $n=kt$.
\lh{Find the principal}Rearrange to $P=\dfrac{A}{(1+i)^{n}}$ to find how much to invest today.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Simple interest}
Find the interest and amount on $\$2000$ at $5\%$/yr simple for $3$ years.\soln $I=2000(0.05)(3)=\$300$; $A=\$2300$.
\end{example}

\begin{example}{ex1}{Example 2 --- Compound annually}
Amount on $\$2000$ at $5\%$/yr compounded annually, $3$ years.\soln $2000(1.05)^{3}\approx\$2315.25$.
\end{example}

\begin{example}{ex2}{Example 3 --- Simple vs compound}
Compare $\$1000$ at $6\%$/yr for $10$ years, simple vs compounded annually.\soln Simple $1000(1+0.6)=\$1600$; compound $1000(1.06)^{10}\approx\$1790.85$.
\end{example}

\begin{example}{ex3}{Example 4 --- Quarterly compounding}
$\$5000$ at $8\%$/yr compounded quarterly, $2$ years.\soln $i=0.02,\ n=8$: $5000(1.02)^{8}\approx\$5858.30$.
\end{example}

\begin{example}{ex4}{Example 5 --- Semi-annual}
$\$6000$ at $5\%$/yr compounded semi-annually, $5$ years.\soln $i=0.025,\ n=10$: $6000(1.025)^{10}\approx\$7680.51$.
\end{example}

\begin{example}{ex5}{Example 6 --- Find the principal}
How much now at $5\%$/yr compounded annually gives $\$3000$ in $4$ years?\soln $P=\dfrac{3000}{1.05^{4}}\approx\$2468.11$.
\end{example}

\begin{example}{ex6}{Example 7 --- Simple interest, amount}
Amount on $\$3000$ at $4\%$/yr simple for $5$ years.\soln $I=3000(0.04)(5)=\$600$; $A=\$3600$.
\end{example}

\begin{example}{ex7}{Example 8 --- Monthly compounding}
$\$1000$ at $6\%$/yr compounded monthly, $5$ years.\soln $i=0.005,\ n=60$: $1000(1.005)^{60}\approx\$1348.85$.
\end{example}

\begin{example}{ex8}{Example 9 --- Compare frequency}
$\$2500$ at $6\%$/yr compounded annually, $4$ years.\soln $2500(1.06)^{4}\approx\$3156.19$.
\end{example}
