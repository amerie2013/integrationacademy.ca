{\small Grade 12 Foundations for College Mathematics (MAP4C) \textperiodcentered{} Unit 3: Geometry \& Trigonometry}
\end{center}

\begin{ideabox}
Prism/cylinder volume $=(\text{base area})\times h$; box $V=lwh$, cylinder $V=\pi r^2h$. Surface area: box $2(lw+lh+wh)$, closed cylinder $2\pi r^2+2\pi rh$. Add joined solids; subtract a drilled hole.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Box $V=lwh$, cylinder $V=\pi r^2h$.
  \item Box $SA=2(lw+lh+wh)$; cylinder $2\pi r^2+2\pi rh$.
  \item Add parts; subtract a hole.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Volume}Prism or cylinder: (base area) $\times$ height — box $V=lwh$, cylinder $V=\pi r^2h$.
\lh{Surface area}Box $SA=2(lw+lh+wh)$; closed cylinder $SA=2\pi r^2+2\pi rh$.
\lh{Composite solids}Add joined volumes, subtract a drilled hole, and ignore hidden internal faces for surface area.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- A box}
Volume and surface area of a $4\times5\times6$ box.\soln $V=120$; $SA=2(20+24+30)=148$.
\end{example}

\begin{example}{ex1}{Example 2 --- A cylinder}
Cylinder radius $3$, height $10$.\soln $V=90\pi\approx282.74$; $SA=18\pi+60\pi=78\pi\approx245.04$.
\end{example}

\begin{example}{ex2}{Example 3 --- A tank}
Volume of a tank, radius $5$, height $8$.\soln $200\pi\approx628.32$.
\end{example}

\begin{example}{ex3}{Example 4 --- Drilled block (subtract)}
A $10\times8\times6$ block with a cylindrical hole of radius $2$ through the height $6$.\soln $480-\pi(4)(6)=480-24\pi\approx404.60$.
\end{example}

\begin{example}{ex4}{Example 5 --- Triangular prism}
Prism with triangular cross-section (base $6$, height $4$), length $10$.\soln $\tfrac12(6)(4)\times10=120$.
\end{example}

\begin{example}{ex5}{Example 6 --- Larger cylinder}
Cylinder radius $4$, height $10$.\soln $V=160\pi\approx502.65$; $SA=32\pi+80\pi=112\pi\approx351.86$.
\end{example}

\begin{example}{ex6}{Example 7 --- A large tank}
Volume of a tank, radius $6$, height $12$.\soln $432\pi\approx1357.17$.
\end{example}

\begin{example}{ex7}{Example 8 --- A box (SA)}
Surface area of a $5\times6\times7$ box.\soln $2(30+35+42)=214$.
\end{example}

\begin{example}{ex8}{Example 9 --- Drilled block 2}
A $12\times8\times5$ block with a cylindrical hole of radius $2$ through the height $5$.\soln $480-\pi(4)(5)=480-20\pi\approx417.17$.
\end{example}
