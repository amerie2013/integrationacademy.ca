\begin{ideabox}
An annuity is a series of equal, regular payments. Its future value — what the deposits grow to — is $FV=R\cdot\dfrac{(1+i)^{n}-1}{i}$, with payment $R$, rate per period $i$, and $n$ payments.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $FV=R\cdot\dfrac{(1+i)^{n}-1}{i}$.
  \item Monthly: $i=\tfrac{r}{12}$, $n=12\times$ years.
  \item Starting early raises the future value sharply.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Future value}An annuity's future value is $FV=R\cdot\dfrac{(1+i)^{n}-1}{i}$, where $R$ is the payment, $i$ the rate per period, $n$ the number of payments.
\lh{Read the schedule}Monthly payments at annual rate $r$ use $i=\tfrac{r}{12}$ and $n=12\times(\text{years})$.
\lh{Start early}Early deposits compound longest, so beginning sooner sharply raises the future value.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Yearly deposits}
$\$500$ at the end of each year at $6\%$/yr for $10$ years.\soln $500\cdot\dfrac{(1.06)^{10}-1}{0.06}\approx\$6590.40$.
\end{example}

\begin{example}{ex1}{Example 2 --- Monthly deposits}
$\$200$/month at $6\%$/yr compounded monthly for $5$ years.\soln $i=0.005,\ n=60$: $\approx\$13\,954.01$.
\end{example}

\begin{example}{ex2}{Example 3 --- Doubling the payment}
How does the future value in Example 1 change at $\$1000$/year?\soln The factor is unchanged; $1000\cdot13.1808\approx\$13\,180.80$ — double.
\end{example}

\begin{example}{ex3}{Example 4 --- Starting early}
$\$100$/month at $6\%$/yr compounded monthly: $40$ years vs $30$ years.\soln $40$ yr $\approx\$199\,149$; $30$ yr $\approx\$100\,452$ — early start nearly doubles it.
\end{example}

\begin{example}{ex4}{Example 5 --- Short plan}
$\$1000$/year at $5\%$/yr for $3$ years.\soln $1000\cdot\dfrac{(1.05)^{3}-1}{0.05}=\$3152.50$.
\end{example}

\begin{example}{ex5}{Example 6 --- Eight-year plan}
$\$250$/year at $5\%$/yr for $8$ years.\soln $250\cdot\dfrac{(1.05)^{8}-1}{0.05}\approx\$2387.28$.
\end{example}

\begin{example}{ex6}{Example 7 --- Deposits vs interest}
For Example 2, how much did you deposit, and how much is interest?\soln Deposits $200\cdot60=\$12\,000$; interest $\approx13\,954-12\,000=\$1954$.
\end{example}

\begin{example}{ex7}{Example 8 --- Monthly, long term}
$\$100$/month at $6\%$/yr compounded monthly for $5$ years.\soln $100\cdot\dfrac{(1.005)^{60}-1}{0.005}\approx\$6977.00$.
\end{example}

\begin{example}{ex8}{Example 9 --- Yearly, larger rate}
$\$300$/year at $4\%$/yr for $6$ years.\soln $300\cdot\dfrac{(1.04)^{6}-1}{0.04}\approx\$1989.89$.
\end{example}
