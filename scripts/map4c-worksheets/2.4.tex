\begin{ideabox}
A mortgage is a present-value annuity: $R=\dfrac{PV\cdot i}{1-(1+i)^{-n}}$, where $PV$ is the amount borrowed (price minus down payment). Each payment's interest is (balance $\times i$); the rest reduces the balance.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Payment $R=\dfrac{PV\cdot i}{1-(1+i)^{-n}}$.
  \item Interest $=$ balance $\times i$; principal $=R-$ interest.
  \item Shorter term $\Rightarrow$ higher payment, less total interest.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Mortgage payment}A mortgage is a present-value annuity: $R=\dfrac{PV\cdot i}{1-(1+i)^{-n}}$, where $PV$ is the amount borrowed (price minus down payment).
\lh{Amortization split}Each payment's interest is (balance $\times i$); the rest reduces the balance. Early payments are mostly interest.
\lh{Term matters}A shorter term means a higher payment but far less total interest.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Monthly payment}
A $\$200\,000$ mortgage at $6\%$/yr ($i=0.005$) over $25$ years ($n=300$).\soln $R=\dfrac{200\,000(0.005)}{1-(1.005)^{-300}}\approx\$1288.60$/month.
\end{example}

\begin{example}{ex1}{Example 2 --- First row of the table}
Split the first payment on that mortgage.\soln Interest $200\,000(0.005)=\$1000$; principal $1288.60-1000=\$288.60$; new balance $\$199\,711.40$.
\end{example}

\begin{example}{ex2}{Example 3 --- Total interest}
Total paid and interest on the $25$-year mortgage.\soln Total $1288.60\times300\approx\$386\,580$; interest $\approx\$186\,580$.
\end{example}

\begin{example}{ex3}{Example 4 --- A shorter term}
Same mortgage over $20$ years ($n=240$).\soln $R\approx\$1432.86$; interest $1432.86(240)-200\,000\approx\$143\,887$.
\end{example}

\begin{example}{ex4}{Example 5 --- Down payment}
A $\$250\,000$ home with a $\$25\,000$ down payment — mortgage amount?\soln $250\,000-25\,000=\$225\,000$.
\end{example}

\begin{example}{ex5}{Example 6 --- Bigger mortgage}
A $\$250\,000$ mortgage at $6\%$/yr ($i=0.005$) over $25$ years.\soln $R\approx\$1610.75$/month.
\end{example}

\begin{example}{ex6}{Example 7 --- Lower rate}
A $\$300\,000$ mortgage at $4.8\%$/yr ($i=0.004$) over $25$ years.\soln $R\approx\$1718.99$/month.
\end{example}

\begin{example}{ex7}{Example 8 --- First interest}
First-payment interest on a $\$180\,000$ balance at $i=0.005$.\soln $180\,000(0.005)=\$900$.
\end{example}

\begin{example}{ex8}{Example 9 --- Second payment}
After the first payment, the balance is $\$199\,711.40$. Split the second payment ($R=\$1288.60$).\soln Interest $199\,711.40(0.005)\approx\$998.56$; principal $\approx\$290.04$.
\end{example}
