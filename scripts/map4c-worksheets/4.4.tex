{\small Grade 12 Foundations for College Mathematics (MAP4C) \textperiodcentered{} Unit 4: Data Management}

\begin{ideabox}
Correlation has a direction (positive/negative/none) and strength (tight line $=$ strong). The line of best fit $y=mx+b$ (from technology) predicts a $y$ from an $x$; interpolate within the data, not far beyond it.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Direction and strength.
  \item Line of best fit $y=mx+b$.
  \item Interpolate (safe) vs extrapolate (risky).
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Correlation}Direction (positive/negative/none) and strength (tight line = strong, loose cloud = weak).
\lh{Line of best fit}The trend line $y=mx+b$ (from technology) predicts a $y$ from an $x$.
\lh{Interpolate vs extrapolate}Predicting within the data is reliable; far beyond it is risky — and correlation is not causation.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Naming a correlation}
Temperature vs ice-cream sales trends clearly upward. Describe the correlation.\soln Strong positive.
\end{example}

\begin{example}{ex1}{Example 2 --- A weak negative}
Hours of TV vs test score trends down but loosely. Describe it.\soln Weak negative.
\end{example}

\begin{example}{ex2}{Example 3 --- Predict from the line}
Line $y=2x+3$ (score vs hours). Predict at $10$ hours.\soln $2(10)+3=23$.
\end{example}

\begin{example}{ex3}{Example 4 --- Interpolate vs extrapolate}
Data covers $0$–$8$ hours. Safer prediction: at $5$ or at $40$ hours?\soln At $5$ (inside the data) — interpolation is reliable.
\end{example}

\begin{example}{ex4}{Example 5 --- Correlation not cause}
Ice-cream sales and drownings rise together. Does ice cream cause drownings?\soln No — hot weather (a hidden variable) drives both.
\end{example}

\begin{example}{ex5}{Example 6 --- Predict again}
Line $y=1.5x+2$. Predict at $x=8$.\soln $1.5(8)+2=14$.
\end{example}

\begin{example}{ex6}{Example 7 --- Strong negative}
Points trend downward in a tight line. Describe the correlation.\soln Strong negative.
\end{example}

\begin{example}{ex7}{Example 8 --- No correlation}
A scattered cloud with no trend. Correlation?\soln None.
\end{example}

\begin{example}{ex8}{Example 9 --- Unrealistic prediction}
Line $y=8x+40$ (score). Predict at $12$ hours; comment.\soln $136$ — impossible ($>100$), so the extrapolation is unreliable.
\end{example}
