\begin{ideabox}
Same-base powers combine by adding, subtracting, or multiplying exponents. A negative exponent is a reciprocal ($a^{-n}=\tfrac{1}{a^n}$); a rational exponent is a root ($a^{m/n}=\sqrt[n]{a^m}$).
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Product/quotient/power laws (same base).
  \item $a^0=1$, $a^{-n}=\dfrac{1}{a^n}$.
  \item $a^{m/n}=\left(\sqrt[n]{a}\right)^m$ â€” root first.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Powers and the laws}Same-base powers combine: $a^m a^n=a^{m+n}$, $\dfrac{a^m}{a^n}=a^{m-n}$, $(a^m)^n=a^{mn}$; also $a^0=1$ and $a^{-n}=\dfrac{1}{a^n}$.
\lh{Rational exponents}A fraction exponent is a root: $a^{m/n}=\left(\sqrt[n]{a}\right)^m$. Take the root first â€” $8^{2/3}=(\sqrt[3]{8})^2=4$.
\lh{Work in steps}Apply one law at a time and clear negative exponents last by moving the factor across the fraction bar.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Combine like bases}
Simplify $\dfrac{x^5\cdot x^2}{x^3}$.\soln $\dfrac{x^7}{x^3}=x^4$.
\end{example}

\begin{example}{ex1}{Example 2 --- Power of a product}
Simplify $(2x^3)^4\cdot x^{-5}$.\soln $16x^{12}\cdot x^{-5}=16x^7$.
\end{example}

\begin{example}{ex2}{Example 3 --- Evaluate a rational power}
Evaluate $27^{2/3}$.\soln $(\sqrt[3]{27})^2=3^2=9$.
\end{example}

\begin{example}{ex3}{Example 4 --- Negative rational power}
Evaluate $16^{-3/4}$.\soln $\dfrac{1}{(\sqrt[4]{16})^3}=\dfrac{1}{2^3}=\dfrac18$.
\end{example}

\begin{example}{ex4}{Example 5 --- Negative exponents, variables}
Simplify with positive exponents: $(4x^{-2}y^5)^{-1}$.\soln $4^{-1}x^2y^{-5}=\dfrac{x^2}{4y^5}$.
\end{example}

\begin{example}{ex5}{Example 6 --- Fifth root power}
Evaluate $32^{3/5}$.\soln $(\sqrt[5]{32})^3=2^3=8$.
\end{example}

\begin{example}{ex6}{Example 7 --- Quotient with variables}
Simplify $\dfrac{6a^5b^2}{2a^2b^5}$.\soln $3a^{3}b^{-3}=\dfrac{3a^3}{b^3}$.
\end{example}

\begin{example}{ex7}{Example 8 --- A growth factor}
Write $(1.05)^2\cdot(1.05)^3$ as one power and state what it models.\soln $(1.05)^5$ â€” five years of $5\%$ annual growth.
\end{example}

\begin{example}{ex8}{Example 9 --- Mixed evaluation}
Evaluate $8^{2/3}\cdot 9^{-1/2}$.\soln $4\cdot\tfrac13=\tfrac43$.
\end{example}
