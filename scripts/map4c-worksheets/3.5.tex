\begin{ideabox}
For a fixed volume, the least-surface-area rectangular prism is a cube; a cylinder is best when $h=2r$. Use a table of dimensions with the right volume and read the minimum surface area.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Fixed volume $\Rightarrow$ least surface area is a cube.
  \item Cylinder best when $h=2r$.
  \item Least surface area $=$ least material.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The cube rule}For a fixed volume, the least surface area is a cube; for a cylinder it is when $h=2r$ (height equals diameter).
\lh{Use a table}List candidate dimensions with the right volume and read off the least surface area.
\lh{Why it matters}Least surface area means least material — real savings in packaging.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Least material for a volume}
A box holds $64$ cm$^3$. Among $2\times2\times16$, $2\times4\times8$, $4\times4\times4$, which uses the least material?\soln $SA$: $136$, $112$, $96$; the cube $4\times4\times4$ ($96$).
\end{example}

\begin{example}{ex1}{Example 2 --- Confirm the cube}
Verify the cube's surface area above.\soln $6(4\times4)=96$ cm$^2$.
\end{example}

\begin{example}{ex2}{Example 3 --- A different volume}
Least-material box for $27$ cm$^3$, and its surface area?\soln Cube $3\times3\times3$; $SA=6(9)=54$ cm$^2$.
\end{example}

\begin{example}{ex3}{Example 4 --- Why it matters}
A $64$ cm$^3$ product ships as $2\times4\times8$ or a cube. Cardboard saved by the cube?\soln $112-96=16$ cm$^2$ per box.
\end{example}

\begin{example}{ex4}{Example 5 --- The cylinder rule}
For a fixed volume, what shape of cylinder uses the least material?\soln Height equals diameter, $h=2r$ (a \emph{square} cylinder).
\end{example}

\begin{example}{ex5}{Example 6 --- Cube for 125}
Least-material box for $125$ cm$^3$.\soln Cube $5\times5\times5$; $SA=6(25)=150$ cm$^2$.
\end{example}

\begin{example}{ex6}{Example 7 --- Cube for 216}
Least-material box for $216$ cm$^3$.\soln Cube $6\times6\times6$; $SA=6(36)=216$ cm$^2$.
\end{example}

\begin{example}{ex7}{Example 8 --- A long box}
Surface area of a $2\times2\times8$ box (volume $32$).\soln $2(4+16+16)=72$ cm$^2$.
\end{example}

\begin{example}{ex8}{Example 9 --- Compare shapes}
For $32$ cm$^3$: compare $2\times2\times8$ with $2\times4\times4$.\soln $2\times2\times8\Rightarrow72$; $2\times4\times4\Rightarrow2(8+8+16)=64$ — the more cube-like box uses less.
\end{example}
