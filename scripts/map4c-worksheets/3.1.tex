\begin{ideabox}
Metric units scale by tens ($1$ m $=100$ cm, $1$ km $=1000$ m). Bridge to imperial with $1$ in $=2.54$ cm, $1$ ft $\approx0.305$ m, $1$ mi $\approx1.609$ km. Multiply by a ratio equal to $1$ so the old unit cancels.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Metric conversions by powers of ten.
  \item Metric $\leftrightarrow$ imperial bridges.
  \item Cancel the unwanted unit.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Metric scaling}Metric units scale by tens: $1\text{ m}=100\text{ cm}$, $1\text{ km}=1000\text{ m}$. To a bigger unit divides; to a smaller unit multiplies.
\lh{Metric $\leftrightarrow$ imperial}Key bridges: $1\text{ in}=2.54\text{ cm}$, $1\text{ ft}\approx0.305\text{ m}$, $1\text{ mi}\approx1.609\text{ km}$.
\lh{Cancel the unit}Multiply by a ratio equal to $1$ arranged so the old unit cancels.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Within metric}
Convert $250$ cm to metres.\soln $250\times\dfrac{1\text{ m}}{100\text{ cm}}=2.5$ m.
\end{example}

\begin{example}{ex1}{Example 2 --- km to m}
Convert $5$ km to metres.\soln $5\times1000=5000$ m.
\end{example}

\begin{example}{ex2}{Example 3 --- Inches to cm}
Convert $12$ in to cm.\soln $12\times2.54=30.48$ cm.
\end{example}

\begin{example}{ex3}{Example 4 --- Feet to metres}
Convert $10$ ft to metres.\soln $10\times0.3048\approx3.05$ m.
\end{example}

\begin{example}{ex4}{Example 5 --- km to miles}
Convert $100$ km to miles.\soln $\dfrac{100}{1.609}\approx62.15$ miles.
\end{example}

\begin{example}{ex5}{Example 6 --- Miles to km}
Convert $3$ miles to km.\soln $3\times1.609\approx4.83$ km.
\end{example}

\begin{example}{ex6}{Example 7 --- Feet to cm}
Convert $6$ ft to cm.\soln $6\times30.48=182.88$ cm.
\end{example}

\begin{example}{ex7}{Example 8 --- Gallons to litres}
Convert $10$ US gallons to litres.\soln $10\times3.785=37.85$ L.
\end{example}

\begin{example}{ex8}{Example 9 --- Kilograms to pounds}
Convert $5$ kg to pounds.\soln $5\times2.205\approx11.03$ lb.
\end{example}
