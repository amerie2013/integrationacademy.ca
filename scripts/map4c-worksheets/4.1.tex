\begin{ideabox}
Summarize data with the mean (average), median (middle value in order), mode (most frequent), and range (highest $-$ lowest). Bar/circle graphs suit categories; histograms suit numeric intervals.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Mean, median, mode.
  \item Range $=$ highest $-$ lowest.
  \item Choose the graph to match the data type.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Centre}Mean (average), median (middle value in order), mode (most frequent). For an even count, average the two middle values.
\lh{Spread and graphs}Range $=$ highest $-$ lowest. Bar/circle graphs suit categories; histograms suit numeric intervals.
\lh{Outliers}An outlier pulls the mean, so the median often better represents a typical value.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- The mean}
Find the mean of $12, 15, 15, 18, 20$.\soln $\dfrac{80}{5}=16$.
\end{example}

\begin{example}{ex1}{Example 2 --- Median (odd)}
Find the median of $7, 3, 9, 5, 11$.\soln Order $3,5,7,9,11$; middle $=7$.
\end{example}

\begin{example}{ex2}{Example 3 --- Median (even)}
Find the median of $4, 8, 10, 14$.\soln Average the middles: $\dfrac{8+10}{2}=9$.
\end{example}

\begin{example}{ex3}{Example 4 --- Mode and range}
For $5, 8, 8, 8, 10, 12$, find the mode and range.\soln Mode $8$; range $12-5=7$.
\end{example}

\begin{example}{ex4}{Example 5 --- Choosing a graph}
Which graph suits (a) favourite sports, (b) test-score distribution?\soln (a) bar/circle; (b) histogram.
\end{example}

\begin{example}{ex5}{Example 6 --- Mean of five}
Find the mean of $3, 6, 9, 12, 15$.\soln $\dfrac{45}{5}=9$.
\end{example}

\begin{example}{ex6}{Example 7 --- Median of six}
Find the median of $2, 4, 6, 8, 10, 12$.\soln $\dfrac{6+8}{2}=7$.
\end{example}

\begin{example}{ex7}{Example 8 --- Mode}
Find the mode of $4, 4, 7, 9, 9, 9$.\soln $9$ (three times).
\end{example}

\begin{example}{ex8}{Example 9 --- Outlier effect}
For $4, 5, 6, 7, 100$, find the mean and median.\soln Mean $\dfrac{122}{5}=24.4$; median $6$ — the median is more typical.
\end{example}
