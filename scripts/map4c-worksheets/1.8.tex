\begin{ideabox}
Undo a power with a root: $x^{n}=a\Rightarrow x=\sqrt[n]{a}$ (even $n$ gives $\pm$). Rearrange formulas by undoing operations in reverse; keep only positive roots for measurements.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $x^{n}=a\Rightarrow x=\sqrt[n]{a}$; even $n$ gives $\pm$.
  \item Isolate, then take the matching root.
  \item Physical lengths use the positive root.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Solving $x^{n}=a$}Undo a power with a root: $x=\sqrt[n]{a}$. An even $n$ (with $a>0$) gives $\pm$; an odd $n$ gives one root.
\lh{Rearranging formulas}Isolate a variable by undoing operations in reverse, treating the other letters as constants; a power is undone by a root.
\lh{Physical roots}For a length or radius, keep only the positive root.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Cube root}
Solve $x^{3}=64$.\soln $x=\sqrt[3]{64}=4$.
\end{example}

\begin{example}{ex1}{Example 2 --- Even power}
Solve $x^{2}=49$.\soln $x=\pm7$.
\end{example}

\begin{example}{ex2}{Example 3 --- Radius of a sphere}
$V=\tfrac43\pi r^{3}=1000$. Find $r$.\soln $r^3=\dfrac{3(1000)}{4\pi}\approx238.7$, so $r\approx6.2$ cm.
\end{example}

\begin{example}{ex3}{Example 4 --- Rearrange for a radius}
Solve $A=\pi r^{2}$ for $r$.\soln $r=\sqrt{\dfrac{A}{\pi}}$ (positive).
\end{example}

\begin{example}{ex4}{Example 5 --- Formula as a relation}
In $A=P(1+i)^n$ with $P,i$ fixed, what relation is $A$ vs $n$?\soln Exponential (the variable is in the exponent).
\end{example}

\begin{example}{ex5}{Example 6 --- Side of a cube}
$V=s^3=216$. Find $s$.\soln $s=\sqrt[3]{216}=6$.
\end{example}

\begin{example}{ex6}{Example 7 --- Rearrange a box}
Solve $V=lwh$ for $h$.\soln $h=\dfrac{V}{lw}$.
\end{example}

\begin{example}{ex7}{Example 8 --- Square root of an area}
A square has area $200$ cm$^2$. Find its side (1 dp).\soln $s=\sqrt{200}\approx14.1$ cm.
\end{example}

\begin{example}{ex8}{Example 9 --- Fifth power}
Solve $x^{5}=32$.\soln $x=\sqrt[5]{32}=2$.
\end{example}
