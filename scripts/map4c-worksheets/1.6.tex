{\LARGE\bfseries\color{titleblue}Worksheet 1.6 --- Rate of Change from Graphs \& Tables}\\[2pt]
{\small Grade 12 Foundations for College Mathematics (MAP4C) \textperiodcentered{} Unit 1: Mathematical Models}
\end{center}

\begin{ideabox}
Rate of change $=\dfrac{\text{change in output}}{\text{change in input}}$ — the slope, with units \emph{output per input}. Equal first differences mean a constant rate; a curve means a changing rate.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Rate $=\dfrac{\Delta\text{output}}{\Delta\text{input}}$, with units.
  \item Constant (line) vs changing (curve).
  \item Zero rate $=$ horizontal graph.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Rate of change}Between two points, rate of change $=\dfrac{\text{change in output}}{\text{change in input}}$ — the slope, with units "output per input".
\lh{Constant vs changing}Equal first differences (straight line) mean a constant rate; unequal differences (a curve) mean a changing rate.
\lh{Zero rate}A flat, horizontal graph has a rate of change of zero.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Rate from two points}
A car travels $90$ km in $1.5$ h at steady speed. Find the rate.\soln $\dfrac{90}{1.5}=60$ km/h.
\end{example}

\begin{example}{ex1}{Example 2 --- Units of a rate}
Earnings go \$0 to \$180 over $12$ h. State the rate.\soln $\dfrac{180}{12}=\$15$/h.
\end{example}

\begin{example}{ex2}{Example 3 --- Constant (table)}
Outputs $2,5,8,11$ at $x=0,1,2,3$. Constant rate?\soln First differences $3,3,3$ — constant rate $3$ (linear).
\end{example}

\begin{example}{ex3}{Example 4 --- Changing (table)}
Outputs $3,6,12,24$. Describe the rate.\soln Differences $3,6,12$ — changing; constant ratio $2$ (exponential).
\end{example}

\begin{example}{ex4}{Example 5 --- Compare graphs}
Car A's graph is a line; car B's curves up. Compare rates.\soln A constant speed; B increasing speed (accelerating).
\end{example}

\begin{example}{ex5}{Example 6 --- Fill rate}
A tank fills $40$ L in $8$ min. Find the rate.\soln $40/8=5$ L/min.
\end{example}

\begin{example}{ex6}{Example 7 --- Quadratic differences}
Outputs $1,4,9,16$. Constant rate?\soln Differences $3,5,7$ — changing (quadratic).
\end{example}

\begin{example}{ex7}{Example 8 --- Zero rate}
A graph sits at $y=25$ for all $x$. Rate of change?\soln Zero — no change in output.
\end{example}

\begin{example}{ex8}{Example 9 --- Interpret intercept}
A plan costs \$20 plus \$0.10/min. Rate and meaning of \$20?\soln Rate \$0.10/min; \$20 is the fixed base fee.
\end{example}
