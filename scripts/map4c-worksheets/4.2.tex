\begin{ideabox}
The population is everyone of interest; a sample is the part studied. A good sample is random, representative, and large enough. Bias makes a sample misrepresent the population.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Population vs sample.
  \item Good sample: random, representative, large enough.
  \item Watch for bias and misleading graphs.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Population vs sample}The population is everyone of interest; a sample is the part studied. A good sample is random, representative, and large enough.
\lh{Bias}Bias makes a sample misrepresent the population — a skewed frame, leading questions, or non-response.
\lh{Misuse}Watch for tiny samples, cherry-picked data, and truncated graph axes that exaggerate.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Population vs sample}
A school has $1200$ students; $100$ are surveyed. Identify each.\soln Population $=1200$; sample $=100$.
\end{example}

\begin{example}{ex1}{Example 2 --- A biased sample}
Surveying only the band class about the whole school's music taste. Good sample?\soln Biased — not representative.
\end{example}

\begin{example}{ex2}{Example 3 --- A leading question}
\emph{Don't you agree our great cafeteria should stay open later?} Why biased?\soln The wording pushes toward \emph{yes} — response bias.
\end{example}

\begin{example}{ex3}{Example 4 --- A tiny sample}
An ad claims \emph{$80\%$ preferred Brand X} from $5$ people. The concern?\soln $80\%$ is only $4$ of $5$ — far too few.
\end{example}

\begin{example}{ex4}{Example 5 --- A better design}
How would you fairly sample the school's lunch preference?\soln Randomly select across all grades, with a large, varied group and neutral questions.
\end{example}

\begin{example}{ex5}{Example 6 --- Naming both}
A city has $50\,000$ voters; a poll asks $800$. Name each.\soln Population $50\,000$; sample $800$.
\end{example}

\begin{example}{ex6}{Example 7 --- Bias check}
Surveying shoppers only at a luxury store about spending. Biased?\soln Yes — not representative of all shoppers.
\end{example}

\begin{example}{ex7}{Example 8 --- Good-sample feature}
Name one feature of a good sample.\soln Random (or representative, or large enough).
\end{example}

\begin{example}{ex8}{Example 9 --- Misleading graph}
A graph's vertical axis starts at $90$, so $92\to96$ looks huge. What is this?\soln A misleading (truncated) scale.
\end{example}
