\begin{center}
{\LARGE\bfseries\color{titleblue}Worksheet 3.7 --- The Sine Law \& the Cosine Law}\\[2pt]
{\small Grade 12 Foundations for College Mathematics (MAP4C) \textperiodcentered{} Unit 3: Geometry \& Trigonometry}
\end{center}

\begin{ideabox}
Sine law $\dfrac{a}{\sin A}=\dfrac{b}{\sin B}=\dfrac{c}{\sin C}$ (matched side–angle pair); cosine law $a^2=b^2+c^2-2bc\cos A$ (SAS or SSS). An obtuse angle has a negative cosine.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Sine law: matched pair.
  \item Cosine law: SAS or SSS.
  \item Obtuse angle $\Rightarrow$ negative cosine.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Two laws}Sine law $\dfrac{a}{\sin A}=\dfrac{b}{\sin B}=\dfrac{c}{\sin C}$ (matched pair); cosine law $a^2=b^2+c^2-2bc\cos A$ (SAS or SSS).
\lh{Obtuse angles}An obtuse angle has a negative cosine, which lengthens the opposite side in the cosine law.
\lh{Choose the law}Use the sine law with a side and its opposite angle; otherwise start with the cosine law.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Sine law, side}
$A=40^\circ$, $B=60^\circ$, $a=10$. Find $b$.\soln $b=\dfrac{10\sin60^\circ}{\sin40^\circ}\approx13.47$.
\end{example}

\begin{example}{ex1}{Example 2 --- Sine law, angle}
$a=8$, $b=6$, $A=50^\circ$. Find $B$.\soln $\sin B=\dfrac{6\sin50^\circ}{8}\approx0.5745\Rightarrow B\approx35.1^\circ$.
\end{example}

\begin{example}{ex2}{Example 3 --- Cosine law, side}
$b=7$, $c=9$, included $A=60^\circ$. Find $a$.\soln $a^2=49+81-2(63)\cos60^\circ=67\Rightarrow a\approx8.19$.
\end{example}

\begin{example}{ex3}{Example 4 --- Cosine law, angle}
Sides $6,8,10$. Find the largest angle (opposite $10$).\soln $\cos C=\dfrac{36+64-100}{2(48)}=0\Rightarrow C=90^\circ$.
\end{example}

\begin{example}{ex4}{Example 5 --- Obtuse triangle}
$b=5$, $c=8$, $A=120^\circ$. Find $a$.\soln $a^2=25+64-80\cos120^\circ=129\Rightarrow a\approx11.36$.
\end{example}

\begin{example}{ex5}{Example 6 --- Sine law, side 2}
$A=40^\circ$, $B=70^\circ$, $a=10$. Find $b$.\soln $b=\dfrac{10\sin70^\circ}{\sin40^\circ}\approx14.62$.
\end{example}

\begin{example}{ex6}{Example 7 --- Cosine law, side 2}
$b=8$, $c=10$, $A=50^\circ$. Find $a$.\soln $a=\sqrt{64+100-160\cos50^\circ}\approx7.82$.
\end{example}

\begin{example}{ex7}{Example 8 --- Largest angle}
Sides $7,9,12$. Find the largest angle (opposite $12$).\soln $\cos C=\dfrac{49+81-144}{2(63)}\approx-0.111\Rightarrow C\approx96.4^\circ$.
\end{example}

\begin{example}{ex8}{Example 9 --- Two roads}
Two roads leave a town at $75^\circ$; cars go $9$ km and $6$ km. Distance apart?\soln $\sqrt{81+36-108\cos75^\circ}\approx9.44$ km.
\end{example}
