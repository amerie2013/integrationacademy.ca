\begin{ideabox}
In a right triangle: $\sin=\tfrac{\text{opp}}{\text{hyp}}$, $\cos=\tfrac{\text{adj}}{\text{hyp}}$, $\tan=\tfrac{\text{opp}}{\text{adj}}$ (SOH-CAH-TOA). Find a side with a ratio; find an angle with an inverse ratio.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item SOH-CAH-TOA.
  \item Side: set up a ratio and solve.
  \item Angle: use $\sin^{-1},\cos^{-1},\tan^{-1}$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The three ratios}$\sin=\dfrac{\text{opp}}{\text{hyp}}$, $\cos=\dfrac{\text{adj}}{\text{hyp}}$, $\tan=\dfrac{\text{opp}}{\text{adj}}$ — SOH-CAH-TOA.
\lh{Sides and angles}For a side, set up a ratio and solve; for an angle, use an inverse ratio ($\sin^{-1}$, etc.).
\lh{Elevation/depression}Angles up or down from the horizontal build the triangle in real problems.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Find the opposite}
Hypotenuse $10$, angle $30^\circ$. Find the opposite side.\soln $10\sin30^\circ=10(0.5)=5$.
\end{example}

\begin{example}{ex1}{Example 2 --- Find the hypotenuse}
Opposite $8$ to a $40^\circ$ angle. Find the hypotenuse.\soln $\dfrac{8}{\sin40^\circ}\approx12.45$.
\end{example}

\begin{example}{ex2}{Example 3 --- Find an angle}
Opposite $5$, adjacent $12$. Find the angle.\soln $\tan^{-1}\!\left(\tfrac{5}{12}\right)\approx22.6^\circ$.
\end{example}

\begin{example}{ex3}{Example 4 --- Angle of elevation}
From $20$ m away, the elevation to a building top is $55^\circ$. Find the height.\soln $20\tan55^\circ\approx28.56$ m.
\end{example}

\begin{example}{ex4}{Example 5 --- Find the adjacent}
A $15$-m ramp makes $25^\circ$ with the ground. Horizontal reach?\soln $15\cos25^\circ\approx13.59$ m.
\end{example}

\begin{example}{ex5}{Example 6 --- Opposite side}
Hypotenuse $15$, angle $30^\circ$. Find the opposite side.\soln $15\sin30^\circ=7.5$.
\end{example}

\begin{example}{ex6}{Example 7 --- Find angle 2}
Opposite $9$, adjacent $12$. Find the angle.\soln $\tan^{-1}\!\left(\tfrac{9}{12}\right)\approx36.87^\circ$.
\end{example}

\begin{example}{ex7}{Example 8 --- Height 2}
From $30$ m, elevation $50^\circ$ to a tower top. Height?\soln $30\tan50^\circ\approx35.75$ m.
\end{example}

\begin{example}{ex8}{Example 9 --- Adjacent 2}
Hypotenuse $12$, angle $40^\circ$. Find the adjacent side.\soln $12\cos40^\circ\approx9.19$.
\end{example}
