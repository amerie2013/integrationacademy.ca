\begin{center}
{\LARGE\bfseries\color{titleblue}Worksheet 1.7 --- Choosing a Model: Linear, Quadratic, or Exponential}\\[2pt]
{\small Grade 12 Foundations for College Mathematics (MAP4C) \textperiodcentered{} Unit 1: Mathematical Models}
\end{center}

\begin{ideabox}
Constant first differences signal a linear model; constant second differences a quadratic; a constant ratio an exponential. A constant \emph{amount} of change is linear; a constant \emph{percentage} is exponential.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Linear: constant first difference.
  \item Quadratic: constant second difference.
  \item Exponential: constant ratio.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The fingerprints}Constant first difference $\Rightarrow$ linear; constant second difference $\Rightarrow$ quadratic; constant ratio $\Rightarrow$ exponential.
\lh{Amount vs percent}A constant \emph{amount} of change per step is linear; a constant \emph{percentage} per step is exponential.
\lh{Write the model}Read the starting value and the step (add for linear, multiply for exponential) to build $y=mx+b$ or $y=a\,b^{x}$.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Spot linear}
$y=7,10,13,16$ at $x=0,1,2,3$. Which model?\soln First differences $3$ — linear, $y=3x+7$.
\end{example}

\begin{example}{ex1}{Example 2 --- Spot exponential}
$y=5,10,20,40$. Which model?\soln Ratio $2$ — exponential, $y=5\cdot2^{x}$.
\end{example}

\begin{example}{ex2}{Example 3 --- Spot quadratic}
$y=1,4,9,16$. Which model?\soln First differences $3,5,7$; second differences $2$ — quadratic ($y=x^2$).
\end{example}

\begin{example}{ex3}{Example 4 --- From a description}
Account A adds \$100/month; account B grows $4\%$/month. Which is which?\soln A linear (constant amount), B exponential (constant percent).
\end{example}

\begin{example}{ex4}{Example 5 --- Build the model}
A colony is $3$ at $x=0$ and triples each step. Write the model; predict $x=4$.\soln $y=3\cdot3^{x}$; $y=3\cdot81=243$.
\end{example}

\begin{example}{ex5}{Example 6 --- Decay model}
Data start at $100$ and drop $20\%$ each step. Write the model.\soln Ratio $0.8$: $y=100(0.8)^{x}$.
\end{example}

\begin{example}{ex6}{Example 7 --- Linear model}
$y=2,6,10,14$. Write the model.\soln First differences $4$: $y=4x+2$.
\end{example}

\begin{example}{ex7}{Example 8 --- Quadratic build}
$y=5,8,13,20,29$. Identify and give the equation.\soln Second differences $2$ — quadratic; $y=x^2+2x+5$.
\end{example}

\begin{example}{ex8}{Example 9 --- Compare growth}
A adds \$50/week from \$0; B doubles from \$1/week. Which is larger after $10$ weeks?\soln A $=\$500$; B $=2^{10}=\$1024$ — B (exponential) wins.
\end{example}
