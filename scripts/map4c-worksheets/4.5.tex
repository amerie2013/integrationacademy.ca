\begin{ideabox}
For equally likely outcomes, $P(\text{event})=\dfrac{\text{favourable}}{\text{total}}$, from $0$ to $1$. Experimental probability comes from trials and approaches the theoretical value over many trials.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $P=\dfrac{\text{favourable}}{\text{total}}$.
  \item Experimental $\to$ theoretical over many trials.
  \item Two dice: $36$ outcomes.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Theoretical probability}$P=\dfrac{\text{favourable}}{\text{total}}$, a number from $0$ to $1$, when outcomes are equally likely.
\lh{Experimental probability}From trials: $\dfrac{\text{successes}}{\text{trials}}$; it approaches the theoretical value over many trials.
\lh{Two dice}There are $36$ equally likely outcomes — count the favourable pairs (six make a sum of $7$).
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- A single die}
Probability of rolling an even number.\soln $\dfrac{3}{6}=\dfrac12$.
\end{example}

\begin{example}{ex1}{Example 2 --- Two dice}
Probability the sum is $7$.\soln $6$ ways out of $36$: $\dfrac{6}{36}=\dfrac16$.
\end{example}

\begin{example}{ex2}{Example 3 --- A card}
Probability of drawing a heart from $52$ cards.\soln $\dfrac{13}{52}=\dfrac14$.
\end{example}

\begin{example}{ex3}{Example 4 --- Experimental vs theoretical}
A coin lands heads $12$ times in $20$ flips. Compare.\soln Experimental $\dfrac{12}{20}=0.6$; theoretical $0.5$ — closer with more flips.
\end{example}

\begin{example}{ex4}{Example 5 --- A decision}
A game costs $\$2$; you win $\$10$ with probability $\tfrac16$. Worth playing?\soln Expected win $\tfrac16(10)\approx\$1.67<\$2$ — a long-run loss.
\end{example}

\begin{example}{ex5}{Example 6 --- Rolling a specific number}
Probability of rolling a $5$.\soln $\dfrac16$.
\end{example}

\begin{example}{ex6}{Example 7 --- A red card}
Probability of drawing a red card.\soln $\dfrac{26}{52}=\dfrac12$.
\end{example}

\begin{example}{ex7}{Example 8 --- Complement}
Probability of NOT rolling a $6$.\soln $1-\tfrac16=\tfrac56$.
\end{example}

\begin{example}{ex8}{Example 9 --- Two dice sum 8}
Probability the sum is $8$.\soln $(2,6),(3,5),(4,4),(5,3),(6,2)=5$ ways: $\dfrac{5}{36}$.
\end{example}
