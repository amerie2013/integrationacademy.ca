{\LARGE\bfseries\color{titleblue}Worksheet 3.2 --- Areas of Composite Shapes}\\[2pt]
{\small Grade 12 Foundations for College Mathematics (MAP4C) \textperiodcentered{} Unit 3: Geometry \& Trigonometry}
\end{center}

\begin{ideabox}
Decompose a region into rectangles ($lw$), triangles ($\tfrac12bh$), and circles ($\pi r^2$), then add the pieces or subtract a cut-out. A semicircle is $\tfrac12\pi r^2$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Rectangle $lw$, triangle $\tfrac12bh$, circle $\pi r^2$.
  \item Add attached parts; subtract holes.
  \item Use the radius (half the diameter).
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Building blocks}Rectangle $lw$, triangle $\tfrac12bh$, circle $\pi r^2$, semicircle $\tfrac12\pi r^2$.
\lh{Decompose}Split a region into simple shapes, then add the pieces or subtract a cut-out.
\lh{Watch the radius}Use the radius (half the diameter) in $\pi r^2$, and keep every length in one unit.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Rectangle}
Area of an $8\times5$ rectangle.\soln $40$.
\end{example}

\begin{example}{ex1}{Example 2 --- Triangle}
Area of a triangle, base $10$, height $6$.\soln $\tfrac12(10)(6)=30$.
\end{example}

\begin{example}{ex2}{Example 3 --- Circle}
Area of a circle of radius $4$.\soln $16\pi\approx50.27$.
\end{example}

\begin{example}{ex3}{Example 4 --- L-shape (subtract)}
A $10\times8$ rectangle with a $4\times3$ corner removed.\soln $80-12=68$.
\end{example}

\begin{example}{ex4}{Example 5 --- Rectangle + semicircle}
A $12\times6$ rectangle capped by a semicircle of radius $3$.\soln $72+\tfrac12\pi(3)^2\approx72+14.14=86.14$.
\end{example}

\begin{example}{ex5}{Example 6 --- Larger triangle}
Area of a triangle, base $14$, height $9$.\soln $\tfrac12(14)(9)=63$.
\end{example}

\begin{example}{ex6}{Example 7 --- Ring (annulus)}
Area between circles of radius $8$ and $5$.\soln $\pi(64-25)=39\pi\approx122.52$.
\end{example}

\begin{example}{ex7}{Example 8 --- Circle, radius 6}
Area of a circle of radius $6$.\soln $36\pi\approx113.1$.
\end{example}

\begin{example}{ex8}{Example 9 --- Rectangle + semicircle end}
A $14\times8$ rectangle with a semicircle of radius $4$ on one end.\soln $112+\tfrac12\pi(4)^2\approx112+25.13=137.13$.
\end{example}
