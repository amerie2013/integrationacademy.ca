{\LARGE\bfseries\color{titleblue}Worksheet 2.5 --- Renting vs Owning Accommodation}\\[2pt]
{\small Grade 12 Foundations for College Mathematics (MAP4C) \textperiodcentered{} Unit 2: Personal Finance}
\end{center}

\begin{ideabox}
Compare the fixed and variable costs of renting and owning, test affordability against about $30\%$ of gross income, and weigh the financial and non-financial trade-offs.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Fixed vs variable housing costs.
  \item Affordability $\le$ about $30\%$ of gross income.
  \item Owning builds equity; renting adds flexibility.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Fixed vs variable}Fixed costs (rent/mortgage, insurance, tax) stay steady; variable costs (utilities, repairs) change with use.
\lh{Affordability}A common guideline: total housing cost $\le$ about $30\%$ of gross monthly income.
\lh{Trade-offs}Owning builds equity but adds tax and repairs; renting costs less up front but builds none.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Cost of owning}
Mortgage $\$1288.60$, tax $\$250$, insurance $\$100$, maintenance $\$150$. Monthly cost?\soln $1288.60+250+100+150=\$1788.60$.
\end{example}

\begin{example}{ex1}{Example 2 --- Cost of renting}
Rent $\$1400$ plus tenant insurance $\$30$. Monthly cost?\soln $1400+30=\$1430$.
\end{example}

\begin{example}{ex2}{Example 3 --- Affordability}
Gross income $\$60\,000$/yr. Is $\$1789$ within the $30\%$ guideline?\soln $0.30(5000)=\$1500$; $\$1789>\$1500$ â€” over the guideline.
\end{example}

\begin{example}{ex3}{Example 4 --- Cost over a year}
How much more does owning ($\$1788.60$) cost than renting ($\$1430$) per year?\soln $358.60\times12\approx\$4303$.
\end{example}

\begin{example}{ex4}{Example 5 --- Trade-offs}
Give one financial and one non-financial reason to own.\soln Financial: builds equity. Non-financial: freedom to renovate (but you handle repairs).
\end{example}

\begin{example}{ex5}{Example 6 --- Variable cost}
Which is variable: property tax or utilities?\soln Utilities (they change with use).
\end{example}

\begin{example}{ex6}{Example 7 --- Affordability limit}
Gross income $\$48\,000$/yr. What is the $30\%$ monthly limit?\soln $0.30(4000)=\$1200$.
\end{example}

\begin{example}{ex7}{Example 8 --- Within the limit?}
Rent $\$1350$ + insurance $\$25$ â€” within the $\$1200$ limit?\soln $\$1375>\$1200$ â€” over.
\end{example}

\begin{example}{ex8}{Example 9 --- Equity adjustment}
Owning is $\$1789$/month, of which $\$289$ builds equity. What is the non-equity cost?\soln $1789-289=\$1500$/month.
\end{example}
