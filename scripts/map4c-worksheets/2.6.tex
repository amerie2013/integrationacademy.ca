\begin{center}
{\LARGE\bfseries\color{titleblue}Worksheet 2.6 --- Designing Budgets}\\[2pt]
{\small Grade 12 Foundations for College Mathematics (MAP4C) \textperiodcentered{} Unit 2: Personal Finance}
\end{center}

\begin{ideabox}
A budget plans income against expenses (fixed and variable) plus savings; it balances when the total does not exceed income. Reach a goal by saving $\dfrac{\text{goal}}{\text{months}}$; cut variable spending first when money is tight.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Income vs expenses (fixed + variable) + savings.
  \item Monthly saving $=\dfrac{\text{goal}}{\text{months}}$.
  \item Cut variable spending first.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Income vs expenses}A budget balances when expenses plus planned savings do not exceed income; the leftover is a surplus.
\lh{Savings plan}To reach a goal, save $\dfrac{\text{goal}}{\text{months}}$ each month.
\lh{Adjusting}When money is tight, cut variable (discretionary) spending first.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- A balanced budget}
Income $\$3200$. Expenses: rent $1100$, food $450$, transport $300$, utilities $160$, phone $60$, insurance $80$, entertainment $200$, savings $300$. Balanced?\soln Total $\$2650$; surplus $3200-2650=\$550$.
\end{example}

\begin{example}{ex1}{Example 2 --- Fixed vs variable}
Total the fixed and variable expenses above (excluding savings).\soln Fixed $1100+60+80=\$1240$; variable $450+300+160+200=\$1110$.
\end{example}

\begin{example}{ex2}{Example 3 --- A savings plan}
Save $\$6000$ for a car in $2$ years. How much per month?\soln $6000/24=\$250$/month.
\end{example}

\begin{example}{ex3}{Example 4 --- Adjusting to a pay cut}
Income drops $\$400$ to $\$2800$. Rebalance the Example 1 budget.\soln Surplus $2800-2650=\$150$; trim variable spending (e.g. entertainment and food) to restore comfort.
\end{example}

\begin{example}{ex4}{Example 5 --- Percent of income}
What fraction of the $\$3200$ income is rent?\soln $\dfrac{1100}{3200}\approx34.4\%$.
\end{example}

\begin{example}{ex5}{Example 6 --- Simple surplus}
Income $\$2500$; expenses $\$2150$. Surplus?\soln $2500-2150=\$350$.
\end{example}

\begin{example}{ex6}{Example 7 --- Classify a cost}
Is groceries a fixed or variable cost?\soln Variable — it changes with use.
\end{example}

\begin{example}{ex7}{Example 8 --- Savings target}
Save $\$4800$ in $16$ months. How much per month?\soln $4800/16=\$300$/month.
\end{example}

\begin{example}{ex8}{Example 9 --- Rent percentage}
Rent $\$900$ on income $\$3000$ — what percent?\soln $900/3000=30\%$.
\end{example}
