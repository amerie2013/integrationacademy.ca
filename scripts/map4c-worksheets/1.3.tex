{\LARGE\bfseries\color{titleblue}Worksheet 1.3 --- Solving Exponential Equations Graphically}\\[2pt]
{\small Grade 12 Foundations for College Mathematics (MAP4C) \textperiodcentered{} Unit 1: Mathematical Models}
\end{center}

\begin{ideabox}
When no common base works, solve $b^{x}=k$ with a graph or table: the solution is where $y=b^{x}$ reaches $k$. Bracket it between two $x$-values, then zoom in.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Graph $y=b^x$ and $y=k$; read the intersection.
  \item Trap the solution, then refine with a table.
  \item Report an estimate to the required decimals.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{No common base}When numbers are not clean powers of one base, solve $b^{x}=k$ by graphing $y=b^x$ and $y=k$ and reading the intersection's $x$-value.
\lh{Trap with a table}Bracket the solution between two $x$-values whose outputs straddle $k$, then zoom in for more decimals.
\lh{State the accuracy}A graphical solution is an estimate, so report it to the required number of decimals.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Read a clean solution}
Use $y=2^{x}$ to solve $2^{x}=8$.\soln The curve reaches $8$ at $x=3$.
\end{example}

\begin{example}{ex1}{Example 2 --- Bracket a solution}
Between which integers is the solution of $3^{x}=5$?\soln $3^1=3$, $3^2=9$, so $1<x<2$.
\end{example}

\begin{example}{ex2}{Example 3 --- Zoom in}
Estimate the solution of $3^{x}=5$ to one decimal.\soln $3^{1.4}\approx4.66$, $3^{1.5}\approx5.20$, so $x\approx1.5$ (closer to $1.46$).
\end{example}

\begin{example}{ex3}{Example 4 --- Interest question}
When does $1.05^{x}=1.276$?\soln $1.05^{5}\approx1.276$, so $x=5$ years.
\end{example}

\begin{example}{ex4}{Example 5 --- Decay to half}
A drug follows $R=0.8^{t}$. Estimate when half remains, $0.8^{t}=0.5$.\soln $0.8^{3}\approx0.512$, $0.8^{4}\approx0.410$; $0.8^{3.1}\approx0.50$, so $t\approx3.1$ h.
\end{example}

\begin{example}{ex5}{Example 6 --- Two curves}
$y=2^{x}$ and $y=3x+4$. Where are they equal for $x>0$?\soln At $x=4$: $2^4=16$ and $3(4)+4=16$; equal at $x=4$.
\end{example}

\begin{example}{ex6}{Example 7 --- Doubling time}
When does $1.08^{x}=2$?\soln $1.08^{9}\approx2.0$, so about $9$ years.
\end{example}

\begin{example}{ex7}{Example 8 --- Estimate to a decimal}
Estimate the solution of $2^{x}=10$.\soln $2^3=8$, $2^4=16$; $2^{3.3}\approx9.85$, so $x\approx3.3$.
\end{example}

\begin{example}{ex8}{Example 9 --- Growth target}
When does $200(1.06)^{t}=1000$? (So $1.06^{t}=5$.)\soln $1.06^{27}\approx4.82$, $1.06^{28}\approx5.11$, so $t\approx27.6$ years.
\end{example}
