\begin{ideabox}
The present value of an annuity is $PV=R\cdot\dfrac{1-(1+i)^{-n}}{i}$ — the single amount today equal to a stream of payments. For a loan, $PV$ is the amount borrowed and $R$ the repayment.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $PV=R\cdot\dfrac{1-(1+i)^{-n}}{i}$.
  \item Loan payment $R=\dfrac{PV\cdot i}{1-(1+i)^{-n}}$.
  \item Total interest $=R\cdot n-PV$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Present value}The present value is $PV=R\cdot\dfrac{1-(1+i)^{-n}}{i}$ — the single amount today equal to a stream of payments; for a loan it is the amount borrowed.
\lh{Payment on a loan}Rearrange to $R=\dfrac{PV\cdot i}{1-(1+i)^{-n}}$.
\lh{Total interest}Total repaid minus the loan: $R\cdot n-PV$.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Value of a stream}
Present value of $\$500$/year for $10$ years at $6\%$/yr.\soln $500\cdot\dfrac{1-(1.06)^{-10}}{0.06}\approx\$3680.04$.
\end{example}

\begin{example}{ex1}{Example 2 --- How big a loan?}
$\$300$/month for $4$ years at $9\%$/yr compounded monthly repays what loan?\soln $i=0.0075,\ n=48$: $\approx\$12\,055.43$.
\end{example}

\begin{example}{ex2}{Example 3 --- Loan payment}
Monthly payment on a $\$10\,000$ loan at $8\%$/yr compounded monthly over $2$ years, and total interest.\soln $R\approx\$452.27$; interest $452.27(24)-10\,000\approx\$854.55$.
\end{example}

\begin{example}{ex3}{Example 4 --- Present vs future}
Why is the present value less than the future value?\soln PV discounts each payment back to today; FV grows each forward — so PV $<$ deposits $<$ FV.
\end{example}

\begin{example}{ex4}{Example 5 --- Short present value}
Present value of $\$1000$/year for $3$ years at $5\%$/yr.\soln $1000\cdot\dfrac{1-(1.05)^{-3}}{0.05}\approx\$2723.25$.
\end{example}

\begin{example}{ex5}{Example 6 --- Yearly present value}
Present value of $\$250$/year for $10$ years at $5\%$/yr.\soln $250\cdot\dfrac{1-(1.05)^{-10}}{0.05}\approx\$1930.43$.
\end{example}

\begin{example}{ex6}{Example 7 --- Four-year stream}
Present value of $\$500$/year for $4$ years at $6\%$/yr.\soln $500\cdot\dfrac{1-(1.06)^{-4}}{0.06}\approx\$1732.55$.
\end{example}

\begin{example}{ex7}{Example 8 --- Payment from a loan}
Monthly payment on an $\$8000$ loan at $7\%$/yr compounded monthly over $3$ years.\soln $i=\tfrac{0.07}{12},\ n=36$: $R\approx\$247.02$.
\end{example}

\begin{example}{ex8}{Example 9 --- Lump sum vs stream}
A prize pays $\$1000$/year for $20$ years at $5\%$/yr, or $\$13\,000$ now. Which is worth more?\soln $PV\approx\$12\,462$; $\$13\,000$ now is worth more.
\end{example}
