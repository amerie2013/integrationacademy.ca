\begin{center}
{\LARGE\bfseries\color{titleblue}Worksheet 4.3 --- Two-Variable Data \& Scatter Plots}\\[2pt]
{\small Grade 12 Foundations for College Mathematics (MAP4C) \textperiodcentered{} Unit 4: Data Management}
\end{center}

\begin{ideabox}
A scatter plot graphs paired data $(x,y)$: the independent variable (input) on the horizontal axis, the dependent variable (output) on the vertical. The shape of the cloud shows whether the two are related.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Independent on $x$, dependent on $y$.
  \item Upward, downward, or no pattern.
  \item Show separate points, not a line.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Paired data}A scatter plot graphs $(x,y)$ pairs; the independent variable goes on the horizontal axis, the dependent on the vertical.
\lh{Read the cloud}An upward cloud is positive, downward is negative, shapeless is none.
\lh{Points, not lines}A scatter plot shows separate points — do not join them.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Choosing the axes}
Hours studied affects test score. Which is independent?\soln Hours studied (the input) — horizontal axis.
\end{example}

\begin{example}{ex1}{Example 2 --- Upward trend}
$(1,55),(2,60),(3,68),(4,75),(5,82)$ (hours vs score). Describe the pattern.\soln As hours rise, score rises — a positive (upward) pattern.
\end{example}

\begin{example}{ex2}{Example 3 --- Downward trend}
$(0,100),(2,80),(4,65),(6,45)$ (hours used vs battery). Describe it.\soln As use rises, battery falls — a negative pattern.
\end{example}

\begin{example}{ex3}{Example 4 --- No pattern}
Shoe size vs math mark gives a scattered cloud. What does it mean?\soln No relationship.
\end{example}

\begin{example}{ex4}{Example 5 --- Reading a point}
On a height–weight plot, a point is $(170,68)$. Interpret it.\soln A person $170$ cm tall weighing $68$ kg.
\end{example}

\begin{example}{ex5}{Example 6 --- Cause vs effect}
Temperature affects ice-cream sales. Which variable is dependent?\soln Sales (the outcome) — vertical axis.
\end{example}

\begin{example}{ex6}{Example 7 --- Positive pattern}
$(1,3),(2,5),(3,7),(4,9)$. Describe the trend.\soln Steady rise — positive.
\end{example}

\begin{example}{ex7}{Example 8 --- Negative pattern}
$(1,20),(2,16),(3,12),(4,8)$. Describe the trend.\soln Steady fall — negative.
\end{example}

\begin{example}{ex8}{Example 9 --- Strength}
Two study–score plots: one tight along a line, one widely scattered. What differs?\soln The strength — the first is strong, the second weak.
\end{example}
