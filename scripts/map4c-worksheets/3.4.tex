\begin{ideabox}
For a fixed perimeter, the maximum-area rectangle is a square; for a fixed area, the minimum-perimeter rectangle is a square. With one side a free wall, the best pen is a $1{:}2$ rectangle. Reduce to one variable, then use a table.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Fixed perimeter $\Rightarrow$ max area is a square.
  \item Fixed area $\Rightarrow$ min perimeter is a square.
  \item Against a wall $\Rightarrow$ best is $1{:}2$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The square rule}For a fixed perimeter the maximum area is a square; for a fixed area the minimum perimeter is a square.
\lh{Reduce to one variable}Use the constraint to write area (or perimeter) in one variable, then read the optimum from a table or graph.
\lh{Against a wall}With one side a free wall, the best pen is a $1{:}2$ rectangle, not a square.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Max area, fixed perimeter}
Perimeter $24$ ($l+w=12$). Which dimensions maximize area?\soln Table: $4\times8=32$, $5\times7=35$, $6\times6=36$; the $6\times6$ square gives $36$.
\end{example}

\begin{example}{ex1}{Example 2 --- Min perimeter, fixed area}
Area $36$. Which dimensions minimize the perimeter?\soln $4\times9\Rightarrow26$, $6\times6\Rightarrow24$; the square, $24$.
\end{example}

\begin{example}{ex2}{Example 3 --- A garden}
$48$ m of fence for a rectangle. Max area?\soln $l+w=24$; square $12\times12$ gives $144$ m$^2$.
\end{example}

\begin{example}{ex3}{Example 4 --- Pen against a wall}
$30$ m of fence for three sides ($2x+y=30$). Max area?\soln $A=x(30-2x)$; peak at $x=7.5,\ y=15$, area $112.5$ m$^2$.
\end{example}

\begin{example}{ex4}{Example 5 --- Why it matters}
Two $36$ m$^2$ gardens: $6\times6$ and $3\times12$. Which needs less fence?\soln $6\times6\Rightarrow24$; $3\times12\Rightarrow30$; the square saves $6$ m.
\end{example}

\begin{example}{ex5}{Example 6 --- Fixed perimeter 32}
Perimeter $32$. Max-area rectangle?\soln Square $8\times8$, area $64$.
\end{example}

\begin{example}{ex6}{Example 7 --- Fixed area 64}
Area $64$. Min-perimeter rectangle?\soln Square $8\times8$, perimeter $32$.
\end{example}

\begin{example}{ex7}{Example 8 --- Fence 20}
$20$ m of fence for a rectangle. Max area?\soln $l+w=10$; square $5\times5$, area $25$.
\end{example}

\begin{example}{ex8}{Example 9 --- Pen against a wall 2}
$40$ m of fence for three sides. Max area?\soln $2x+y=40$; $x=10,\ y=20$, area $200$ m$^2$.
\end{example}
