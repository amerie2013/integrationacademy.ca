\begin{ideabox}
Probability runs from $0$ (impossible) to $1$ (certain). Theoretical comes from counting outcomes; experimental comes from trials.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Theoretical $P=\dfrac{\text{favourable}}{\text{total}}$.
  \item Experimental $P=\dfrac{\text{successes}}{\text{trials}}$; complement $P(\text{not }A)=1-P(A)$.
  \item Independent events: $P(A\text{ and }B)=P(A)\times P(B)$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Measuring chance}\textbf{Probability} is $\dfrac{\text{favourable outcomes}}{\text{total outcomes}}$, a number from $0$ (impossible) to $1$ (certain).
\lh{Theoretical vs experimental}\textbf{Theoretical} probability comes from counting; \textbf{experimental} probability comes from trials, and the two agree over many repetitions.
\lh{The complement}The chance an event does \emph{not} happen is $1$ minus the chance it does.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Single outcome}
$P$ of rolling a $4$ on a die.\soln $\dfrac16$.
\end{example}

\begin{example}{ex1}{Example 2 --- Several favourable}
$P$ of an even number on a die.\soln $2,4,6$: $\dfrac36=\dfrac12$.
\end{example}

\begin{example}{ex2}{Example 3 --- Complement}
$P$ of not rolling a $4$.\soln $1-\dfrac16=\dfrac56$.
\end{example}

\begin{example}{ex3}{Example 4 --- Experimental vs theoretical}
$50$ flips give $30$ heads.\soln Experimental $=\dfrac{30}{50}=0.6$ vs theoretical $0.5$; more flips drift toward $0.5$.
\end{example}

\begin{example}{ex4}{Example 5 --- Independent events}
$P$ of a $6$ then another $6$.\soln $\dfrac16\times\dfrac16=\dfrac1{36}$.
\end{example}

\begin{example}{ex5}{Example 6 --- From a deck}
$P$ of drawing a red card.\soln $\dfrac{26}{52}=\dfrac12$.
\end{example}

\begin{example}{ex6}{Example 7 --- Several favourable}
$P$ of a number less than $3$ on a die.\soln $1,2$: $\dfrac26=\dfrac13$.
\end{example}

\begin{example}{ex7}{Example 8 --- Complement}
If $P(\text{rain})=0.3$, find $P(\text{no rain})$.\soln $1-0.3=0.7$.
\end{example}

\begin{example}{ex8}{Example 9 --- Independent events}
$P$ of two heads in a row.\soln $\dfrac12\times\dfrac12=\dfrac14$.
\end{example}
