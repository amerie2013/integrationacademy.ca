\begin{ideabox}
Solve $ax^2+bx+c=0$ by factoring and the zero product property; the solutions are the x-intercepts.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Rearrange so one side is $0$.
  \item Factor (common factor first), then set each factor to $0$.
  \item Solutions are where the parabola meets the x-axis.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Set to zero}Rearrange the equation to $ax^2+bx+c=0$, then solve by factoring or the quadratic formula.
\lh{Zero-product property}Once factored, a product is zero only when a factor is zero, so each factor gives a solution.
\lh{The quadratic formula}$x=\dfrac{-b\pm\sqrt{b^2-4ac}}{2a}$ solves any quadratic, even one that will not factor.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Trinomial}
Solve $x^2-5x+6=0$.\soln $(x-2)(x-3)=0\Rightarrow x=2$ or $x=3$ — the x-intercepts:\eplot{-1}{6}{-2}{8}{\addplot[exblue,very thick,domain=0:5]{x^2-5*x+6};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(2,0) (3,0)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Difference of squares}
Solve $x^2-9=0$.\soln $(x-3)(x+3)=0\Rightarrow x=\pm3$.
\end{example}

\begin{example}{ex2}{Example 3 --- Negative constant}
Solve $x^2+2x-8=0$.\soln $(x+4)(x-2)=0\Rightarrow x=-4,2$.
\end{example}

\begin{example}{ex3}{Example 4 --- Common factor}
Solve $x^2-6x=0$.\soln $x(x-6)=0\Rightarrow x=0$ or $x=6$:\eplot{-2}{8}{-12}{8}{\addplot[exblue,very thick,domain=-1:7]{x^2-6*x};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(0,0) (6,0)};}
\end{example}

\begin{example}{ex4}{Example 5 --- Break-even}
Profit (in $\$1000$s) is $P=x^2-7x+10$. Where is $P=0$?\soln $(x-2)(x-5)=0\Rightarrow x=2$ or $x=5$.
\end{example}

\begin{example}{ex5}{Example 6 --- Trinomial}
Solve $x^2-7x+10=0$.\soln $(x-2)(x-5)=0\Rightarrow x=2,5$.
\end{example}

\begin{example}{ex6}{Example 7 --- Mixed signs}
Solve $x^2-x-12=0$.\soln $(x-4)(x+3)=0\Rightarrow x=4,-3$.
\end{example}

\begin{example}{ex7}{Example 8 --- Both negative}
Solve $x^2+5x+6=0$.\soln $(x+2)(x+3)=0\Rightarrow x=-2,-3$.
\end{example}

\begin{example}{ex8}{Example 9 --- Larger squares}
Solve $x^2-16=0$.\soln $(x-4)(x+4)=0\Rightarrow x=\pm4$.
\end{example}
