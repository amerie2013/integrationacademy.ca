{\LARGE\bfseries\color{titleblue}Worksheet 4.4 --- The Cosine Law}\\[2pt]
{\small Grade 11 Foundations for College Mathematics (MBF3C) \textperiodcentered{} Unit 4: Geometry and Trigonometry}
\end{center}

\begin{ideabox}
The cosine law solves SAS and SSS triangles: $c^2=a^2+b^2-2ab\cos C$ — Pythagoras plus a correction for the angle.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Find a side (SAS): $c^2=a^2+b^2-2ab\cos C$.
  \item Find an angle (SSS): $\cos C=\dfrac{a^2+b^2-c^2}{2ab}$.
  \item When $C=90^\circ$ it becomes $c^2=a^2+b^2$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{A wider tool}The \textbf{cosine law} $a^2=b^2+c^2-2bc\cos A$ solves triangles the sine law cannot start.
\lh{Two cases}Use it with two sides and their included angle (SAS), or with all three sides (SSS) to find an angle.
\lh{A familiar limit}At $90^\circ$ the cosine term vanishes and the law becomes the Pythagorean theorem.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Find a side (SAS)}
$a=7,\ b=9,\ C=50^\circ$. Find $c$.\soln $c^2=49+81-2(7)(9)\cos50^\circ\approx130-80.99=49.0\Rightarrow c\approx7.0$.\gtri{$A$}{$B$}{$50^\circ$}{$7$}{$9$}{$c$}
\end{example}

\begin{example}{ex1}{Example 2 --- Find an angle (SSS)}
$a=5,\ b=6,\ c=7$. Find $C$.\soln $\cos C=\dfrac{25+36-49}{2(5)(6)}=\dfrac{12}{60}=0.2\Rightarrow C\approx78.5^\circ$.\gtri{$A$}{$B$}{$C$}{$5$}{$6$}{$7$}
\end{example}

\begin{example}{ex2}{Example 3 --- Largest angle}
Sides $4,5,6$; find the largest angle (opposite $6$).\soln $\cos C=\dfrac{16+25-36}{2(4)(5)}=\dfrac{5}{40}=0.125\Rightarrow C\approx82.8^\circ$.
\end{example}

\begin{example}{ex3}{Example 4 --- Choose the law}
Given $a=8,\ c=11,\ B=35^\circ$, which law finds $b$?\soln Two sides and the angle between them (SAS): the cosine law, $b^2=8^2+11^2-2(8)(11)\cos35^\circ$.
\end{example}

\begin{example}{ex4}{Example 5 --- Application (two roads)}
Roads leave a town at $60^\circ$; cars drive $40$ km and $30$ km. Distance apart?\soln $d^2=1600+900-2400\cos60^\circ=1300\Rightarrow d=\sqrt{1300}\approx36.1$ km.
\end{example}

\begin{example}{ex5}{Example 6 --- Find a side (SAS)}
$a=6,\ b=8,\ C=60^\circ$. Find $c$.\soln $c^2=36+64-96(0.5)=52\Rightarrow c\approx7.21$.
\end{example}

\begin{example}{ex6}{Example 7 --- Find an angle (SSS)}
$a=4,\ b=5,\ c=6$. Find the angle opposite $6$.\soln $\cos C=\dfrac{16+25-36}{40}=0.125\Rightarrow C\approx82.8^\circ$.
\end{example}

\begin{example}{ex7}{Example 8 --- Find a side (SAS)}
$a=10,\ b=7,\ C=45^\circ$. Find $c$.\soln $c^2=100+49-140\cos45^\circ\approx49.0\Rightarrow c\approx7.0$.
\end{example}

\begin{example}{ex8}{Example 9 --- Why a correction}
Why is the cosine law "Pythagoras with a correction"?\soln The $-2ab\cos C$ term is $0$ when $C=90^\circ$, leaving $c^2=a^2+b^2$.
\end{example}
