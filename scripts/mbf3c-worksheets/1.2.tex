\begin{ideabox}
Vertex form $y=a(x-h)^2+k$ has vertex $(h,k)$; $a$ stretches/reflects, $h$ shifts horizontally, $k$ vertically.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $a$: stretch by $|a|$; reflect if $a<0$.
  \item $h$: horizontal shift (opposite to the sign). $k$: vertical shift.
  \item Standard form comes from expanding vertex form.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{The parabola}A \textbf{quadratic relation} graphs as a U-shaped \textbf{parabola}, with a vertex, an axis of symmetry, and up to two $x$-intercepts.
\lh{Opening and vertex}A positive leading coefficient opens the parabola up (a minimum); a negative one opens it down (a maximum).
\lh{Second differences}In a table, quadratics have equal \emph{second} differences, which distinguishes them from straight lines.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Read the vertex}
Vertex of $y=(x-3)^2+2$.\soln $h=3,\ k=2$: vertex $(3,2)$:\eplot{-1}{7}{-1}{10}{\addplot[exblue,very thick,domain=0.5:5.5]{(x-3)^2+2};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(3,2)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Describe}
Describe $y=-2(x+1)^2+5$ from $y=x^2$.\soln $a=-2$ reflect and stretch 2; $h=-1$ left 1; $k=5$ up 5.
\end{example}

\begin{example}{ex2}{Example 3 --- Vertex to standard}
Expand $y=(x-1)^2+4$.\soln $x^2-2x+1+4=x^2-2x+5$.
\end{example}

\begin{example}{ex3}{Example 4 --- Direction \& max}
For $y=-(x-2)^2+3$, state the direction and max.\soln $a<0$ opens down; vertex $(2,3)$: maximum 3:\eplot{-2}{6}{-6}{5}{\addplot[exblue,very thick,domain=-0.5:4.5]{-(x-2)^2+3};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(2,3)};}
\end{example}

\begin{example}{ex4}{Example 5 --- Write the equation}
Write $y=x^2$ shifted right 3 and down 2.\soln $h=3,\ k=-2$: $y=(x-3)^2-2$.
\end{example}

\begin{example}{ex5}{Example 6 --- Negative h}
Vertex of $y=(x+4)^2-2$.\soln $(-4,-2)$.
\end{example}

\begin{example}{ex6}{Example 7 --- Direction}
Does $y=3(x-1)^2$ open up or down?\soln $a=3>0$: up.
\end{example}

\begin{example}{ex7}{Example 8 --- Max value}
Max value of $y=-(x-5)^2+8$.\soln Vertex $(5,8)$, opens down: $8$.
\end{example}

\begin{example}{ex8}{Example 9 --- Standard form}
Write $y=(x+2)^2-7$ in standard form.\soln $x^2+4x+4-7=x^2+4x-3$.
\end{example}
