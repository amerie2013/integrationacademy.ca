\begin{center}
{\LARGE\bfseries\color{titleblue}Worksheet 3.1 --- Simple \& Compound Interest}\\[2pt]
{\small Grade 11 Foundations for College Mathematics (MBF3C) \textperiodcentered{} Unit 3: Personal Finance}
\end{center}

\begin{ideabox}
Simple interest $I=Prt$ grows linearly; compound interest $A=P(1+i)^n$ grows exponentially.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Simple: $I=Prt$, $A=P(1+rt)$.
  \item Compound: $A=P(1+i)^n$, with $i=\dfrac{r}{\text{periods/yr}}$, $n=(\text{periods/yr})\times\text{years}$.
  \item Total interest $I=A-P$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Simple interest}\textbf{Simple interest} is earned only on the principal, $I=Prt$, growing the balance in a straight line.
\lh{Compound interest}\textbf{Compound interest} earns interest on the interest too, $A=P(1+i)^n$, curving upward over time.
\lh{Comparing}Compound interest eventually beats simple interest because its base keeps growing.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Simple interest}
Interest on \$500 at $4\%$ for 3 years.\soln $I=Prt=500(0.04)(3)=\$60$.
\end{example}

\begin{example}{ex1}{Example 2 --- Compound (annual)}
\$1000 at $5\%$ compounded annually for 3 years.\soln $A=1000(1.05)^3\approx\$1157.63$.
\end{example}

\begin{example}{ex2}{Example 3 --- Quarterly}
\$1000 at $6\%$ compounded quarterly for 1 year.\soln $i=\dfrac{0.06}{4}=0.015,\ n=4$: $1000(1.015)^4\approx\$1061.36$.
\end{example}

\begin{example}{ex3}{Example 4 --- Total interest}
Total interest in the previous annual example.\soln $I=A-P=1157.63-1000\approx\$157.63$.
\end{example}

\begin{example}{ex4}{Example 5 --- Simple vs compound}
Compare \$1000 at $5\%$ for 5 years, simple vs compound.\soln Simple $1000(1.25)=\$1250$; compound $1000(1.05)^5\approx\$1276.28$ — the curve beats the line:\eplot{0}{20}{0}{4}{\addplot[qorange,very thick,domain=0:20]{1+0.05*x};\addplot[exblue,very thick,domain=0:20]{1.05^x};}
\end{example}

\begin{example}{ex5}{Example 6 --- Periodic rate}
Find $i$ for $6\%$ compounded monthly.\soln $i=\dfrac{0.06}{12}=0.005$.
\end{example}

\begin{example}{ex6}{Example 7 --- Simple amount}
Amount: \$1000 at $5\%$ simple for 4 years.\soln $A=1000(1+0.20)=\$1200$.
\end{example}

\begin{example}{ex7}{Example 8 --- Semi-annual}
\$2000 at $4\%$ compounded semi-annually for 1 year.\soln $i=0.02,\ n=2$: $2000(1.02)^2\approx\$2080.80$.
\end{example}

\begin{example}{ex8}{Example 9 --- Number of periods}
Find $n$ for $6\%$ compounded monthly over 2 years.\soln $n=12\times2=24$.
\end{example}
