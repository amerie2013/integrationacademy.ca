\begin{ideabox}
Compare account fees and investments (risk vs return), and find the real cost of buying on credit.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Monthly interest rate $=\dfrac{\text{annual rate}}{12}$; for partial periods use $I=Prt$ ($t$ in years).
  \item Higher potential return usually means higher risk.
  \item Total each option's cost and pick the lower.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Saving and investing}Money set aside earns interest, and the longer it compounds, the more it grows.
\lh{The cost of credit}Borrowing on credit charges interest, so a balance carried from month to month grows quickly.
\lh{Compare the total}When choosing a loan or card, compare the \emph{total} you repay, not just the monthly amount.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Card interest (1 month)}
A \$1500 balance at $19.9\%$/yr is unpaid for one month.\soln Monthly rate $=\dfrac{0.199}{12}\approx0.01658$; interest $=1500(0.01658)\approx\$24.88$.
\end{example}

\begin{example}{ex1}{Example 2 --- Cost on credit}
A \$1500 purchase is paid 55 days late at $19.9\%$/yr (simple).\soln $I=Prt=1500(0.199)\left(\dfrac{55}{365}\right)\approx\$45.00$.
\end{example}

\begin{example}{ex2}{Example 3 --- A GIC}
\$2000 in a GIC at $3\%$ compounded annually for 2 years.\soln $2000(1.03)^2\approx\$2121.80$.
\end{example}

\begin{example}{ex3}{Example 4 --- Choosing an account}
Account A: flat \$10/month. Account B: \$0.50/transaction, 25 transactions.\soln B $=25(0.50)=\$12.50>\$10$, so Account A.
\end{example}

\begin{example}{ex4}{Example 5 --- Comparing two cards}
On a \$1000 balance unpaid one month, compare $19.9\%$ and $21.5\%$.\soln A: $1000\cdot\tfrac{0.199}{12}\approx\$16.58$; B: $1000\cdot\tfrac{0.215}{12}\approx\$17.92$. Card A is cheaper.
\end{example}

\begin{example}{ex5}{Example 6 --- Card interest}
Interest on a \$2000 balance at $18\%$/yr for one month.\soln $2000\cdot\tfrac{0.18}{12}=\$30$.
\end{example}

\begin{example}{ex6}{Example 7 --- A GIC}
\$3000 in a GIC at $4\%$ compounded annually for 2 years.\soln $3000(1.04)^2\approx\$3244.80$.
\end{example}

\begin{example}{ex7}{Example 8 --- Risk vs return}
Which is riskier: a savings account or stocks?\soln Stocks — higher potential return, higher risk.
\end{example}

\begin{example}{ex8}{Example 9 --- Monthly rate}
Find the monthly rate for a $24\%$/yr card.\soln $\dfrac{0.24}{12}=0.02=2\%$ per month.
\end{example}
