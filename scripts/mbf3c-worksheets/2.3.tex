{\LARGE\bfseries\color{titleblue}Worksheet 2.3 --- Exponential Growth \& Decay Applications}\\[2pt]
{\small Grade 11 Foundations for College Mathematics (MBF3C) \textperiodcentered{} Unit 2: Exponential Relations}
\end{center}

\begin{ideabox}
Model with $y=a\cdot b^t$: $a$ is the initial amount, $b=1+r$ for growth or $1-r$ for decay; half-life uses $\left(\tfrac12\right)^{t/h}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Growth factor $1+r$; decay factor $1-r$.
  \item Doubling: $b=2$. Half-life $h$: $b=\tfrac12$ over each period.
  \item Compound interest $A=P(1+i)^n$ is exponential.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Percentage change}Repeated percentage change follows $A=A_0 b^{\,t}$, with $b=1+r$ for growth and $b=1-r$ for decay.
\lh{Half-life and doubling}Halving uses base $\tfrac12$ and doubling uses base $2$, scaled by the time interval.
\lh{Using the model}Substitute a time to predict an amount; the base shows how fast the quantity changes.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Doubling}
$P=200(2)^t$. Find $P$ after 3 years.\soln $200(2)^3=200(8)=1600$. Doubling has the shape of $y=2^x$:\eplot{-1}{4}{-1}{9}{\addplot[exblue,very thick,domain=-1:3.2]{2^x};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(0,1) (1,2) (2,4) (3,8)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Half-life}
80 g halves every 5 years. How much after 10 years?\soln $10\div5=2$ half-lives: $80\left(\tfrac12\right)^2=20$ g.
\end{example}

\begin{example}{ex2}{Example 3 --- Growth factor}
A town grows $8\%$ per year. State the growth factor.\soln $b=1+0.08=1.08$.
\end{example}

\begin{example}{ex3}{Example 4 --- Decay factor}
A car loses $15\%$ each year. State the decay factor.\soln $b=1-0.15=0.85$.
\end{example}

\begin{example}{ex4}{Example 5 --- Bouncing ball}
Height after $n$ bounces: $h=100(0.6)^n$ cm. Find $h$ after 3 bounces.\soln $100(0.6)^3=100(0.216)=21.6$ cm.
\end{example}

\begin{example}{ex5}{Example 6 --- Compound interest}
\$1000 grows at $5\%$/year for 2 years.\soln $1000(1.05)^2\approx\$1102.50$.
\end{example}

\begin{example}{ex6}{Example 7 --- Doubling}
$P=200(2)^t$ after 4 years.\soln $200(16)=3200$.
\end{example}

\begin{example}{ex7}{Example 8 --- Half-life}
100 g halves every 3 h. After 9 h?\soln $9\div3=3$: $100\left(\tfrac12\right)^3=12.5$ g.
\end{example}

\begin{example}{ex8}{Example 9 --- Growth factor}
Growth of $5\%$ per year.\soln $b=1.05$.
\end{example}
