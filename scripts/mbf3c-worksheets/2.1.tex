{\small Grade 11 Foundations for College Mathematics (MBF3C) \textperiodcentered{} Unit 2: Exponential Relations}
\end{center}

\begin{ideabox}
Exponent laws combine powers, including zero and negative exponents.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Product $x^m x^n=x^{m+n}$; quotient $\dfrac{x^m}{x^n}=x^{m-n}$; power $(x^m)^n=x^{mn}$.
  \item $x^0=1$ and $x^{-n}=\dfrac{1}{x^n}$.
  \item A negative exponent means reciprocal, not a negative number.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Powers}A power $a^n$ is repeated multiplication; same-base powers combine as $a^m a^n=a^{m+n}$ and $(a^m)^n=a^{mn}$.
\lh{Zero and negatives}$a^0=1$ and $a^{-n}=\dfrac{1}{a^n}$ handle the special cases.
\lh{Simplify step by step}Apply one law at a time, and rewrite negative exponents as fractions last.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Product}
Simplify $x^5\cdot x^3$.\soln Add the exponents: $x^{5+3}=x^8$.
\end{example}

\begin{example}{ex1}{Example 2 --- Quotient}
Simplify $\dfrac{x^9}{x^4}$.\soln Subtract the exponents: $x^{9-4}=x^5$.
\end{example}

\begin{example}{ex2}{Example 3 --- Power of a power}
Simplify $(x^4)^3$.\soln Multiply the exponents: $x^{4\cdot3}=x^{12}$.
\end{example}

\begin{example}{ex3}{Example 4 --- Power of a product}
Simplify $(2x^3)^2$.\soln The exponent reaches both factors: $2^2 x^{6}=4x^6$.
\end{example}

\begin{example}{ex4}{Example 5 --- Zero exponent}
Evaluate $5^0$.\soln Any non-zero base to the power 0 is $1$.
\end{example}

\begin{example}{ex5}{Example 6 --- Negative exponent}
Evaluate $2^{-3}$.\soln Reciprocal: $\dfrac{1}{2^3}=\dfrac18$.
\end{example}

\begin{example}{ex6}{Example 7 --- Negative exponent}
Evaluate $5^{-2}$.\soln $\dfrac{1}{5^2}=\dfrac{1}{25}$.
\end{example}

\begin{example}{ex7}{Example 8 --- Combine laws}
Simplify $x^{-3}\cdot x^{7}$.\soln Add exponents: $x^{-3+7}=x^4$.
\end{example}

\begin{example}{ex8}{Example 9 --- Growth vs decay}
Sketch the shapes of $y=2^x$ and $y=0.5^x$.\soln Base $>1$ grows; base $<1$ decays; both pass $(0,1)$:\eplot{-3}{3}{-1}{9}{\addplot[exblue,very thick,domain=-3:3]{2^x};\addplot[qorange,very thick,domain=-3:3]{0.5^x};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(0,1)};}
\end{example}
