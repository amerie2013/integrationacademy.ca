\begin{ideabox}
The vertex gives the maximum (if $a<0$) or minimum (if $a>0$); the x-intercepts give break-even or landing times.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $a>0$ minimum; $a<0$ maximum, at the vertex.
  \item Find the vertex with $x=-\dfrac{b}{2a}$, then substitute.
  \item Set the relation to $0$ for break-even / landing.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Modelling with quadratics}Heights, areas, and revenue often follow a quadratic, so their maximum or minimum sits at the vertex.
\lh{Vertex or zeros}A greatest value points to the \textbf{vertex}; a start or a landing points to the \textbf{zeros}.
\lh{Interpret in context}Translate the answer back into the situation and keep only sensible (positive) values.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Read a max}
Find the maximum of $y=-(x-3)^2+9$.\soln Vertex $(3,9)$, opens down: maximum 9 at $x=3$:\eplot{-1}{7}{-2}{11}{\addplot[exblue,very thick,domain=0:6]{-(x-3)^2+9};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(3,9)};}
\end{example}

\begin{example}{ex1}{Example 2 --- Max height}
A ball: $h=-5t^2+20t$. Max height?\soln $t=-\dfrac{20}{2(-5)}=2$; $h=-20+40=20$ m:\eplot{-0.5}{4.5}{-2}{24}{\addplot[exblue,very thick,domain=0:4]{-5*x^2+20*x};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(2,20)};}
\end{example}

\begin{example}{ex2}{Example 3 --- When it lands}
Same ball: when does it hit the ground?\soln $-5t(t-4)=0\Rightarrow t=0$ or $t=4$ s.
\end{example}

\begin{example}{ex3}{Example 4 --- Max profit}
Profit $P=-2x^2+12x-10$. Max profit?\soln $x=3$; $P=-18+36-10=8$.
\end{example}

\begin{example}{ex4}{Example 5 --- Max area}
Perimeter 20 m rectangle. Max area?\soln $A=w(10-w)$, vertex $w=5$: $5\times5$, area 25 m\textsuperscript{2}.
\end{example}

\begin{example}{ex5}{Example 6 --- Find a min}
Min of $y=x^2-6x+5$.\soln $x=3$; $y=9-18+5=-4$.
\end{example}

\begin{example}{ex6}{Example 7 --- Max height}
$h=-5t^2+30t$. Max height?\soln $t=3$; $h=45$ m.
\end{example}

\begin{example}{ex7}{Example 8 --- Break-even}
$P=x^2-8x+12$. Break-even values?\soln $(x-2)(x-6)=0\Rightarrow x=2,6$.
\end{example}

\begin{example}{ex8}{Example 9 --- Max area}
Perimeter 24 m rectangle. Max area?\soln $6\times6=36$ m\textsuperscript{2}.
\end{example}
