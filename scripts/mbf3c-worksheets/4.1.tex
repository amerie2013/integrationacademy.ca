\begin{ideabox}
Volume measures the space inside; surface area measures the outside. Prisms and cylinders multiply base by height; cones and pyramids take a third; spheres have their own formulas.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Cylinder: $V=\pi r^2 h$, $SA=2\pi r^2+2\pi r h$.
  \item Cone: $V=\tfrac13\pi r^2 h$, $SA=\pi r^2+\pi r\ell$. Sphere: $V=\tfrac43\pi r^3$, $SA=4\pi r^2$.
  \item Composite solids: add (or subtract) the pieces.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Volume}\textbf{Volume} measures the space a solid fills: prisms and cylinders use base area times height, and a sphere uses $\tfrac43\pi r^3$.
\lh{Surface area}\textbf{Surface area} totals the outside faces and is measured in square units.
\lh{Match the solid}Pick the formula that fits the shape, and keep every measurement in the same unit.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Cylinder volume}
A can has $r=5$ cm, $h=12$ cm.\soln $V=\pi r^2 h=\pi(25)(12)=300\pi\approx942.5$ cm³.
\end{example}

\begin{example}{ex1}{Example 2 --- Cylinder surface area}
Surface area of the same can.\soln $SA=2\pi r^2+2\pi rh=50\pi+120\pi=170\pi\approx534.1$ cm².
\end{example}

\begin{example}{ex2}{Example 3 --- Cone (slant height first)}
A cone has $r=6$, $h=8$. Find its surface area.\soln Slant $\ell=\sqrt{r^2+h^2}=\sqrt{36+64}=10$, then $SA=\pi r^2+\pi r\ell=36\pi+60\pi=96\pi\approx301.6$.\rtri{6}{8}{$h=8$}{$r=6$}{$\ell=10$}
\end{example}

\begin{example}{ex3}{Example 4 --- Sphere volume}
Volume of a sphere with $r=6$.\soln $V=\tfrac43\pi(216)=288\pi\approx904.8$.
\end{example}

\begin{example}{ex4}{Example 5 --- Work backwards}
A sphere has volume $36\pi$. Find $r$.\soln $\tfrac43\pi r^3=36\pi\Rightarrow r^3=27\Rightarrow r=3$.
\end{example}

\begin{example}{ex5}{Example 6 --- Composite solid}
A silo: cylinder ($r=3,h=10$) topped by a cone ($r=3,h=4$).\soln $V=\pi(9)(10)+\tfrac13\pi(9)(4)=90\pi+12\pi=102\pi\approx320.4$.
\end{example}

\begin{example}{ex6}{Example 7 --- Rectangular prism}
Volume of a $4\times3\times2$ cm box.\soln $V=lwh=4\cdot3\cdot2=24$ cm³.
\end{example}

\begin{example}{ex7}{Example 8 --- Cone volume}
Volume of a cone with $r=3$, $h=4$.\soln $V=\tfrac13\pi(9)(4)=12\pi\approx37.7$.
\end{example}

\begin{example}{ex8}{Example 9 --- Cube — reverse}
A cube has volume $125$ cm³. Find its side.\soln $s=\sqrt[3]{125}=5$ cm.
\end{example}
