{\small Grade 11 Foundations for College Mathematics (MBF3C) \textperiodcentered{} Unit 2: Exponential Relations}
\end{center}

\begin{ideabox}
$y=a\cdot b^x$ has y-intercept $(0,a)$, horizontal asymptote $y=0$, domain all reals, and range $y>0$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $b>1$ grows; $0<b<1$ decays; a larger base is steeper.
  \item The y-intercept is $a$ (since $b^0=1$).
  \item Exponential beats linear and quadratic for large $x$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Repeated multiplying}An \textbf{exponential relation} $y=a\,b^x$ multiplies by $b$ each step, curving sharply rather than rising steadily.
\lh{The graph}It passes through $(0,a)$ and levels off toward the asymptote $y=0$ without ever crossing it.
\lh{Growth or decay}A base $b>1$ grows, while a base between $0$ and $1$ decays.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Evaluate}
For $y=2^x$, find the value at $x=3$.\soln $2^3=8$.
\end{example}

\begin{example}{ex1}{Example 2 --- y-intercept}
State the y-intercept of $y=3\cdot2^x$.\soln At $x=0$, $2^0=1$, so $y=3$. y-intercept $(0,3)$.
\end{example}

\begin{example}{ex2}{Example 3 --- Growth or decay}
Is $y=\left(\tfrac12\right)^x$ growth or decay?\soln Base between 0 and 1, so decay:\eplot{-3}{3}{-1}{9}{\addplot[exblue,very thick,domain=-3:3]{0.5^x};\addplot[red,only marks,mark=*,mark size=1.6pt] coordinates {(0,1)};}
\end{example}

\begin{example}{ex3}{Example 4 --- Asymptote}
State the horizontal asymptote of $y=2^x$.\soln As $x\to-\infty$, $2^x\to0$. Asymptote $y=0$:\eplot{-4}{3}{-1}{9}{\addplot[exblue,very thick,domain=-4:3]{2^x};}
\end{example}

\begin{example}{ex4}{Example 5 --- Distinguish relations}
Which grows fastest for large $x$: $2x$, $x^2$, or $2^x$?\soln Multiplying beats adding or squaring eventually: $2^x$.
\end{example}

\begin{example}{ex5}{Example 6 --- Compare bases}
Which grows faster, $2^x$ or $5^x$?\soln Both pass $(0,1)$; the larger base is steeper: $5^x$.
\end{example}

\begin{example}{ex6}{Example 7 --- Domain \& range}
Domain and range of $y=2^x$.\soln Any $x$; outputs positive. Domain all reals; range $y>0$.
\end{example}

\begin{example}{ex7}{Example 8 --- Negative input}
Find $2^x$ at $x=-1$.\soln $2^{-1}=\dfrac12$.
\end{example}

\begin{example}{ex8}{Example 9 --- y-intercept}
State the y-intercept of $y=5\cdot2^x$.\soln $5\cdot1=5$: $(0,5)$.
\end{example}
