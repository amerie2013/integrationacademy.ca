{\LARGE\bfseries\color{titleblue}Worksheet 4.3 --- The Sine Law}\\[2pt]
{\small Grade 11 Foundations for College Mathematics (MBF3C) \textperiodcentered{} Unit 4: Geometry and Trigonometry}
\end{center}

\begin{ideabox}
The sine law solves any triangle where a side is paired with its opposite angle: $\dfrac{a}{\sin A}=\dfrac{b}{\sin B}=\dfrac{c}{\sin C}$.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $\dfrac{a}{\sin A}=\dfrac{b}{\sin B}=\dfrac{c}{\sin C}$.
  \item Use it with a side and its opposite angle, plus one more piece.
  \item Angles sum to $180^\circ$.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Any triangle}The \textbf{sine law} $\dfrac{a}{\sin A}=\dfrac{b}{\sin B}=\dfrac{c}{\sin C}$ links each side to the sine of its opposite angle.
\lh{When to use it}Use it when you know a side and its opposite angle, plus one more piece of information.
\lh{Match the pair}Always pair a side with the angle across from it to keep the ratios valid.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Find a side}
$A=40^\circ,\ a=10,\ B=60^\circ$. Find $b$.\soln $b=\dfrac{a\sin B}{\sin A}=\dfrac{10\sin60^\circ}{\sin40^\circ}\approx13.5$.\gtri{$40^\circ$}{$60^\circ$}{$C$}{$10$}{$b$}{$c$}
\end{example}

\begin{example}{ex1}{Example 2 --- Find an angle}
$a=8,\ A=50^\circ,\ b=6$. Find $B$.\soln $\sin B=\dfrac{b\sin A}{a}=\dfrac{6\sin50^\circ}{8}\approx0.5745$, so $B=\sin^{-1}(0.5745)\approx35.1^\circ$.\gtri{$50^\circ$}{$B$}{$C$}{$8$}{$6$}{$c$}
\end{example}

\begin{example}{ex2}{Example 3 --- Third angle first}
$A=45^\circ,\ B=60^\circ,\ a=12$. Find $c$.\soln $C=75^\circ$; $c=\dfrac{12\sin75^\circ}{\sin45^\circ}\approx16.4$.
\end{example}

\begin{example}{ex3}{Example 4 --- Two angles}
$A=40^\circ,\ B=75^\circ$. Find $C$.\soln $C=180^\circ-40^\circ-75^\circ=65^\circ$.
\end{example}

\begin{example}{ex4}{Example 5 --- Find a side}
$A=30^\circ,\ a=5,\ B=80^\circ$. Find $b$.\soln $b=\dfrac{5\sin80^\circ}{\sin30^\circ}\approx9.85$.
\end{example}

\begin{example}{ex5}{Example 6 --- Find an angle}
$a=10,\ A=70^\circ,\ b=8$. Find $B$.\soln $\sin B=\dfrac{8\sin70^\circ}{10}\approx0.7518\Rightarrow B\approx48.8^\circ$.
\end{example}

\begin{example}{ex6}{Example 7 --- Application}
From two points $50$ m apart, the angles to a tree are $70^\circ$ and $60^\circ$ (tree angle $50^\circ$). Distance from the first point?\soln $d=\dfrac{50\sin60^\circ}{\sin50^\circ}\approx56.5$ m.
\end{example}

\begin{example}{ex7}{Example 8 --- Find a side}
$A=60^\circ,\ a=10,\ B=40^\circ$. Find $b$.\soln $b=\dfrac{10\sin40^\circ}{\sin60^\circ}\approx7.42$.
\end{example}

\begin{example}{ex8}{Example 9 --- Set it up}
For $A=35^\circ,\ a=9,\ b=7$, write the equation for $B$.\soln $\dfrac{\sin B}{7}=\dfrac{\sin35^\circ}{9}\Rightarrow\sin B=\dfrac{7\sin35^\circ}{9}$.
\end{example}
