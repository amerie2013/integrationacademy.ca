\begin{ideabox}
Study a population through a representative sample, chosen by a method that avoids bias; then pick the graph that suits the data.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Population = everyone of interest; sample = the part surveyed.
  \item Sampling: simple random, systematic (every $k$-th), stratified (by group, in proportion).
  \item Bar = compare categories; circle = parts of a whole; histogram = distribution; scatter = relationship.
\end{itemize}
\end{ideabox}

\begin{learnbox}
\lh{Gathering data}Data is collected through surveys, experiments, or existing records; a fair sample represents the whole group.
\lh{Choosing a graph}\textbf{Bar graphs} compare categories, \textbf{histograms} show numeric ranges, and \textbf{circle graphs} show parts of a whole.
\lh{Read carefully}A well-labelled graph makes patterns clear, but watch for a misleading scale.
\end{learnbox}

\sech{Worked Examples}

\begin{example}{ex0}{Example 1 --- Population vs sample}
A principal surveys $200$ of $1500$ students.\soln Population $=1500$ students; sample $=200$ students.
\end{example}

\begin{example}{ex1}{Example 2 --- Name the method}
A factory tests every $50$th item.\soln Fixed interval $\Rightarrow$ systematic sampling.
\end{example}

\begin{example}{ex2}{Example 3 --- Spot the bias}
Asking only hockey fans the town's favourite sport.\soln Biased — the sample over-represents hockey fans (see the survey below); poll a random cross-section instead.\begin{center}\begin{tikzpicture}[scale=0.9]
\draw[->] (0,0)--(0,4.2) node[above]{\small frequency};
\draw[->] (0,0)--(6,0) node[right]{\small sport};
\foreach \y in {1,2,3,4}\draw (0,\y)--(-0.08,\y) node[left]{\small\y0};
\fill[exblue] (0.6,0) rectangle (1.4,3.0);
\fill[exblue] (1.9,0) rectangle (2.7,4.0);
\fill[exblue] (3.2,0) rectangle (4.0,2.0);
\fill[exblue] (4.5,0) rectangle (5.3,1.0);
\node[below] at (1.0,0){\small Soccer};
\node[below] at (2.3,0){\small Hockey};
\node[below] at (3.6,0){\small Basket};
\node[below] at (4.9,0){\small Other};
\end{tikzpicture}\end{center}
\end{example}

\begin{example}{ex3}{Example 4 --- Circle-graph angle}
$25\%$ chose pizza. Find its central angle.\soln $0.25\times360^\circ=90^\circ$.
\end{example}

\begin{example}{ex4}{Example 5 --- Choose the graph}
(a) sales by store, (b) fraction of a budget per item, (c) height vs shoe size.\soln (a) bar, (b) circle, (c) scatter.
\end{example}

\begin{example}{ex5}{Example 6 --- Population vs sample}
A poll surveys $500$ of $40\,000$ voters.\soln Population $40\,000$; sample $500$.
\end{example}

\begin{example}{ex6}{Example 7 --- Name the method}
Drawing names from a hat.\soln Simple random sampling.
\end{example}

\begin{example}{ex7}{Example 8 --- Circle-graph angle}
A category is $40\%$ of the data.\soln $0.40\times360^\circ=144^\circ$.
\end{example}

\begin{example}{ex8}{Example 9 --- Choose the graph}
Best graph for the distribution of $100$ test scores?\soln A histogram (grouped numeric data).
\end{example}
